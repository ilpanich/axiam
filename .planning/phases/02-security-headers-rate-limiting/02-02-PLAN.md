    ```rust
                register_per_min: 5,
                token_per_min: 20,
                password_reset_per_min: 3,
            }
        }
    ```

    ```rust
    //! Rate-limit key extractor using X-Forwarded-For header (per D-02).

    use actix_governor::KeyExtractor;
    use actix_web::dev::ServiceRequest;
    use std::net::IpAddr;

    /// Extracts client IP from X-Forwarded-For header, falls back to peer address.
    #[derive(Debug, Clone)]
    pub struct XForwardedForKeyExtractor;

    impl KeyExtractor for XForwardedForKeyExtractor {
        type Key = IpAddr;
        type KeyExtractionError = actix_web::Error;

        fn extract(&self, req: &ServiceRequest) -> Result<Self::Key, Self::KeyExtractionError> {
            if let Some(forwarded_for) = req.headers().get("X-Forwarded-For") {
                if let Ok(val) = forwarded_for.to_str() {
                    if let Some(first) = val.split(',').next() {
                        if let Ok(ip) = first.trim().parse::<IpAddr>() {
                            return Ok(ip);
                        }
                    }
                }
            }
            req.peer_addr()
                .map(|addr| addr.ip())
                .ok_or_else(|| actix_web::error::ErrorInternalServerError("no peer address"))
        }
    }
    ```

    In `crates/axiam-api-rest/src/server.rs`, add helper functions that build Governor instances from RateLimitConfig. Each endpoint gets its own GovernorConfig with its own in-memory store — NEVER share configs between endpoints.

    ```rust
    use actix_governor::{Governor, GovernorConfigBuilder};
    use crate::extractors::rate_limit::XForwardedForKeyExtractor;
    use crate::config::RateLimitConfig;

    fn build_governor(requests_per_min: u32) -> Governor<XForwardedForKeyExtractor> {
        let replenish_interval = 60 / requests_per_min.max(1);
        let config = GovernorConfigBuilder::default()
            .per_second(replenish_interval as u64)
            .burst_size(requests_per_min)
            .key_extractor(XForwardedForKeyExtractor)
            .finish()
            .expect("valid governor config");
        Governor::new(&config)
    }
    ```

    IMPORTANT: Check the exact actix-governor 0.10 API. `GovernorConfigBuilder` may use different method names. Consult `cargo doc -p actix-governor --open` if needed. The key is: `per_second()` sets the replenishment rate, `burst_size()` sets max tokens.
