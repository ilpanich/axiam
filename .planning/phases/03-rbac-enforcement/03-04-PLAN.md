  <action>
    **1. Create `frontend/src/hooks/usePermissions.ts`** (per UI-SPEC New Hook Contract):

    ```typescript
    import { useAuthStore } from '@/stores/auth';

    export function usePermissions() {
      const permissions = useAuthStore((s) => s.user?.permissions ?? []);
      const isLoading = useAuthStore((s) => s.isInitializing);

      const can = (permission: string): boolean => {
        return permissions.includes(permission) || permissions.includes('*');
      };

      return { can, permissions, isLoading };
    }
    ```

    **2. Update `frontend/src/components/layout/Sidebar.tsx`** (per UI-SPEC Sidebar Permission Gating Rules):
