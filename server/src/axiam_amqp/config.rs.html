3use axiam_core::error::AxiamError;
4use serde::Deserialize;
5
6/// Documented dev/test-only default AMQP master signing key (SECHRD-08 /
7/// D-05c). Used ONLY as a fallback in debug builds (`cfg!(debug_assertions)`
8/// — i.e. never in the `cargo build --release` binary that ships in the
9/// production container image, see `docker/Dockerfile.server`) when
10/// `AXIAM__AMQP__SIGNING_KEY` is unset, so local dev/test runs work without
11/// extra setup. This key MUST NOT be used in production — see
12/// [`AmqpConfig::resolve_signing_key`], which fails closed instead of
13/// falling back to this constant in a release build.
14const DEV_DEFAULT_SIGNING_KEY: &[u8] = b"axiam-dev-only-amqp-signing-key-DO-NOT-USE-IN-PROD";
15
16/// Configuration for connecting to RabbitMQ.
17#[derive(Debug, Clone, Deserialize)]
18#[serde(default)]
19pub struct AmqpConfig {
20    /// AMQP connection URI (e.g., `amqp://localhost:5672`).
21    pub url: String,
22    /// Channel prefetch count for consumers.
23    pub prefetch_count: u16,
24    /// Delay between reconnection attempts in milliseconds.
25    pub reconnect_delay_ms: u64,
26    /// Maximum number of connection retries before giving up.
27    pub max_retries: u32,
28    /// HMAC-SHA256 master signing key for authenticating AMQP message
29    /// payloads (SEC-022/055, SECHRD-08). Set via
30    /// `AXIAM__AMQP__SIGNING_KEY` (hex-encoded key). Signing is mandatory —
31    /// there is no unsigned code path (D-05c). Use
32    /// [`AmqpConfig::resolve_signing_key`] to resolve this to the mandatory
33    /// key that consumers/producers must use: a debug-build-only dev
34    /// default is used when unset in development, but a release build
35    /// fails closed (`AxiamError::ServiceUnavailable`) if unset.
36    #[serde(default)]
37    pub signing_key: Option<String>,
38    /// Freshness skew window (seconds) for AMQP replay protection (NEW-4). An
39    /// `AuthzRequest`/`AuditEventMessage` is accepted only when its `issued_at`
40    /// lies within ±this many seconds of the consumer's clock. Set via
41    /// `AXIAM__AMQP__REPLAY_SKEW_SECS`; defaults to
42    /// [`crate::messages::DEFAULT_FRESHNESS_SKEW_SECS`] (5 minutes).
43    #[serde(default = "default_replay_skew_secs")]
44    pub replay_skew_secs: u64,
45}
46
47fn default_replay_skew_secs() -> u64 {
48    crate::messages::DEFAULT_FRESHNESS_SKEW_SECS as u64
49}
50
51impl Default for AmqpConfig {
52    fn default() -> Self {
53        Self {
54            url: "amqp://localhost:5672".into(),
55            prefetch_count: 10,
56            reconnect_delay_ms: 5000,
57            max_retries: 5,
58            signing_key: None,
59            replay_skew_secs: default_replay_skew_secs(),
60        }
61    }
62}
63
64impl AmqpConfig {
65    /// Resolve the configured signing key to a mandatory master key
66    /// (SECHRD-08 / D-05c).
67    ///
68    /// - If `signing_key` is set, hex-decode and return it (an invalid hex
69    ///   value is a misconfiguration and fails closed).
70    /// - If unset in a debug build (`cfg!(debug_assertions)` — dev/test),
71    ///   fall back to the documented [`DEV_DEFAULT_SIGNING_KEY`] so local
72    ///   runs work without extra setup.
73    /// - If unset in a release build (the production container image builds
74    ///   with `cargo build --release`), fail closed: return
75    ///   `AxiamError::ServiceUnavailable` — AMQP signing is mandatory in
76    ///   production, there is no unsigned/zero-key code path.
77    pub fn resolve_signing_key(&self) -> Result<Vec<u8>, AxiamError> {
78        match &self.signing_key {
79            Some(hex_key) => hex::decode(hex_key).map_err(|e| {
80                AxiamError::ServiceUnavailable(format!(
81                    "AXIAM__AMQP__SIGNING_KEY is not valid hex: {e}"
82                ))
83            }),
84            None if cfg!(debug_assertions) => {
85                tracing::warn!(
86                    "AXIAM__AMQP__SIGNING_KEY not set — using dev-only default AMQP signing key \
87                     (NOT valid in a release/production build, SECHRD-08 / D-05c)"
88                );
89                Ok(DEV_DEFAULT_SIGNING_KEY.to_vec())
90            }
91            None => Err(AxiamError::ServiceUnavailable(
92                "AMQP signing key not configured (AXIAM__AMQP__SIGNING_KEY) — mandatory in \
93                 production (SECHRD-08 / D-05c)"
94                    .to_string(),
95            )),
96        }
97    }
98
99    /// Resolve the configured freshness skew to a `chrono::Duration` for the
100    /// AMQP replay-protection gate (NEW-4).
101    pub fn replay_skew(&self) -> chrono::Duration {
102        chrono::Duration::seconds(self.replay_skew_secs as i64)
103    }
104}

106#[cfg(test)]
107mod tests {
108    use super::*;
109
110    #[test]
111    fn resolve_signing_key_decodes_configured_hex() {
112        let cfg = AmqpConfig {
113            signing_key: Some(hex::encode(b"a-32-byte-ish-test-signing-key!")),
114            ..AmqpConfig::default()
115        };
116        let key = cfg
117            .resolve_signing_key()
118            .expect("valid hex key must resolve");
119        assert_eq!(key, b"a-32-byte-ish-test-signing-key!".to_vec());
120    }
121
122    #[test]
123    fn resolve_signing_key_rejects_invalid_hex() {
124        let cfg = AmqpConfig {
125            signing_key: Some("not-valid-hex!!".to_string()),
126            ..AmqpConfig::default()
127        };
128        assert!(
129            cfg.resolve_signing_key().is_err(),
130            "invalid hex signing key must fail closed, not silently ignore"
131        );
132    }
133
134    #[test]
135    fn resolve_signing_key_falls_back_to_dev_default_when_unset_in_debug_build() {
136        let cfg = AmqpConfig {
137            signing_key: None,
138            ..AmqpConfig::default()
139        };
140        // This test itself only runs in a debug build (`cargo test`), so the
141        // dev-default fallback branch applies here.
142        if cfg!(debug_assertions) {
143            let key = cfg
144                .resolve_signing_key()
145                .expect("debug build must fall back to the documented dev default");
146            assert_eq!(key, DEV_DEFAULT_SIGNING_KEY.to_vec());
147        }
148    }
