11/// Well-known queue names used by AXIAM.
12pub mod queues {
13    /// Inbound async authorization check requests.
14    pub const AUTHZ_REQUEST: &str = "axiam.authz.request";
15    /// Outbound authorization decision responses.
16    pub const AUTHZ_RESPONSE: &str = "axiam.authz.response";
17    /// Dead-letter queue for [`AUTHZ_REQUEST`] poison messages (CQ-B05).
18    pub const AUTHZ_REQUEST_DLQ: &str = "axiam.authz.request.dlq";
19    /// Inbound audit events from external services.
20    pub const AUDIT_EVENTS: &str = "axiam.audit.events";
21    /// Dead-letter queue for [`AUDIT_EVENTS`] poison messages (CQ-B05).
22    pub const AUDIT_EVENTS_DLQ: &str = "axiam.audit.events.dlq";
23    /// Outbound real-time event notifications.
24    pub const NOTIFICATIONS: &str = "axiam.notifications";
25    /// Outbound async mail delivery queue (D-14).
26    ///
27    /// Messages dead-letter to [`MAIL_OUTBOUND_DLQ`] when exhausted.
28    pub const MAIL_OUTBOUND: &str = "axiam.mail.outbound";
29    /// Dead-letter queue for [`MAIL_OUTBOUND`] exhausted-retry messages (D-14).
30    pub const MAIL_OUTBOUND_DLQ: &str = "axiam.mail.outbound.dlq";
31    /// Primary webhook delivery queue (CORR-03/D-07).
32    ///
33    /// A terminal nack (requeue=false, attempts exhausted) dead-letters to
34    /// [`WEBHOOK_DLQ`]. Declared and consumed via [`AmqpManager::declare_webhook_topology`].
35    pub const WEBHOOK: &str = "axiam.webhook";
36    /// Webhook retry-delay queue (CORR-03/D-07). No consumer is ever
37    /// attached to this queue — a message published here with a per-message
38    /// TTL (`x-message-ttl`/`BasicProperties::with_expiration`) dead-letters
39    /// back to [`WEBHOOK`] via the default exchange once the TTL expires,
40    /// giving RabbitMQ-native delayed retry without an in-process sleep
41    /// tying up a consumer slot (D-07/Pitfall 5).
42    pub const WEBHOOK_RETRY: &str = "axiam.webhook.retry";
43    /// Terminal dead-letter queue for exhausted webhook deliveries
44    /// (CORR-03/D-07). Messages here are real and replayable — not silently
45    /// dropped (see the DLX-wiring note on [`WEBHOOK`]/[`WEBHOOK_RETRY`]).
46    pub const WEBHOOK_DLQ: &str = "axiam.webhook.dlq";
47}

49/// Queues declared via the plain durable loop (no special arguments).
50///
51/// These are declared first so that the DLQ targets already exist when the
52/// primary queues with `x-dead-letter-exchange` are declared below.
53const ALL_QUEUES: &[&str] = &[
54    queues::AUTHZ_RESPONSE,
55    queues::NOTIFICATIONS,
56    // DLQs are plain-durable (no DLQ args of their own).
57    queues::AUDIT_EVENTS_DLQ,
58    queues::AUTHZ_REQUEST_DLQ,
59    queues::MAIL_OUTBOUND_DLQ,
60    // Primary queues with DLX are declared separately below.
61];

63/// Manages a RabbitMQ connection and channel.
64pub struct AmqpManager {
65    connection: Connection,
66    channel: Channel,
67    prefetch_count: u16,
68}
69
70impl AmqpManager {
71    /// Establish a connection to RabbitMQ and create a channel.
72    pub async fn connect(config: &AmqpConfig) -> Result<Self, AmqpError> {
73        info!("Connecting to RabbitMQ");
74
75        let connection = Connection::connect(&config.url, ConnectionProperties::default())
76            .await
77            .map_err(AmqpError::Connection)?;
78
79        let channel = connection
80            .create_channel()
81            .await
82            .map_err(AmqpError::Channel)?;
83
84        channel
85            .basic_qos(config.prefetch_count, BasicQosOptions::default())
86            .await
87            .map_err(AmqpError::Channel)?;
88
89        info!("Successfully connected to RabbitMQ");
90
91        Ok(Self {
92            connection,
93            channel,
94            prefetch_count: config.prefetch_count,
95        })
96    }

98    /// Connect with automatic retry on failure.
99    ///
100    /// Always attempts at least once. `max_retries` controls how many
101    /// additional attempts are made after the first failure.
102    pub async fn connect_with_retry(config: &AmqpConfig) -> Result<Self, AmqpError> {
103        let total_attempts = config.max_retries.saturating_add(1);
104        for attempt in 1..=total_attempts {
105            match Self::connect(config).await {
106                Ok(manager) => return Ok(manager),
107                Err(e) => {
108                    if attempt == total_attempts {
109                        let lapin_err = e.into_lapin_error();
110                        tracing::error!(
111                            error = %lapin_err,
112                            attempts = total_attempts,
113                            "Failed to connect to RabbitMQ after all retries"
114                        );
115                        return Err(AmqpError::MaxRetriesExhausted(lapin_err));
116                    }
117                    warn!(
118                        error = %e,
119                        attempt,
120                        max_retries = config.max_retries,
121                        delay_ms = config.reconnect_delay_ms,
122                        "RabbitMQ connection failed, retrying"
123                    );
124                    tokio::time::sleep(std::time::Duration::from_millis(config.reconnect_delay_ms))
125                        .await;
126                }
127            }
128        }
129        unreachable!("loop always returns")
130    }

132    /// Declare all AXIAM queues as durable.
133    ///
134    /// DLQ queues are declared first (plain durable, no DLQ args of their own),
135    /// then the primary queues are declared with `x-dead-letter-exchange`
136    /// pointing at their corresponding DLQs so that exhausted/rejected messages
137    /// dead-letter rather than being silently dropped or hot-looping (CQ-B05).
138    ///
139    /// Queues with dead-letter routing:
140    /// - `AUDIT_EVENTS`  → `AUDIT_EVENTS_DLQ`   (CQ-B05 / REQ-14 AC-5)
141    /// - `AUTHZ_REQUEST` → `AUTHZ_REQUEST_DLQ`   (CQ-B05 / REQ-14 AC-5)
142    /// - `MAIL_OUTBOUND` → `MAIL_OUTBOUND_DLQ`   (D-14)
143    pub async fn declare_queues(&self) -> Result<(), AmqpError> {
144        let options = QueueDeclareOptions {
145            durable: true,
146            ..QueueDeclareOptions::default()
147        };
148
149        // Declare plain-durable queues (DLQs and non-primary queues) first.
150        for queue in ALL_QUEUES {
151            self.channel
152                .queue_declare((*queue).into(), options, FieldTable::default())
153                .await
154                .map_err(AmqpError::Declaration)?;
155            info!(queue, "Declared queue");
156        }
157
158        // Declare AUDIT_EVENTS with dead-letter routing (CQ-B05).
159        let mut audit_args = FieldTable::default();
160        audit_args.insert(
161            "x-dead-letter-exchange".into(),
162            lapin::types::AMQPValue::LongString(queues::AUDIT_EVENTS_DLQ.into()),
163        );
164        self.channel
165            .queue_declare(queues::AUDIT_EVENTS.into(), options, audit_args)
166            .await
167            .map_err(AmqpError::Declaration)?;
168        info!(
169            queue = queues::AUDIT_EVENTS,
170            "Declared queue (with dead-letter routing)"
171        );
172
173        // Declare AUTHZ_REQUEST with dead-letter routing (CQ-B05).
174        let mut authz_args = FieldTable::default();
175        authz_args.insert(
176            "x-dead-letter-exchange".into(),
177            lapin::types::AMQPValue::LongString(queues::AUTHZ_REQUEST_DLQ.into()),
178        );
179        self.channel
180            .queue_declare(queues::AUTHZ_REQUEST.into(), options, authz_args)
181            .await
182            .map_err(AmqpError::Declaration)?;
183        info!(
184            queue = queues::AUTHZ_REQUEST,
185            "Declared queue (with dead-letter routing)"
186        );
187
188        // Declare MAIL_OUTBOUND with dead-letter routing (D-14).
189        let mut mail_args = FieldTable::default();
190        mail_args.insert(
191            "x-dead-letter-exchange".into(),
192            lapin::types::AMQPValue::LongString(queues::MAIL_OUTBOUND_DLQ.into()),
193        );
194        self.channel
195            .queue_declare(queues::MAIL_OUTBOUND.into(), options, mail_args)
196            .await
197            .map_err(AmqpError::Declaration)?;
198        info!(
199            queue = queues::MAIL_OUTBOUND,
200            "Declared queue (with dead-letter routing)"
201        );
202
203        Ok(())
204    }

206    /// Declare the durable webhook AMQP topology (primary + retry + DLQ,
207    /// CORR-03/D-07).
208    ///
209    /// **Correct DLX form (Pitfall 4):** unlike [`Self::declare_queues`]'s
210    /// existing `AUDIT_EVENTS`/`AUTHZ_REQUEST`/`MAIL_OUTBOUND` wiring above
211    /// (which sets `x-dead-letter-exchange` to a queue NAME with no matching
212    /// `exchange_declare` anywhere in this crate — RabbitMQ silently drops
213    /// dead-lettered messages in that case), this topology uses the default
214    /// (nameless, `""`) exchange plus an explicit `x-dead-letter-routing-key`
215    /// for both dead-letter hops. The default exchange's implicit
216    /// per-queue-name routing makes this the well-known-correct,
217    /// minimal-surface form — never a bare undeclared exchange name.
218    ///
219    /// Declaration order (DLQ target first, same discipline as
220    /// [`Self::declare_queues`]'s `ALL_QUEUES` convention):
221    /// 1. [`queues::WEBHOOK_DLQ`] — terminal, plain durable, no DLX of its own.
222    /// 2. [`queues::WEBHOOK`] — primary; a terminal nack (requeue=false,
223    ///    attempts exhausted per `AXIAM__WEBHOOK__MAX_ATTEMPTS`) dead-letters
224    ///    to [`queues::WEBHOOK_DLQ`].
225    /// 3. [`queues::WEBHOOK_RETRY`] — no consumer ever attached; a message
226    ///    published here with a per-message TTL dead-letters back to
227    ///    [`queues::WEBHOOK`] once the TTL expires, so RabbitMQ (not an
228    ///    in-process `tokio::time::sleep`) schedules the delay
229    ///    (D-07/Pitfall 5).
230    pub async fn declare_webhook_topology(&self) -> Result<(), AmqpError> {
231        let options = QueueDeclareOptions {
232            durable: true,
233            ..QueueDeclareOptions::default()
234        };
235
236        // 1. Terminal DLQ — plain durable, no DLX args of its own.
237        self.channel
238            .queue_declare(queues::WEBHOOK_DLQ.into(), options, FieldTable::default())
239            .await
240            .map_err(AmqpError::Declaration)?;
241        info!(queue = queues::WEBHOOK_DLQ, "Declared queue");
242
243        // 2. Primary webhook queue: terminal-exhaustion nacks dead-letter to
244        // WEBHOOK_DLQ via the default exchange + explicit routing key.
245        let mut webhook_args = FieldTable::default();
246        webhook_args.insert(
247            "x-dead-letter-exchange".into(),
248            lapin::types::AMQPValue::LongString("".into()),
249        );
250        webhook_args.insert(
251            "x-dead-letter-routing-key".into(),
252            lapin::types::AMQPValue::LongString(queues::WEBHOOK_DLQ.into()),
253        );
254        self.channel
255            .queue_declare(queues::WEBHOOK.into(), options, webhook_args)
256            .await
257            .map_err(AmqpError::Declaration)?;
258        info!(
259            queue = queues::WEBHOOK,
260            "Declared queue (with dead-letter routing to WEBHOOK_DLQ)"
261        );
262
263        // 3. Retry queue: per-message TTL (set at publish time) dead-letters
264        // back to the primary WEBHOOK queue via the default exchange — no
265        // consumer attached here, no in-process sleep holding a slot.
266        let mut retry_args = FieldTable::default();
267        retry_args.insert(
268            "x-dead-letter-exchange".into(),
269            lapin::types::AMQPValue::LongString("".into()),
270        );
271        retry_args.insert(
272            "x-dead-letter-routing-key".into(),
273            lapin::types::AMQPValue::LongString(queues::WEBHOOK.into()),
274        );
275        self.channel
276            .queue_declare(queues::WEBHOOK_RETRY.into(), options, retry_args)
277            .await
278            .map_err(AmqpError::Declaration)?;
279        info!(
280            queue = queues::WEBHOOK_RETRY,
281            "Declared queue (with dead-letter routing back to WEBHOOK)"
282        );
283
284        Ok(())
285    }

292    /// Create a new channel on the existing connection with QoS applied.
293    pub async fn create_channel(&self) -> Result<Channel, AmqpError> {
294        let channel = self
295            .connection
296            .create_channel()
297            .await
298            .map_err(AmqpError::Channel)?;
299        channel
300            .basic_qos(self.prefetch_count, BasicQosOptions::default())
301            .await
302            .map_err(AmqpError::Channel)?;
303        Ok(channel)
304    }

306    /// Create a new channel with QoS and publisher confirms enabled.
307    ///
308    /// Use this for channels that will publish messages and need broker
309    /// acknowledgement (`Confirmation::Ack`/`Nack` instead of `NotRequested`).
310    pub async fn create_publisher_channel(&self) -> Result<Channel, AmqpError> {
311        let channel = self.create_channel().await?;
312        channel
313            .confirm_select(ConfirmSelectOptions::default())
314            .await
315            .map_err(AmqpError::Channel)?;
316        Ok(channel)
317    }
