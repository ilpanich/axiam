1//! AMQP message types for serialization/deserialization.
2//!
3//! `MailType` and `OutboundMailMessage` are defined in `axiam-core::models::mail`
4//! so that crates which cannot depend on `axiam-amqp` (e.g. `axiam-audit`) can
5//! still build outbound mail messages.  They are re-exported here for convenience.
6//!
7//! SEC-022/055: `AuthzRequest` and `AuditEventMessage` carry an HMAC-SHA256
8//! signature over the serialized payload body. Publishers set it; consumers
9//! verify it. The signing key is a per-tenant AMQP signing secret.
10//!
11//! SEC-055: The `to_address` field in `OutboundMailMessage` is treated as
12//! advisory; the actual recipient is always resolved server-side from
13//! `user_id` + `tenant_id` to prevent recipient hijacking.
14

26// ---------------------------------------------------------------------------
27// HKDF per-tenant key derivation (SECHRD-08 / D-05a/b)
28// ---------------------------------------------------------------------------
29
30/// Fixed application-level salt for HKDF tenant-key derivation. HKDF salts
31/// are not secret; domain separation comes from the `info` parameter below.
32const APP_SALT: &[u8] = b"axiam-amqp-hkdf-salt-v1";
33
34/// Fixed domain-separation tag mixed into HKDF's `info` parameter so a
35/// derived AMQP signing subkey can never collide with a subkey derived for
36/// any other purpose from the same master key.
37const DOMAIN_TAG: &[u8] = b"axiam-amqp-v1";
38
39/// Current message envelope key version (SECHRD-08 / D-05b, NEW-4). Bump this
40/// when rotating the master signing key OR when changing the signed envelope
41/// shape; `key_version` travels on the wire so a verifier always derives the
42/// subkey the publisher actually used.
43///
44/// NEW-4 (v2): the signed body now MUST carry a `nonce` + `issued_at` for
45/// replay protection. Consumers **reject** (nack, requeue:false) any message
46/// with `key_version < 2` — this is a hard cutover, there is no v1 grace path.
47pub const CURRENT_KEY_VERSION: u8 = 2;
48
49/// Minimum accepted envelope key version (NEW-4 hard cutover). Any message
50/// with `key_version` below this is rejected outright — it predates the
51/// mandatory `nonce`/`issued_at` replay-protection fields.
52pub const MIN_ACCEPTED_KEY_VERSION: u8 = 2;
53
54/// Default freshness skew for the `issued_at` acceptance window (NEW-4).
55/// A message is accepted only when its `issued_at` lies within ±5 minutes of
56/// the consumer's current clock. This bounds both clock drift and how long a
57/// captured message stays replay-eligible before the freshness gate rejects it.
58pub const DEFAULT_FRESHNESS_SKEW_SECS: i64 = 300;
59
60fn default_key_version() -> u8 {
61    CURRENT_KEY_VERSION
62}
63
64/// Serde default for `nonce` (NEW-4). A v1 message lacks this field; it
65/// deserializes to the nil UUID and is then rejected by the `key_version < 2`
66/// gate before the nonce is ever consulted, so the concrete value is inert.
67fn default_nonce() -> Uuid {
68    Uuid::nil()
69}
70
71/// Serde default for `issued_at` (NEW-4). A v1 message lacks this field; it
72/// deserializes to the Unix epoch — far outside any freshness window — so a
73/// message that somehow bypassed the `key_version` gate would still fail the
74/// freshness check. `from_timestamp(0, 0)` is infallible for a constant.
75fn default_issued_at() -> DateTime<Utc> {
76    DateTime::<Utc>::from_timestamp(0, 0).expect("epoch is a valid timestamp")
77}
78
79/// Return `true` when `issued_at` lies within `±skew` of `now` (NEW-4).
80///
81/// This is the freshness gate for AMQP replay protection: a captured message
82/// can only be replayed successfully inside this window, and the durable
83/// nonce store rejects duplicates within it. `skew` bounds legitimate clock
84/// drift between producer and consumer; [`DEFAULT_FRESHNESS_SKEW_SECS`] is the
85/// default.
86pub fn is_fresh(issued_at: DateTime<Utc>, now: DateTime<Utc>, skew: chrono::Duration) -> bool {
87    now.signed_duration_since(issued_at).abs() <= skew
88}

90/// Derive a per-tenant AMQP signing subkey from the shared master key
91/// (SECHRD-08 / D-05a/b, T-25-19).
92///
93/// Uses HKDF-SHA256 (`hkdf::Hkdf<Sha256>`) — never hand-rolled concatenation
94/// hashing. The `info` parameter is domain-separated (fixed tag) and
95/// versioned (`key_version`) then tenant-scoped (`tenant_id`), so:
96/// - A signature produced with tenant A's subkey never verifies under
97///   tenant B's subkey, even though both derive from the same master key.
98/// - Bumping `key_version` (master key rotation) yields an entirely
99///   different subkey for the same tenant, without breaking in-flight
100///   messages signed under the prior version (D-05b).
101pub fn derive_tenant_key(master: &[u8], tenant_id: Uuid, key_version: u8) -> [u8; 32] {
102    let hk = Hkdf::<Sha256>::new(Some(APP_SALT), master);
103    let mut info = Vec::with_capacity(DOMAIN_TAG.len() + 1 + 16);
104    info.extend_from_slice(DOMAIN_TAG);
105    info.push(key_version);
106    info.extend_from_slice(tenant_id.as_bytes());
107    let mut okm = [0u8; 32];
108    hk.expand(&info, &mut okm)
109        .expect("32 bytes is a valid HKDF-SHA256 output length");
110    okm
111}

113// ---------------------------------------------------------------------------
114// HMAC helpers (SEC-022)
115// ---------------------------------------------------------------------------
116
117/// Compute HMAC-SHA256 of the JSON-serialized message payload.
118///
119/// The `hmac_signature` field must be set to `None` (or absent) before
120/// serializing `payload_json` — otherwise the signature is over a message
121/// that includes a placeholder signature, making verification impossible.
122/// Use [`sign_payload`] to produce the canonical JSON before signing.
123pub fn sign_payload(key: &[u8], payload_json: &[u8]) -> String {
124    let mut mac =
125        <HmacSha256 as hmac::KeyInit>::new_from_slice(key).expect("HMAC accepts any key length");
126    mac.update(payload_json);
127    hex::encode(mac.finalize().into_bytes())
128}
129
130/// Verify an HMAC-SHA256 signature over the canonical payload bytes.
131///
132/// Returns `true` if the signature matches. Uses constant-time comparison
133/// internally (via the `hmac` crate's `verify_slice`).
134pub fn verify_payload(key: &[u8], payload_json: &[u8], signature_hex: &str) -> bool {
135    let mut mac =
136        <HmacSha256 as hmac::KeyInit>::new_from_slice(key).expect("HMAC accepts any key length");
137    mac.update(payload_json);
138    let expected = hex::decode(signature_hex).unwrap_or_default();
139    mac.verify_slice(&expected).is_ok()
140}
141
142/// Verify a signed AMQP envelope against its per-tenant derived subkey
143/// (SECHRD-08 / D-05a/b/c).
144///
145/// Derives the tenant's subkey from `master_key` + `tenant_id` +
146/// `key_version`, then verifies `signature` over `canonical_bytes`.
147///
148/// Returns `false` when `signature` is `None` (unsigned) OR the signature
149/// does not verify — there is no accept-when-absent code path. Consumers
150/// MUST reject (nack, never process) whenever this returns `false`
151/// (T-25-20).
152pub fn verify_tenant_signature(
153    master_key: &[u8],
154    tenant_id: Uuid,
155    key_version: u8,
156    canonical_bytes: &[u8],
157    signature: Option<&str>,
158) -> bool {
159    let subkey = derive_tenant_key(master_key, tenant_id, key_version);
160    signature.is_some_and(|sig| verify_payload(&subkey, canonical_bytes, sig))
161}

167/// Authorization check request received from `axiam.authz.request`.
168#[derive(Debug, Deserialize, Serialize)]
169pub struct AuthzRequest {
170    /// Caller-provided ID to correlate request with response.
171    pub correlation_id: Uuid,
172    pub tenant_id: Uuid,
173    pub subject_id: Uuid,
174    pub action: String,
175    pub resource_id: Uuid,
176    #[serde(default, skip_serializing_if = "Option::is_none")]
177    pub scope: Option<String>,
178    /// HKDF master-key rotation version (SECHRD-08 / D-05b). Selects which
179    /// per-tenant subkey derivation the signature was produced with.
180    #[serde(default = "default_key_version")]
181    pub key_version: u8,
182    /// Per-message unique nonce for replay protection (NEW-4). ALWAYS emitted
183    /// (no `skip_serializing_if`) so it is inside the signed HMAC body. The
184    /// consumer records it in the durable `amqp_nonce_replay` store; a
185    /// duplicate within the freshness window is a replay and is rejected.
186    #[serde(default = "default_nonce")]
187    pub nonce: Uuid,
188    /// Producer-side send time for the freshness gate (NEW-4). ALWAYS emitted
189    /// (no `skip_serializing_if`) so it is inside the signed HMAC body. The
190    /// consumer rejects the message if this lies outside ±skew of its clock.
191    #[serde(default = "default_issued_at")]
192    pub issued_at: DateTime<Utc>,
193    /// HMAC-SHA256 of the JSON-serialized message body (this field set to null).
194    /// Computed with the per-tenant subkey derived via [`derive_tenant_key`]
195    /// (SECHRD-08 / D-05a/b). Mandatory — the consumer rejects both unsigned
196    /// and invalid-signature messages; there is no fail-open path (D-05c).
197    #[serde(default, skip_serializing_if = "Option::is_none")]
198    pub hmac_signature: Option<String>,
199}

264/// Webhook delivery message carried on the `axiam.webhook` /
265/// `axiam.webhook.retry` queues (CORR-03/D-07).
266///
267/// Produced by `WebhookDeliveryService::emit` (`axiam-api-rest`), consumed
268/// by the webhook AMQP consumer (wired in 26-07) which drives
269/// `WebhookDeliveryService::deliver_once` per (re)delivery. `attempt` is
270/// incremented by the consumer on each republish to the retry queue and
271/// checked against `AXIAM__WEBHOOK__MAX_ATTEMPTS` before a terminal nack
272/// routes the message to `WEBHOOK_DLQ` (D-07).
273///
274/// `tenant_id` is not in the plan's original DTO sketch, but is required
275/// for `deliver_once` to resolve the webhook via the tenant-scoped

278/// be tenant-scoped.

280pub struct WebhookMessage {
281    pub webhook_id: Uuid,
282    pub delivery_id: Uuid,
283    pub tenant_id: Uuid,
284    pub event_type: String,
285    pub payload: serde_json::Value,
286    pub attempt: u32,
287}
