7/// gRPC rate-limit bucket-key mode (D8 parity with
8/// `axiam_api_rest::config::rate_limit::RateLimitKeyMode`).
9///
10/// **Currently a no-op / reserved for forward compatibility.** The only
11/// gRPC surface wrapped by the rate-limit layers today
12/// (`middleware::rate_limit::{build_grpc_governor_layer,
13/// GrpcSharedRateLimitLayer}`) is the low-latency, service-mesh-wide authz
14/// check (`AuthorizationService`) — those layers are `Server::builder()`-
15/// wide `tower::Layer`s that run BEFORE tonic resolves any per-RPC
16/// authenticated identity (`ValidatedClaims`, inserted by each service's
17/// own `with_interceptor(...)` auth interceptor — see `server.rs`). There is
18/// therefore no client identity available at the point these layers key a
19/// request, structurally identical to why REST's `/auth/login` always stays
20/// per-IP (see `axiam_api_rest::config::rate_limit::RateLimitKeyMode`
21/// docs). Setting this to anything other than `Ip` has no observable effect
22/// yet; it exists so `AXIAM__GRPC__KEY` round-trips through config the same
23/// way `AXIAM__RATE_LIMIT__KEY` does on the REST side, and so a future
24/// per-RPC client-identity-aware interceptor (re-ordering the auth
25/// interceptor ahead of the rate-limit layer for `TokenService`'s
26/// `introspect_token`/`validate_token`, which DO have a caller identity via
27/// `ValidatedClaims.sub`) has a config surface to land on without another
28/// env var rename.
29#[derive(Debug, Clone, Copy, PartialEq, Eq, Deserialize, Default)]
30#[serde(rename_all = "snake_case")]
31pub enum GrpcRateLimitKeyMode {
32    /// Key on source IP only (current and only implemented behavior).
33    #[default]
34    Ip,
35    /// Reserved (D8 parity) — not yet wired to any gRPC surface; behaves
36    /// identically to `Ip` today.
37    ClientId,
38    /// Reserved (D8 parity) — not yet wired to any gRPC surface; behaves
39    /// identically to `Ip` today.
40    IpClientId,
41}
42
43/// Configuration for the gRPC server.
44#[derive(Debug, Clone, Deserialize)]
45pub struct GrpcConfig {
46    #[serde(default = "default_host")]
47    pub host: String,
48    #[serde(default = "default_port")]
49    pub port: u16,
50    /// Max gRPC authz requests per second per IP (default: 100).
51    /// Generous for service-mesh patterns where authz is called per-request.
52    /// Configure via AXIAM__GRPC__GRPC_AUTHZ_PER_SEC env var.
53    #[serde(default = "default_grpc_authz_per_sec")]
54    pub grpc_authz_per_sec: u32,
55    /// D8 parity field — see [`GrpcRateLimitKeyMode`]. Currently always
56    /// behaves as `Ip` regardless of value; reserved for a future per-RPC
57    /// client-identity-aware rate limiter. Configure via `AXIAM__GRPC__KEY`.
58    #[serde(default)]
59    pub key: GrpcRateLimitKeyMode,
60}
61
62impl Default for GrpcConfig {
63    fn default() -> Self {
64        Self {
65            host: default_host(),
66            port: default_port(),
67            grpc_authz_per_sec: default_grpc_authz_per_sec(),
68            key: GrpcRateLimitKeyMode::Ip,
69        }
70    }
71}
72
73impl GrpcConfig {
74    pub fn bind_address(&self) -> SocketAddr {
75        let addr = format!("{}:{}", self.host, self.port);
76        addr.parse()
77            .unwrap_or_else(|e| panic!("invalid gRPC bind address '{addr}': {e}"))
78    }
79}
80
81/// Default gRPC bind host.
82///
83/// Binds to loopback (`127.0.0.1`) so the gRPC API is not exposed on
84/// all interfaces unless explicitly configured. Deploy behind mTLS or
85/// an internal network when binding to `0.0.0.0`.
86fn default_host() -> String {
87    "127.0.0.1".into()
88}
89
90fn default_port() -> u16 {
91    50051
92}
93
94fn default_grpc_authz_per_sec() -> u32 {
95    100
96}

98#[cfg(test)]
99mod tests {
100    use super::*;
101
102    #[test]
103    fn default_key_mode_is_ip() {
104        assert_eq!(GrpcConfig::default().key, GrpcRateLimitKeyMode::Ip);
105        assert_eq!(GrpcRateLimitKeyMode::default(), GrpcRateLimitKeyMode::Ip);
106    }
107
108    #[test]
109    fn key_mode_deserializes_from_documented_env_values() {
110        assert_eq!(
111            serde_json::from_str::<GrpcRateLimitKeyMode>("\"ip\"").unwrap(),
112            GrpcRateLimitKeyMode::Ip
113        );
114        assert_eq!(
115            serde_json::from_str::<GrpcRateLimitKeyMode>("\"client_id\"").unwrap(),
116            GrpcRateLimitKeyMode::ClientId
117        );
118        assert_eq!(
119            serde_json::from_str::<GrpcRateLimitKeyMode>("\"ip_client_id\"").unwrap(),
120            GrpcRateLimitKeyMode::IpClientId
121        );
122    }
