1//! gRPC rate limiting via tower-governor (per D-10, D-11), brought to parity
2//! with the fixed REST limiter (SECHRD-03, D-01c).
3//!
4//! Two layers cooperate here (mirroring
5//! `axiam-api-rest::middleware::rate_limit_shared` / `extractors::rate_limit`):
6//!
7//! - [`GrpcTrustedHopsKeyExtractor`] — a custom `tower_governor::KeyExtractor`
8//!   that replaces `SmartIpKeyExtractor`. `SmartIpKeyExtractor` unconditionally
9//!   trusts the LEFTMOST `X-Forwarded-For` hop and, more fundamentally, can
10//!   never find tonic's real peer address at all (it only looks for an
11//!   `axum::extract::ConnectInfo<SocketAddr>` extension or a bare `SocketAddr`
12//!   extension — tonic inserts `TcpConnectInfo`/`TlsConnectInfo<TcpConnectInfo>`
13//!   instead). This extractor mirrors the fixed REST
14//!   `XForwardedForKeyExtractor` (plan 24-03): a configured `trusted_hops`
15//!   selects the rightmost trusted XFF hop, and when there are NOT enough
16//!   hops to trust, XFF is ignored entirely and the verified tonic connection
17//!   peer address is used instead (never `hops[0]` — SECHRD-03/D-01d).
18//! - [`GrpcSharedRateLimitLayer`] — an async pre-check `tower::Layer` that
19//!   reuses the plan-24-04 `SurrealRateLimitBucketRepository` shared-store
20//!   counter, run BEFORE the existing per-replica in-memory `GovernorLayer`
21//!   (kept byte-for-byte unchanged as the fail-open fallback, D-01b). This is
22//!   deliberately NOT a `governor::StateStore` impl and never calls
23//!   `block_on` — `StateStore::measure_and_replace` is a *synchronous* trait
24//!   method (RESEARCH Pitfall 1), so the shared-store check is a separate
25//!   async tower service that performs its own SurrealDB round-trip and then
26//!   delegates to the inner service.
27//!
28//! HARD CONSTRAINT (D-01c coordination note): the `Quota::per_second(...)
29//! .burst_size(...)` throughput/quota math in [`build_grpc_governor_layer`]
30//! is untouched by this module — CORR-01/Phase 26 owns it.

57/// Reads the SAME `AXIAM__RATE_LIMIT__TRUSTED_HOPS` env var the REST shared
58/// pre-check (`axiam-api-rest::middleware::rate_limit_shared::trusted_hops`)
59/// and `server.rs::build_governor` use, so the gRPC key and the REST key are
60/// derived identically from the same deployment-topology configuration.
61///
62/// `pub(crate)` so `server.rs::start_grpc_server` can pass the SAME value to
63/// both [`build_grpc_governor_layer`] (via this function internally) and
64/// [`GrpcSharedRateLimitLayer::new`] — the key-extraction logic in the
65/// shared-store pre-check and the in-memory governor MUST stay in lockstep.
66pub(crate) fn trusted_hops_from_env() -> usize {
67    std::env::var("AXIAM__RATE_LIMIT__TRUSTED_HOPS")
68        .ok()
69        .and_then(|v| v.parse().ok())
70        .unwrap_or(0)
71}

73/// Extracts the gRPC caller's IP from `X-Forwarded-For`, falling back to the
74/// verified tonic connection peer address (`TcpConnectInfo`/
75/// `TlsConnectInfo<TcpConnectInfo>`).
76///
77/// `trusted_hops` controls how many rightmost XFF entries are trusted
78/// reverse-proxy hops to skip (mirrors
79/// `axiam-api-rest::extractors::rate_limit::XForwardedForKeyExtractor`).
80/// When `trusted_hops >= hops.len()`, the header is NOT trusted at all and
81/// XFF is ignored entirely — the key is derived from the verified peer
82/// address instead (a client cannot manufacture extra hops to force a
83/// fallback to an attacker-controlled `hops[0]` — SECHRD-03/D-01d).
84#[derive(Debug, Clone, Default)]
85pub struct GrpcTrustedHopsKeyExtractor {
86    pub trusted_hops: usize,
87}
88
89impl GrpcTrustedHopsKeyExtractor {
90    /// Create an extractor keyed by `trusted_hops` trusted reverse-proxy
91    /// hops (0 = no trusted hops; XFF is only ever used when it has more
92    /// than 0 entries beyond what's trusted).
93    pub fn new(trusted_hops: usize) -> Self {
94        Self { trusted_hops }
95    }
96}

98impl KeyExtractor for GrpcTrustedHopsKeyExtractor {
99    type Key = IpAddr;
100
101    fn extract<T>(&self, req: &Request<T>) -> Result<Self::Key, GovernorError> {
102        if let Some(forwarded_for) = req.headers().get("x-forwarded-for")
103            && let Ok(val) = forwarded_for.to_str()
104        {
105            let hops: Vec<&str> = val.split(',').map(str::trim).collect();
106            // Select the rightmost-untrusted hop (parity with REST).
107            if self.trusted_hops < hops.len() {
108                let idx = hops.len() - 1 - self.trusted_hops;
109                if let Ok(ip) = hops[idx].parse::<IpAddr>() {
110                    return Ok(ip);
111                }
112            }
113            // Fewer hops than trusted_hops requires: the header cannot be
114            // trusted. Fall through to the verified peer address below
115            // instead of indexing into XFF (SECHRD-03 — no `hops[0]`
116            // fallback that would let a rotating XFF mint a fresh bucket).
117        }
118
119        grpc_peer_addr(req)
120            .map(|addr| addr.ip())
121            .ok_or(GovernorError::UnableToExtractKey)
122    }
123}

125/// Reads the verified tonic connection peer address from request
126/// extensions, mirroring `tonic::Request::remote_addr()`'s own lookup
127/// (`TcpConnectInfo` for plaintext, `TlsConnectInfo<TcpConnectInfo>` for TLS)
128/// — the extension types tonic's `ConnectInfoLayer` actually inserts (NOT
129/// the `axum::extract::ConnectInfo<SocketAddr>` / bare `SocketAddr`
130/// extensions `SmartIpKeyExtractor` looks for, which tonic never sets).
131fn grpc_peer_addr<T>(req: &Request<T>) -> Option<SocketAddr> {
132    req.extensions()
133        .get::<TcpConnectInfo>()
134        .and_then(TcpConnectInfo::remote_addr)
135        .or_else(|| {
136            req.extensions()
137                .get::<TlsConnectInfo<TcpConnectInfo>>()
138                .and_then(|info| info.get_ref().remote_addr())
139        })
140}

142/// Concrete type alias for the gRPC GovernorLayer.
143///
144/// - `K` = [`GrpcTrustedHopsKeyExtractor`] — trusted_hops-aware, keys off the
145///   verified peer address when XFF hops are insufficient (SECHRD-03).
146/// - `M` = NoOpMiddleware — no rate-limit headers injected into responses (gRPC transport)
147/// - `RespBody` = tonic::body::Body — tonic's native streaming body type
148pub type GrpcGovernorLayer =
149    GovernorLayer<GrpcTrustedHopsKeyExtractor, NoOpMiddleware<QuantaInstant>, tonic::body::Body>;

151/// Build a [`GovernorLayer`] for gRPC server-wide rate limiting.
152///
153/// The layer uses token-bucket semantics (via the `governor` crate) with
154/// `authz_per_sec` tokens replenished per second and a burst allowance equal
155/// to `authz_per_sec` (one second's worth of tokens). For service-mesh
156/// patterns where the authz endpoint is called on every request, the default
157/// of 100 tokens/sec is intentionally generous.
158///
159/// # Panics
160///
161/// Panics at startup if `authz_per_sec` is 0 — the governor crate requires a
162/// non-zero burst size.
163pub fn build_grpc_governor_layer(authz_per_sec: u32) -> GrpcGovernorLayer {
164    assert!(authz_per_sec >= 1, "grpc_authz_per_sec must be >= 1");
165
166    // CORR-01 (D-01): the PREVIOUS "fix" here (CQ-B44) called
167    // `tower_governor`'s OWN `GovernorConfigBuilder::per_second(n)` /
168    // `.burst_size(authz_per_sec * 2)` as if `.per_second(n)` meant "n tokens
169    // per second." It does NOT — `tower_governor`'s builder methods set the
170    // replenish PERIOD (i.e. "one token every n seconds"), so with the
171    // default `grpc_authz_per_sec=100` the governor replenished 1 token every
172    // 100 seconds, and raising the configured rate made throughput WORSE, not
173    // better. Fixed by bypassing `tower_governor`'s confusing
174    // `.per_second()`/`.per_millisecond()` convenience methods entirely and
175    // constructing the underlying `governor::Quota` directly via
176    // `Quota::per_second`, whose `replenish_interval()` is `1s /
177    // authz_per_sec` and whose `burst_size()` is `authz_per_sec` — then
178    // feeding that quota into the builder via `.const_period()`/
179    // `.const_burst_size()`. Burst is `authz_per_sec` (not `* 2`), per D-01.
180    //
181    // SECHRD-03/D-01c: key extractor is GrpcTrustedHopsKeyExtractor for
182    // IP-spoofing resistance parity with REST — unrelated to this quota math.
183    let burst = NonZeroU32::new(authz_per_sec).expect("authz_per_sec >= 1 asserted above");
184    let quota = Quota::per_second(burst);
185
186    let config = Arc::new(
187        GovernorConfigBuilder::default()
188            .const_period(quota.replenish_interval())
189            .const_burst_size(quota.burst_size().get())
190            .key_extractor(GrpcTrustedHopsKeyExtractor::new(trusted_hops_from_env()))
191            .finish()
192            .expect("valid GovernorConfig for gRPC rate limiter"),
193    );
194
195    GovernorLayer::new(config)
196}

202/// Fixed-window duration (seconds) for the shared bucket — same value and
203/// rationale as the REST shared pre-check
204/// (`axiam-api-rest::middleware::rate_limit_shared::WINDOW_SECS`): a simple
205/// fixed-window counter is acceptable here since this layer only needs to be
206/// *approximately* right (it fails open by design).
207const WINDOW_SECS: i64 = 60;
208
209/// Truncates `now` down to the start of the current fixed
210/// [`WINDOW_SECS`]-second window.
211fn window_start(now: DateTime<Utc>) -> DateTime<Utc> {
212    let epoch = now.timestamp();
213    let start_epoch = epoch - epoch.rem_euclid(WINDOW_SECS);
214    DateTime::<Utc>::from_timestamp(start_epoch, 0).unwrap_or(now)
215}

217/// Builds a gRPC `RESOURCE_EXHAUSTED` response — the same status the
218/// in-memory `GovernorLayer` returns for `GovernorError::TooManyRequests`
219/// (see `tower_governor::errors::GovernorError`'s `Response<tonic::body::Body>`
220/// conversion) — so clients see one consistent rate-limit contract
221/// regardless of which layer rejected the request.
222fn too_many_requests_response() -> Response<tonic::body::Body> {
223    tonic::Status::resource_exhausted("rate limit exceeded").into_http()
224}

226/// Async SurrealDB-backed shared rate-limit pre-check `tower::Layer` for the
227/// gRPC path (SECHRD-03 / D-01a, D-01b, D-01c, D-01d).
228///
229/// Reuses [`SurrealRateLimitBucketRepository::increment`] (plan 24-04) —
230/// there is no reimplemented counter here. Wire this layer BEFORE (i.e.
231/// `.layer()` it FIRST — tower's `ServiceBuilder`/`Server::builder()`
232/// executes the FIRST-added layer with the request FIRST, the opposite of
233/// actix's last-`.wrap()`-is-outermost rule) the existing
234/// [`build_grpc_governor_layer`] `GovernorLayer` on the same
235/// `Server::builder()`, e.g.:
236///
237/// ```ignore
238/// Server::builder()
239///     .layer(GrpcSharedRateLimitLayer::new(db, "grpc_authz", grpc_config.grpc_authz_per_sec, trusted_hops))
240///     .layer(build_grpc_governor_layer(grpc_config.grpc_authz_per_sec))
241///     .add_service(authz_svc)
242/// ```
243///
244/// **Fail-open (D-01b, T-24-73 accepted risk):** when the shared store is
245/// unreachable, or no client IP can be extracted, this layer logs a
246/// `warn`-level alarm and forwards the request unchanged so the existing
247/// in-memory governor makes the decision instead — a counter-store outage
248/// must never hard-block gRPC authz traffic.
249///
250/// **CRITICAL (RESEARCH Pitfall 1):** `governor::StateStore::measure_and_replace`
251/// is a *synchronous* trait method. This layer is deliberately NOT a
252/// `StateStore` implementation and never calls `block_on` — it performs its
253/// own async SurrealDB round-trip as a plain `tower::Layer`/`Service`, then
254/// delegates to the inner service.
255pub struct GrpcSharedRateLimitLayer<C: Connection> {
256    db: Surreal<C>,
257    endpoint: &'static str,
258    limit: u32,
259    trusted_hops: usize,
260}
261
262impl<C: Connection> GrpcSharedRateLimitLayer<C> {
263    /// `endpoint` MUST be unique per rate-limited gRPC surface so the shared

265    /// never collapse distinct surfaces into one global bucket.
266    pub fn new(db: Surreal<C>, endpoint: &'static str, limit: u32, trusted_hops: usize) -> Self {
267        Self {
268            db,
269            endpoint,
270            limit,
271            trusted_hops,
272        }
273    }
274}

339impl<S, C> Service<Request<tonic::body::Body>> for GrpcSharedRateLimitService<S, C>
340where
341    S: Service<Request<tonic::body::Body>, Response = Response<tonic::body::Body>>
342        + Clone
343        + Send
344        + 'static,
345    S::Future: Send + 'static,
346    S::Error: Send + 'static,
347    C: Connection + 'static,
348{
349    type Response = Response<tonic::body::Body>;
350    type Error = S::Error;
351    type Future = Pin<Box<dyn Future<Output = Result<Self::Response, Self::Error>> + Send>>;
352
353    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
354        self.inner.poll_ready(cx)
355    }
356
357    fn call(&mut self, req: Request<tonic::body::Body>) -> Self::Future {
358        // Standard clone-and-swap so the returned future owns an
359        // independent, ready-to-call copy of the inner service (mirrors
360        // tower-http's convention for async-wrapping middlewares).
361        let clone = self.inner.clone();
362        let mut inner = std::mem::replace(&mut self.inner, clone);
363
364        let db = self.db.clone();
365        let endpoint = self.endpoint;
366        let limit = self.limit;
367        let key_extractor = GrpcTrustedHopsKeyExtractor::new(self.trusted_hops);
368
369        Box::pin(async move {
370            let ip = key_extractor.extract(&req).ok();
371
372            let allow = match ip {
373                Some(ip) => {
374                    let repo = SurrealRateLimitBucketRepository::new(db);
375                    let key = format!("{endpoint}:{ip}");
376                    let window = window_start(Utc::now());
377                    match repo.increment(&key, window).await {
378                        Ok(count) => count <= limit as u64,
379                        Err(err) => {
380                            // Fail OPEN (D-01b): a counter-store outage must
381                            // never hard-block gRPC authz traffic. Do NOT
382                            // log the raw key (endpoint:ip) at info+ (mirrors
383                            // the REST T-24-43 note) — this warn-level alarm
384                            // omits it.
385                            tracing::warn!(
386                                endpoint,
387                                error = %err,
388                                "shared gRPC rate-limit store unreachable; falling back \
389                                 to per-replica in-memory governor"
390                            );
391                            true
392                        }
393                    }
394                }
395                // No client-IP key available — fail open; the in-memory
396                // governor still makes the real decision.
397                None => true,
398            };
399
400            if allow {
401                inner.call(req).await
402            } else {
403                Ok(too_many_requests_response())
404            }
405        })
406    }
407}

459    // -----------------------------------------------------------------
460    // CORR-01/D-02: sustained-throughput + monotonicity regression test
461    // -----------------------------------------------------------------
462    //
463    // Drives sustained load against the SAME `governor::Quota::per_second`
464    // construction `build_grpc_governor_layer` uses (see the production
465    // function above) via a `FakeRelativeClock`, and asserts the permitted
466    // count over a simulated window approximates the configured rate — the
467    // only reliable way to distinguish a correctly-throttling governor from
468    // one that is inverted (RESEARCH Pitfall 1: a naive "does the first
469    // burst get through" smoke test looks identical either way).
470    //
471    // This test would FAIL against the old, inverted construction: with
472    // `tower_governor`'s `.per_second(n)` treated as "n tokens/sec" (the
473    // CQ-B44 bug), the replenish period becomes `n` SECONDS, so over a
474    // 1-second simulated window virtually no additional tokens regenerate
475    // past the initial burst — the sustained (post-burst) count would be
476    // ~0, not approximately `n`.
477    fn permitted_over_window(rate: u32, window: std::time::Duration) -> u64 {
478        use governor::RateLimiter;
479        use governor::clock::FakeRelativeClock;
480
481        let burst = NonZeroU32::new(rate).expect("rate >= 1");
482        let quota = Quota::per_second(burst);
483        let clock = FakeRelativeClock::default();
484        let limiter = RateLimiter::direct_with_clock(quota, clock.clone());
485
486        // Drain the initial burst (the bucket starts full) — burst capacity
487        // is a separate, already-asserted property (D-01: burst ==
488        // authz_per_sec); it must NOT be counted as "sustained" throughput,
489        // or a fully-inverted governor with a large burst could still look
490        // like it's passing this test on the first drain alone.
491        while limiter.check().is_ok() {}
492
493        // Now simulate sustained load: advance the fake clock in small
494        // steps across the window, greedily draining whatever new tokens
495        // have accumulated at each step — this proves tokens keep
496        // replenishing throughout the window (sustained throughput), not
497        // just once at t=0.
498        let step = std::time::Duration::from_millis(5);
499        let mut elapsed = std::time::Duration::ZERO;
500        let mut permitted: u64 = 0;
501        while elapsed < window {
502            clock.advance(step);
503            elapsed += step;
504            while limiter.check().is_ok() {
505                permitted += 1;
506            }
507        }
508        permitted
509    }

511    #[test]
512    fn governor_sustained_throughput_matches_configured_rate() {
513        let window = std::time::Duration::from_secs(1);
514        let rate = 50u32;
515
516        let permitted = permitted_over_window(rate, window);
517
518        // Tolerance band accounts for the fake clock's discrete 5ms step
519        // granularity relative to the token replenish interval; well
520        // outside this band would mean the governor is not throttling at
521        // approximately the configured rate (CORR-01's ~1/100s inversion
522        // would put this number at ~0, nowhere close to 50).
523        let lower = (rate as f64 * 0.7) as u64;
524        let upper = (rate as f64 * 1.3) as u64;
525        assert!(
526            permitted >= lower && permitted <= upper,
527            "expected sustained throughput approximately {rate} req/s over a \
528             1s window (allowed range {lower}..={upper}), got {permitted} — \
529             the governor quota construction may be inverted again (CORR-01)"
530        );
531        assert!(
532            permitted > 1,
533            "sustained throughput of {permitted} over 1s is consistent with \
534             the ~1-token/100s inversion bug this test guards against"
535        );

543        // this relationship backwards (raising the configured rate makes
544        // the governor SLOWER) or flat (both effectively ~0).
545        let window = std::time::Duration::from_secs(1);
546
547        let low = permitted_over_window(10, window);
548        let high = permitted_over_window(100, window);
549
550        assert!(
551            high > low,
552            "expected a higher configured rate (100/s) to permit strictly \
553             more sustained requests than a lower rate (10/s) over the same \
554             1s window, got low={low} high={high} — the quota construction \
555             may be inverted (CORR-01)"
556        );
557    }
