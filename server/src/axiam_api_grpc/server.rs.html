1//! Tonic gRPC server setup.
2//!
3//! CQ-B20: Server builder sets message-size limits, per-connection timeout,
4//! and concurrency limit to prevent resource exhaustion.
5//! TLS (REQ-15 AC-1): When `AXIAM__GRPC_TLS_CERT_PATH` and
6//! `AXIAM__GRPC_TLS_KEY_PATH` env vars are set, the server is configured with
7//! `ServerTlsConfig`; otherwise plaintext mode is used (suitable for in-mesh
8//! mutual-TLS handled at the sidecar/service-mesh layer).
9

32/// Start the gRPC server with all registered services.
33///
34/// Applies two cooperating rate-limit layers (SECHRD-03, D-01a/b/c — gap
35/// closure for 24-07): the SurrealDB-backed [`GrpcSharedRateLimitLayer`]
36/// shared-store pre-check runs FIRST (outermost), failing OPEN on any DB
37/// error to the per-replica in-memory `GovernorLayer` (per D-10) built via
38/// `grpc_authz_per_sec` from `GrpcConfig`. Both layers derive their client-IP
39/// key from the SAME `trusted_hops` value so gRPC keying stays in lockstep
40/// across the shared store and the in-memory fallback.
41///
42/// Transport limits (CQ-B20):
43/// - Max message size: 4 MiB decode / 4 MiB encode
44/// - Per-connection timeout: 30 s
45/// - Concurrency limit: 256 streams per connection
46///
47/// TLS (REQ-15 AC-1): env-gated via `AXIAM__GRPC_TLS_CERT_PATH` /
48/// `AXIAM__GRPC_TLS_KEY_PATH`. When absent, TLS is disabled and a warning
49/// is logged (acceptable for in-mesh/loopback deployments).
50pub async fn start_grpc_server<R, P, Res, S, G, U, C>(
51    addr: SocketAddr,
52    engine: AuthorizationEngine<R, P, Res, S, G>,
53    user_repo: U,
54    auth_config: AuthConfig,
55    grpc_config: &GrpcConfig,
56    db: Surreal<C>,
57    batch_max_concurrency: usize,
58) -> Result<(), tonic::transport::Error>
59where
60    R: RoleRepository + 'static,
61    P: PermissionRepository + 'static,
62    Res: ResourceRepository + 'static,
63    S: ScopeRepository + 'static,
64    G: GroupRepository + 'static,
65    U: UserRepository + 'static,
66    C: Connection + 'static,
67{
68    tracing::info!(
69        bind = %addr,
70        grpc_authz_per_sec = grpc_config.grpc_authz_per_sec,
71        "Starting gRPC server",
72    );
73
74    // SECHRD-03 gap closure (24-07 follow-up): the shared-store pre-check
75    // MUST use the same trusted_hops value as the in-memory governor's key
76    // extractor (both ultimately key off GrpcTrustedHopsKeyExtractor logic)
77    // so a rotating XFF cannot mint a fresh bucket in one layer while being
78    // correctly collapsed in the other.
79    let trusted_hops = trusted_hops_from_env();
80    let shared_rate_limit_layer = GrpcSharedRateLimitLayer::new(
81        db,
82        "grpc_authz",
83        grpc_config.grpc_authz_per_sec,
84        trusted_hops,
85    );
86    let governor_layer = build_grpc_governor_layer(grpc_config.grpc_authz_per_sec);
87
88    let authz_svc = AuthorizationServiceServer::with_interceptor(
89        AuthorizationServiceImpl::new(engine, batch_max_concurrency),
90        AuthInterceptor::new(auth_config.clone()),
91    );
92    // SECFIX-01: UserService and TokenService previously had zero auth —
93    // any unauthenticated mesh peer could call GetUser/ValidateCredentials/
94    // IntrospectToken. Wrap them with the same AuthInterceptor chokepoint
95    // as AuthorizationService so every gRPC call requires a verified bearer JWT.
96    let user_svc = UserServiceServer::with_interceptor(
97        UserServiceImpl::new(user_repo, auth_config.clone()),
98        AuthInterceptor::new(auth_config.clone()),
99    );
100    let token_svc = TokenServiceServer::with_interceptor(
101        TokenServiceImpl::new(auth_config.clone()),
102        AuthInterceptor::new(auth_config),
103    );
104
105    // CQ-B20: Apply transport limits to the gRPC server builder.
106    // Note: tonic 0.14 does not expose max_decoding_message_size / max_encoding_message_size
107    // at the Server level (added in tonic 0.12+ via the Router API). The HTTP/2 frame-size
108    // limit (max_frame_size) is the closest available per-connection cap in 0.14. Upgrade
109    // to tonic ≥0.12 to use per-service max_decoding_message_size / max_encoding_message_size.
110    // Tracked: max_decoding_message_size (CQ-B20, pending tonic upgrade in Phase 19).
111    // SECHRD-03 (D-01a/b/c): the shared-store pre-check is `.layer()`'d
112    // FIRST so it is OUTERMOST (tower's ServiceBuilder/Server::builder()
113    // convention — first `.layer()` call = outermost = runs first, the
114    // opposite of actix's last-`.wrap()`-is-outermost rule). It fails OPEN
115    // to `governor_layer` on any SurrealDB error or missing key, so a DB
116    // blip degrades to the in-memory limiter rather than hard-blocking.
117    let mut builder = Server::builder()
118        .max_frame_size(4 * 1024 * 1024) // CQ-B20: 4 MiB frame cap (tonic-0.14 equivalent of max_decoding_message_size)
119        .timeout(Duration::from_secs(30))
120        .concurrency_limit_per_connection(256)
121        .layer(shared_rate_limit_layer)
122        .layer(governor_layer);
123
124    // REQ-15 AC-1 / CQ-B20: Env-gated TLS.
125    // Set AXIAM__GRPC_TLS_CERT_PATH and AXIAM__GRPC_TLS_KEY_PATH to enable.
126    let cert_path = std::env::var("AXIAM__GRPC_TLS_CERT_PATH").ok();
127    let key_path = std::env::var("AXIAM__GRPC_TLS_KEY_PATH").ok();
128    match (cert_path, key_path) {
129        (Some(cert_path), Some(key_path)) => {
130            let cert_pem = std::fs::read(&cert_path).unwrap_or_else(|e| {
131                panic!("AXIAM__GRPC_TLS_CERT_PATH set but file not readable at '{cert_path}': {e}")
132            });
133            let key_pem = std::fs::read(&key_path).unwrap_or_else(|e| {
134                panic!("AXIAM__GRPC_TLS_KEY_PATH set but file not readable at '{key_path}': {e}")
135            });
136
137            let identity = Identity::from_pem(cert_pem, key_pem);
138            let tls_config = ServerTlsConfig::new().identity(identity);
139            tracing::info!("gRPC server TLS enabled (AXIAM__GRPC_TLS_CERT_PATH)");
140            builder
141                .tls_config(tls_config)?
142                .add_service(authz_svc)
143                .add_service(user_svc)
144                .add_service(token_svc)
145                .serve(addr)
146                .await
147        }
148        _ => {
149            tracing::warn!(
150                "gRPC TLS is DISABLED — set AXIAM__GRPC_TLS_CERT_PATH + \
151                 AXIAM__GRPC_TLS_KEY_PATH to enable (acceptable for in-mesh deployments)"
152            );
153            builder
154                .add_service(authz_svc)
155                .add_service(user_svc)
156                .add_service(token_svc)
157                .serve(addr)
158                .await
159        }
160    }
