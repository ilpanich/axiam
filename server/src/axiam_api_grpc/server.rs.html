1//! Tonic gRPC server setup.
2//!
3//! CQ-B20: Server builder sets message-size limits, per-connection timeout,
4//! and concurrency limit to prevent resource exhaustion.
5//! TLS (REQ-15 AC-1): When `AXIAM__GRPC_TLS_CERT_PATH` and
6//! `AXIAM__GRPC_TLS_KEY_PATH` env vars are set, the server is configured with
7//! `ServerTlsConfig`; otherwise plaintext mode is used (suitable for in-mesh
8//! mutual-TLS handled at the sidecar/service-mesh layer).
9//!
10//! D2 (benchmark plan — native gRPC TLS termination): this is the same
11//! mechanism the p2-tls13 native-TLS bench overlay
12//! (`benchmarks/targets/axiam/docker-compose.native-tls.yml`) turns on,
13//! pointed at the SAME server cert/key files the REST listener uses
14//! (`crates/axiam-server/src/tls.rs` / `AXIAM__SERVER__TLS__CERT_PATH`+
15//! `KEY_PATH`) so the two protocols present identical PKI material. No new
16//! `TlsConfig`-style struct was introduced here — the existing flat
17//! `AXIAM__GRPC_TLS_CERT_PATH`/`KEY_PATH` env-var pair (already shipped in
18//! phase 11) is reused as-is per D2's "least invasive" guidance, rather than
19//! adding a parallel `AXIAM__GRPC__TLS__*` nested config surface.
20//!
21//! Caveat vs. the REST listener: `crates/axiam-server/src/tls.rs` builds a
22//! custom rustls `ServerConfig` restricted to TLS 1.3 only
23//! (`with_protocol_versions(&[&rustls::version::TLS13])`). Tonic 0.14's
24//! `ServerTlsConfig` (see `tonic::transport::server::tls`) does not expose a
25//! protocol-version knob — it always builds `rustls::ServerConfig::builder()`
26//! with the crate's default versions (TLS 1.2 negotiable in addition to 1.3).
27//! There is no way to force TLS-1.3-only through tonic's public API without
28//! hand-rolling the accept loop, which is out of scope here; the gRPC
29//! listener is TLS 1.3-*capable* (matching REST posture) but not TLS
30//! 1.3-*exclusive*.
31

54/// Start the gRPC server with all registered services.
55///
56/// Applies two cooperating rate-limit layers (SECHRD-03, D-01a/b/c — gap
57/// closure for 24-07): the SurrealDB-backed [`GrpcSharedRateLimitLayer`]
58/// shared-store pre-check runs FIRST (outermost), failing OPEN on any DB
59/// error to the per-replica in-memory `GovernorLayer` (per D-10) built via
60/// `grpc_authz_per_sec` from `GrpcConfig`. Both layers derive their client-IP
61/// key from the SAME `trusted_hops` value so gRPC keying stays in lockstep
62/// across the shared store and the in-memory fallback.
63///
64/// Transport limits (CQ-B20):
65/// - Max message size: 4 MiB decode / 4 MiB encode
66/// - Per-connection timeout: 30 s
67/// - Concurrency limit: 256 streams per connection
68///
69/// TLS (REQ-15 AC-1): env-gated via `AXIAM__GRPC_TLS_CERT_PATH` /
70/// `AXIAM__GRPC_TLS_KEY_PATH`. When absent, TLS is disabled and a warning
71/// is logged (acceptable for in-mesh/loopback deployments).
72pub async fn start_grpc_server<R, P, Res, S, G, U, C>(
73    addr: SocketAddr,
74    engine: AuthorizationEngine<R, P, Res, S, G>,
75    user_repo: U,
76    auth_config: AuthConfig,
77    grpc_config: &GrpcConfig,
78    db: Surreal<C>,
79    batch_max_concurrency: usize,
80) -> Result<(), tonic::transport::Error>
81where
82    R: RoleRepository + 'static,
83    P: PermissionRepository + 'static,
84    Res: ResourceRepository + 'static,
85    S: ScopeRepository + 'static,
86    G: GroupRepository + 'static,
87    U: UserRepository + 'static,
88    C: Connection + 'static,
89{
90    tracing::info!(
91        bind = %addr,
92        grpc_authz_per_sec = grpc_config.grpc_authz_per_sec,
93        "Starting gRPC server",
94    );
95
96    // SECHRD-03 gap closure (24-07 follow-up): the shared-store pre-check
97    // MUST use the same trusted_hops value as the in-memory governor's key
98    // extractor (both ultimately key off GrpcTrustedHopsKeyExtractor logic)
99    // so a rotating XFF cannot mint a fresh bucket in one layer while being
100    // correctly collapsed in the other.
101    let trusted_hops = trusted_hops_from_env();
102    let shared_rate_limit_layer = GrpcSharedRateLimitLayer::new(
103        db,
104        "grpc_authz",
105        grpc_config.grpc_authz_per_sec,
106        trusted_hops,
107    );
108    let governor_layer = build_grpc_governor_layer(grpc_config.grpc_authz_per_sec);
109
110    let authz_svc = AuthorizationServiceServer::with_interceptor(
111        AuthorizationServiceImpl::new(engine, batch_max_concurrency),
112        AuthInterceptor::new(auth_config.clone()),
113    );
114    // SECFIX-01: UserService and TokenService previously had zero auth —
115    // any unauthenticated mesh peer could call GetUser/ValidateCredentials/
116    // IntrospectToken. Wrap them with the same AuthInterceptor chokepoint
117    // as AuthorizationService so every gRPC call requires a verified bearer JWT.
118    let user_svc = UserServiceServer::with_interceptor(
119        UserServiceImpl::new(user_repo, auth_config.clone()),
120        AuthInterceptor::new(auth_config.clone()),
121    );
122    let token_svc = TokenServiceServer::with_interceptor(
123        TokenServiceImpl::new(auth_config.clone()),
124        AuthInterceptor::new(auth_config),
125    );
126
127    // CQ-B20: Apply transport limits to the gRPC server builder.
128    // Note: tonic 0.14 does not expose max_decoding_message_size / max_encoding_message_size
129    // at the Server level (added in tonic 0.12+ via the Router API). The HTTP/2 frame-size
130    // limit (max_frame_size) is the closest available per-connection cap in 0.14. Upgrade
131    // to tonic ≥0.12 to use per-service max_decoding_message_size / max_encoding_message_size.
132    // Tracked: max_decoding_message_size (CQ-B20, pending tonic upgrade in Phase 19).
133    // SECHRD-03 (D-01a/b/c): the shared-store pre-check is `.layer()`'d
134    // FIRST so it is OUTERMOST (tower's ServiceBuilder/Server::builder()
135    // convention — first `.layer()` call = outermost = runs first, the
136    // opposite of actix's last-`.wrap()`-is-outermost rule). It fails OPEN
137    // to `governor_layer` on any SurrealDB error or missing key, so a DB
138    // blip degrades to the in-memory limiter rather than hard-blocking.
139    let mut builder = Server::builder()
140        .max_frame_size(4 * 1024 * 1024) // CQ-B20: 4 MiB frame cap (tonic-0.14 equivalent of max_decoding_message_size)
141        .timeout(Duration::from_secs(30))
142        .concurrency_limit_per_connection(256)
143        .layer(shared_rate_limit_layer)
144        .layer(governor_layer);
145
146    // REQ-15 AC-1 / CQ-B20: Env-gated TLS.
147    // Set AXIAM__GRPC_TLS_CERT_PATH and AXIAM__GRPC_TLS_KEY_PATH to enable.
148    let cert_path = std::env::var("AXIAM__GRPC_TLS_CERT_PATH").ok();
149    let key_path = std::env::var("AXIAM__GRPC_TLS_KEY_PATH").ok();
150    match (cert_path, key_path) {
151        (Some(cert_path), Some(key_path)) => {
152            let cert_pem = std::fs::read(&cert_path).unwrap_or_else(|e| {
153                panic!("AXIAM__GRPC_TLS_CERT_PATH set but file not readable at '{cert_path}': {e}")
154            });
155            let key_pem = std::fs::read(&key_path).unwrap_or_else(|e| {
156                panic!("AXIAM__GRPC_TLS_KEY_PATH set but file not readable at '{key_path}': {e}")
157            });
158
159            let identity = Identity::from_pem(cert_pem, key_pem);
160            let tls_config = ServerTlsConfig::new().identity(identity);
161            tracing::info!("gRPC server TLS enabled (AXIAM__GRPC_TLS_CERT_PATH)");
162            builder
163                .tls_config(tls_config)?
164                .add_service(authz_svc)
165                .add_service(user_svc)
166                .add_service(token_svc)
167                .serve(addr)
168                .await
169        }
170        _ => {
171            tracing::warn!(
172                "gRPC TLS is DISABLED — set AXIAM__GRPC_TLS_CERT_PATH + \
173                 AXIAM__GRPC_TLS_KEY_PATH to enable (acceptable for in-mesh deployments)"
174            );
175            builder
176                .add_service(authz_svc)
177                .add_service(user_svc)
178                .add_service(token_svc)
179                .serve(addr)
180                .await
181        }
182    }
