11/// Configuration for the Actix-Web REST API server.
12#[derive(Debug, Clone, Deserialize)]
13#[serde(default)]
14pub struct ServerConfig {
15    /// Bind address (default: "127.0.0.1").
16    pub host: String,
17    /// Bind port (default: 8090).
18    pub port: u16,
19    /// Allowed CORS origins. Empty disables cross-origin requests (restrictive).
20    pub cors_allowed_origins: Vec<String>,
21    /// Optional direct-TLS termination. Disabled by default — the recommended
22    /// deployment terminates TLS at the proxy/load-balancer layer (ASVS V9.1.x,
23    /// D-06). When enabled, the server binds with rustls restricted to TLS 1.3
24    /// (see `axiam-server`).
25    pub tls: TlsConfig,
26}
27
28impl Default for ServerConfig {
29    fn default() -> Self {
30        Self {
31            host: "127.0.0.1".into(),
32            port: 8090,
33            cors_allowed_origins: Vec::new(),
34            tls: TlsConfig::default(),
35        }
36    }
37}

39/// Native client-certificate (mTLS) authentication policy for the direct-TLS
40/// listener (D3).
41///
42/// When client auth is enabled, rustls verifies the presented client
43/// certificate against the CA bundle at [`TlsConfig::client_ca_path`] during
44/// the TLS handshake, and the *verified* leaf certificate is exposed to request
45/// handlers via the connection extensions — the certificate-auth flow then
46/// consumes that verified cert rather than a spoofable proxy header.
47#[derive(Debug, Clone, Copy, PartialEq, Eq, Default, Deserialize)]
48#[serde(rename_all = "lowercase")]
49pub enum ClientAuth {
50    /// No client certificate requested (server-auth only). Backward-compatible
51    /// default — the rustls config is built with `with_no_client_auth()`.
52    #[default]
53    Off,
54    /// Request a client certificate but allow anonymous clients. A presented
55    /// certificate is still verified against the CA bundle; connections without
56    /// one are accepted (`WebPkiClientVerifier::builder(..).allow_unauthenticated()`).
57    Optional,
58    /// Require a client certificate verified against the CA bundle. The TLS
59    /// handshake is rejected if the client presents no certificate or an
60    /// unverifiable one (`WebPkiClientVerifier::builder(..).build()`).
61    Required,
62}
63
64/// Direct-TLS configuration for the REST API listener.
65///
66/// TLS is **opt-in**: the default (`enabled = false`) preserves the plaintext
67/// bind used behind a TLS-terminating proxy. When `enabled = true`, both
68/// `cert_path` and `key_path` must point at readable PEM files; the server
69/// negotiates TLS 1.3 only (ASVS V9.1.2), whose cipher suites are all
70/// ASVS-approved (V9.1.3).
71#[derive(Debug, Clone, Deserialize)]
72#[serde(default)]
73pub struct TlsConfig {
74    /// Enable direct TLS termination in the server process.
75    pub enabled: bool,
76    /// Path to the PEM-encoded certificate chain (leaf first).
77    pub cert_path: Option<PathBuf>,
78    /// Path to the PEM-encoded private key (PKCS#8, PKCS#1, or SEC1).
79    pub key_path: Option<PathBuf>,
80    /// Offer HTTP/2 (`h2`) over ALPN alongside `http/1.1`. Default `true`
81    /// (backward-compatible; matches what the actix-web rustls bind advertises).
82    ///
83    /// Set `false` to build the rustls listener advertising `http/1.1` only —
84    /// used to run the p2 benchmark apples-to-apples with the plaintext p0
85    /// listener (which is HTTP/1.1), isolating the h2-vs-h1.1 throughput effect
86    /// (B2). See `axiam-server`'s `tls` module for the important caveat that the
87    /// actix-web `HttpServer` bind re-adds `h2` to ALPN regardless.
88    pub http2: bool,
89    /// Native client-certificate (mTLS) policy (D3). Default `off` keeps the
90    /// server-auth-only behaviour. `optional`/`required` build the rustls config
91    /// with a `WebPkiClientVerifier` over [`Self::client_ca_path`].
92    pub client_auth: ClientAuth,
93    /// Path to the PEM CA bundle used to verify client certificates. Required
94    /// (and must be readable) when `client_auth` is `optional` or `required`;
95    /// ignored when `off`.
96    pub client_ca_path: Option<PathBuf>,
97}
98
99impl Default for TlsConfig {
100    fn default() -> Self {
101        Self {
102            enabled: false,
103            cert_path: None,
104            key_path: None,
105            http2: true,
106            client_auth: ClientAuth::Off,
107            client_ca_path: None,
108        }
109    }
110}
111
112impl ServerConfig {
113    /// Returns the socket bind address as "host:port".
114    pub fn bind_address(&self) -> String {
115        format!("{}:{}", self.host, self.port)
116    }
117}

119#[cfg(test)]
120mod tests {
121    use super::*;
122
123    #[test]
124    fn client_auth_defaults_to_off() {
125        assert_eq!(ClientAuth::default(), ClientAuth::Off);
126        assert_eq!(TlsConfig::default().client_auth, ClientAuth::Off);
127        assert!(TlsConfig::default().client_ca_path.is_none());
128    }
129
130    #[test]
131    fn client_auth_parses_all_three_values() {
132        for (raw, expected) in [
133            ("\"off\"", ClientAuth::Off),
134            ("\"optional\"", ClientAuth::Optional),
135            ("\"required\"", ClientAuth::Required),
136        ] {
137            let parsed: ClientAuth =
138                serde_json::from_str(raw).unwrap_or_else(|e| panic!("{raw} must parse: {e}"));
139            assert_eq!(parsed, expected, "{raw}");
140        }
141    }
142
143    #[test]
144    fn client_auth_rejects_invalid_value() {
145        let err = serde_json::from_str::<ClientAuth>("\"enabled\"");
146        assert!(err.is_err(), "unknown client_auth value must be rejected");
147    }
148
149    #[test]
150    fn tls_config_deserializes_client_auth_fields() {
151        let cfg: TlsConfig = serde_json::from_value(serde_json::json!({
152            "enabled": true,
153            "cert_path": "/etc/axiam/server.crt",
154            "key_path": "/etc/axiam/server.key",
155            "client_auth": "required",
156            "client_ca_path": "/etc/axiam/ca.crt",
157        }))
158        .expect("TlsConfig with client-auth fields must deserialize");
159        assert_eq!(cfg.client_auth, ClientAuth::Required);
160        assert_eq!(
161            cfg.client_ca_path.as_deref(),
162            Some(std::path::Path::new("/etc/axiam/ca.crt"))
163        );
164    }
