5/// Rate-limit bucket-key derivation mode (D8).
6///
7/// Environment variable: `AXIAM__RATE_LIMIT__KEY` = `ip` | `client_id` |
8/// `ip_client_id` (default `ip`).
9///
10/// **Why this exists (NAT'd-fleet lesson):** the bucket key was
11/// unconditionally `"{endpoint}:{ip}"` (see
12/// `middleware::rate_limit_shared::RateLimitShared` and
13/// `extractors::rate_limit::XForwardedForKeyExtractor`). Behind a NAT/proxy
14/// fleet (many distinct OAuth2 clients — e.g. IoT devices or microservices —
15/// egressing through one shared IP), every client sharing that IP collided
16/// into a SINGLE bucket, so one noisy/misbehaving client could exhaust the
17/// `/oauth2/token`, `/oauth2/revoke`, or `/oauth2/introspect` quota for every
18/// other client behind the same NAT gateway. `client_id` and `ip_client_id`
19/// give each OAuth2 client its own bucket (optionally still scoped per-IP)
20/// on the endpoints where a client identity is actually known.
21///
22/// **Scope — where this setting applies:** ONLY the three endpoints where an
23/// OAuth2 client authenticates itself via a form-encoded `client_id`
24/// (`client_secret_post`, RFC 6749 §2.3.1): `/oauth2/token`,
25/// `/oauth2/revoke`, `/oauth2/introspect` (see `handlers::oauth2` and
26/// `server.rs`'s wiring of `RateLimitShared::new_client_identity_aware` /
27/// the client-aware governor for exactly those three resources).
28///
29/// **`/auth/login` (and every other rate-limited endpoint) ALWAYS keys
30/// per-IP, regardless of this setting.** Login authenticates a *user* via
31/// username/password — there is no OAuth2 client identity anywhere in that
32/// request, so there is nothing meaningful to key on besides the source IP.
33/// This is intentional and NOT a bug: switching this config value never
34/// changes login's (or MFA's, or password-reset's, etc.) rate-limit
35/// behavior. See `server.rs` — those resources are wired with the plain
36/// `build_governor`/`RateLimitShared::new` constructors, which never read
37/// this field.
38#[derive(Debug, Clone, Copy, PartialEq, Eq, Deserialize, Default)]
39#[serde(rename_all = "snake_case")]
40pub enum RateLimitKeyMode {
41    /// Key on source IP only (current/default behavior, unchanged).
42    #[default]
43    Ip,
44    /// Key on the OAuth2 `client_id` alone — independent buckets per client,
45    /// regardless of which IP(s) it connects from.
46    ClientId,
47    /// Key on the `(ip, client_id)` pair — independent buckets per client
48    /// AND per IP, so a compromised/leaked client credential rate-limited
49    /// from one IP doesn't automatically throttle the same client_id
50    /// operating legitimately from a different IP.
51    IpClientId,
52}
53
54/// Rate limit configuration.
55/// Environment variables: AXIAM__RATE_LIMIT__LOGIN_PER_MIN, etc.
56#[derive(Debug, Clone, Deserialize)]
57#[serde(default)]
58pub struct RateLimitConfig {
59    /// Max login requests per minute per IP (default: 10).
60    pub login_per_min: u32,
61    /// Max register requests per minute per IP (default: 5).
62    pub register_per_min: u32,
63    /// Max oauth2/token requests per minute per client (default: 20).
64    pub token_per_min: u32,
65    /// Max password-reset requests per minute per IP (default: 3).
66    pub password_reset_per_min: u32,
67    /// Max MFA requests per minute per IP (default: 5).
68    /// Covers /auth/mfa/enroll, /confirm, /verify, /setup/enroll, /setup/confirm (SEC-020).
69    pub mfa_per_min: u32,
70    /// Max oauth2/introspect requests per minute per IP (default: 10).
71    /// SEC-020: introspect endpoint rate-limited to prevent token probing.
72    pub introspect_per_min: u32,
73    /// Max oauth2/revoke requests per minute per IP (default: 10).
74    /// SEC-020: revoke endpoint rate-limited to prevent DoS via token flooding.
75    pub revoke_per_min: u32,
76    /// Max authz-check requests per minute per IP (default: 300).
77    /// Authz checks are read-only and high-frequency — used by UI permission gating.
78    /// Kept in a dedicated bucket so heavy UI use does not consume the login/token limit (D-07).
79    pub authz_check_per_min: u32,
80    /// Rate-limit bucket-key derivation mode (D8, default: `Ip` — current
81    /// behavior, unchanged). See [`RateLimitKeyMode`] for the full
82    /// rationale and scope (only `/oauth2/token`, `/oauth2/revoke`,
83    /// `/oauth2/introspect` honor this; `/auth/login` and every other
84    /// endpoint always stay per-IP).
85    pub key: RateLimitKeyMode,
86}
87
88impl Default for RateLimitConfig {
89    fn default() -> Self {
90        Self {
91            login_per_min: 10,
92            register_per_min: 5,
93            token_per_min: 20,
94            password_reset_per_min: 3,
95            mfa_per_min: 5,
96            introspect_per_min: 10,
97            revoke_per_min: 10,
98            authz_check_per_min: 300,
99            key: RateLimitKeyMode::Ip,
100        }
101    }
102}
103
104impl RateLimitConfig {
105    /// Validates all rate limits are >= 1 (governor panics on zero).
106    pub fn validate(&self) {
107        assert!(self.login_per_min >= 1, "login_per_min must be >= 1");
108        assert!(self.register_per_min >= 1, "register_per_min must be >= 1");
109        assert!(self.token_per_min >= 1, "token_per_min must be >= 1");
110        assert!(
111            self.password_reset_per_min >= 1,
112            "password_reset_per_min must be >= 1"
113        );
114        assert!(self.mfa_per_min >= 1, "mfa_per_min must be >= 1");
115        assert!(
116            self.introspect_per_min >= 1,
117            "introspect_per_min must be >= 1"
118        );
119        assert!(self.revoke_per_min >= 1, "revoke_per_min must be >= 1");
120        assert!(
121            self.authz_check_per_min >= 1,
122            "authz_check_per_min must be >= 1"
123        );
124    }
125}

127#[cfg(test)]
128mod tests {
129    use super::*;
130
131    /// D8 acceptance: default behavior (`ip`) is unchanged — a
132    /// freshly-defaulted config must key exactly the way it did before this
133    /// field existed.
134    #[test]
135    fn default_key_mode_is_ip() {
136        assert_eq!(RateLimitConfig::default().key, RateLimitKeyMode::Ip);
137        assert_eq!(RateLimitKeyMode::default(), RateLimitKeyMode::Ip);
138    }
139
140    /// `AXIAM__RATE_LIMIT__KEY` values must map onto exactly `ip`,
141    /// `client_id`, `ip_client_id` via serde `snake_case` — this is what the
142    /// `config` crate's `Environment` source (see `axiam-server::main`)
143    /// deserializes the raw env var string against.
144    #[test]
145    fn key_mode_deserializes_from_documented_env_values() {
146        assert_eq!(
147            serde_json::from_str::<RateLimitKeyMode>("\"ip\"").unwrap(),
148            RateLimitKeyMode::Ip
149        );
150        assert_eq!(
151            serde_json::from_str::<RateLimitKeyMode>("\"client_id\"").unwrap(),
152            RateLimitKeyMode::ClientId
153        );
154        assert_eq!(
155            serde_json::from_str::<RateLimitKeyMode>("\"ip_client_id\"").unwrap(),
156            RateLimitKeyMode::IpClientId
157        );
158    }
