1//! Rate limiting configuration loaded from environment variables.
2
3use serde::Deserialize;
4
5/// Rate limit configuration.
6/// Environment variables: AXIAM__RATE_LIMIT__LOGIN_PER_MIN, etc.
7#[derive(Debug, Clone, Deserialize)]
8#[serde(default)]
9pub struct RateLimitConfig {
10    /// Max login requests per minute per IP (default: 10).
11    pub login_per_min: u32,
12    /// Max register requests per minute per IP (default: 5).
13    pub register_per_min: u32,
14    /// Max oauth2/token requests per minute per client (default: 20).
15    pub token_per_min: u32,
16    /// Max password-reset requests per minute per IP (default: 3).
17    pub password_reset_per_min: u32,
18    /// Max MFA requests per minute per IP (default: 5).
19    /// Covers /auth/mfa/enroll, /confirm, /verify, /setup/enroll, /setup/confirm (SEC-020).
20    pub mfa_per_min: u32,
21    /// Max oauth2/introspect requests per minute per IP (default: 10).
22    /// SEC-020: introspect endpoint rate-limited to prevent token probing.
23    pub introspect_per_min: u32,
24    /// Max oauth2/revoke requests per minute per IP (default: 10).
25    /// SEC-020: revoke endpoint rate-limited to prevent DoS via token flooding.
26    pub revoke_per_min: u32,
27    /// Max authz-check requests per minute per IP (default: 300).
28    /// Authz checks are read-only and high-frequency — used by UI permission gating.
29    /// Kept in a dedicated bucket so heavy UI use does not consume the login/token limit (D-07).
30    pub authz_check_per_min: u32,
31}
32
33impl Default for RateLimitConfig {
34    fn default() -> Self {
35        Self {
36            login_per_min: 10,
37            register_per_min: 5,
38            token_per_min: 20,
39            password_reset_per_min: 3,
40            mfa_per_min: 5,
41            introspect_per_min: 10,
42            revoke_per_min: 10,
43            authz_check_per_min: 300,
44        }
45    }
46}
47
48impl RateLimitConfig {
49    /// Validates all rate limits are >= 1 (governor panics on zero).
50    pub fn validate(&self) {
51        assert!(self.login_per_min >= 1, "login_per_min must be >= 1");
52        assert!(self.register_per_min >= 1, "register_per_min must be >= 1");
53        assert!(self.token_per_min >= 1, "token_per_min must be >= 1");
54        assert!(
55            self.password_reset_per_min >= 1,
56            "password_reset_per_min must be >= 1"
57        );
58        assert!(self.mfa_per_min >= 1, "mfa_per_min must be >= 1");
59        assert!(
60            self.introspect_per_min >= 1,
61            "introspect_per_min must be >= 1"
62        );
63        assert!(self.revoke_per_min >= 1, "revoke_per_min must be >= 1");
64        assert!(
65            self.authz_check_per_min >= 1,
66            "authz_check_per_min must be >= 1"
67        );
68    }
69}
