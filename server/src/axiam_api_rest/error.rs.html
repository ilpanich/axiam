1//! HTTP error response mapping for AXIAM API.
2//!
3//! Provides [`AxiamApiError`], a newtype around [`AxiamError`] that
4//! implements Actix-Web's [`ResponseError`] trait.
5
6use actix_web::HttpResponse;
7use actix_web::http::StatusCode;
8use axiam_core::error::AxiamError;
9use serde::Serialize;
10use tracing::error;
11
12/// Newtype wrapper so we can implement Actix-Web's `ResponseError`
13/// for the core `AxiamError` (orphan rule).
14#[derive(Debug)]
15pub struct AxiamApiError(pub AxiamError);
16
17impl std::fmt::Display for AxiamApiError {
18    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
19        write!(f, "{}", self.0)
20    }
21}
22
23impl From<AxiamError> for AxiamApiError {
24    fn from(err: AxiamError) -> Self {
25        Self(err)
26    }
27}
28
29#[derive(Serialize)]
30struct ErrorBody {
31    error: String,
32    message: String,
33    /// The action being checked (e.g. `"users:create"`), when known.
34    /// Populated only for `authorization_denied` (403) responses;
35    /// omitted from the JSON body otherwise (SDK-Q02).
36    #[serde(skip_serializing_if = "Option::is_none")]
37    action: Option<String>,
38    /// The resource id being checked, when known and not the "global"
39    /// nil-UUID sentinel. Populated only for `authorization_denied` (403)
40    /// responses; omitted from the JSON body otherwise (SDK-Q02).
41    #[serde(skip_serializing_if = "Option::is_none")]
42    resource_id: Option<String>,
43}
44
45impl actix_web::ResponseError for AxiamApiError {
46    fn status_code(&self) -> StatusCode {
47        match &self.0 {
48            AxiamError::NotFound { .. } => StatusCode::NOT_FOUND,
49            AxiamError::AlreadyExists { .. } => StatusCode::CONFLICT,
50            AxiamError::AuthenticationFailed { .. } | AxiamError::ReplayDetected => {
51                StatusCode::UNAUTHORIZED
52            }
53            AxiamError::AuthorizationDenied { .. } => StatusCode::FORBIDDEN,
54            AxiamError::Validation { .. } | AxiamError::TenantContext => StatusCode::BAD_REQUEST,
55            AxiamError::PasswordPolicy { .. } => StatusCode::UNPROCESSABLE_ENTITY,
56            AxiamError::RateLimited => StatusCode::TOO_MANY_REQUESTS,
57            AxiamError::EmailConfig(_) => StatusCode::BAD_REQUEST,
58            AxiamError::ServiceUnavailable(_) => StatusCode::SERVICE_UNAVAILABLE,
59            AxiamError::Database(_)
60            | AxiamError::Certificate(_)
61            | AxiamError::Crypto(_)
62            | AxiamError::EmailDelivery(_)
63            | AxiamError::WebhookDelivery(_)
64            | AxiamError::Internal(_) => StatusCode::INTERNAL_SERVER_ERROR,
65            // Note: ReplayDetected is handled above in the UNAUTHORIZED arm.
66        }
67    }
68
69    fn error_response(&self) -> HttpResponse {
70        // Client-facing error code slug.
71        let error_code = match &self.0 {
72            AxiamError::NotFound { .. } => "not_found",
73            AxiamError::AlreadyExists { .. } => "already_exists",
74            AxiamError::AuthenticationFailed { .. } | AxiamError::ReplayDetected => {
75                "authentication_failed"
76            }
77            AxiamError::AuthorizationDenied { .. } => "authorization_denied",
78            AxiamError::Validation { .. } => "validation_error",
79            AxiamError::PasswordPolicy { .. } => "password_policy_violation",
80            AxiamError::TenantContext => "tenant_context",
81            AxiamError::RateLimited => "rate_limited",
82            AxiamError::EmailConfig(_) => "email_config_error",
83            AxiamError::ServiceUnavailable(_) => "service_unavailable",
84            // Server-error variants: log detail, return generic message.
85            _ => "internal_error",
86        };
87
88        // Client-facing message: echo for known client errors; generic for 5xx.
89        // SEC-011/SEC-039/CQ-B33: internal detail (DB strings, crypto messages,
90        // stack traces) MUST NOT appear in the response body.
91        let message = match &self.0 {
92            AxiamError::NotFound { .. }
93            | AxiamError::AlreadyExists { .. }
94            | AxiamError::AuthenticationFailed { .. }
95            | AxiamError::ReplayDetected
96            | AxiamError::AuthorizationDenied { .. }
97            | AxiamError::Validation { .. }
98            | AxiamError::PasswordPolicy { .. }
99            | AxiamError::TenantContext
100            | AxiamError::RateLimited
101            | AxiamError::EmailConfig(_)
102            | AxiamError::ServiceUnavailable(_) => self.0.to_string(),
103            // 5xx variants: log the detail server-side, return only a generic message.
104            _ => {
105                error!(
106                    error = %self.0,
107                    "internal server error"
108                );
109                "An internal error occurred".to_string()
110            }
111        };
112
113        // SDK-Q02: surface the checked action/resource on authorization
114        // denials so SDKs can parse them from the response body (CONTRACT
115        // §2 "if available from the response body"). Every other error
116        // kind omits both fields (skip_serializing_if above).
117        let (action, resource_id) = match &self.0 {
118            AxiamError::AuthorizationDenied {
119                action,
120                resource_id,
121                ..
122            } => (action.clone(), resource_id.clone()),
123            _ => (None, None),
124        };
125
126        HttpResponse::build(self.status_code()).json(ErrorBody {
127            error: error_code.into(),
128            message,
129            action,
130            resource_id,
131        })
132    }
133}
