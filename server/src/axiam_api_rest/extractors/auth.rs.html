26/// Object-safe per-request session-validity check (D-15 / REQ-7).
27///
28/// Access tokens are stateless JWTs, so revoking a session (deleting its row)
29/// has no effect unless every authenticated request re-checks that the session
30/// behind the token's `jti` (= `session.id`) is still active. This trait is the
31/// object-safe seam that lets the connection-agnostic [`AuthenticatedUser`]
32/// extractor perform that check without being generic over the DB `Connection`.
33///
34/// Mirrors the [`crate::authz::AuthzChecker`] boxed-future pattern because the
35/// underlying repository methods are native `async fn` (RPITIT, not dyn-safe).
36pub trait SessionValidator: Send + Sync {
37    /// Returns `true` if a non-expired session with `session_id` exists for
38    /// `tenant_id`. A revoked session (row deleted) or an expired one is
39    /// considered inactive.
40    fn is_session_active<'a>(
41        &'a self,
42        tenant_id: Uuid,
43        session_id: Uuid,
44    ) -> Pin<Box<dyn Future<Output = bool> + Send + 'a>>;
45}
46
47impl<C: Connection> SessionValidator for SurrealSessionRepository<C> {
48    fn is_session_active<'a>(
49        &'a self,
50        tenant_id: Uuid,
51        session_id: Uuid,
52    ) -> Pin<Box<dyn Future<Output = bool> + Send + 'a>> {
53        Box::pin(async move {
54            match self.get_by_id(tenant_id, session_id).await {
55                Ok(session) => session.expires_at > chrono::Utc::now(),
56                Err(_) => false,
57            }
58        })
59    }
60}

62/// Authenticated user context extracted from a valid JWT.
63///
64/// Use this as a handler parameter to require authentication.
65/// If the audit middleware has already validated the token, the cached
66/// claims are reused to avoid double verification.
67///
68/// **Audience narrowing (D-19 / D-21):** only tokens with `aud = axiam:user`
69/// are accepted on user-facing routes. M2M tokens are rejected with 401.
70/// When `aud` is absent and `AuthConfig.allow_missing_aud_as_user` is `true`,
71/// the token is accepted with a `WARN` log on every request. No in-process
72/// rate limit is applied — operator-side log filtering (Loki dedup, fluent-bit
73/// rewrite_tag, etc.) handles burst noise. This trade-off keeps the extractor
74/// allocation-free and avoids shared mutable state.
75#[derive(Debug, Clone)]
76pub struct AuthenticatedUser {
77    pub user_id: Uuid,
78    pub tenant_id: Uuid,
79    pub org_id: Uuid,
80    /// The session ID — equals the JWT `jti` claim which is set to
81    /// `session.id` for user-facing tokens (D-15). Use this for
82    /// selective session invalidation on password change.
83    pub session_id: Uuid,
84    pub claims: ValidatedClaims,
85}
86
87impl actix_web::FromRequest for AuthenticatedUser {
88    type Error = AxiamApiError;
89    type Future = Pin<Box<dyn Future<Output = Result<Self, Self::Error>>>>;
90
91    fn from_request(req: &HttpRequest, _payload: &mut Payload) -> Self::Future {
92        // Synchronous JWT/aud/jti extraction first (ends the `req` borrow before
93        // the async block). Clone the optional session validator (an Arc) so the
94        // returned future is `'static`.
95        let user_result = extract_user(req);
96        let validator = req
97            .app_data::<web::Data<Arc<dyn SessionValidator>>>()
98            .map(|d| d.get_ref().clone());
99
100        Box::pin(async move {
101            let user = user_result?;
102            // REQ-7 / D-15: reject access tokens whose session has been revoked
103            // (row deleted on password change/reset/MFA reset) or expired. The
104            // validator is optional so non-session test harnesses are unaffected;
105            // the production server (and session-security tests) always register it.
106            if let Some(validator) = validator
107                && !validator
108                    .is_session_active(user.tenant_id, user.session_id)
109                    .await
110            {
111                return Err(AxiamError::AuthenticationFailed {
112                    reason: "session revoked or expired".into(),
113                }
114                .into());
115            }
116            Ok(user)
117        })
118    }
119}

121/// Service-account context extracted from a valid M2M JWT.
122///
123/// Use this as a handler parameter to require M2M authentication.
124/// Accepts only tokens with `aud = axiam:m2m`; user tokens are rejected.
125#[derive(Debug, Clone)]
126pub struct AuthenticatedServiceAccount {
127    /// OAuth2 `client_id` (the token `sub` for M2M tokens).
128    pub client_id: String,
129    pub tenant_id: Uuid,
130    pub claims: ValidatedClaims,
131}

133impl actix_web::FromRequest for AuthenticatedServiceAccount {

146/// Decode and validate the JWT from the request — shared between both
147/// extractors. Does not apply audience narrowing.
148fn parse_validated_claims(req: &HttpRequest) -> Result<ValidatedClaims, AxiamApiError> {
149    let config = req
150        .app_data::<web::Data<AuthConfig>>()
151        .ok_or(AxiamError::Internal("missing auth config".into()))?;
152
153    // Try cookie first (browser clients), then Authorization header (service clients).
154    let token = if let Some(cookie) = req.cookie("axiam_access") {
155        cookie.value().to_owned()
156    } else {
157        // Fall back to Authorization: Bearer header for non-browser clients.
158        let header = req
159            .headers()
160            .get("Authorization")
161            .and_then(|v| v.to_str().ok())
162            .ok_or(AxiamError::AuthenticationFailed {
163                reason: "missing authentication credentials".into(),
164            })?;
165
166        // Parse `Authorization` as case-insensitive Bearer with flexible whitespace.
167        let header = header.trim();
168        let mut parts = header.splitn(2, char::is_whitespace);
169        let scheme = parts.next().unwrap_or("");
170        let credentials = parts.next().unwrap_or("").trim();
171
172        if !scheme.eq_ignore_ascii_case("bearer") || credentials.is_empty() {
173            return Err(AxiamError::AuthenticationFailed {
174                reason: "invalid Authorization scheme, expected Bearer".into(),
175            }
176            .into());
177        }
178        credentials.to_owned()
179    };
180
181    let validated = validate_access_token(&token, config).map_err(AxiamError::from)?;
182    Ok(validated)
183}

189fn extract_user(req: &HttpRequest) -> Result<AuthenticatedUser, AxiamApiError> {
190    // Try to reuse claims cached by the audit middleware.
191    if let Some(cached) = req.extensions().get::<Arc<CachedUserIdentity>>() {
192        let config = req
193            .app_data::<web::Data<AuthConfig>>()
194            .ok_or(AxiamError::Internal("missing auth config".into()))?;
195
196        let session_id = check_user_aud_and_parse_jti(&cached.claims, config)?;
197        return Ok(AuthenticatedUser {
198            user_id: cached.user_id,
199            tenant_id: cached.tenant_id,
200            org_id: cached.org_id,
201            session_id,
202            claims: cached.claims.clone(),
203        });
204    }
205
206    let config = req
207        .app_data::<web::Data<AuthConfig>>()
208        .ok_or(AxiamError::Internal("missing auth config".into()))?;
209
210    let validated = parse_validated_claims(req)?;
211
212    let session_id = check_user_aud_and_parse_jti(&validated, config)?;
213
214    let user_id =
215        Uuid::parse_str(&validated.0.sub).map_err(|_| AxiamError::AuthenticationFailed {
216            reason: "invalid sub claim".into(),
217        })?;
218
219    let tenant_id =
220        Uuid::parse_str(&validated.0.tenant_id).map_err(|_| AxiamError::AuthenticationFailed {
221            reason: "invalid tenant_id claim".into(),
222        })?;
223
224    let org_id =
225        Uuid::parse_str(&validated.0.org_id).map_err(|_| AxiamError::AuthenticationFailed {
226            reason: "invalid org_id claim".into(),
227        })?;
228
229    Ok(AuthenticatedUser {
230        user_id,
231        tenant_id,
232        org_id,
233        session_id,
234        claims: validated,
235    })
236}

238/// Enforce the audience narrowing policy for user-facing routes and
239/// parse the `jti` claim as a `Uuid` (= session ID).
240///
241/// Audience rules (D-19, D-21):
242/// - `aud = axiam:user` → accepted.
243/// - `aud` absent AND `allow_missing_aud_as_user = true` → accepted; a
244///   `WARN` is emitted on every such request. No in-process rate limiting is
245///   applied here — operator-side log deduplication (Loki, fluent-bit, etc.)
246///   handles burst noise, keeping this hot path allocation-free and
247///   lock-free.
248/// - `aud` absent AND `allow_missing_aud_as_user = false` → 401.
249/// - `aud = axiam:m2m` → 401 (audience mismatch — wrong token type for
250///   this route).
251/// - Any other `aud` value → 401 (unknown audience).
252fn check_user_aud_and_parse_jti(
253    validated: &ValidatedClaims,
254    config: &AuthConfig,
255) -> Result<Uuid, AxiamApiError> {
256    match validated.0.aud.as_deref() {
257        Some(AUD_USER) => {
258            // Expected audience — accept.
259        }
260        None => {
261            if config.allow_missing_aud_as_user {
262                // Back-compat window: treat as axiam:user, but warn unconditionally.
263                // TRADE-OFF: no in-process rate limit on this warn. Operator-side
264                // log filtering (e.g. Loki `rate()` alert, fluent-bit rewrite_tag
265                // dedup) is the intended mechanism for suppressing burst noise
266                // during the rollout window. Keeping this path lock-free is
267                // intentional — auth hot paths must not contend on shared atomics.
268                let jti = &validated.0.jti;
269                tracing::warn!(
270                    token_jti = %jti,
271                    "accepted access token without aud — backward-compat window active"
272                );
273            } else {
274                return Err(AxiamError::AuthenticationFailed {
275                    reason: "aud required".into(),
276                }
277                .into());
278            }
279        }
280        Some(AUD_M2M) => {
281            return Err(AxiamError::AuthenticationFailed {
282                reason: "audience mismatch — this route requires axiam:user audience".into(),
283            }
284            .into());
285        }
286        Some(_) => {
287            return Err(AxiamError::AuthenticationFailed {
288                reason: "unknown audience".into(),
289            }
290            .into());
291        }
292    }
293
294    Uuid::parse_str(&validated.0.jti).map_err(|_| {
295        AxiamError::AuthenticationFailed {
296            reason: "invalid jti".into(),
297        }
298        .into()
299    })
300}

305
306fn extract_service_account(
307    req: &HttpRequest,
308) -> Result<AuthenticatedServiceAccount, AxiamApiError> {
309    let validated = parse_validated_claims(req)?;
310
311    match validated.0.aud.as_deref() {
312        Some(AUD_M2M) => {
313            // Expected audience — accept.
314        }
315        Some(AUD_USER) => {
316            return Err(AxiamError::AuthenticationFailed {
317                reason: "audience mismatch — this route requires axiam:m2m audience".into(),
318            }
319            .into());
320        }
321        Some(_) | None => {
322            return Err(AxiamError::AuthenticationFailed {
323                reason: "audience mismatch — this route requires axiam:m2m audience".into(),
324            }
325            .into());
326        }
327    }
328
329    let tenant_id =
330        Uuid::parse_str(&validated.0.tenant_id).map_err(|_| AxiamError::AuthenticationFailed {
331            reason: "invalid tenant_id claim".into(),
332        })?;
333
334    Ok(AuthenticatedServiceAccount {
335        client_id: validated.0.sub.clone(),
336        tenant_id,
337        claims: validated,
338    })
339}

393    fn make_user_token(config: &AuthConfig, jti: Option<String>) -> String {
394        let jti = jti.unwrap_or_else(|| Uuid::new_v4().to_string());
395        issue_access_token(
396            Uuid::new_v4(),
397            Uuid::new_v4(),
398            Uuid::new_v4(),
399            &[],
400            config,
401            jti,
402            AUD_USER,
403        )
404        .unwrap()
405    }

407    fn make_m2m_token(config: &AuthConfig) -> String {
408        issue_client_credentials_token("my-service", Uuid::new_v4(), Uuid::new_v4(), &[], config)
409            .unwrap()
410    }

412    fn make_no_aud_token(config: &AuthConfig) -> String {
413        // Manually craft via issue_access_token then patch — actually simpler to
414        // issue a user token and strip aud. We don't have that API, so instead
415        // we use the fact that decode_access_token does not require aud.
416        // The simplest approach: issue with AUD_USER but override aud to None
417        // by constructing claims manually.
418        use axiam_auth::token::{AccessTokenClaims, SubjectKind};
419        use chrono::Utc;
420        use jsonwebtoken::{Algorithm, EncodingKey, Header};
421
422        let now = Utc::now().timestamp();
423        let claims = AccessTokenClaims {
424            sub: Uuid::new_v4().to_string(),
425            tenant_id: Uuid::new_v4().to_string(),
426            org_id: Uuid::new_v4().to_string(),
427            iss: "axiam-test".into(),
428            iat: now,
429            exp: now + 900,
430            jti: Uuid::new_v4().to_string(),
431            aud: None, // no audience
432            scope: None,
433            sub_kind: SubjectKind::User,
434        };
435        let key = EncodingKey::from_ed_pem(config.jwt_private_key_pem.as_bytes()).unwrap();
436        let header = Header::new(Algorithm::EdDSA);
437        jsonwebtoken::encode(&header, &claims, &key).unwrap()
438    }
