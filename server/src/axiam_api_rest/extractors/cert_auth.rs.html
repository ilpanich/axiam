1//! Certificate-based authentication extractor for mTLS device auth.
2//!
3//! Reads the `X-Client-Certificate` header (URL-encoded PEM forwarded
4//! by the TLS-terminating reverse proxy) and validates it via
5//! [`DeviceAuthService`].
6
7use actix_web::HttpRequest;
8use actix_web::web;
9use axiam_core::error::AxiamError;
10use axiam_core::models::certificate::DeviceIdentity;
11use surrealdb::Connection;
12use uuid::Uuid;
13
14use crate::error::AxiamApiError;
15use crate::state::AppState;
16
17/// Authenticated device context extracted from a client certificate.
18///
19/// Use this as a handler parameter to require certificate-based auth.
20#[derive(Debug, Clone)]
21pub struct CertificateAuthenticated {
22    pub service_account_id: Uuid,
23    pub tenant_id: Uuid,
24    pub certificate_id: Uuid,
25}
26
27impl CertificateAuthenticated {
28    /// Extract and validate the client certificate from the request.
29    ///
30    /// This is called manually from the handler rather than via
31    /// `FromRequest`, because the concrete `DeviceAuthService` type
32    /// depends on the DB connection generic `C`.
33    pub async fn extract<C: Connection + Clone>(req: &HttpRequest) -> Result<Self, AxiamApiError> {
34        let state = req
35            .app_data::<web::Data<AppState<C>>>()
36            .ok_or(AxiamError::Internal("missing AppState".into()))?;
37        let service = &state.device_auth_service;
38
39        let header = req
40            .headers()
41            .get("X-Client-Certificate")
42            .and_then(|v| v.to_str().ok())
43            .ok_or(AxiamError::AuthenticationFailed {
44                reason: "missing X-Client-Certificate header".into(),
45            })?;
46
47        // URL-decode the PEM (reverse proxy URL-encodes it)
48        let pem = urldecode(header).map_err(|_| AxiamError::AuthenticationFailed {
49            reason: "invalid URL encoding in X-Client-Certificate".into(),
50        })?;
51
52        let identity: DeviceIdentity = service.authenticate(&pem).await.map_err(|e| match &e {
53            // Map certificate/NotFound errors to proper 401/403 status codes
54            AxiamError::Certificate(msg) if msg.contains("not bound to a service account") => {
55                AxiamError::AuthorizationDenied {
56                    reason: msg.clone(),
57                    action: None,
58                    resource_id: None,
59                }
60            }
61            AxiamError::Certificate(msg) => AxiamError::AuthenticationFailed {
62                reason: msg.clone(),
63            },
64            AxiamError::NotFound { .. } => AxiamError::AuthenticationFailed {
65                reason: "unknown client certificate".into(),
66            },
67            _ => e,
68        })?;
69
70        Ok(CertificateAuthenticated {
71            service_account_id: identity.service_account_id,
72            tenant_id: identity.tenant_id,
73            certificate_id: identity.certificate_id,
74        })
75    }
76}

78/// Simple percent-decoding for the `X-Client-Certificate` header.
79fn urldecode(input: &str) -> Result<String, ()> {
80    let mut result = Vec::with_capacity(input.len());
81    let mut bytes = input.bytes();
82    while let Some(b) = bytes.next() {
83        if b == b'%' {
84            let hi = bytes.next().ok_or(())?;
85            let lo = bytes.next().ok_or(())?;
86            let hi = hex_val(hi)?;
87            let lo = hex_val(lo)?;
88            result.push((hi << 4) | lo);
89        } else {
90            result.push(b);
91        }
92    }
93    String::from_utf8(result).map_err(|_| ())
94}
95
96fn hex_val(b: u8) -> Result<u8, ()> {
97    match b {
98        b'0'..=b'9' => Ok(b - b'0'),
99        b'a'..=b'f' => Ok(b - b'a' + 10),
100        b'A'..=b'F' => Ok(b - b'A' + 10),
101        _ => Err(()),
102    }
103}
