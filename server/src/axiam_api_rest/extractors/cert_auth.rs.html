1//! Certificate-based authentication extractor for mTLS device auth.
2//!
3//! Two identity sources are supported, in order of trust:
4//!
5//! 1. **Native mTLS (D3):** when the server terminates TLS in-process with
6//!    client-cert auth enabled (`AXIAM__SERVER__TLS__CLIENT_AUTH`), rustls
7//!    cryptographically verifies the client certificate against the configured
8//!    CA bundle *during the handshake*. `axiam-server`'s `on_connect` hook then
9//!    stores the verified leaf certificate as a [`VerifiedClientCert`] in the
10//!    connection extensions. This extractor consumes that verified certificate —
11//!    it cannot be forged by a request header.
12//! 2. **Legacy proxy path:** the `X-Client-Certificate` header (URL-encoded PEM
13//!    forwarded by a TLS-terminating reverse proxy). Used only when no verified
14//!    certificate is present on the connection (i.e. TLS terminated upstream).
15//!
16//! Both paths validate the certificate via [`DeviceAuthService`].
17

31/// A client certificate that rustls **verified** against the configured client
32/// CA bundle during the TLS 1.3 handshake (D3).
33///
34/// `axiam-server`'s `HttpServer::on_connect` hook builds this from the rustls
35/// connection's `peer_certificates()` and inserts it into the per-connection
36/// [`actix_web::dev::Extensions`]; handlers read it back with
37/// `HttpRequest::conn_data::<VerifiedClientCert>()`. Because it originates from
38/// the verified peer chain — not a header — it is a trusted identity assertion.
39#[derive(Debug, Clone)]
40pub struct VerifiedClientCert {
41    /// DER encoding of the verified leaf certificate.
42    pub der: Vec<u8>,
43    /// Subject Alternative Names (DNS/URI/RFC822/IP) parsed from the leaf, in
44    /// certificate order. Empty if the leaf carries no SAN extension.
45    pub sans: Vec<String>,
46    /// Lowercase hex SHA-256 of the leaf's SubjectPublicKeyInfo (the SPKI
47    /// fingerprint) — a stable key-identity handle for cert-mapped identities.
48    pub spki_sha256: String,
49}
50
51impl VerifiedClientCert {
52    /// Parse SAN entries and the SPKI fingerprint from a DER-encoded leaf
53    /// certificate. Returns an error string only if the DER cannot be parsed as
54    /// an X.509 certificate (rustls has already verified the chain by this
55    /// point, so this parse is expected to succeed).
56    pub fn from_der(der: &[u8]) -> Result<Self, String> {
57        let (_, cert) =
58            parse_x509_certificate(der).map_err(|e| format!("parse client cert DER: {e}"))?;
59
60        let mut sans = Vec::new();
61        if let Ok(Some(ext)) = cert.subject_alternative_name() {
62            for name in &ext.value.general_names {
63                match name {
64                    GeneralName::DNSName(s) => sans.push(format!("DNS:{s}")),
65                    GeneralName::RFC822Name(s) => sans.push(format!("email:{s}")),
66                    GeneralName::URI(s) => sans.push(format!("URI:{s}")),
67                    GeneralName::IPAddress(b) => sans.push(format!("IP:{}", fmt_ip(b))),
68                    _ => {}
69                }
70            }
71        }
72
73        // The `.raw` field of SubjectPublicKeyInfo is the DER of the full SPKI
74        // structure (RFC 5280 §4.1) — the standard input for an SPKI fingerprint.
75        let spki_sha256 = hex::encode(Sha256::digest(cert.public_key().raw));
76
77        Ok(Self {
78            der: der.to_vec(),
79            sans,
80            spki_sha256,
81        })
82    }
83}

85/// Best-effort textual rendering of a SAN IP address (4-byte v4 / 16-byte v6).
86fn fmt_ip(bytes: &[u8]) -> String {
87    match bytes.len() {
88        4 => bytes
89            .iter()
90            .map(|b| b.to_string())
91            .collect::<Vec<_>>()
92            .join("."),
93        16 => bytes
94            .chunks(2)
95            .map(|c| format!("{:02x}{:02x}", c[0], c[1]))
96            .collect::<Vec<_>>()
97            .join(":"),
98        _ => hex::encode(bytes),
99    }
100}

102/// Authenticated device context extracted from a client certificate.
103///
104/// Use this as a handler parameter to require certificate-based auth.
105#[derive(Debug, Clone)]
106pub struct CertificateAuthenticated {
107    pub service_account_id: Uuid,
108    pub tenant_id: Uuid,
109    pub certificate_id: Uuid,
110}
111
112impl CertificateAuthenticated {
113    /// Extract and validate the client certificate from the request.
114    ///
115    /// This is called manually from the handler rather than via
116    /// `FromRequest`, because the concrete `DeviceAuthService` type
117    /// depends on the DB connection generic `C`.
118    pub async fn extract<C: Connection + Clone>(req: &HttpRequest) -> Result<Self, AxiamApiError> {
119        let state = req
120            .app_data::<web::Data<AppState<C>>>()
121            .ok_or(AxiamError::Internal("missing AppState".into()))?;
122        let service = &state.device_auth_service;
123
124        // Prefer the VERIFIED client certificate captured at TLS handshake time
125        // (D3 native mTLS): rustls has already checked it against the client-CA
126        // bundle, so it is authoritative and cannot be spoofed by a header. Only
127        // fall back to the `X-Client-Certificate` proxy header when TLS was
128        // terminated upstream (no verified cert on this connection).
129        let identity_result = if let Some(verified) = req.conn_data::<VerifiedClientCert>() {
130            service.authenticate_der(&verified.der).await
131        } else {
132            let header = req
133                .headers()
134                .get("X-Client-Certificate")
135                .and_then(|v| v.to_str().ok())
136                .ok_or(AxiamError::AuthenticationFailed {
137                    reason: "missing client certificate (no native mTLS peer cert and no \
138                             X-Client-Certificate header)"
139                        .into(),
140                })?;
141
142            // URL-decode the PEM (reverse proxy URL-encodes it)
143            let pem = urldecode(header).map_err(|_| AxiamError::AuthenticationFailed {
144                reason: "invalid URL encoding in X-Client-Certificate".into(),
145            })?;
146
147            service.authenticate(&pem).await
148        };
149
150        let identity: DeviceIdentity = identity_result.map_err(|e| match &e {
151            // Map certificate/NotFound errors to proper 401/403 status codes
152            AxiamError::Certificate(msg) if msg.contains("not bound to a service account") => {
153                AxiamError::AuthorizationDenied {
154                    reason: msg.clone(),
155                    action: None,
156                    resource_id: None,
157                }
158            }
159            AxiamError::Certificate(msg) => AxiamError::AuthenticationFailed {
160                reason: msg.clone(),
161            },
162            AxiamError::NotFound { .. } => AxiamError::AuthenticationFailed {
163                reason: "unknown client certificate".into(),
164            },
165            _ => e,
166        })?;
167
168        Ok(CertificateAuthenticated {
169            service_account_id: identity.service_account_id,
170            tenant_id: identity.tenant_id,
171            certificate_id: identity.certificate_id,
172        })
173    }
174}

176/// Simple percent-decoding for the `X-Client-Certificate` header.
177fn urldecode(input: &str) -> Result<String, ()> {
178    let mut result = Vec::with_capacity(input.len());
179    let mut bytes = input.bytes();
180    while let Some(b) = bytes.next() {
181        if b == b'%' {
182            let hi = bytes.next().ok_or(())?;
183            let lo = bytes.next().ok_or(())?;
184            let hi = hex_val(hi)?;
185            let lo = hex_val(lo)?;
186            result.push((hi << 4) | lo);
187        } else {
188            result.push(b);
189        }
190    }
191    String::from_utf8(result).map_err(|_| ())
192}
193
194fn hex_val(b: u8) -> Result<u8, ()> {
195    match b {
196        b'0'..=b'9' => Ok(b - b'0'),
197        b'a'..=b'f' => Ok(b - b'a' + 10),
198        b'A'..=b'F' => Ok(b - b'A' + 10),
199        _ => Err(()),
200    }
201}
