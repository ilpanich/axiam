1//! Rate-limit key extractor using X-Forwarded-For header (per D-02, SEC-048).
2//!
3//! SEC-048: The extractor selects the *rightmost untrusted* hop from the
4//! X-Forwarded-For header to prevent IP-spoofing based rate-limit evasion.
5//! The `trusted_hops` value should equal the number of trusted reverse-proxy
6//! hops (e.g. 1 for a single nginx/ingress in front of the server).
7//!
8//! Example:
9//!   X-Forwarded-For: <attacker-ip>, <real-client-ip>, <trusted-proxy-ip>
10//!   trusted_hops = 1 → selects <real-client-ip> (skip 1 from right)
11//!
12//! Falls back to the direct peer address when the header is absent, unparseable,
13//! or has fewer hops than `trusted_hops` (a client cannot manufacture extra
14//! trusted hops to force a fallback to an attacker-controlled entry: when there
15//! are not enough hops to trust, the XFF header is ignored entirely and the key
16//! is derived from `peer_addr()`).
17//!
18//! **nginx/ingress requirement**: The upstream proxy MUST append the real client
19//! IP to the RIGHT of X-Forwarded-For (not inject it at position 0) for this to
20//! be effective. With `proxy_add_x_forwarded_for` in nginx this is the default
21//! behaviour — the real client is the RIGHTMOST trusted entry, not the leftmost.
22
23use actix_governor::governor::NotUntil;
24use actix_governor::governor::clock::{Clock, DefaultClock, QuantaInstant};
25use actix_governor::{KeyExtractor, SimpleKeyExtractionError};
26use actix_web::HttpResponse;
27use actix_web::dev::ServiceRequest;
28use actix_web::http::header::{ContentType, RETRY_AFTER};
29use std::net::IpAddr;
30
31/// Extracts client IP from X-Forwarded-For header, falls back to peer address.
32///
33/// `trusted_hops` controls how many rightmost entries in the XFF header to skip
34/// (they come from trusted proxies). The selected entry is the **rightmost
35/// untrusted** hop — `idx = len - 1 - trusted_hops` — which is correct for a
36/// single trusted reverse proxy that right-appends the real client IP (nginx
37/// `proxy_add_x_forwarded_for`). `trusted_hops = 1` selects the client IP a
38/// single proxy appended; `trusted_hops = 0` selects the rightmost entry
39/// verbatim. (SEC-070: earlier doc text wrongly described this as using the
40/// *leftmost* entry — the leftmost is the most attacker-controllable position
41/// and is never used.)
42#[derive(Debug, Clone, Default)]
43pub struct XForwardedForKeyExtractor {
44    /// Number of trusted reverse-proxy hops to skip from the right of
45    /// the X-Forwarded-For list. Set to the number of load-balancers/
46    /// ingress proxies between the client and this server.
47    ///
48    /// Default: 0 (rightmost entry). NOTE: when the server is exposed directly
49    /// (no proxy), a client can still set XFF to mint fresh rate-limit buckets;
50    /// deploy behind a proxy that overwrites/right-appends XFF and set
51    /// `trusted_hops` to the proxy count so the untrusted client value is
52    /// skipped.
53    pub trusted_hops: usize,
54}
55
56impl XForwardedForKeyExtractor {
57    /// Create an extractor that uses the rightmost-untrusted hop.
58    /// `trusted_hops` = number of trusted proxy entries from the right.
59    pub fn with_trusted_hops(trusted_hops: usize) -> Self {
60        Self { trusted_hops }
61    }
62}
63
64impl KeyExtractor for XForwardedForKeyExtractor {
65    type Key = IpAddr;
66    type KeyExtractionError = SimpleKeyExtractionError<&'static str>;
67
68    fn extract(&self, req: &ServiceRequest) -> Result<Self::Key, Self::KeyExtractionError> {
69        if let Some(forwarded_for) = req.headers().get("X-Forwarded-For")
70            && let Ok(val) = forwarded_for.to_str()
71        {
72            let hops: Vec<&str> = val.split(',').map(str::trim).collect();
73            // Select the rightmost-untrusted hop.
74            // hops = [client, proxy1, ..., trusted_proxy]
75            // trusted_hops=1 → index = len - 1 - 1 = len - 2
76            if self.trusted_hops < hops.len() {
77                let idx = hops.len() - 1 - self.trusted_hops;
78                if let Ok(ip) = hops[idx].parse::<IpAddr>() {
79                    return Ok(ip);
80                }
81            }
82            // Fewer hops than trusted_hops requires: the header cannot be
83            // trusted at all (an attacker could otherwise rotate it to get a
84            // fresh bucket per request via the old `hops[0]` fallback — SECHRD-03).
85            // Fall through to peer_addr() below instead of indexing into XFF.
86        }
87        req.peer_addr()
88            .map(|addr| addr.ip())
89            .ok_or_else(|| SimpleKeyExtractionError::new("no peer address"))
90    }
91
92    /// Returns a JSON 429 response with Retry-After header per D-03.
93    ///
94    /// Body: `{"error":"rate_limit_exceeded","retry_after":<seconds>}`
95    fn exceed_rate_limit_response(
96        &self,
97        negative: &NotUntil<QuantaInstant>,
98        mut response: actix_web::HttpResponseBuilder,
99    ) -> HttpResponse {
100        let wait_secs = negative
101            .wait_time_from(DefaultClock::default().now())
102            .as_secs();
103        response
104            .content_type(ContentType::json())
105            .insert_header((RETRY_AFTER, wait_secs.to_string()))
106            .body(format!(
107                r#"{{"error":"rate_limit_exceeded","retry_after":{}}}"#,
108                wait_secs
109            ))
110    }
111}
