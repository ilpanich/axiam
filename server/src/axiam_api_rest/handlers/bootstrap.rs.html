1//! Admin bootstrap endpoint — first-run setup.
2//!
3//! On a brand-new deployment the database is empty: there are no organizations,
4//! tenants or users. This endpoint performs the entire first-run provisioning
5//! in one call — it creates the **organization**, the **default tenant**, seeds
6//! the permission set and default roles, and creates the initial **admin user**
7//! bound to the super-admin role. Org/tenant creation is get-or-create by slug
8//! so a retry after a transient failure reuses them.
9//!
10//! It is a one-shot operation. First-super-admin creation is atomic (SECHRD-04
11//! / SEC-049 / D-03c): a uniqueness-invariant `bootstrap_lock:global` CREATE
12//! inside the same transaction that creates the admin user means two concurrent
13//! first-run requests can create AT MOST ONE super-admin — the loser gets
14//! `AxiamError::AlreadyExists` (409) and its whole transaction rolls back.
15//! After the first admin is created, every subsequent call also hits the same
16//! `bootstrap_lock:global` uniqueness violation and is refused with 409.
17//! Additional organizations/tenants are created afterwards through the
18//! authenticated admin API.
19//!
20//! The endpoint is also gated (D-03a): a request is refused (fail-closed,
21//! no admin created) unless EITHER `AXIAM_BOOTSTRAP_ADMIN_EMAIL` is set and
22//! matches the request email, OR the request carries a valid one-time
23//! setup token (server-minted at first boot, logged once, consumed once —
24//! D-03b). An unset/absent gate never allows bootstrap.
25

49#[derive(Debug, Deserialize, utoipa::ToSchema)]
50pub struct BootstrapRequest {
51    /// Display name of the organization to create (required). On a fresh
52    /// deployment nothing exists yet, so bootstrap provisions the org and its
53    /// default tenant itself rather than requiring pre-existing IDs.
54    pub organization_name: String,
55    /// URL-safe slug for the organization. Derived from `organization_name`
56    /// when omitted or blank.
57    #[serde(default)]
58    pub organization_slug: Option<String>,
59    /// Display name of the default tenant. Defaults to `"Default"`.
60    #[serde(default)]
61    pub tenant_name: Option<String>,
62    /// URL-safe slug for the default tenant. Defaults to `"default"`.
63    #[serde(default)]
64    pub tenant_slug: Option<String>,
65    /// Admin email address.
66    pub email: String,
67    /// Admin username.
68    pub username: String,
69    /// Admin password (hashed with Argon2id before storage).
70    pub password: String,
71    /// One-time first-run setup token (D-03a/D-03b). Required when
72    /// `AXIAM_BOOTSTRAP_ADMIN_EMAIL` is not set; ignored otherwise.
73    #[serde(default)]
74    pub setup_token: Option<String>,
75}

77/// Convert an arbitrary display name into a URL-safe slug: ASCII alphanumerics
78/// are lower-cased, every other run of characters collapses to a single `-`,
79/// and leading/trailing dashes are trimmed.
80fn slugify(input: &str) -> String {
81    let mut slug = String::with_capacity(input.len());
82    for ch in input.trim().chars() {
83        if ch.is_ascii_alphanumeric() {
84            slug.push(ch.to_ascii_lowercase());
85        } else if !slug.ends_with('-') && !slug.is_empty() {
86            slug.push('-');
87        }
88    }
89    while slug.ends_with('-') {
90        slug.pop();
91    }
92    slug
93}

99#[derive(Debug, SurrealValue)]
100struct SetupTokenRow {
101    #[allow(dead_code)]
102    created_at: DateTime<Utc>,
103}
104
105#[derive(Debug, SurrealValue)]
106struct ConsumedTokenRow {
107    #[allow(dead_code)]
108    consumed_at: DateTime<Utc>,
109}
110
111#[derive(Debug, SurrealValue)]
112struct BootstrapLockRow {
113    #[allow(dead_code)]
114    locked_at: DateTime<Utc>,
115}
116
117/// sha256 hex hash of `token`.
118fn hash_setup_token(token: &str) -> String {
119    let mut hasher = Sha256::new();
120    hasher.update(token.as_bytes());
121    hex::encode(hasher.finalize())
122}
123
124/// Returns `Ok(true)` iff `token`'s hash exists in `bootstrap_setup_token`
125/// and has NOT already been consumed. This is a fast-fail pre-check only —
126/// the authoritative single-use guarantee is the `bootstrap_setup_token_consumed`
127/// CREATE inside the same atomic transaction that creates the admin user
128/// (Task 3): a replayed token still loses to a UNIQUE-index violation even
129/// if two requests race past this pre-check simultaneously.
130async fn setup_token_is_valid<C: Connection + Clone>(
131    db: &Surreal<C>,
132    token_hash: &str,
133) -> Result<bool, AxiamApiError> {
134    let minted: Vec<SetupTokenRow> = db
135        .query("SELECT created_at FROM type::record('bootstrap_setup_token', $hash)")
136        .bind(("hash", token_hash.to_string()))
137        .await
138        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?
139        .take(0)
140        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?;
141    if minted.is_empty() {
142        return Ok(false);
143    }
144
145    let consumed: Vec<ConsumedTokenRow> = db
146        .query("SELECT consumed_at FROM type::record('bootstrap_setup_token_consumed', $hash)")
147        .bind(("hash", token_hash.to_string()))
148        .await
149        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?
150        .take(0)
151        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?;
152
153    Ok(consumed.is_empty())
154}

170/// `POST /api/v1/admin/bootstrap` — first-run admin setup.
171///
172/// Creates the initial admin user with the super-admin role and seeds the
173/// default permission set. First-super-admin creation is atomic (SECHRD-04
174/// / D-03c): a uniqueness-invariant `bootstrap_lock` CREATE inside the same
175/// transaction means a second call — concurrent OR sequential — against a
176/// tenant that already has an admin is refused with 409 Conflict; no
177/// partial admin or orphan role RELATE can result. No token is issued —
178/// the user must authenticate via `/api/v1/auth/login` (D-11).
179///
180/// Mandatory gate (D-03a): requires EITHER `AXIAM_BOOTSTRAP_ADMIN_EMAIL` set
181/// and matching the request email (403 on mismatch), OR a valid setup_token
182/// (403 if missing/invalid/already consumed).
183#[utoipa::path(
184    post,
185    path = "/api/v1/admin/bootstrap",
186    tag = "admin",
187    request_body = BootstrapRequest,
188    responses(
189        (status = 201, description = "Organization, tenant and admin user created", body = BootstrapResponse),
190        (status = 400, description = "Invalid request (missing required fields)"),
191        (status = 403, description = "Bootstrap gate not satisfied (email mismatch or invalid/missing setup token)"),
192        (status = 409, description = "Bootstrap already completed"),
193    )
194)]
195pub async fn bootstrap<C: Connection + Clone>(
196    state: web::Data<AppState<C>>,
197    body: web::Json<BootstrapRequest>,
198) -> Result<HttpResponse, AxiamApiError> {
199    let req = body.into_inner();
200
201    // 1. Mandatory fail-closed gate (SECHRD-04 / D-03a): EITHER
202    //    AXIAM_BOOTSTRAP_ADMIN_EMAIL must be set and match the request
203    //    email (D-10, existing behavior — preserved verbatim), OR the
204    //    request must carry a setup token whose hash is minted and not yet
205    //    consumed. Both unset/invalid => refuse. An unset gate never
206    //    allows bootstrap.
207    let mut consumed_token_hash: Option<String> = None;
208    match std::env::var("AXIAM_BOOTSTRAP_ADMIN_EMAIL") {
209        Ok(expected) => {
210            // Env gate IS set — preserve the existing email-match behavior.
211            if req.email != expected {
212                return Err(AxiamApiError(AxiamError::AuthorizationDenied {
213                    reason: "email does not match AXIAM_BOOTSTRAP_ADMIN_EMAIL".into(),
214                    action: None,
215                    resource_id: None,
216                }));
217            }
218        }
219        Err(_) => {
220            // Env gate NOT set — fall back to the one-time setup token.
221            let token = req.setup_token.as_deref().filter(|t| !t.is_empty());
222            let token_hash = match token {
223                Some(t) => hash_setup_token(t),
224                None => {
225                    return Err(AxiamApiError(AxiamError::AuthorizationDenied {
226                        reason: "bootstrap gate not satisfied: set \
227                                 AXIAM_BOOTSTRAP_ADMIN_EMAIL or provide a valid \
228                                 setup_token"
229                            .into(),
230                        action: None,
231                        resource_id: None,
232                    }));
233                }
234            };
235            if !setup_token_is_valid(&state.db, &token_hash).await? {
236                return Err(AxiamApiError(AxiamError::AuthorizationDenied {
237                    reason: "setup token is invalid, unknown, or already consumed".into(),
238                    action: None,
239                    resource_id: None,
240                }));
241            }
242            consumed_token_hash = Some(token_hash);
243        }
244    }
245
246    // 2. Validate required fields and derive slugs.
247    if req.organization_name.trim().is_empty()
248        || req.email.trim().is_empty()
249        || req.username.trim().is_empty()
250        || req.password.is_empty()
251    {
252        return Err(AxiamApiError(AxiamError::Validation {
253            message: "organization_name, email, username and password are required".into(),
254        }));
255    }
256    let org_slug = req
257        .organization_slug
258        .as_deref()
259        .map(slugify)
260        .filter(|s| !s.is_empty())
261        .unwrap_or_else(|| slugify(&req.organization_name));
262    if org_slug.is_empty() {
263        return Err(AxiamApiError(AxiamError::Validation {
264            message: "organization_name must contain at least one alphanumeric character".into(),
265        }));
266    }
267    let tenant_name = req
268        .tenant_name
269        .as_deref()
270        .map(str::trim)
271        .filter(|s| !s.is_empty())
272        .unwrap_or("Default")
273        .to_string();
274    let tenant_slug = req
275        .tenant_slug
276        .as_deref()
277        .map(slugify)
278        .filter(|s| !s.is_empty())
279        .unwrap_or_else(|| {
280            let s = slugify(&tenant_name);
281            if s.is_empty() { "default".into() } else { s }
282        });
283
284    // 3. Fast-fail one-shot gate: if the global bootstrap lock already exists
285    //    the system has been initialized. The authoritative guard is the
286    //    `bootstrap_lock:global` CREATE inside the atomic transaction below;
287    //    this pre-check just avoids creating a duplicate org on a call that is
288    //    going to be refused anyway.
289    let existing_lock: Vec<BootstrapLockRow> = state
290        .db
291        .query("SELECT locked_at FROM type::record('bootstrap_lock', 'global')")
292        .await
293        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?
294        .take(0)
295        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?;
296    if !existing_lock.is_empty() {
297        return Err(AxiamApiError(AxiamError::AlreadyExists {
298            entity: "bootstrap".into(),
299        }));
300    }
301
302    // 4. Provision the organization and default tenant. Get-or-create by slug
303    //    so a retry after a transient mid-flight failure reuses the same
304    //    org/tenant instead of leaking duplicates (the `bootstrap_lock:global`
305    //    invariant below still guarantees exactly one successful bootstrap).
306    let org = match state.org_repo.get_by_slug(&org_slug).await {
307        Ok(o) => o,
308        Err(_) => state
309            .org_repo
310            .create(CreateOrganization {
311                name: req.organization_name.trim().to_string(),
312                slug: org_slug.clone(),
313                metadata: None,
314            })
315            .await
316            .map_err(|e| {
317                AxiamApiError(AxiamError::Internal(format!("create organization: {e}")))
318            })?,
319    };
320    let tenant = match state.tenant_repo.get_by_slug(org.id, &tenant_slug).await {
321        Ok(t) => t,
322        Err(_) => state
323            .tenant_repo
324            .create(CreateTenant {
325                organization_id: org.id,
326                name: tenant_name,
327                slug: tenant_slug.clone(),
328                metadata: None,
329            })
330            .await
331            .map_err(|e| AxiamApiError(AxiamError::Internal(format!("create tenant: {e}"))))?,
332    };
333    let tenant_id = tenant.id;
334
335    // 5. Seed permissions (idempotent).
336    seed_permissions(&state.db, tenant_id, PERMISSION_REGISTRY)
337        .await
338        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?;
339
340    // 6. Seed default roles and get their IDs.
341    let seed_result = seed_default_roles(&state.db, tenant_id, PERMISSION_REGISTRY)
342        .await
343        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?;
344
345    // 5+6. SECHRD-04 / D-03c: create the admin user and assign the
346    // super-admin role atomically, keyed on a uniqueness invariant instead
347    // of the old SELECT-then-branch TOCTOU check.
348    //
349    // `CREATE type::record('bootstrap_lock', $tenant_id)` is the uniqueness
350    // invariant: a concurrent OR sequential second request racing/retrying
351    // on the SAME tenant_id hits a UNIQUE-index violation on this CREATE
352    // (the record ID itself IS the constraint) and its WHOLE transaction
353    // rolls back — no partial admin, no orphan role RELATE. When a setup
354    // token was used, its hash is consumed in the SAME transaction
355    // (`bootstrap_setup_token_consumed`), so a replay of the same token
356    // also loses to the same violation. The admin's initial password hash
357    // is seeded into `password_history` in the same transaction too
358    // (Pitfall 5 — bootstrap bypasses `create_with_consent`).
359    //
360    // Password hashing is Argon2id and must happen before the transaction.
361    //
362    // SurrealDB v3 quirk: BEGIN TRANSACTION occupies result slot 0;
363    // the first statement result is at .take(1). (See MEMORY.md)
364    let user_id = Uuid::new_v4();
365    let user_id_str = user_id.to_string();
366    let role_id_str = seed_result.super_admin_role_id.to_string();
367    let tenant_id_str = tenant_id.to_string();
368    let ph_id_str = Uuid::new_v4().to_string();
369
370    // Apply the server-configured password pepper (REQ-14 AC-1) — the same
371    // pepper the user repository uses at login-time verification. Previously
372    // this hashed with `None`, so a deployment that set AXIAM__AUTH__PEPPER
373    // could never authenticate the bootstrap admin (login verifies WITH the
374    // pepper, but this stored a hash computed WITHOUT it → "invalid credentials").
375    let pepper = state.auth_config.pepper.as_ref().map(|p| p.expose_secret());
376    let password_hash = password::hash_password(&req.password, pepper)
377        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?;
378
379    // The RELATE uses backtick record IDs (required when type::record() is not
380    // supported inside RELATE per SurrealDB v3 quirk).
381    // The `bootstrap_lock:global` CREATE is the global one-shot invariant: a
382    // concurrent OR sequential second bootstrap hits a UNIQUE record-ID
383    // violation here and its whole transaction rolls back — no partial admin,
384    // no duplicate super-admin.
385    let mut txn_stmts = vec![
386        "BEGIN TRANSACTION".to_string(),
387        "CREATE type::record('bootstrap_lock', 'global') SET locked_at = time::now()".to_string(),
388        "CREATE type::record('user', $user_id) SET \
389           tenant_id = $tenant_id, \
390           username = $username, email = $email, \
391           password_hash = $password_hash, \
392           status = 'Active', \
393           mfa_enabled = false, \
394           failed_login_attempts = 0, \
395           last_failed_login_at = NONE, \
396           locked_until = NONE, \
397           email_verified_at = NONE, \
398           metadata = {}"
399            .to_string(),
400        format!(
401            "RELATE user:`{user_id_str}` -> has_role -> role:`{role_id_str}` \
402             SET resource_id = NONE"
403        ),
404        "CREATE type::record('password_history', $ph_id) SET \
405           tenant_id = $tenant_id, user_id = $user_id, password_hash = $password_hash"
406            .to_string(),
407    ];
408    if consumed_token_hash.is_some() {
409        txn_stmts.push(
410            "CREATE type::record('bootstrap_setup_token_consumed', $token_hash) \
411             SET consumed_at = time::now()"
412                .to_string(),
413        );
414    }
415    txn_stmts.push("COMMIT TRANSACTION".to_string());
416    let txn_query = txn_stmts.join("; ");
417
418    let mut query = state
419        .db
420        .query(txn_query)
421        .bind(("tenant_id", tenant_id_str))
422        .bind(("user_id", user_id_str.clone()))
423        .bind(("username", req.username))
424        .bind(("email", req.email))
425        .bind(("password_hash", password_hash))
426        .bind(("ph_id", ph_id_str));
427    if let Some(token_hash) = consumed_token_hash {
428        query = query.bind(("token_hash", token_hash));
429    }
430
431    let result = query
432        .await
433        .map_err(|e| AxiamApiError(AxiamError::Internal(e.to_string())))?;
434
435    result.check().map_err(|e| {
436        let msg = e.to_string();
437        // SurrealDB v3 UNIQUE index violation message contains "already
438        // contains" (e.g. bootstrap_lock's implicit record-ID uniqueness
439        // constraint). Also match "already exists" and "unique" as
440        // fallback patterns (mirrors saml_replay.rs::insert_assertion).
441        if msg.contains("already contains")
442            || msg.contains("already exists")
443            || msg.contains("unique")
444        {
445            AxiamApiError(AxiamError::AlreadyExists {
446                entity: "bootstrap".into(),
447            })
448        } else {
449            AxiamApiError(AxiamError::Internal(format!(
450                "bootstrap transaction: {msg}"
451            )))
452        }
453    })?;
454
455    // 7. Return 201 — no token (user must login via /api/v1/auth/login, per D-11).
456    Ok(HttpResponse::Created().json(BootstrapResponse {
457        message: "Organization, tenant and admin user created. Login via /api/v1/auth/login."
458            .into(),
459        organization_id: org.id,
460        organization_slug: org_slug,
461        tenant_id,
462        tenant_slug,
463        user_id,
464    }))
