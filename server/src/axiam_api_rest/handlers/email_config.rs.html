1//! Admin email-config endpoints for organizations and tenants (FUNC-03 / D-13).
2//!
3//! Six scope-nested singleton handlers (GET/PUT/DELETE for org and tenant
4//! scopes), gated by `email_config:read` (GET) and `email_config:write`
5//! (PUT/DELETE) per D-03, each enforcing an org/tenant ownership check
6//! before touching the repository (T-28-01 IDOR mitigation).
7//!
8//! GET always returns the raw own-scope row (`get_org_config` /
9//! `get_tenant_override`) — never the merged `get_effective_config` (D-14).
10//! Secrets (`SmtpConfig.password` / `ApiProviderConfig.api_key`) never
11//! appear in any response body: both `EmailConfig` and `EmailConfigOverride`
12//! carry `#[serde(skip_serializing)]` secrets end-to-end (D-01, 28-01).
13//!
14//! PUT performs structural validation only (D-15) — no live SMTP/API
15//! connectivity check is ever made at write time.
16

33/// D-02 fail-closed guard: `email_config_repo` is `None` when
34/// `AXIAM__EMAIL_ENCRYPTION_KEY` is unset. Every email-config handler must
35/// go through this instead of unwrapping directly, preserving the
36/// pre-AppState "App data is not configured" 500 behavior with a clearer
37/// error path.
38fn require_email_config_repo<C: Connection + Clone>(
39    state: &AppState<C>,
40) -> Result<&SurrealEmailConfigRepository<C>, AxiamApiError> {
41    state.email_config_repo.as_ref().ok_or_else(|| {
42        AxiamApiError(AxiamError::Internal(
43            "email configuration is disabled (AXIAM__EMAIL_ENCRYPTION_KEY not set)".into(),
44        ))
45    })
46}

48/// Structural-only validation for a tenant email-config override (D-15).
49///
50/// Mirrors [`validate_email_config`]'s checks but only applies them to
51/// fields the caller actually supplied — `None` fields inherit from the
52/// org baseline and are not validated here. No live SMTP/API connectivity
53/// check is ever performed.
54fn validate_email_config_override(input: &EmailConfigOverride) -> AxiamResult<()> {
55    let mut violations = Vec::new();
56
57    if let Some(ref from_email) = input.from_email
58        && (from_email.is_empty() || !from_email.contains('@'))
59    {
60        violations.push("from_email must be a valid email address".to_string());
61    }
62
63    if let Some(ref from_name) = input.from_name
64        && from_name.is_empty()
65    {
66        violations.push("from_name must not be empty".to_string());
67    }
68
69    if let Some(Some(ref reply_to)) = input.reply_to
70        && (reply_to.is_empty() || !reply_to.contains('@'))
71    {
72        violations.push("reply_to must be a valid email address if provided".to_string());
73    }
74
75    if let Some(ref provider) = input.provider
76        && let ProviderConfig::Smtp(smtp) = provider
77    {
78        if smtp.host.is_empty() {
79            violations.push("SMTP host must not be empty".to_string());
80        }
81        if smtp.port == 0 {
82            violations.push("SMTP port must be > 0".to_string());
83        }
84    }
85    // API-provider kinds: an empty api_key is the D-02 "omit — preserve
86    // stored value" sentinel, not a structural violation (mirrors
87    // validate_email_config's org-level handling).
88
89    if violations.is_empty() {
90        Ok(())
91    } else {
92        Err(AxiamError::Validation {
93            message: format!("Invalid email config override: {}", violations.join("; ")),
94        })
95    }
96}

102/// `GET /api/v1/organizations/{org_id}/email-config`
103#[utoipa::path(
104    get,
105    path = "/api/v1/organizations/{org_id}/email-config",
106    tag = "email-config",
107    params(
108        ("org_id" = Uuid, Path, description = "Organization ID"),
109    ),
110    responses(
111        (status = 200, description = "Organization email configuration (secrets omitted)",
112         body = EmailConfig),
113        (status = 404, description = "No email configuration set for this organization"),
114    ),
115    security(("bearer" = []))
116)]
117pub async fn get_org_email_config<C: Connection + Clone>(
118    user: AuthenticatedUser,
119    authz: AuthzData,
120    state: web::Data<AppState<C>>,
121    path: web::Path<Uuid>,
122) -> Result<HttpResponse, AxiamApiError> {
123    RequirePermission::new("email_config:read", Uuid::nil())
124        .check(&user, authz.get_ref().as_ref())
125        .await?;
126    let org_id = path.into_inner();
127
128    if org_id != user.org_id {
129        return Err(AxiamApiError(AxiamError::AuthorizationDenied {
130            reason: "cannot read email configuration for a different organization".into(),
131            action: None,
132            resource_id: None,
133        }));
134    }
135
136    let repo = require_email_config_repo(&state)?;
137    match repo.get_org_config(org_id).await? {
138        Some(config) => Ok(HttpResponse::Ok().json(config)),
139        None => Err(AxiamApiError(AxiamError::NotFound {
140            entity: "email_config".into(),
141            id: org_id.to_string(),
142        })),
143    }
144}

161pub async fn set_org_email_config<C: Connection + Clone>(
162    user: AuthenticatedUser,
163    authz: AuthzData,
164    state: web::Data<AppState<C>>,
165    path: web::Path<Uuid>,
166    body: web::Json<SetOrgEmailConfig>,
167) -> Result<HttpResponse, AxiamApiError> {
168    RequirePermission::new("email_config:write", Uuid::nil())
169        .check(&user, authz.get_ref().as_ref())
170        .await?;
171    let org_id = path.into_inner();
172
173    if org_id != user.org_id {
174        return Err(AxiamApiError(AxiamError::AuthorizationDenied {
175            reason: "cannot modify email configuration for a different organization".into(),
176            action: None,
177            resource_id: None,
178        }));
179    }
180
181    let input = body.into_inner();
182    validate_email_config(&input)?;
183    let repo = require_email_config_repo(&state)?;
184    let config = repo.set_org_config(org_id, input).await?;
185    Ok(HttpResponse::Ok().json(config))
186}

374        let input = EmailConfigOverride {
375            from_email: Some("not-an-email".to_string()),
376            ..Default::default()
377        };
378        let err = validate_email_config_override(&input).unwrap_err();
379        assert!(err.to_string().contains("from_email"));
380    }
