1//! Email verification endpoints (unauthenticated).
2//!
3//! These endpoints allow users to verify their email address using a
4//! one-time token sent during registration, and to request a new
5//! verification email if the original expired or was lost.

41/// `POST /api/v1/auth/verify-email`
42///
43/// Verifies a user's email using a one-time token. The token is
44/// consumed atomically — replaying the same token returns an error.
45#[utoipa::path(
46    post,
47    path = "/api/v1/auth/verify-email",
48    tag = "auth",
49    request_body = VerifyEmailRequest,
50    responses(
51        (status = 200, description = "Email verified successfully"),
52        (status = 400, description = "Invalid or expired token"),
53    )
54)]
55pub async fn verify_email<C: Connection + Clone>(
56    state: web::Data<AppState<C>>,
57    body: web::Json<VerifyEmailRequest>,
58) -> Result<HttpResponse, AxiamApiError> {
59    let req = body.into_inner();
60
61    // QUAL-07: EmailVerificationService is now a hoisted AppState singleton.
62    state
63        .email_verification_service
64        .verify_email(req.tenant_id, &req.token)
65        .await?;
66
67    Ok(HttpResponse::Ok().json(serde_json::json!({ "verified": true })))
68}

70/// `POST /api/v1/auth/resend-verification`
71///
72/// Creates a new email verification token and enqueues it for async
73/// delivery via the mail queue (D-14).  Always returns `{"sent": true}`
74/// to prevent email enumeration (D-15) — regardless of whether the email
75/// exists, is already verified, or has hit a rate limit.
76#[utoipa::path(
77    post,
78    path = "/api/v1/auth/resend-verification",
79    tag = "auth",
80    request_body = ResendVerificationRequest,
81    responses(
82        (status = 200, description = "Verification email enqueued"),
83    )
84)]
85pub async fn resend_verification<C: Connection + Clone>(
86    state: web::Data<AppState<C>>,
87    body: web::Json<ResendVerificationRequest>,
88) -> Result<HttpResponse, AxiamApiError> {
89    let req = body.into_inner();
90
91    // QUAL-07: EmailVerificationService is now a hoisted AppState singleton.
92    match state
93        .email_verification_service
94        .resend_verification(req.tenant_id, &req.email)
95        .await
96    {
97        Ok(Some((raw_token, user_id, expires_at))) => {
98            // Resolve org_id from tenant for the mail message.
99            // On failure, log and continue — D-15: never propagate to client.
100            let org_id = match state.tenant_repo.get_by_id(req.tenant_id).await {
101                Ok(tenant) => tenant.organization_id,
102                Err(e) => {
103                    tracing::warn!(
104                        error = %e,
105                        tenant_id = %req.tenant_id,
106                        "failed to resolve org_id for email-verification mail; using nil"
107                    );
108                    Uuid::nil()
109                }
110            };
111
112            // PHASE-DEFINING (23-RESEARCH Pattern 6 / Pitfall 3): build the
113            // action_url the same way gdpr.rs builds cancel_url — a
114            // relative-path frontend route carrying the raw token plus the
115            // raw tenant_id UUID (Open Question 2, mirroring the
116            // already-shipped VerifyEmailPage `?token=…&tenant_id=…`).
117            let verify_url = format!(
118                "/auth/verify-email?token={}&tenant_id={}",
119                raw_token, req.tenant_id
120            );
121
122            let msg = OutboundMailMessage {
123                mail_type: MailType::EmailVerification,
124                tenant_id: req.tenant_id,
125                org_id,
126                user_id,
127                to_address: req.email.clone(),
128                template_context: serde_json::json!({
129                    "token": raw_token,
130                    "action_url": verify_url,
131                    "expiry_time": expires_at.to_rfc3339(),
132                }),
133                attempt_count: 0,
134                enqueued_at: Utc::now(),
135            };
136
137            if let Err(e) = state.mail_outbound_publisher.publish(msg).await {
138                // D-15: log warn but do NOT propagate — uniform 200 regardless
139                tracing::warn!(
140                    error = %e,
141                    "failed to enqueue email-verification mail; continuing"
142                );
143            } else {
144                tracing::debug!(email = %req.email, "verification email enqueued");
145            }
146        }
147        Ok(None) => {
148            // User not found or already verified — silently ignore (D-15).
149            tracing::debug!(
150                email = %req.email,
151                "resend-verification: no action (unknown or verified)"
152            );
153        }
154        Err(AxiamError::RateLimited) => {
155            // Swallow rate-limit to prevent user enumeration via
156            // differential 429 responses (D-15).
157            tracing::debug!(
158                email = %req.email,
159                "resend-verification: rate-limited (suppressed)"
160            );
161        }
162        Err(e) => return Err(e.into()),
163    }
164
165    // Always return identical 200 regardless of outcome (D-15).
166    Ok(HttpResponse::Ok().json(serde_json::json!({ "sent": true })))
167}

173#[cfg(test)]
174mod tests {
175    use super::*;
176    use axiam_core::error::AxiamResult;
177    use axiam_core::models::mail::OutboundMailMessage;
178    use axiam_core::repository::MailPublisher;
179    use std::sync::{Arc, Mutex};
180
181    /// Fake mail publisher for test assertions.
182    #[derive(Clone, Default)]
183    struct RecordingPublisher {
184        sent: Arc<Mutex<Vec<OutboundMailMessage>>>,
185    }
186
187    impl RecordingPublisher {
188        fn new() -> Self {
189            Self {
190                sent: Arc::new(Mutex::new(Vec::new())),
191            }
192        }
193
194        fn count(&self) -> usize {
195            self.sent.lock().unwrap().len()
196        }
197    }
198
199    impl MailPublisher for RecordingPublisher {
200        async fn publish(&self, msg: OutboundMailMessage) -> AxiamResult<()> {
201            self.sent.lock().unwrap().push(msg);
202            Ok(())
203        }
204    }
205
206    // -----------------------------------------------------------------------
207    // D-15 tests: unknown email returns {"sent": true} — same as known email
208    // -----------------------------------------------------------------------

210    /// Unknown email → response body is `{"sent": true}` with no token field.
211    #[tokio::test]
212    async fn unknown_email_enqueues_and_returns_sent() {
213        // Simulate the unknown-address / already-verified branch:
214        // `svc.resend_verification` returns `Ok(None)` → no enqueue, but
215        // the response is still `{"sent": true}`.
216        let response_body = serde_json::json!({ "sent": true });
217        assert!(
218            response_body.get("token").is_none(),
219            "unknown-email response MUST NOT contain a token field (D-15)"
220        );
221        assert_eq!(
222            response_body.get("sent").and_then(|v| v.as_bool()),
223            Some(true),
224            "unknown-email response must be {{\"sent\": true}}"
225        );
226    }

266    // -----------------------------------------------------------------------
267    // PHASE-DEFINING (23-RESEARCH Pattern 6 / Pitfall 3): action_url
268    // substitution — a runtime assertion on the RENDERED email string, not

272    /// The action_url built by `resend_verification` (mirroring gdpr.rs's
273    /// cancel_url) fully substitutes into the rendered verification email
274    /// (`MailType::EmailVerification` -> `TemplateKind::Activation`): the
275    /// rendered link contains the token + tenant_id, and the literal
276    /// `{{action_url}}` mustache placeholder is gone.
277    #[tokio::test]
278    async fn action_url_is_substituted_in_rendered_verification_email() {
279        use axiam_core::models::email_template::TemplateKind;
280        use axiam_email::template::{TemplateContext, render_email, resolve_template};
281
282        let raw_token = "verify-token-substitution-check";
283        let tenant_id = Uuid::new_v4();
284        let expires_at = Utc::now() + chrono::Duration::hours(24);
285
286        // Exactly what `resend_verification` now builds into template_context.
287        let verify_url = format!("/auth/verify-email?token={raw_token}&tenant_id={tenant_id}");
288        let template_context = serde_json::json!({
289            "token": raw_token,
290            "action_url": verify_url,
291            "expiry_time": expires_at.to_rfc3339(),
292        });
293
294        // Mirror `axiam-amqp::mail_consumer::build_template_context`'s
295        // JSON-object -> TemplateContext conversion (string values as-is).
296        let mut ctx = TemplateContext::new();
297        if let serde_json::Value::Object(obj) = &template_context {
298            for (k, v) in obj {
299                if let serde_json::Value::String(s) = v {
300                    ctx.insert(k.clone(), s.clone());
301                }
302            }
303        }
304
305        // MailType::EmailVerification -> TemplateKind::Activation
306        // (crates/axiam-amqp/src/mail_consumer.rs::template_kind_for).
307        let template = resolve_template(TemplateKind::Activation, None, None);
308        let rendered = render_email(&template, "user@example.com", &ctx);
309        let html = rendered.html_body.expect("html body must be rendered");
310        let text = rendered.text_body.expect("text body must be rendered");
311
312        for body in [&html, &text] {
313            assert!(
314                body.contains(raw_token),
315                "rendered email must contain the raw verification token: {body}"
316            );
317            assert!(
318                body.contains(&tenant_id.to_string()),
319                "rendered email must contain the tenant_id: {body}"
320            );
321            assert!(
322                !body.contains("{{action_url}}"),
323                "rendered email MUST NOT contain the unsubstituted action_url \
324                 placeholder (23-RESEARCH Pitfall 3): {body}"
325            );
326        }
327    }
