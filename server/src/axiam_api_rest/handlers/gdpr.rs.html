85/// SHA-256 hex of `raw_token`.
86pub fn sha256_hex(raw: &str) -> String {
87    let mut h = Sha256::new();
88    h.update(raw.as_bytes());
89    hex::encode(h.finalize())
90}

92/// Generate a cryptographically-random 256-bit (32-byte) cancel token,
93/// hex-encoded as a 64-character string (CQ-B39).
94///
95/// Replaces the previous `Uuid::new_v4().to_string()` (128-bit) token.
96fn generate_cancel_token() -> String {
97    use rand::RngExt;
98    let bytes: [u8; 32] = rand::rng().random();
99    hex::encode(bytes)
100}

102/// Append a GDPR audit log entry, logging any failure without propagating it.
103///
104/// Factored out of the individual GDPR handlers to eliminate the repeated
105/// audit-append block pattern (CQ-B39).  The fire-and-forget `let _ = …`
106/// pattern is intentional: an audit failure must not block the user response,
107/// but failures are logged at `error!` level since GDPR audit trails are legally
108/// significant (CQ-B31 / T-12-01).
109async fn append_gdpr_audit<C: Connection + Clone>(
110    audit_repo: &SurrealAuditLogRepository<C>,
111    tenant_id: Uuid,
112    actor_id: Uuid,
113    action: &str,
114    resource_id: Option<Uuid>,
115    metadata: Option<serde_json::Value>,
116) {
117    if let Err(e) = audit_repo
118        .append(axiam_core::models::audit::CreateAuditLogEntry {
119            tenant_id,
120            actor_id,
121            actor_type: ActorType::User,
122            action: action.into(),
123            resource_id,
124            outcome: AuditOutcome::Success,
125            ip_address: None,
126            metadata,
127        })
128        .await
129    {
130        tracing::error!(
131            error = %e,
132            %tenant_id,
133            "gdpr: failed to write audit log for GDPR request (legally significant)"
134        );
135    }
136}

142/// Environment variable naming the append-only dead-letter file for the GDPR
143/// erasure audit event. Intended to point at a mounted volume so the file
144/// survives container restarts. Absent = the file sink is skipped for this
145/// event (the structured tracing event sink below still fires
146/// unconditionally on failure).
147pub const GDPR_AUDIT_DLQ_FILE_ENV: &str = "AXIAM__GDPR_AUDIT_DLQ_FILE";

149/// Injectable seam for the erasure audit DB-write (SECHRD-12).
150///
151/// `SurrealAuditLogRepository<C>` is the production implementation (forwards
152/// to [`AuditLogRepository::append`]); tests inject a failing double to drive
153/// the dead-letter path in [`write_erasure_audit_with_dlq`] without a live
154/// broken database. Defined here (not in `axiam-core::repository`) so it
155/// stays out of the generic repository trait surface owned by another plan.
156pub trait AuditWriteSink: Send + Sync {
157    fn write(
158        &self,
159        entry: CreateAuditLogEntry,
160    ) -> impl Future<Output = AxiamResult<AuditLogEntry>> + Send;
161}
162
163impl<C: Connection> AuditWriteSink for SurrealAuditLogRepository<C> {
164    fn write(
165        &self,
166        entry: CreateAuditLogEntry,
167    ) -> impl Future<Output = AxiamResult<AuditLogEntry>> + Send {
168        self.append(entry)
169    }
170}

172/// Write the erasure audit record via `sink`. If the DB write fails, the
173/// record is dead-lettered to BOTH an append-only local file AND a
174/// structured `tracing` audit event (D-02), so a legally-significant
175/// erasure event is never silently lost to a transient DB failure (T-24-61).
176///
177/// Never propagates the DB error — the caller (the cleanup ticker, T-04-36)
178/// must not panic or abort the sweep on this failure path; the two
179/// dead-letter sinks ARE the durability guarantee for this branch.
180pub async fn write_erasure_audit_with_dlq<S: AuditWriteSink>(sink: &S, entry: CreateAuditLogEntry) {
181    let dlq_entry = entry.clone();
182    if let Err(e) = sink.write(entry).await {
183        tracing::error!(
184            error = %e,
185            tenant_id = %dlq_entry.tenant_id,
186            action = %dlq_entry.action,
187            "gdpr: erasure audit DB-write failed — dead-lettering to append-only file + \
188             structured event (legally significant, SECHRD-12)"
189        );
190        dead_letter_erasure_audit(&dlq_entry, &e);
191    }
192}
193
194/// Dead-letter a failed erasure audit record to the two durable sinks (D-02).
195fn dead_letter_erasure_audit(entry: &CreateAuditLogEntry, db_error: &AxiamError) {
196    // Sink 1: append-only local file on a mounted volume (T-24-62). Opened
197    // with `.append(true)` — an existing file is never truncated/rewritten,
198    // matching AXIAM's append-only audit posture.
199    match std::env::var(GDPR_AUDIT_DLQ_FILE_ENV) {
200        Ok(path) => match serde_json::to_string(entry) {
201            Ok(line) => match OpenOptions::new().create(true).append(true).open(&path) {
202                Ok(mut file) => {
203                    if let Err(e) = writeln!(file, "{line}") {
204                        tracing::error!(
205                            error = %e,
206                            path = %path,
207                            "gdpr: failed to append erasure audit dead-letter file"
208                        );
209                    }
210                }
211                Err(e) => {
212                    tracing::error!(
213                        error = %e,
214                        path = %path,
215                        "gdpr: failed to open erasure audit dead-letter file"
216                    );
217                }
218            },
219            Err(e) => {
220                tracing::error!(
221                    error = %e,
222                    "gdpr: failed to serialize erasure audit dead-letter record"
223                );
224            }
225        },
226        Err(_) => {
227            tracing::warn!(
228                env_var = GDPR_AUDIT_DLQ_FILE_ENV,
229                "gdpr: erasure audit dead-letter FILE sink skipped (env var not set); \
230                 structured event sink below still fires"
231            );
232        }
233    }
234
235    // Sink 2: structured `tracing` audit event — the "audit syslog" sink
236    // (RESEARCH Assumption A4: satisfied by a structured tracing JSON event
237    // captured by the container log driver, NOT a literal syslog(3) socket;
238    // the distroless deployment has no local syslogd, per 06-01 decisions).
239    tracing::error!(
240        target: "axiam.audit.dlq",
241        tenant_id = %entry.tenant_id,
242        actor_id = %entry.actor_id,
243        action = %entry.action,
244        resource_id = ?entry.resource_id,
245        db_error = %db_error,
246        "gdpr_audit_dlq"
247    );
248}

254/// `POST /api/v1/account/export`
255///
256/// Enqueues an async GDPR Art. 15 data-export job.  Returns immediately with
257/// `{"queued": true}`; the cleanup sweep generates the encrypted JSON, stores
258/// it, and emails a single-use 24 h download link (D-12).
259///
260/// Self-service (own account) or admin with `gdpr:export` permission.
261#[utoipa::path(
262    post,
263    path = "/api/v1/account/export",
264    tag = "gdpr",
265    request_body = inline(serde_json::Value),
266    responses(
267        (status = 200, description = "Export job enqueued"),
268        (status = 403, description = "Forbidden"),
269    ),
270    security(("bearer" = []))
271)]
272#[allow(clippy::too_many_arguments)]
273pub async fn request_account_export<C: Connection + Clone>(
274    auth_user: AuthenticatedUser,
275    authz: AuthzData,
276    state: web::Data<AppState<C>>,
277    body: web::Json<ExportRequest>,
278) -> Result<HttpResponse, AxiamApiError> {
279    let req = body.into_inner();
280    let target_id = req.user_id.unwrap_or(auth_user.user_id);
281
282    // Ownership check: self OR gdpr:export permission.
283    if !is_own_resource(&auth_user, target_id) {
284        RequirePermission::new("gdpr:export", Uuid::nil())
285            .check(&auth_user, authz.get_ref().as_ref())
286            .await?;
287    }
288
289    // CQ-B39: Deduplicate concurrent export requests — reject if a queued
290    // export already exists for this user to avoid duplicate processing.
291    if state
292        .export_job_repo
293        .has_pending_for_user(auth_user.tenant_id, target_id)
294        .await?
295    {
296        return Err(AxiamError::AlreadyExists {
297            entity: "export_job".into(),
298        }
299        .into());
300    }
301
302    // Create a queued export job.
303    state
304        .export_job_repo
305        .create(CreateExportJob {
306            tenant_id: auth_user.tenant_id,
307            user_id: target_id,
308        })
309        .await?;
310
311    // Audit: gdpr.data_export_requested.
312    append_gdpr_audit(
313        &state.audit_repo,
314        auth_user.tenant_id,
315        auth_user.user_id,
316        "gdpr.data_export_requested",
317        Some(target_id),
318        Some(serde_json::json!({ "subject_id": target_id.to_string() })),
319    )
320    .await;
321
322    Ok(HttpResponse::Ok().json(QueuedResponse { queued: true }))
323}

325/// `GET /api/v1/account/export/{token}`
326///
327/// Single-use download of the GDPR export blob (D-13).  Looks up the job by
328/// SHA-256(token), verifies it is ready and not expired/used, decrypts the
329/// blob, marks it as downloaded, and deletes the record.
330///
331/// Requires authentication (self or admin with `gdpr:export`).
332#[utoipa::path(
333    get,
334    path = "/api/v1/account/export/{token}",
335    tag = "gdpr",
336    params(("token" = String, Path, description = "Single-use download token")),
337    responses(
338        (status = 200, description = "Export JSON"),
339        (status = 404, description = "Token not found or expired"),
340    ),
341    security(("bearer" = []))
342)]
343#[allow(clippy::too_many_arguments)]
344pub async fn download_account_export<C: Connection + Clone>(
345    auth_user: AuthenticatedUser,
346    authz: AuthzData,
347    state: web::Data<AppState<C>>,
348    path: web::Path<DownloadPath>,
349) -> Result<HttpResponse, AxiamApiError> {
350    let token = path.into_inner().token;
351    let token_hash = sha256_hex(&token);
352
353    let job = state
354        .export_job_repo
355        .find_by_download_token_hash(auth_user.tenant_id, &token_hash)
356        .await?
357        .ok_or_else(|| AxiamError::NotFound {
358            entity: "export_job".into(),
359            id: "by-token".into(),
360        })?;
361
362    // Ownership check.
363    if !is_own_resource(&auth_user, job.user_id) {
364        RequirePermission::new("gdpr:export", Uuid::nil())
365            .check(&auth_user, authz.get_ref().as_ref())
366            .await?;
367    }
368
369    // Validate status and expiry (D-13: single-use, 24h TTL).
370    use axiam_core::models::gdpr::ExportJobStatus;
371    if job.status != ExportJobStatus::Ready {
372        return Err(AxiamError::AuthorizationDenied {
373            reason: "export token already used or not ready".into(),
374            action: None,
375            resource_id: None,
376        }
377        .into());
378    }
379    if let Some(expires_at) = job.expires_at
380        && expires_at < Utc::now()
381    {
382        return Err(AxiamError::AuthorizationDenied {
383            reason: "export token expired".into(),
384            action: None,
385            resource_id: None,
386        }
387        .into());
388    }
389
390    // Decrypt the blob.
391    let (nonce, ciphertext) = match (&job.blob_nonce, &job.encrypted_blob) {
392        (Some(n), Some(ct)) => (n.clone(), ct.clone()),
393        _ => {
394            return Err(AxiamError::Internal("export blob missing on ready job".into()).into());
395        }
396    };
397    let key = state
398        .email_encryption_key
399        .as_ref()
400        .ok_or_else(|| AxiamError::Internal("export encryption key not configured".into()))?;
401
402    let plaintext_bytes = decrypt_separate(key, &nonce, &ciphertext)
403        .map_err(|e| AxiamError::Internal(format!("export decrypt failed: {e}")))?;
404
405    // Atomic single-use consume: UPDATE WHERE status = 'ready' + DELETE.
406    // If 0 rows were updated the token was already consumed (TOCTTOU-safe,
407    // D-13 / CQ-B38 / REQ-14 AC-5).
408    let consumed = state
409        .export_job_repo
410        .consume_ready_and_delete(job.id)
411        .await?;
412    if !consumed {
413        return Err(AxiamError::AuthorizationDenied {
414            reason: "export token already consumed".into(),
415            action: None,
416            resource_id: None,
417        }
418        .into());
419    }
420
421    let plaintext = String::from_utf8(plaintext_bytes)
422        .map_err(|e| AxiamError::Internal(format!("export UTF-8 decode: {e}")))?;
423
424    Ok(HttpResponse::Ok()
425        .content_type("application/json")
426        .body(plaintext))
427}

429/// `POST /api/v1/account/delete`
430///
431/// Initiates Art. 17 erasure: immediately disables the account, revokes all
432/// sessions, emails a single-use cancel link, and schedules purge at +30 d
433/// (D-07/D-08/D-09).
434///
435/// Self-service (own account) or admin with `users:erase` permission.
436#[utoipa::path(
437    post,
438    path = "/api/v1/account/delete",
439    tag = "gdpr",
440    request_body = inline(serde_json::Value),
441    responses(
442        (status = 200, description = "Deletion scheduled"),
443        (status = 403, description = "Forbidden"),
444    ),
445    security(("bearer" = []))
446)]
447#[allow(clippy::too_many_arguments)]
448pub async fn request_account_delete<C: Connection + Clone>(
449    auth_user: AuthenticatedUser,
450    authz: AuthzData,
451    state: web::Data<AppState<C>>,
452    body: web::Json<DeleteRequest>,
453) -> Result<HttpResponse, AxiamApiError> {
454    let req = body.into_inner();
455    let target_id = req.user_id.unwrap_or(auth_user.user_id);
456
457    // Ownership check: self OR users:erase permission.
458    if !is_own_resource(&auth_user, target_id) {
459        RequirePermission::new("users:erase", Uuid::nil())
460            .check(&auth_user, authz.get_ref().as_ref())
461            .await?;
462    }
463
464    // Load target user to get email for the cancel mail.
465    let target_user = state
466        .user_repo
467        .get_by_id(auth_user.tenant_id, target_id)
468        .await?;
469
470    // Schedule purge at +30 days (D-08).
471    let scheduled_purge_at = Utc::now() + chrono::Duration::days(30);
472
473    // Generate cancel token: 256-bit random token, hex-encoded (CQ-B39).
474    // Only the SHA-256 hash is stored in the DB; the raw token is emailed.
475    let raw_cancel_token = generate_cancel_token();
476    let cancel_token_hash = sha256_hex(&raw_cancel_token);
477
478    // D-14: mark the user deletion-pending AND create the account_deletion
479    // row (holding the cancel_token_hash) in ONE transaction — a create
480    // failure (e.g. a duplicate pending-deletion conflict) must never
481    // strand the user in deletion_pending=true with no account_deletion row
482    // to hold a cancellable token (CQ-B39 residual / uncancellable purge).
483    state
484        .account_deletion_repo
485        .create_with_pending_flag(
486            auth_user.tenant_id,
487            target_id,
488            scheduled_purge_at,
489            cancel_token_hash,
490        )
491        .await?;
492
493    // Revoke all sessions immediately (D-08: account disabled). Kept as a
494    // separate, subsequent call — not part of the D-14 strand-risk
495    // transaction: a failure here leaves an unrevoked session, a lesser and
496    // pre-existing concern, not an uncancellable purge.
497    state
498        .auth_service
499        .revoke_all_sessions(auth_user.tenant_id, target_id)
500        .await?;
501
502    // Resolve org_id for the mail message.
503    let org_id = match state.tenant_repo.get_by_id(auth_user.tenant_id).await {
504        Ok(tenant) => tenant.organization_id,
505        Err(e) => {
506            tracing::warn!(error = %e, "failed to resolve org_id for delete-cancel mail");
507            Uuid::nil()
508        }
509    };
510
511    // Enqueue the cancel-link email (D-09).
512    let cancel_url = format!(
513        "/api/v1/auth/account/delete/cancel?token={}",
514        raw_cancel_token
515    );
516    let msg = OutboundMailMessage {
517        mail_type: MailType::DeletionCancel,
518        tenant_id: auth_user.tenant_id,
519        org_id,
520        user_id: target_id,
521        to_address: target_user.email.clone(),
522        template_context: serde_json::json!({
523            "action_url": cancel_url,
524            "expiry_time": scheduled_purge_at.to_rfc3339(),
525        }),
526        attempt_count: 0,
527        enqueued_at: Utc::now(),
528    };
529    if let Err(e) = state.mail_outbound_publisher.publish(msg).await {
530        tracing::warn!(error = %e, "failed to enqueue delete-cancel email; continuing");
531    }
532
533    // Audit: gdpr.erasure_requested.
534    append_gdpr_audit(
535        &state.audit_repo,
536        auth_user.tenant_id,
537        auth_user.user_id,
538        "gdpr.erasure_requested",
539        Some(target_id),
540        Some(serde_json::json!({
541            "subject_id": target_id.to_string(),
542            "scheduled_purge_at": scheduled_purge_at.to_rfc3339(),
543        })),
544    )
545    .await;
546
547    Ok(HttpResponse::Ok().json(ScheduledResponse { scheduled: true }))
548}

550/// `GET /api/v1/auth/account/delete/cancel?token=<opaque>`
551///
552/// **Public endpoint** (token-authenticated; listed in `PUBLIC_PATHS`).
553///
554/// Validates the single-use cancel token; if the deletion is still pending
555/// and within the grace window, aborts it and re-enables the account (D-09).
556/// A second call with the same token is rejected (single-use).
557#[utoipa::path(
558    get,
559    path = "/api/v1/auth/account/delete/cancel",
560    tag = "gdpr",
561    params(("token" = String, Query, description = "Single-use cancel token")),
562    responses(
563        (status = 200, description = "Deletion cancelled"),
564        (status = 400, description = "Token invalid or expired"),
565    )
566)]
567pub async fn cancel_account_delete<C: Connection + Clone>(
568    state: web::Data<AppState<C>>,
569    query: web::Query<CancelQuery>,
570) -> Result<HttpResponse, AxiamApiError> {
571    let token_hash = sha256_hex(&query.token);
572
573    // Global lookup by token hash (no auth context — public endpoint, D-09).
574    let deletion = state
575        .account_deletion_repo
576        .find_by_token_hash_global(&token_hash)
577        .await?
578        .ok_or_else(|| AxiamError::NotFound {
579            entity: "account_deletion".into(),
580            id: "by-token".into(),
581        })?;
582
583    // Validate: must be pending and within grace window.
584    if deletion.status != AccountDeletionStatus::Pending {
585        return Err(AxiamError::AuthorizationDenied {
586            reason: "cancel token already used".into(),
587            action: None,
588            resource_id: None,
589        }
590        .into());
591    }
592    if deletion.scheduled_purge_at < Utc::now() {
593        return Err(AxiamError::AuthorizationDenied {
594            reason: "grace window expired — cannot cancel".into(),
595            action: None,
596            resource_id: None,
597        }
598        .into());
599    }
600
601    // Abort deletion: mark cancelled (single-use) + re-enable account.
602    state
603        .account_deletion_repo
604        .mark_cancelled(deletion.tenant_id, deletion.id)
605        .await?;
606
607    state
608        .user_repo
609        .clear_deletion_pending(deletion.tenant_id, deletion.user_id)
610        .await?;
611
612    Ok(HttpResponse::Ok().json(CancelledResponse { cancelled: true }))
613}
