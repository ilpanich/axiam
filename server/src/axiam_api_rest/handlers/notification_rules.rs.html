77/// `POST /api/v1/notification-rules`
78#[utoipa::path(
79    post,
80    path = "/api/v1/notification-rules",
81    tag = "notification_rules",
82    request_body = CreateNotificationRuleRequest,
83    responses(
84        (status = 201, description = "Notification rule created",
85         body = NotificationRuleResponse),
86    ),
87    security(("bearer" = []))
88)]
89pub async fn create<C: Connection + Clone>(
90    user: AuthenticatedUser,
91    authz: AuthzData,
92    state: web::Data<AppState<C>>,
93    body: web::Json<CreateNotificationRuleRequest>,
94) -> Result<HttpResponse, AxiamApiError> {
95    RequirePermission::new("notification_rules:create", Uuid::nil())
96        .check(&user, authz.get_ref().as_ref())
97        .await?;
98    let req = body.into_inner();
99
100    validate_name(&req.name)?;
101    validate_events(&req.events)?;
102    validate_recipient_emails(&req.recipient_emails)?;
103
104    let rule = state
105        .notification_rule_repo
106        .create(CreateNotificationRule {
107            tenant_id: user.tenant_id,
108            name: req.name,
109            description: req.description,
110            events: req.events,
111            recipient_emails: req.recipient_emails,
112        })
113        .await?;
114    Ok(HttpResponse::Created().json(NotificationRuleResponse::from(rule)))
115}

184/// `PUT /api/v1/notification-rules/{id}`
185#[utoipa::path(
186    put,
187    path = "/api/v1/notification-rules/{id}",
188    tag = "notification_rules",
189    params(("id" = Uuid, Path, description = "Rule ID")),
190    request_body = UpdateNotificationRuleRequest,
191    responses(
192        (status = 200, description = "Notification rule updated",
193         body = NotificationRuleResponse),
194    ),
195    security(("bearer" = []))
196)]
197pub async fn update<C: Connection + Clone>(
198    user: AuthenticatedUser,
199    authz: AuthzData,
200    path: web::Path<Uuid>,
201    state: web::Data<AppState<C>>,
202    body: web::Json<UpdateNotificationRuleRequest>,
203) -> Result<HttpResponse, AxiamApiError> {
204    RequirePermission::new("notification_rules:update", Uuid::nil())
205        .check(&user, authz.get_ref().as_ref())
206        .await?;
207    let id = path.into_inner();
208    let req = body.into_inner();
209
210    if let Some(ref name) = req.name {
211        validate_name(name)?;
212    }
213    if let Some(ref events) = req.events {
214        validate_events(events)?;
215    }
216    if let Some(ref emails) = req.recipient_emails {
217        validate_recipient_emails(emails)?;
218    }
219
220    let rule = state
221        .notification_rule_repo
222        .update(
223            user.tenant_id,
224            id,
225            UpdateNotificationRule {
226                name: req.name,
227                description: req.description,
228                events: req.events,
229                recipient_emails: req.recipient_emails,
230                enabled: req.enabled,
231            },
232        )
233        .await?;
234    Ok(HttpResponse::Ok().json(NotificationRuleResponse::from(rule)))
235}

269fn validation_err(msg: impl Into<String>) -> AxiamApiError {
270    axiam_core::error::AxiamError::Validation {
271        message: msg.into(),
272    }
273    .into()
274}
275
276fn validate_name(name: &str) -> Result<(), AxiamApiError> {
277    if name.trim().is_empty() {
278        return Err(validation_err("name must not be empty"));
279    }
280    Ok(())
281}
282
283fn validate_events(events: &[NotificationEventType]) -> Result<(), AxiamApiError> {
284    if events.is_empty() {
285        return Err(validation_err("events must not be empty"));
286    }
287    Ok(())
288}
289
290fn validate_recipient_emails(emails: &[String]) -> Result<(), AxiamApiError> {
291    if emails.is_empty() {
292        return Err(validation_err("recipient_emails must not be empty"));
293    }
294    if emails.len() > 20 {
295        return Err(validation_err(
296            "recipient_emails must have at most 20 entries",
297        ));
298    }
299    for email in emails {
300        if !email.contains('@') {
301            return Err(validation_err(format!("invalid email address: {email}")));
302        }
303    }
304    Ok(())
305}
