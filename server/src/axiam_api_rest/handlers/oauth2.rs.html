56/// `GET /oauth2/authorize` -- OAuth2 authorization endpoint.
57///
58/// The user must be authenticated (redirected to login first if not).
59/// On success, redirects to `redirect_uri?code=...&state=...`.
60/// On error, redirects with `?error=...&error_description=...&state=...`.
61#[utoipa::path(
62    get,
63    path = "/oauth2/authorize",
64    tag = "oauth2",
65    params(AuthorizeQuery),
66    responses(
67        (status = 302, description = "Redirect with authorization code"),
68    ),
69    security(("bearer" = []))
70)]
71pub async fn authorize<C: Connection + Clone>(
72    user: AuthenticatedUser,
73    query: web::Query<AuthorizeQuery>,
74    state: web::Data<AppState<C>>,
75) -> HttpResponse {
76    let q = query.into_inner();
77
78    let req = AuthorizeRequest {
79        tenant_id: user.tenant_id,
80        user_id: user.user_id,
81        response_type: q.response_type,
82        client_id: q.client_id,
83        redirect_uri: q.redirect_uri.clone(),
84        scope: q.scope,
85        state: q.state.clone(),
86        code_challenge: q.code_challenge,
87        code_challenge_method: q.code_challenge_method,
88        nonce: q.nonce,
89    };
90
91    match state.authorize_service.authorize(req).await {
92        Ok(resp) => {
93            match url::Url::parse(&resp.redirect_uri) {
94                Ok(mut url) => {
95                    url.query_pairs_mut().append_pair("code", &resp.code);
96                    if let Some(ref state) = resp.state {
97                        url.query_pairs_mut().append_pair("state", state);
98                    }
99                    HttpResponse::Found()
100                        .append_header(("Location", url.to_string()))
101                        .finish()
102                }
103                Err(_) => {
104                    // Never leak the authorization code to an
105                    // unknown host.
106                    HttpResponse::InternalServerError().json(OAuth2ErrorResponse {
107                        error: "server_error".into(),
108                        error_description: "invalid redirect_uri in \
109                                 authorization response"
110                            .into(),
111                    })
112                }
113            }
114        }
115        Err(e) => {
116            // Per RFC 6749: only redirect when the redirect_uri has
117            // been validated. For InvalidClient / redirect_uri
118            // errors, return a direct HTTP error instead.
119            match &e {
120                OAuth2Error::InvalidClient(_) | OAuth2Error::InvalidRedirectUri(_) => {
121                    build_oauth2_error_response(&e)
122                }
123                _ => {
124                    // These errors occur after client+redirect_uri
125                    // were validated — safe to redirect.
126                    build_error_redirect(&q.redirect_uri, &e, q.state.as_deref())
127                }
128            }
129        }
130    }
131}

133/// `POST /oauth2/token` -- OAuth2 token endpoint.
134///
135/// Accepts form-encoded body per RFC 6749. Supports authorization_code,
136/// client_credentials, and refresh_token grant types. The `tenant_id` is
137/// passed as a query parameter since the token endpoint is unauthenticated
138/// (the client is authenticating itself here).
139#[utoipa::path(
140    post,
141    path = "/oauth2/token",
142    tag = "oauth2",
143    params(TenantQuery),
144    request_body(
145        content_type = "application/x-www-form-urlencoded",
146        content = TokenRequest,
147    ),
148    responses(
149        (status = 200, description = "Token response",
150         body = TokenResponse),
151        (status = 400, description = "OAuth2 error",
152         body = OAuth2ErrorResponse),
153    ),
154)]
155pub async fn token<C: Connection + Clone>(
156    tenant_query: web::Query<TenantQuery>,
157    form: web::Form<TokenRequest>,
158    state: web::Data<AppState<C>>,
159) -> HttpResponse {
160    let tenant_id = tenant_query.into_inner().tenant_id;
161
162    match state
163        .token_service
164        .exchange(tenant_id, form.into_inner())
165        .await
166    {
167        Ok(resp) => HttpResponse::Ok()
168            .append_header(("Cache-Control", "no-store"))
169            .append_header(("Pragma", "no-cache"))
170            .json(resp),
171        Err(e) => build_oauth2_error_response(&e),
172    }
173}

255/// `GET /.well-known/openid-configuration` -- OIDC Discovery document.
256///
257/// Returns the OpenID Provider metadata per OpenID Connect Discovery 1.0.
258/// The issuer URL is taken from `AuthConfig::oauth2_issuer_url` when set,
259/// falling back to `AuthConfig::jwt_issuer` otherwise.
260#[utoipa::path(
261    get,
262    path = "/.well-known/openid-configuration",
263    tag = "oidc",
264    responses(
265        (status = 200, description = "OpenID Connect Discovery document",
266         body = OidcDiscoveryDocument),
267    ),
268)]
269pub async fn discovery(auth_config: web::Data<AuthConfig>) -> HttpResponse {
270    let issuer = auth_config.effective_issuer();
271    // Guard: effective_issuer must be a valid URL for a compliant
272    // discovery document.  Startup validation should catch this, but
273    // defend in depth at the endpoint level.
274    if url::Url::parse(issuer).is_err() {
275        return HttpResponse::InternalServerError().json(OAuth2ErrorResponse {
276            error: "server_error".into(),
277            error_description: "OIDC issuer is not configured as a \
278                valid URL"
279                .into(),
280        });
281    }
282    let doc = build_discovery_document(issuer);
283    HttpResponse::Ok().json(doc)
284}

286/// `GET /oauth2/jwks` -- JSON Web Key Set.
287///
288/// Returns the public signing keys used by the authorization server
289/// so that relying parties can verify JWTs without sharing a secret.
290///
291/// B3: served from an in-process cache (`state.oauth2_jwks_cache`) keyed by
292/// a hash of the source PEM, with a `Cache-Control: public, max-age=<n>`
293/// header (configurable, default 300s) and a strong `ETag`. Clients that
294/// send a matching `If-None-Match` get `304 Not Modified` with no body;
295/// clients that ignore caching headers entirely still get the identical
296/// `200 OK` + JWKS body they always did -- this is a pure additive change.
297/// See `axiam_oauth2::jwks_cache` module docs for the cache design and the
298/// documented limitations (no key-rotation mechanism exists yet; the
299/// endpoint serves one global, not per-tenant, key set).
300#[utoipa::path(
301    get,
302    path = "/oauth2/jwks",
303    tag = "oidc",
304    responses(
305        (status = 200, description = "JWKS document", body = JwksDocument),
306        (status = 304, description = "Not Modified -- ETag matches If-None-Match"),
307        (status = 500, description = "Key parsing error"),
308    ),
309)]
310pub async fn jwks<C: Connection + Clone>(
311    req: HttpRequest,
312    state: web::Data<AppState<C>>,
313) -> HttpResponse {
314    let if_none_match = req
315        .headers()
316        .get(actix_web::http::header::IF_NONE_MATCH)
317        .and_then(|v| v.to_str().ok());
318
319    let cache_control = state.oauth2_jwks_cache_config.cache_control_header();
320
321    match state
322        .oauth2_jwks_cache
323        .get(&state.auth_config.jwt_public_key_pem, if_none_match)
324    {
325        Ok(JwksCacheResponse::Fresh { body, etag }) => HttpResponse::Ok()
326            .content_type("application/json")
327            .insert_header(("Cache-Control", cache_control))
328            .insert_header(("ETag", etag))
329            .body(body),
330        Ok(JwksCacheResponse::NotModified { etag }) => HttpResponse::NotModified()
331            .insert_header(("Cache-Control", cache_control))
332            .insert_header(("ETag", etag))
333            .finish(),
334        Err(e) => HttpResponse::InternalServerError().json(OAuth2ErrorResponse {
335            error: "server_error".into(),
336            error_description: e,
337        }),
338    }
339}

341/// `GET /oauth2/userinfo` -- OIDC UserInfo endpoint.
342///
343/// Returns claims about the authenticated user. Requires a valid
344/// Bearer access token. Email and username are included based on
345/// the scopes present in the access token.
346#[utoipa::path(
347    get,
348    path = "/oauth2/userinfo",
349    tag = "oidc",
350    responses(
351        (status = 200, description = "UserInfo response",
352         body = UserInfoResponse),
353        (status = 401, description = "Invalid or missing access token"),
354    ),
355    security(("bearer" = []))
356)]
357pub async fn userinfo<C: Connection + Clone>(
358    user: AuthenticatedUser,
359    state: web::Data<AppState<C>>,
360) -> HttpResponse {
361    let scopes: Vec<String> = user
362        .claims
363        .0
364        .scope
365        .as_deref()
366        .unwrap_or("")
367        .split_whitespace()
368        .map(String::from)
369        .collect();
370
371    let has_scope = |s: &str| scopes.iter().any(|sc| sc == s);
372
373    // Fetch user details for email/username when the relevant
374    // scopes are present.
375    let (email, preferred_username) = if has_scope("email") || has_scope("profile") {
376        match state
377            .user_repo
378            .get_by_id(user.tenant_id, user.user_id)
379            .await
380        {
381            Ok(u) => (
382                if has_scope("email") {
383                    Some(u.email)
384                } else {
385                    None
386                },
387                if has_scope("profile") {
388                    Some(u.username)
389                } else {
390                    None
391                },
392            ),
393            Err(e) => {
394                tracing::error!(
395                    user_id = %user.user_id,
396                    tenant_id = %user.tenant_id,
397                    error = %e,
398                    "userinfo: failed to fetch user for scoped claims"
399                );
400                return HttpResponse::InternalServerError().json(OAuth2ErrorResponse {
401                    error: "server_error".into(),
402                    error_description: "failed to retrieve user claims".into(),
403                });
404            }
405        }
406    } else {
407        (None, None)
408    };
409
410    HttpResponse::Ok().json(UserInfoResponse {
411        sub: user.user_id.to_string(),
412        email,
413        preferred_username,
414        tenant_id: user.tenant_id.to_string(),
415        org_id: user.org_id.to_string(),
416    })
417}

423/// Build a redirect response with error parameters per RFC 6749
424/// section 4.1.2.1.
425fn build_error_redirect(
426    redirect_uri: &str,
427    error: &OAuth2Error,
428    state: Option<&str>,
429) -> HttpResponse {
430    let mut url = match url::Url::parse(redirect_uri) {
431        Ok(u) => u,
432        Err(_) => {
433            // Unparseable redirect_uri — return direct error
434            return HttpResponse::BadRequest().json(OAuth2ErrorResponse {
435                error: error.error_code().to_string(),
436                error_description: error.error_description(),
437            });
438        }
439    };
440    url.query_pairs_mut()
441        .append_pair("error", error.error_code())
442        .append_pair("error_description", &error.error_description());
443    if let Some(state) = state {
444        url.query_pairs_mut().append_pair("state", state);
445    }
446    HttpResponse::Found()
447        .append_header(("Location", url.to_string()))
448        .finish()
449}

451/// Build an OAuth2 JSON error response with the appropriate HTTP status.
452///
453/// `invalid_client` returns 401 with a `WWW-Authenticate` header per
454/// RFC 6749 §5.2.  Although the token endpoint uses `client_secret_post`,
455/// RFC 6749 §5.2 still requires the 401 response to include
456/// `WWW-Authenticate` indicating the authentication scheme.
457fn build_oauth2_error_response(e: &OAuth2Error) -> HttpResponse {
458    let status = match e {
459        OAuth2Error::InvalidClient(_) => actix_web::http::StatusCode::UNAUTHORIZED,
460        OAuth2Error::ServerError(_) => actix_web::http::StatusCode::INTERNAL_SERVER_ERROR,
461        _ => actix_web::http::StatusCode::BAD_REQUEST,
462    };
463    let mut builder = HttpResponse::build(status);
464    builder
465        .append_header(("Cache-Control", "no-store"))
466        .append_header(("Pragma", "no-cache"));
467    // RFC 6749 §5.2: 401 responses MUST include WWW-Authenticate
468    if status == actix_web::http::StatusCode::UNAUTHORIZED {
469        builder.append_header(("WWW-Authenticate", "Bearer realm=\"axiam\""));
470    }
471    builder.json(OAuth2ErrorResponse {
472        error: e.error_code().to_string(),
473        error_description: e.error_description(),
474    })
475}

480//
481// The full RFC-conformance/e2e coverage for `/oauth2/jwks` lives in
482// `tests/oauth2_*.rs`. These handler-level tests are deliberately narrow:

486// `web::Data`) to prove the `If-None-Match` -> 304 wiring and the
487// `Cache-Control`/`ETag` headers actually reach the HTTP response. The
488// cache/ETag/304 *logic itself* (RFC 7232 comparison, rotation, malformed
489// keys, etc.) is exhaustively covered by `axiam_oauth2::jwks_cache`'s own
490// unit tests -- this module does not re-test that logic, only the plumbing.
