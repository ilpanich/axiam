23/// Body for the request-reset endpoint.
24///
25/// `tenant_id` is optional (Open Question 1 / Assumption A4): the public
26/// forgot-password page has no prior tenant context, so it may instead
27/// supply `org_slug`/`tenant_slug` (mirroring `handlers/auth.rs`'s
28/// `(Option<Uuid>, Option<&str>)` login pattern). Slug-resolution failure
29/// is enumeration-safe (D-05) — it funnels into the SAME uniform
30/// `{"sent": true}` response as an unknown account, never a distinct
31/// error status.
32#[derive(Debug, Deserialize, utoipa::ToSchema)]
33pub struct RequestResetBody {
34    #[serde(default)]
35    pub tenant_id: Option<Uuid>,
36    pub email: String,
37    #[serde(default)]
38    pub org_slug: Option<String>,
39    #[serde(default)]
40    pub tenant_slug: Option<String>,
41}
42
43/// Body for the confirm-reset endpoint.
44#[derive(Debug, Deserialize, utoipa::ToSchema)]
45pub struct ConfirmResetBody {
46    pub tenant_id: Uuid,
47    pub token: String,
48    pub new_password: String,
49}

55/// Resolve the effective tenant context for `request_reset`.
56///
57/// Accepts either a raw `tenant_id` UUID or an `(org_slug, tenant_slug)`
58/// pair, mirroring `handlers/auth.rs`'s login resolution
59/// (`(Option<Uuid>, Option<&str>)`). UNLIKE login, a resolution failure is
60/// represented here as `None` rather than a distinct error — the caller
61/// MUST treat `None` as an enumeration-safe no-op (D-05 / Pitfall 4 /
62/// T-23-06-A), funneling into the SAME uniform `{"sent": true}` response
63/// as an unknown account, NEVER a distinct 400/404.
64async fn resolve_reset_tenant_id<O, T>(
65    org_repo: &O,
66    tenant_repo: &T,
67    tenant_id: Option<Uuid>,
68    org_slug: Option<&str>,
69    tenant_slug: Option<&str>,
70) -> Option<Uuid>
71where
72    O: OrganizationRepository,
73    T: TenantRepository,
74{
75    match (tenant_id, tenant_slug) {
76        (Some(id), _) => Some(id),
77        (None, Some(tenant_slug)) => match org_slug {
78            Some(org_slug) => match org_repo.get_by_slug(org_slug).await {
79                Ok(org) => tenant_repo
80                    .get_by_slug(org.id, tenant_slug)
81                    .await
82                    .ok()
83                    .map(|t| t.id),
84                Err(_) => None,
85            },
86            None => None,
87        },
88        (None, None) => None,
89    }
90}

96/// `POST /api/v1/auth/reset`
97///
98/// Initiates a password reset by enqueuing an `OutboundMailMessage` to the
99/// async mail queue.  Always returns `{"sent": true}` to prevent email
100/// enumeration (D-15) — regardless of whether the email exists, the user is
101/// federated, delivery later succeeds, or the rate limit is exceeded.
102#[utoipa::path(
103    post,
104    path = "/api/v1/auth/reset",
105    tag = "auth",
106    request_body = RequestResetBody,
107    responses(
108        (status = 200, description = "Password reset email enqueued"),
109    )
110)]
111pub async fn request_reset<C: Connection + Clone>(
112    state: web::Data<AppState<C>>,
113    body: web::Json<RequestResetBody>,
114) -> Result<HttpResponse, AxiamApiError> {
115    let req = body.into_inner();
116
117    // Resolve tenant context — accept either a raw tenant_id UUID or a
118    // (org_slug, tenant_slug) pair (Open Question 1 / Assumption A4).
119    // Resolution failure NEVER `?`-propagates: it funnels into the SAME
120    // uniform enumeration-safe `{"sent": true}` response as an unknown
121    // account (D-05 / Pitfall 4 / T-23-06-A).
122    let tenant_id = resolve_reset_tenant_id(
123        &state.org_repo,
124        &state.tenant_repo,
125        req.tenant_id,
126        req.org_slug.as_deref(),
127        req.tenant_slug.as_deref(),
128    )
129    .await;
130
131    let Some(tenant_id) = tenant_id else {
132        // D-05: unresolvable/missing tenant context is enumeration-safe —
133        // identical to the unknown-account branch below. Never disclose
134        // whether an org/tenant slug exists.
135        tracing::debug!(
136            email = %req.email,
137            "password-reset: no action (tenant unresolved)"
138        );
139        return Ok(HttpResponse::Ok().json(serde_json::json!({ "sent": true })));
140    };
141
142    // QUAL-07: PasswordResetService is now a hoisted AppState singleton
143    // (was constructed per-request here).
144    let expiry_hours = state.auth_config.password_reset_token_expiry_hours;
145    let pepper = state.auth_config.pepper.as_ref().map(|p| p.expose_secret());
146
147    match state
148        .password_reset_service
149        .initiate_reset(tenant_id, &req.email, expiry_hours, pepper)
150        .await
151    {
152        Ok(Some((raw_token, user_id, expires_at))) => {
153            // Resolve org_id from tenant for the mail message.
154            // On failure, log and continue — D-15: never propagate to client.
155            let org_id = match state.tenant_repo.get_by_id(tenant_id).await {
156                Ok(tenant) => tenant.organization_id,
157                Err(e) => {
158                    tracing::warn!(
159                        error = %e,
160                        tenant_id = %tenant_id,
161                        "failed to resolve org_id for password-reset mail; using nil"
162                    );
163                    Uuid::nil()
164                }
165            };
166
167            // PHASE-DEFINING (23-RESEARCH Pattern 6 / Pitfall 3): build the
168            // action_url the same way gdpr.rs builds cancel_url — a
169            // relative-path frontend route carrying the raw token plus the
170            // resolved tenant_id UUID (Open Question 2, mirroring the
171            // already-shipped VerifyEmailPage `?token=…&tenant_id=…`).
172            let reset_url = format!("/auth/reset-password?token={raw_token}&tenant_id={tenant_id}");
173
174            let msg = OutboundMailMessage {
175                mail_type: MailType::PasswordReset,
176                tenant_id,
177                org_id,
178                user_id,
179                to_address: req.email.clone(),
180                template_context: serde_json::json!({
181                    "token": raw_token,
182                    "action_url": reset_url,
183                    "expiry_time": expires_at.to_rfc3339(),
184                }),
185                attempt_count: 0,
186                enqueued_at: Utc::now(),
187            };
188
189            if let Err(e) = state.mail_outbound_publisher.publish(msg).await {
190                // D-15: log warn but do NOT propagate — uniform 200 regardless
191                tracing::warn!(
192                    error = %e,
193                    "failed to enqueue password-reset email; continuing"
194                );
195            } else {
196                tracing::debug!(email = %req.email, "password reset email enqueued");
197            }
198        }
199        Ok(None) => {
200            // User not found or federated — silently ignore (D-15).
201            tracing::debug!(
202                email = %req.email,
203                "password-reset: no action (unknown or federated)"
204            );
205        }
206        Err(AxiamError::RateLimited) => {
207            // Swallow rate-limit to prevent user enumeration via
208            // differential 429 responses (D-15).
209            tracing::debug!(
210                email = %req.email,
211                "password-reset: rate-limited (suppressed)"
212            );
213        }
214        Err(e) => return Err(e.into()),
215    }
216
217    // Always return identical 200 regardless of outcome (D-15).
218    Ok(HttpResponse::Ok().json(serde_json::json!({ "sent": true })))
219}

221/// `POST /api/v1/auth/reset/confirm`
222///
223/// Confirms a password reset using a one-time token and a new
224/// password. The token is consumed atomically.
225///
226/// Returns `{"reset": true}` on success, or 400 with policy
227/// violations if the new password is too weak.
228#[utoipa::path(
229    post,
230    path = "/api/v1/auth/reset/confirm",
231    tag = "auth",
232    request_body = ConfirmResetBody,
233    responses(
234        (status = 200, description = "Password reset successfully"),
235        (status = 400, description = "Invalid token or password policy violation"),
236    )
237)]
238pub async fn confirm_reset<C: Connection + Clone>(
239    state: web::Data<AppState<C>>,
240    body: web::Json<ConfirmResetBody>,
241) -> Result<HttpResponse, AxiamApiError> {
242    use axiam_core::repository::{SettingsRepository, TenantRepository};
243
244    let req = body.into_inner();
245
246    // Resolve the tenant to get its org_id for settings.
247    let tenant = state.tenant_repo.get_by_id(req.tenant_id).await?;
248
249    // Resolve effective password policy.
250    let settings = state
251        .settings_repo
252        .get_effective_settings(tenant.organization_id, req.tenant_id)
253        .await?;
254
255    // QUAL-07: PasswordResetService is now a hoisted AppState singleton.
256    state
257        .password_reset_service
258        .confirm_reset(
259            req.tenant_id,
260            &req.token,
261            &req.new_password,
262            &settings.password,
263            state.auth_config.pepper.as_ref().map(|p| p.expose_secret()),
264            Some(&state.http_client),
265        )
266        .await?;
267
268    Ok(HttpResponse::Ok().json(serde_json::json!({ "reset": true })))
269}

283    /// Fake mail publisher that records messages for test assertions.
284    #[derive(Clone, Default)]
285    struct RecordingPublisher {
286        sent: Arc<Mutex<Vec<OutboundMailMessage>>>,
287        fail: bool,
288    }
289
290    impl RecordingPublisher {
291        fn new() -> Self {
292            Self {
293                sent: Arc::new(Mutex::new(Vec::new())),
294                fail: false,
295            }
296        }
297
298        fn failing() -> Self {
299            Self {
300                sent: Arc::new(Mutex::new(Vec::new())),
301                fail: true,
302            }
303        }
304
305        fn messages(&self) -> Vec<OutboundMailMessage> {
306            self.sent.lock().unwrap().clone()
307        }
308
309        fn count(&self) -> usize {
310            self.sent.lock().unwrap().len()
311        }
312    }

314    impl MailPublisher for RecordingPublisher {
315        async fn publish(&self, msg: OutboundMailMessage) -> AxiamResult<()> {
316            if self.fail {
317                return Err(AxiamError::Internal("mock publish failure".into()));
318            }
319            self.sent.lock().unwrap().push(msg);
320            Ok(())
321        }
322    }
