54/// `POST /api/v1/permissions`
55#[utoipa::path(
56    post,
57    path = "/api/v1/permissions",
58    tag = "permissions",
59    request_body = CreatePermissionRequest,
60    responses(
61        (status = 201, description = "Permission created", body = Permission),
62    ),
63    security(("bearer" = []))
64)]
65pub async fn create<C: Connection + Clone>(
66    user: AuthenticatedUser,
67    authz: AuthzData,
68    state: web::Data<AppState<C>>,
69    body: web::Json<CreatePermissionRequest>,
70) -> Result<HttpResponse, AxiamApiError> {
71    RequirePermission::new("permissions:create", Uuid::nil())
72        .check(&user, authz.get_ref().as_ref())
73        .await?;
74    let req = body.into_inner();
75    let input = CreatePermission {
76        tenant_id: user.tenant_id,
77        action: req.action,
78        description: req.description,
79    };
80    let permission = state.permission_repo.create(input).await?;
81    Ok(HttpResponse::Created().json(permission))
82}
