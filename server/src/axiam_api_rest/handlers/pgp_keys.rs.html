31/// `POST /api/v1/pgp-keys`
32#[utoipa::path(
33    post,
34    path = "/api/v1/pgp-keys",
35    tag = "pgp-keys",
36    request_body = CreatePgpKeyRequest,
37    responses(
38        (status = 201, description = "PGP key generated",
39         body = GeneratedPgpKey),
40    ),
41    security(("bearer" = []))
42)]
43pub async fn generate<C: Connection + Clone>(
44    user: AuthenticatedUser,
45    authz: AuthzData,
46    state: web::Data<AppState<C>>,
47    body: web::Json<CreatePgpKeyRequest>,
48) -> Result<HttpResponse, AxiamApiError> {
49    RequirePermission::new("pgp_keys:generate", Uuid::nil())
50        .check(&user, authz.get_ref().as_ref())
51        .await?;
52    let req = body.into_inner();
53    let input = CreatePgpKey {
54        tenant_id: user.tenant_id,
55        name: req.name,
56        purpose: req.purpose,
57        algorithm: req.algorithm,
58        email: req.email,
59    };
60    let result = state.pgp_service.generate(input).await?;
61    Ok(HttpResponse::Created().json(result))
62}

142/// `POST /api/v1/pgp-keys/sign-audit-batch`
143#[utoipa::path(
144    post,
145    path = "/api/v1/pgp-keys/sign-audit-batch",
146    tag = "pgp-keys",
147    request_body = SignAuditBatchRequest,
148    responses(
149        (status = 200, description = "Audit batch signed",
150         body = SignedAuditBatch),
151    ),
152    security(("bearer" = []))
153)]
154pub async fn sign_audit_batch<C: Connection + Clone>(
155    user: AuthenticatedUser,
156    authz: AuthzData,
157    state: web::Data<AppState<C>>,
158    body: web::Json<SignAuditBatchRequest>,
159) -> Result<HttpResponse, AxiamApiError> {
160    RequirePermission::new("pgp_keys:sign_audit_batch", Uuid::nil())
161        .check(&user, authz.get_ref().as_ref())
162        .await?;
163    let request = body.into_inner();
164
165    // Reject empty or duplicate entry IDs
166    if request.entry_ids.is_empty() {
167        return Err(axiam_core::error::AxiamError::Validation {
168            message: "entry_ids must not be empty".into(),
169        }
170        .into());
171    }
172    let unique_ids: std::collections::HashSet<_> = request.entry_ids.iter().collect();
173    if unique_ids.len() != request.entry_ids.len() {
174        return Err(axiam_core::error::AxiamError::Validation {
175            message: "entry_ids must not contain duplicates".into(),
176        }
177        .into());
178    }
179
180    let entries = state
181        .audit_repo
182        .get_by_ids(user.tenant_id, &request.entry_ids)
183        .await?;
184
185    if entries.len() != request.entry_ids.len() {
186        let found_ids: std::collections::HashSet<_> = entries.iter().map(|e| e.id).collect();
187        let missing: Vec<_> = request
188            .entry_ids
189            .iter()
190            .filter(|id| !found_ids.contains(id))
191            .collect();
192        return Err(axiam_core::error::AxiamError::Validation {
193            message: format!("audit entries not found: {missing:?}"),
194        }
195        .into());
196    }
197
198    let result = state
199        .pgp_service
200        .sign_audit_batch(user.tenant_id, entries)
201        .await?;
202    Ok(HttpResponse::Ok().json(result))
203}

205/// `POST /api/v1/pgp-keys/{id}/encrypt`
206#[utoipa::path(
207    post,
208    path = "/api/v1/pgp-keys/{id}/encrypt",
209    tag = "pgp-keys",
210    request_body = EncryptRequest,
211    params(("id" = Uuid, Path, description = "PGP key ID")),
212    responses(
213        (status = 200, description = "Data encrypted",
214         body = EncryptedExport),
215    ),
216    security(("bearer" = []))
217)]
218pub async fn encrypt<C: Connection + Clone>(
219    user: AuthenticatedUser,
220    authz: AuthzData,
221    path: web::Path<Uuid>,
222    state: web::Data<AppState<C>>,
223    body: web::Json<EncryptRequest>,
224) -> Result<HttpResponse, AxiamApiError> {
225    RequirePermission::new("pgp_keys:encrypt", Uuid::nil())
226        .check(&user, authz.get_ref().as_ref())
227        .await?;
228    let key_id = path.into_inner();
229    let plaintext = base64::Engine::decode(
230        &base64::engine::general_purpose::STANDARD,
231        &body.data_base64,
232    )
233    .map_err(|e| axiam_core::error::AxiamError::Validation {
234        message: format!("invalid base64: {e}"),
235    })?;
236
237    let result = state
238        .pgp_service
239        .encrypt_for_export(user.tenant_id, key_id, &plaintext)
240        .await?;
241    Ok(HttpResponse::Ok().json(result))
242}
