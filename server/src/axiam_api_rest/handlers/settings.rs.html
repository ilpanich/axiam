17/// `GET /api/v1/organizations/{org_id}/settings`
18#[utoipa::path(
19    get,
20    path = "/api/v1/organizations/{org_id}/settings",
21    tag = "settings",
22    params(
23        ("org_id" = Uuid, Path, description = "Organization ID"),
24    ),
25    responses(
26        (status = 200, description = "Organization security settings",
27         body = SecuritySettings),
28    ),
29    security(("bearer" = []))
30)]
31pub async fn get_org_settings<C: Connection + Clone>(
32    user: AuthenticatedUser,
33    authz: AuthzData,
34    state: web::Data<AppState<C>>,
35    path: web::Path<Uuid>,
36) -> Result<HttpResponse, AxiamApiError> {
37    // SECURITY (NEW-2): org-baseline settings are a higher privilege than the
38    // per-tenant `settings:*` self-management permissions. Enforce the dedicated
39    // org-level permission declared in ROUTE_PERMISSION_MAP so a tenant-scoped
40    // `settings:get` grant cannot read the org baseline that cascades to every
41    // sibling tenant.
42    RequirePermission::new("organizations:get_settings", Uuid::nil())
43        .check(&user, authz.get_ref().as_ref())
44        .await?;
45    let org_id = path.into_inner();
46
47    // Authorization: only allow reads for the authenticated user's own org.
48    if org_id != user.org_id {
49        return Err(AxiamApiError(
50            axiam_core::error::AxiamError::AuthorizationDenied {
51                reason: "cannot read settings for a different organization".into(),
52                action: None,
53                resource_id: None,
54            },
55        ));
56    }
57
58    let settings = state.settings_repo.get_org_settings(org_id).await?;
59    Ok(HttpResponse::Ok().json(settings))
60}

62/// `PUT /api/v1/organizations/{org_id}/settings`
63#[utoipa::path(
64    put,
65    path = "/api/v1/organizations/{org_id}/settings",
66    tag = "settings",
67    params(
68        ("org_id" = Uuid, Path, description = "Organization ID"),
69    ),
70    request_body = SetOrgSettings,
71    responses(
72        (status = 200, description = "Organization settings updated",
73         body = SecuritySettings),
74    ),
75    security(("bearer" = []))
76)]
77pub async fn set_org_settings<C: Connection + Clone>(
78    user: AuthenticatedUser,
79    authz: AuthzData,
80    state: web::Data<AppState<C>>,
81    path: web::Path<Uuid>,
82    body: web::Json<SetOrgSettings>,
83) -> Result<HttpResponse, AxiamApiError> {
84    // SECURITY (NEW-2): enforce the dedicated org-level permission (declared in
85    // ROUTE_PERMISSION_MAP) rather than the tenant-scoped `settings:update`, so a
86    // tenant admin cannot rewrite the org security baseline inherited by every
87    // sibling tenant.
88    RequirePermission::new("organizations:update_settings", Uuid::nil())
89        .check(&user, authz.get_ref().as_ref())
90        .await?;
91    let org_id = path.into_inner();
92
93    // Authorization: only allow writes to the authenticated user's own org.
94    if org_id != user.org_id {
95        return Err(AxiamApiError(
96            axiam_core::error::AxiamError::AuthorizationDenied {
97                reason: "cannot modify settings for a different organization".into(),
98                action: None,
99                resource_id: None,
100            },
101        ));
102    }
103
104    let input = body.into_inner();
105    validate_org_settings(&input)?;
106    let settings = state.settings_repo.set_org_settings(org_id, input).await?;
107    Ok(HttpResponse::Ok().json(settings))
108}

110/// `GET /api/v1/settings`
111///
112/// Returns the effective (merged) security settings for the
113/// authenticated user's tenant. Org baseline + tenant overrides.
114#[utoipa::path(
115    get,
116    path = "/api/v1/settings",
117    tag = "settings",
118    responses(
119        (status = 200, description = "Effective tenant security settings",
120         body = SecuritySettings),
121    ),
122    security(("bearer" = []))
123)]
124pub async fn get_tenant_settings<C: Connection + Clone>(
125    user: AuthenticatedUser,
126    authz: AuthzData,
127    state: web::Data<AppState<C>>,
128) -> Result<HttpResponse, AxiamApiError> {
129    RequirePermission::new("settings:get", Uuid::nil())
130        .check(&user, authz.get_ref().as_ref())
131        .await?;
132    let settings = state
133        .settings_repo
134        .get_effective_settings(user.org_id, user.tenant_id)
135        .await?;
136    Ok(HttpResponse::Ok().json(settings))
137}

139/// `PUT /api/v1/settings`
140///
141/// Set tenant-level overrides. Only fields that are **more restrictive**
142/// than the org baseline are accepted. Omit a field (set to `null`) to
143/// inherit from the org.
144#[utoipa::path(
145    put,
146    path = "/api/v1/settings",
147    tag = "settings",
148    request_body = TenantSettingsOverride,
149    responses(
150        (status = 200, description = "Tenant settings updated",
151         body = SecuritySettings),
152        (status = 400, description = "Override violates org baseline"),
153    ),
154    security(("bearer" = []))
155)]
156pub async fn set_tenant_settings<C: Connection + Clone>(
157    user: AuthenticatedUser,
158    authz: AuthzData,
159    state: web::Data<AppState<C>>,
160    body: web::Json<TenantSettingsOverride>,
161) -> Result<HttpResponse, AxiamApiError> {
162    RequirePermission::new("settings:update", Uuid::nil())
163        .check(&user, authz.get_ref().as_ref())
164        .await?;
165    let org = state.settings_repo.get_org_settings(user.org_id).await?;
166    let overrides = body.into_inner();
167
168    // Validate: tenant can only be more restrictive than org
169    validate_tenant_override(&org, &overrides)?;
170
171    // Merge org baseline + overrides into a complete settings row.
172    // The ID passed here is just a placeholder; the underlying
173    // SurrealSettingsRepository derives a deterministic UUID (v5)
174    // from (scope, scope_id) and uses that as the canonical ID.
175    let merged = effective_settings(&org, &overrides, user.tenant_id, Uuid::nil());
176
177    let result = state
178        .settings_repo
179        .store_effective_tenant_settings(user.tenant_id, merged)
180        .await?;
181    Ok(HttpResponse::Ok().json(result))
182}
