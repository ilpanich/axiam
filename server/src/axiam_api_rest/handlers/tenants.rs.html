19/// Request body for tenant creation (organization_id comes from the URL path).
20#[derive(Debug, Deserialize, utoipa::ToSchema)]
21pub struct CreateTenantRequest {
22    pub name: String,
23    pub slug: String,
24    pub metadata: Option<serde_json::Value>,
25}

40/// `POST /api/v1/organizations/{org_id}/tenants`
41#[utoipa::path(
42    post,
43    path = "/api/v1/organizations/{org_id}/tenants",
44    tag = "tenants",
45    params(("org_id" = Uuid, Path, description = "Organization ID")),
46    request_body = CreateTenantRequest,
47    responses(
48        (status = 201, description = "Tenant created", body = Tenant),
49    ),
50    security(("bearer" = []))
51)]
52pub async fn create<C: Connection + Clone>(
53    user: AuthenticatedUser,
54    authz: AuthzData,
55    state: web::Data<AppState<C>>,
56    path: web::Path<OrgPath>,
57    body: web::Json<CreateTenantRequest>,
58) -> Result<HttpResponse, AxiamApiError> {
59    RequirePermission::new("tenants:create", Uuid::nil())
60        .check(&user, authz.get_ref().as_ref())
61        .await?;
62
63    // Authorization: only allow creating tenants under the caller's own org.
64    if path.org_id != user.org_id {
65        return Err(AxiamApiError(
66            axiam_core::error::AxiamError::AuthorizationDenied {
67                reason: "cannot access a different organization".into(),
68                action: None,
69                resource_id: None,
70            },
71        ));
72    }
73
74    let req = body.into_inner();
75    let input = CreateTenant {
76        organization_id: path.org_id,
77        name: req.name,
78        slug: req.slug,
79        metadata: req.metadata,
80    };
81    let tenant = state.tenant_repo.create(input).await?;
82
83    // Auto-seed permissions for the new tenant so RBAC works immediately.
84    seed_permissions(&state.db, tenant.id, PERMISSION_REGISTRY)
85        .await
86        .map_err(|e| {
87            tracing::error!(
88                "Failed to seed permissions for new tenant {}: {}",
89                tenant.id,
90                e
91            );
92            AxiamApiError(AxiamError::Internal(
93                "Failed to seed permissions for tenant".into(),
94            ))
95        })?;
96
97    Ok(HttpResponse::Created().json(tenant))
98}

114pub async fn list<C: Connection + Clone>(

123        .await?;
124
125    // Authorization: only allow listing tenants under the caller's own org.

129                reason: "cannot access a different organization".into(),

131                resource_id: None,

158pub async fn get<C: Connection + Clone>(
159    user: AuthenticatedUser,
160    authz: AuthzData,
161    state: web::Data<AppState<C>>,
162    path: web::Path<TenantPath>,
163) -> Result<HttpResponse, AxiamApiError> {
164    RequirePermission::new("tenants:get", Uuid::nil())
165        .check(&user, authz.get_ref().as_ref())
166        .await?;
167
168    // Authorization: reject cross-org probing before touching the DB.
169    if path.org_id != user.org_id {
170        return Err(AxiamApiError(
171            axiam_core::error::AxiamError::AuthorizationDenied {
172                reason: "cannot access a different organization".into(),
173                action: None,
174                resource_id: None,
175            },
176        ));
177    }
178
179    let tenant = state.tenant_repo.get_by_id(path.tenant_id).await?;
180    if tenant.organization_id != path.org_id {
181        return Err(AxiamError::NotFound {
182            entity: "Tenant".into(),
183            id: path.tenant_id.to_string(),
184        }
185        .into());
186    }
187    Ok(HttpResponse::Ok().json(tenant))
188}
