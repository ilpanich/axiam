1//! Global authorization middleware — default-deny for all non-public paths.
2//!
3//! [`AuthzMiddleware`] implements the first layer of defense-in-depth (D-01,
4//! D-03):
5//!
6//! - Public paths (from [`PUBLIC_PATHS`]) pass through without any credential
7//!   check.
8//! - All other paths require a JWT to be present in either the `axiam_access`
9//!   cookie or the `Authorization` header. Missing credentials → **401**.
10//! - When credentials are present the request is forwarded to the handler.
11//!   The per-handler [`RequirePermission`] guard then performs the actual
12//!   permission check (D-02), returning **403** on denial.
13//!
14//! [`RequirePermission`]: crate::authz::RequirePermission
15

32/// Normalizes `path` for the public-path allowlist check (SECHRD-11 / T-24-11).
33///
34/// - Collapses repeated `/` into a single `/`.
35/// - Rejects (returns `None`) any path with a literal `..` path segment.
36///
37/// This function performs **no** rewriting/canonicalization for routing
38/// purposes — it exists solely to produce a fail-closed input to
39/// [`is_public_path`]. A `None` result must always translate to "not
40/// public" (deny → falls through to the normal credential check), never an
41/// implicit allow.
42fn normalize_for_public_check(path: &str) -> Option<String> {
43    // Reject any `..` segment outright (path-traversal attempt). A plain
44    // per-segment comparison is sufficient here because the only decision
45    // this function feeds is a boolean allow/deny, not a rewritten path
46    // used for filesystem or routing access.
47    if path.split('/').any(|segment| segment == "..") {
48        return None;
49    }
50
51    // Collapse repeated slashes (`//` -> `/`) without altering leading or
52    // trailing slash semantics.
53    let mut normalized = String::with_capacity(path.len());
54    let mut prev_was_slash = false;
55    for c in path.chars() {
56        if c == '/' {
57            if prev_was_slash {
58                continue;
59            }
60            prev_was_slash = true;
61        } else {
62            prev_was_slash = false;
63        }
64        normalized.push(c);
65    }
66    Some(normalized)
67}

69/// Returns `true` if `path` matches one of `entries` under the
70/// segment-boundary-aware allowlist rule.
71///
72/// Matching rules:
73/// - `path` is first normalized (see [`normalize_for_public_check`]); a
74///   normalization failure (a `..` segment present) fails closed — this
75///   function returns `false`, never an implicit allow.
76/// - Entries ending with `*` are **segment-boundary prefix-matched**: the
77///   trailing `*` (and any single `/` immediately before it) is stripped to
78///   form the prefix, and a match requires the remainder of `path` after
79///   that prefix to be either empty or to begin with `/`. This prevents
80///   prefix confusion (e.g. an `/api/v1/auth/*` entry must NOT match
81///   `/api/v1/authz/...`).
82/// - All other entries are compared with `==` (exact match, against the
83///   normalized path).
84///
85/// Extracted from [`is_public_path`] so tests can prove the matching
86/// property against a synthetic allowlist (Pitfall 6: the live
87/// [`PUBLIC_PATHS`] registry has no adjacent-prefix wildcard pair to
88/// exploit today) while exercising the exact same code path production
89/// traffic runs through.
90fn matches_public_allowlist(path: &str, entries: &[&str]) -> bool {
91    let Some(path) = normalize_for_public_check(path) else {
92        // Ambiguous/non-canonical path — fail closed, never public.
93        return false;
94    };
95    let path = path.as_str();
96
97    for &entry in entries {
98        if let Some(prefix) = entry.strip_suffix('*') {
99            let prefix = prefix.strip_suffix('/').unwrap_or(prefix);
100            if let Some(remainder) = path.strip_prefix(prefix)
101                && (remainder.is_empty() || remainder.starts_with('/'))
102            {
103                return true;
104            }
105        } else if path == entry {
106            return true;
107        }
108    }
109    false
110}
111
112/// Returns `true` if `path` is in the public-path allowlist and should be
113/// allowed through without credential validation.
114///
115/// See [`matches_public_allowlist`] for the matching rules.
116pub fn is_public_path(path: &str) -> bool {
117    matches_public_allowlist(path, PUBLIC_PATHS)
118}

124/// Global authorization middleware.
125///
126/// Wrap API scopes with `.wrap(AuthzMiddleware)` in `server.rs`.
127pub struct AuthzMiddleware;
128
129impl<S, B> Transform<S, ServiceRequest> for AuthzMiddleware
130where
131    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error> + 'static,
132    B: 'static,
133{
134    type Response = ServiceResponse<EitherBody<B>>;
135    type Error = Error;
136    type Transform = AuthzMiddlewareService<S>;
137    type InitError = ();
138    type Future = Ready<Result<Self::Transform, Self::InitError>>;
139
140    fn new_transform(&self, service: S) -> Self::Future {
141        ready(Ok(AuthzMiddlewareService { inner: service }))
142    }
143}

145// ---------------------------------------------------------------------------
146// Inner service
147// ---------------------------------------------------------------------------
148
149pub struct AuthzMiddlewareService<S> {
150    inner: S,
151}
152
153impl<S, B> Service<ServiceRequest> for AuthzMiddlewareService<S>
154where
155    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error> + 'static,
156    B: 'static,
157{
158    type Response = ServiceResponse<EitherBody<B>>;
159    type Error = Error;
160    type Future = Pin<Box<dyn Future<Output = Result<Self::Response, Self::Error>>>>;
161
162    fn poll_ready(
163        &self,
164        cx: &mut std::task::Context<'_>,
165    ) -> std::task::Poll<Result<(), Self::Error>> {
166        self.inner.poll_ready(cx)
167    }
168
169    fn call(&self, req: ServiceRequest) -> Self::Future {
170        let path = req.path().to_owned();
171
172        // 1. Public paths pass through unconditionally.
173        if is_public_path(&path) {
174            let fut = self.inner.call(req);
175            return Box::pin(async move {
176                let res = fut.await?;
177                Ok(res.map_into_left_body())
178            });
179        }
180
181        // 2. Check for credential presence (cookie OR Authorization header).
182        let has_cookie = req.cookie(COOKIE_ACCESS).is_some();
183        let has_bearer = req.headers().contains_key("Authorization");
184
185        if !has_cookie && !has_bearer {
186            // No credentials → 401 Unauthorized.
187            let error: actix_web::Error = AxiamApiError(AxiamError::AuthenticationFailed {
188                reason: "authentication required".into(),
189            })
190            .into();
191            return Box::pin(async move {
192                let res = req.error_response(error);
193                Ok(res.map_into_right_body())
194            });
195        }
196
197        // 3. Credentials present — forward to handler for permission check.
198        let fut = self.inner.call(req);
199        Box::pin(async move {
200            let res = fut.await?;
201            Ok(res.map_into_left_body())
202        })
203    }
204}

214    #[test]
215    fn public_paths_are_recognized() {
216        assert!(is_public_path("/health"));
217        assert!(is_public_path("/api/v1/auth/login"));
218        // Refresh authenticates via its own opaque cookie and must bypass the
219        // access-token credential check (the access cookie is gone post-expiry).
220        assert!(is_public_path("/api/v1/auth/refresh"));
221        assert!(is_public_path("/.well-known/openid-configuration"));
222        assert!(is_public_path("/oauth2/token"));
223        assert!(is_public_path("/api/docs/openapi.json")); // prefix match via /api/docs/*
224        assert!(is_public_path("/api/v1/admin/bootstrap"));
225    }

245    // -----------------------------------------------------------------
246    // SECHRD-11 / T-24-11 / T-24-12 — segment-boundary + normalization
247    // -----------------------------------------------------------------
248    //
249    // These tests assert properties of `is_public_path` / `normalize_for_
250    // public_check` directly, using a synthetic wildcard entry, per
251    // RESEARCH.md Pitfall 6: today's real `PUBLIC_PATHS` has only one
252    // wildcard entry (`/api/docs/*`) with no adjacent-prefix collision to
253    // exploit, so the property must be proven against the matching
254    // function itself, not the live registry.
255
256    #[test]
257    fn wildcard_prefix_confusion_is_rejected() {
258        // A `/api/v1/auth/*`-style entry must require a segment boundary:
259        // `/api/v1/authz/...` shares the literal character prefix
260        // "/api/v1/auth" but is a DIFFERENT path segment ("authz", not
261        // "auth"), and must never be classified public by that entry.
262        //
263        // Calls `matches_public_allowlist` (the exact code `is_public_path`
264        // delegates to) against a synthetic allowlist, per Pitfall 6: the
265        // live PUBLIC_PATHS registry has no adjacent-prefix wildcard pair
266        // to exploit today, so this proves the property of the matching
267        // function itself, not of the current registry contents.
268        const SYNTHETIC_PUBLIC_PATHS: &[&str] = &["/api/v1/auth/*"];
269        assert!(!matches_public_allowlist(
270            "/api/v1/authz/check",
271            SYNTHETIC_PUBLIC_PATHS
272        ));
273        assert!(!matches_public_allowlist(
274            "/api/v1/authzzz",
275            SYNTHETIC_PUBLIC_PATHS
276        ));
277        // But the legitimate child path under the wildcard must still match.
278        assert!(matches_public_allowlist(
279            "/api/v1/auth/login",
280            SYNTHETIC_PUBLIC_PATHS
281        ));
282        // And the bare prefix itself (no trailing segment) must match too.
283        assert!(matches_public_allowlist(
284            "/api/v1/auth",
285            SYNTHETIC_PUBLIC_PATHS
286        ));
287    }
288
289    #[test]
290    fn real_wildcard_entry_still_matches_legitimate_paths() {
291        // /api/docs/* is the one real wildcard entry today (Pitfall 6) —
292        // confirm the hardened matcher doesn't regress it.
293        assert!(is_public_path("/api/docs/openapi.json"));
294        assert!(is_public_path("/api/docs/"));
295        assert!(is_public_path("/api/docs"));
296        // A sibling path that merely shares the "/api/doc" character
297        // prefix must NOT match.
298        assert!(!is_public_path("/api/documents/secret"));
299    }
300
301    #[test]
302    fn double_slash_is_collapsed_before_matching() {
303        // "//" must be collapsed to "/" before the allowlist check so an
304        // attacker cannot dodge an exact-match entry with a non-canonical
305        // path, nor smuggle a false negative/positive via slash-doubling.
306        assert!(is_public_path("/api/v1/auth//login"));
307        assert!(is_public_path("//health"));
308        assert!(!is_public_path("//api//v1//users"));
309    }
310
311    #[test]
312    fn dot_dot_segment_is_rejected_fail_closed() {
313        // Any path containing a literal `..` segment must be denied
314        // outright (fail-closed), never resolved/canonicalized into an
315        // implicit allow.
316        assert!(!is_public_path("/api/docs/../v1/users"));
317        assert!(!is_public_path("/api/v1/auth/../../v1/users"));
318        assert!(!is_public_path("/.."));
319        // A segment that merely CONTAINS ".." as a substring (not an exact
320        // ".." segment) is not path traversal and must not be penalized.
321        assert!(is_public_path("/api/v1/auth/login"));
322    }

324    #[test]
325    fn exact_match_entry_still_matches_canonical_path() {
326        assert!(is_public_path("/health"));
327        assert!(is_public_path("/api/v1/auth/login"));
328        assert!(!is_public_path("/health/"));
