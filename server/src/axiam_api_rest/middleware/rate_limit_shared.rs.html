1//! Async SurrealDB-backed shared rate-limit pre-check middleware
2//! (SECHRD-03 / D-01a, D-01b, D-01d).
3//!
4//! [`RateLimitShared`] runs an async windowed-CAS increment against the
5//! `rate_limit_bucket` table (`SurrealRateLimitBucketRepository`) BEFORE
6//! the existing per-replica in-memory `Governor`/`GovernorLayer`
7//! (`server.rs::build_governor`, kept byte-for-byte unchanged as the
8//! fail-open fallback). This closes the multi-replica HPA gap where
9//! per-replica in-memory buckets otherwise multiply the effective rate
10//! limit by the replica count.
11//!
12//! **Fail-open (D-01b, T-24-42 accepted risk):** when the shared store is
13//! unreachable (or no DB handle / no client-IP key is available), this
14//! middleware logs a `warn`-level alarm and forwards the request unchanged
15//! so the existing in-memory governor makes the decision instead. A
16//! counter-store outage must never hard-block auth traffic — this is the
17//! ONE deliberate fail-open exception in this phase; every other control
18//! fails closed.
19//!
20//! **CRITICAL (RESEARCH Pitfall 1):** `governor::StateStore::measure_and_replace`
21//! is a *synchronous* trait method. This middleware is deliberately NOT a
22//! `StateStore` implementation and never calls `block_on`/`futures::executor::
23//! block_on` — it is a separate async Actix `Transform`/`Service` (mirroring
24//! [`crate::middleware::authz::AuthzMiddleware`]'s scaffold) that performs
25//! its own async SurrealDB round-trip, then delegates to the inner service.
26

46/// Fixed-window duration (seconds) for the shared bucket. A simple
47/// fixed-window counter is acceptable here (unlike the in-memory GCRA
48/// governor) because this layer only needs to be *approximately* right —
49/// it fails open by design (RESEARCH.md "Don't Hand-Roll").
50const WINDOW_SECS: i64 = 60;
51
52/// Reads the SAME `AXIAM__RATE_LIMIT__TRUSTED_HOPS` env var
53/// `server.rs::build_governor` uses, so the shared-store bucket key and the
54/// in-memory governor's key are derived from the identical client IP
55/// (D-01d parity — a rotating XFF must not yield a fresh bucket in either
56/// layer).
57fn trusted_hops() -> usize {
58    std::env::var("AXIAM__RATE_LIMIT__TRUSTED_HOPS")
59        .ok()
60        .and_then(|v| v.parse().ok())
61        .unwrap_or(0)
62}
63
64/// Truncates `now` down to the start of the current fixed
65/// [`WINDOW_SECS`]-second window.
66fn window_start(now: DateTime<Utc>) -> DateTime<Utc> {
67    let epoch = now.timestamp();
68    let start_epoch = epoch - epoch.rem_euclid(WINDOW_SECS);
69    DateTime::<Utc>::from_timestamp(start_epoch, 0).unwrap_or(now)
70}
71
72/// Builds the same-shaped 429 response the in-memory governor returns
73/// (`extractors::rate_limit::XForwardedForKeyExtractor::exceed_rate_limit_response`),
74/// so clients see one consistent rate-limit contract regardless of which
75/// layer rejected the request.
76fn too_many_requests_response() -> HttpResponse {
77    HttpResponse::TooManyRequests()
78        .content_type(ContentType::json())
79        .insert_header((RETRY_AFTER, WINDOW_SECS.to_string()))
80        .body(r#"{"error":"rate_limit_exceeded"}"#)
81}

87/// Shared SurrealDB-backed rate-limit pre-check middleware.
88///
89/// Generic over the SurrealDB connection type `C` so it can be wired
90/// against both the production `DbClient` and the in-memory `Mem` engine
91/// used in tests. `endpoint` MUST be unique per rate-limited resource so
92/// the shared bucket key (`"{endpoint}:{ip}"`) preserves per-endpoint
93/// granularity — never collapse distinct endpoints into one global bucket.
94///
95/// Wire this BEFORE (i.e. `.wrap()` AFTER, since the last `.wrap()` call is
96/// the outermost layer and therefore executes first) `build_governor(...)`
97/// on the same resource, e.g.:
98///
99/// ```ignore
100/// web::resource("/login")
101///     .wrap(build_governor(rate_limit_cfg.login_per_min))
102///     .wrap(RateLimitShared::<C>::new("login", rate_limit_cfg.login_per_min))
103///     .route(web::post().to(handlers::auth::login::<C>))
104/// ```
105pub struct RateLimitShared<C: Connection + Clone> {
106    endpoint: &'static str,
107    limit: u32,
108    /// D8: bucket-key derivation mode. Always [`RateLimitKeyMode::Ip`] for
109    /// instances built via [`RateLimitShared::new`] — only
110    /// [`RateLimitShared::new_client_identity_aware`] ever sets this to
111    /// something else.
112    key_mode: RateLimitKeyMode,
113    /// D8: whether this resource has a form-encoded OAuth2 `client_id` to
114    /// peek at (`/oauth2/token`, `/oauth2/revoke`, `/oauth2/introspect`
115    /// ONLY — see [`RateLimitShared::new_client_identity_aware`]). `false`
116    /// for every other resource (including `/auth/login`), which never
117    /// touches the request body and always keys per-IP.
118    client_identity_aware: bool,
119    _marker: PhantomData<C>,
120}
121
122impl<C: Connection + Clone> RateLimitShared<C> {
123    /// Plain IP-keyed constructor — unchanged since before D8. Use this for
124    /// every endpoint EXCEPT `/oauth2/token`, `/oauth2/revoke`, and
125    /// `/oauth2/introspect`. In particular, `/auth/login` MUST keep using
126    /// this constructor: it authenticates a user via username/password, so
127    /// there is no OAuth2 `client_id` in the request to key on, and this
128    /// type never reads `AXIAM__RATE_LIMIT__KEY` for it.
129    pub fn new(endpoint: &'static str, limit: u32) -> Self {
130        Self {
131            endpoint,
132            limit,
133            key_mode: RateLimitKeyMode::Ip,
134            client_identity_aware: false,
135            _marker: PhantomData,
136        }
137    }
138
139    /// D8 constructor for the three endpoints where an OAuth2 client
140    /// authenticates itself via a form-encoded `client_id`
141    /// (`client_secret_post`, RFC 6749 §2.3.1): `/oauth2/token`,
142    /// `/oauth2/revoke`, `/oauth2/introspect`.
143    ///
144    /// When `key_mode` is [`RateLimitKeyMode::ClientId`] or
145    /// [`RateLimitKeyMode::IpClientId`], this middleware peeks (and
146    /// restores) the request body to read `client_id`, and stashes it into
147    /// the request extensions (as [`RateLimitClientId`]) so the downstream
148    /// in-memory `Governor`'s `ClientAwareKeyExtractor` on the SAME
149    /// resource can reuse it without a second body read. When `key_mode` is
150    /// [`RateLimitKeyMode::Ip`] this is exactly as if `new()` had been
151    /// called — the body is never touched.
152    ///
153    /// **Do not use this constructor for `/auth/login` or any other
154    /// endpoint without a client identity** — see
155    /// [`crate::config::rate_limit::RateLimitKeyMode`] docs.
156    pub fn new_client_identity_aware(
157        endpoint: &'static str,
158        limit: u32,
159        key_mode: RateLimitKeyMode,
160    ) -> Self {
161        Self {
162            endpoint,
163            limit,
164            key_mode,
165            client_identity_aware: true,
166            _marker: PhantomData,
167        }
168    }
169}

195    fn new_transform(&self, service: S) -> Self::Future {
196        ready(Ok(RateLimitSharedService {
197            inner: Rc::new(service),
198            endpoint: self.endpoint,
199            limit: self.limit,
200            key_mode: self.key_mode,
201            client_identity_aware: self.client_identity_aware,
202            _marker: PhantomData,
203        }))
204    }
205}

211pub struct RateLimitSharedService<S, C: Connection + Clone> {
212    inner: Rc<S>,
213    endpoint: &'static str,
214    limit: u32,
215    key_mode: RateLimitKeyMode,
216    client_identity_aware: bool,
217    _marker: PhantomData<C>,
218}
219
220impl<S, B, C> Service<ServiceRequest> for RateLimitSharedService<S, C>
221where
222    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error> + 'static,
223    B: 'static,
224    C: Connection + Clone + 'static,
225{
226    type Response = ServiceResponse<EitherBody<B>>;
227    type Error = Error;
228    type Future = std::pin::Pin<Box<dyn Future<Output = Result<Self::Response, Self::Error>>>>;
229
230    fn poll_ready(
231        &self,
232        cx: &mut std::task::Context<'_>,
233    ) -> std::task::Poll<Result<(), Self::Error>> {
234        self.inner.poll_ready(cx)
235    }
236
237    fn call(&self, mut req: ServiceRequest) -> Self::Future {
238        let endpoint = self.endpoint;
239        let limit = self.limit;
240        let key_mode = self.key_mode;
241        let client_identity_aware = self.client_identity_aware;
242        let inner = Rc::clone(&self.inner);
243
244        // Reuse the EXACT fixed IP-key extraction logic (D-01d) so the
245        // shared bucket key and the in-memory governor's key agree.
246        let extractor = XForwardedForKeyExtractor::with_trusted_hops(trusted_hops());
247        let ip = extractor.extract(&req).ok();
248
249        // The SurrealDB handle is read from `web::Data<AppState<C>>` (QUAL-01
250        // — was a standalone `web::Data<Surreal<C>>` registration). Its
251        // absence is treated exactly like a DB error below — fail open to
252        // the in-memory governor.
253        let db = req
254            .app_data::<web::Data<AppState<C>>>()
255            .map(|d| d.db.clone());
256
257        Box::pin(async move {
258            // D8: for the client-identity-aware endpoints only
259            // (`/oauth2/token`, `/oauth2/revoke`, `/oauth2/introspect` —
260            // never `/auth/login`), peek the form-encoded body for
261            // `client_id` and restore the payload so the handler's
262            // `web::Form<..>` extractor still sees the full body
263            // untouched. This ALSO stashes the result into request
264            // extensions for the downstream in-memory `Governor`'s
265            // `ClientAwareKeyExtractor` to reuse (see
266            // `extractors::rate_limit::RateLimitClientId`).
267            //
268            // Skipped entirely in `ip` mode (the default) — no body read,
269            // byte-for-byte the pre-D8 code path.
270            let client_id = if client_identity_aware {
271                let client_id = if key_mode != RateLimitKeyMode::Ip {
272                    let bytes = req.extract::<web::Bytes>().await.unwrap_or_default();
273                    let client_id = extract_form_client_id(&bytes);
274                    // Restore the body EXACTLY as read so the handler's
275                    // `web::Form<..>` extraction downstream is unaffected —
276                    // this middleware must be transparent to the request.
277                    req.set_payload(Payload::from(bytes));
278                    client_id
279                } else {
280                    None
281                };
282                req.extensions_mut()
283                    .insert(RateLimitClientId(client_id.clone()));
284                client_id
285            } else {
286                None
287            };
288
289            // The bucket-key "identity" part: IP alone (`ip` mode, or
290            // fail-safe fallback when no client_id was found), the
291            // client_id alone, or the `(ip, client_id)` pair. Mirrors
292            // `extractors::rate_limit::ClientAwareKeyExtractor` exactly so
293            // this shared-store pre-check and the in-memory governor never
294            // disagree about which bucket a request belongs to.
295            let key_part = match (key_mode, &client_id) {
296                (RateLimitKeyMode::Ip, _) | (_, None) => ip.map(|ip| ip.to_string()),
297                (RateLimitKeyMode::ClientId, Some(cid)) => Some(format!("client:{cid}")),
298                (RateLimitKeyMode::IpClientId, Some(cid)) => {
299                    ip.map(|ip| format!("{ip}:client:{cid}"))
300                }
301            };
302
303            let allow = match (key_part, db) {
304                (Some(key_part), Some(db)) => {
305                    let repo = SurrealRateLimitBucketRepository::new(db);
306                    let key = format!("{endpoint}:{key_part}");
307                    let window = window_start(Utc::now());
308                    match repo.increment(&key, window).await {
309                        Ok(count) => count <= limit as u64,
310                        Err(err) => {
311                            // Fail OPEN (D-01b): a counter-store outage must
312                            // never hard-block auth traffic. T-24-43: do NOT
313                            // log the raw key (endpoint:ip/client_id) at
314                            // info+ — this warn-level alarm omits it.
315                            tracing::warn!(
316                                endpoint,
317                                error = %err,
318                                "shared rate-limit store unreachable; falling back \
319                                 to per-replica in-memory governor"
320                            );
321                            true
322                        }
323                    }
324                }
325                // No key part (IP unavailable) or no DB handle available —
326                // fail open; the in-memory governor still makes the real
327                // decision.
328                _ => true,
329            };
330
331            if allow {
332                let res = inner.call(req).await?;
333                Ok(res.map_into_left_body())
334            } else {
335                let res = req.into_response(too_many_requests_response());
336                Ok(res.map_into_right_body())
337            }
338        })
339    }
