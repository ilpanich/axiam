1//! Actix-Web middleware that injects OWASP-recommended security headers into
2//! every HTTP response.
3//!
4//! Headers added:
5//! - `X-Content-Type-Options: nosniff` — prevent MIME-type sniffing
6//! - `X-Frame-Options: DENY` — prevent clickjacking
7//! - `Referrer-Policy: strict-origin-when-cross-origin` — limit referrer leakage
8//! - `Content-Security-Policy` — restrict resource origins (ASVS V14.4.4). The
9//!   policy mirrors the frontend Nginx policy (`docker/nginx.conf`) so the one
10//!   middleware covers both the JSON API responses and the same-origin Swagger UI
11//!   served at `/api/docs/` (swagger-ui 5.x is CSP-friendly: scripts load from
12//!   `'self'`; `style-src` allows `'unsafe-inline'` for its runtime-injected
13//!   styles and `img-src` allows `data:` for its inline icons).
14
15use std::future::{Future, Ready, ready};
16use std::pin::Pin;
17
18use actix_web::Error;
19use actix_web::dev::{Service, ServiceRequest, ServiceResponse, Transform};
20use actix_web::http::header::{HeaderName, HeaderValue};
21
22/// Middleware that appends OWASP security headers to every response.
23pub struct SecurityHeadersMiddleware;
24
25impl<S, B> Transform<S, ServiceRequest> for SecurityHeadersMiddleware
26where
27    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error> + 'static,
28    B: 'static,
29{
30    type Response = ServiceResponse<B>;
31    type Error = Error;
32    type Transform = SecurityHeadersService<S>;
33    type InitError = ();
34    type Future = Ready<Result<Self::Transform, Self::InitError>>;
35
36    fn new_transform(&self, service: S) -> Self::Future {
37        ready(Ok(SecurityHeadersService { inner: service }))
38    }
39}
40
41/// The inner service produced by [`SecurityHeadersMiddleware`].
42pub struct SecurityHeadersService<S> {
43    inner: S,
44}
45
46impl<S, B> Service<ServiceRequest> for SecurityHeadersService<S>
47where
48    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error> + 'static,
49    B: 'static,
50{
51    type Response = ServiceResponse<B>;
52    type Error = Error;
53    type Future = Pin<Box<dyn Future<Output = Result<Self::Response, Self::Error>>>>;
54
55    fn poll_ready(
56        &self,
57        cx: &mut std::task::Context<'_>,
58    ) -> std::task::Poll<Result<(), Self::Error>> {
59        self.inner.poll_ready(cx)
60    }
61
62    fn call(&self, req: ServiceRequest) -> Self::Future {
63        let fut = self.inner.call(req);
64        Box::pin(async move {
65            let mut res = fut.await?;
66            let headers = res.headers_mut();
67            headers.insert(
68                HeaderName::from_static("x-content-type-options"),
69                HeaderValue::from_static("nosniff"),
70            );
71            headers.insert(
72                HeaderName::from_static("x-frame-options"),
73                HeaderValue::from_static("DENY"),
74            );
75            headers.insert(
76                HeaderName::from_static("referrer-policy"),
77                HeaderValue::from_static("strict-origin-when-cross-origin"),
78            );
79            headers.insert(
80                HeaderName::from_static("content-security-policy"),
81                HeaderValue::from_static(
82                    "default-src 'self'; \
83                     script-src 'self'; \
84                     style-src 'self' 'unsafe-inline'; \
85                     img-src 'self' data:; \
86                     frame-ancestors 'none'; \
87                     form-action 'self'; \
88                     base-uri 'self'",
89                ),
90            );
91            Ok(res)
92        })
93    }
94}
