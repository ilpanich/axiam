1//! Actix-Web middleware that injects OWASP-recommended security headers into
2//! every HTTP response.
3//!
4//! Headers added:
5//! - `X-Content-Type-Options: nosniff` — prevent MIME-type sniffing
6//! - `X-Frame-Options: DENY` — prevent clickjacking
7//! - `Referrer-Policy: strict-origin-when-cross-origin` — limit referrer leakage
8
9use std::future::{Future, Ready, ready};
10use std::pin::Pin;
11
12use actix_web::Error;
13use actix_web::dev::{Service, ServiceRequest, ServiceResponse, Transform};
14use actix_web::http::header::{HeaderName, HeaderValue};
15
16/// Middleware that appends OWASP security headers to every response.
17pub struct SecurityHeadersMiddleware;
18
19impl<S, B> Transform<S, ServiceRequest> for SecurityHeadersMiddleware
20where
21    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error> + 'static,
22    B: 'static,
23{
24    type Response = ServiceResponse<B>;
25    type Error = Error;
26    type Transform = SecurityHeadersService<S>;
27    type InitError = ();
28    type Future = Ready<Result<Self::Transform, Self::InitError>>;
29
30    fn new_transform(&self, service: S) -> Self::Future {
31        ready(Ok(SecurityHeadersService { inner: service }))
32    }
33}
34
35/// The inner service produced by [`SecurityHeadersMiddleware`].
36pub struct SecurityHeadersService<S> {
37    inner: S,
38}
39
40impl<S, B> Service<ServiceRequest> for SecurityHeadersService<S>
41where
42    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error> + 'static,
43    B: 'static,
44{
45    type Response = ServiceResponse<B>;
46    type Error = Error;
47    type Future = Pin<Box<dyn Future<Output = Result<Self::Response, Self::Error>>>>;
48
49    fn poll_ready(
50        &self,
51        cx: &mut std::task::Context<'_>,
52    ) -> std::task::Poll<Result<(), Self::Error>> {
53        self.inner.poll_ready(cx)
54    }
55
56    fn call(&self, req: ServiceRequest) -> Self::Future {
57        let fut = self.inner.call(req);
58        Box::pin(async move {
59            let mut res = fut.await?;
60            let headers = res.headers_mut();
61            headers.insert(
62                HeaderName::from_static("x-content-type-options"),
63                HeaderValue::from_static("nosniff"),
64            );
65            headers.insert(
66                HeaderName::from_static("x-frame-options"),
67                HeaderValue::from_static("DENY"),
68            );
69            headers.insert(
70                HeaderName::from_static("referrer-policy"),
71                HeaderValue::from_static("strict-origin-when-cross-origin"),
72            );
73            Ok(res)
74        })
75    }
76}
