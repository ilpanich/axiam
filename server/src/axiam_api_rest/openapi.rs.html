449/// Build the full OpenAPI document, merging in the SAML SP endpoints when the
450/// `saml` feature is enabled. Use this instead of `ApiDoc::openapi()` directly.
451pub fn api_doc() -> utoipa::openapi::OpenApi {
452    let doc = ApiDoc::openapi();
453    #[cfg(feature = "saml")]
454    let doc = doc.merge_from(SamlApiDoc::openapi());
455    doc
456}

458/// Adds Bearer JWT security scheme to the OpenAPI spec.
459struct SecurityAddon;
460
461impl utoipa::Modify for SecurityAddon {
462    fn modify(&self, openapi: &mut utoipa::openapi::OpenApi) {
463        let components = openapi.components.get_or_insert_with(Default::default);
464        components.add_security_scheme(
465            "bearer",
466            utoipa::openapi::security::SecurityScheme::Http(
467                utoipa::openapi::security::HttpBuilder::new()
468                    .scheme(utoipa::openapi::security::HttpAuthScheme::Bearer)
469                    .bearer_format("JWT")
470                    .build(),
471            ),
472        );
473    }
