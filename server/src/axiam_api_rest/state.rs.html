1//! `AppState<C>` — the single composition root for every REST handler
2//! dependency (QUAL-01).
3//!
4//! Before this module, `axiam-server/src/main.rs` registered ~49 individual
5//! `.app_data(web::Data::new(...))` calls and every handler extracted its own
6//! subset of `web::Data<Repo<C>>` / `web::Data<Service<...>>` params. Now
7//! `main.rs` builds ONE `AppState<C>` and registers it once; handlers extract
8//! `web::Data<AppState<C>>` and access dependencies as fields
9//! (`state.user_repo`, `state.auth_service`, ...).
10//!
11//! ## Three dependencies stay registered outside `AppState<C>` (documented deviation)
12//!
13//! `AuthConfig`, `Arc<dyn SessionValidator>`, and `Arc<dyn AuthzChecker>` are
14//! deliberately KEPT as their own standalone `web::Data<T>` registrations in
15//! `main.rs`, in addition to being available as `AppState<C>` fields
16//! (`state.auth_config`, `state.session_validator`) for handler-level
17//! consistency where it is convenient. This is not an oversight:
18//!
19//! - `impl actix_web::FromRequest for AuthenticatedUser`
20//!   (`extractors/auth.rs`) is a fixed, non-generic impl. It cannot know the
21//!   concrete `C` used by `AppState<C>` at the call site, so it cannot do
22//!   `req.app_data::<web::Data<AppState<C>>>()` — there is no `C` in scope.
23//!   It must keep looking up the connection-agnostic `web::Data<AuthConfig>`
24//!   and `web::Data<Arc<dyn SessionValidator>>` types directly.
25//! - `axiam-audit::middleware::AuditMiddleware` (a different crate, wrapping
26//!   the *entire* `App` in `main.rs`, with no dependency on
27//!   `axiam-api-rest::AppState`) independently does
28//!   `req.app_data::<web::Data<AuthConfig>>()`. It has no way to resolve a
29//!   generic `AppState<C>` either.
30//! - 118 handler call sites across the crate use the `AuthzData` type alias
31//!   (`web::Data<Arc<dyn AuthzChecker>>`) directly via
32//!   `authz.get_ref().as_ref()` — none of that call-site shape changes in
33//!   this plan (only the 283 `web::Data<T>` sites named in 29-03-PLAN.md's
34//!   scope note are migrated), so `Arc<dyn AuthzChecker>` must remain
35//!   registered under its own type too.
36//!
37//! `SurrealPasswordResetTokenRepository<C>` and
38//! `SurrealEmailVerificationTokenRepository<C>` were NEVER registered in
39//! `main.rs` at all (a pre-existing bug — see 29-03-SUMMARY.md) despite being
40//! required by `password_reset.rs`/`email_verification.rs`; since both
41//! services are now hoisted singletons, these two repos are constructed
42//! once in `main.rs` purely to build `password_reset_service` /
43//! `email_verification_service` and do not need their own `AppState` field.

81// ---------------------------------------------------------------------------
82// DynMailPublisher — object-safe seam over axiam_core::repository::MailPublisher
83// ---------------------------------------------------------------------------
84//
85// `MailPublisher::publish` returns `impl Future` (RPITIT), which is not
86// dyn-safe on its own. `AppState<C>` needs ONE concrete, non-generic field
87// type that works for both the real AMQP-backed `MailOutboundPublisher`
88// (production — requires a live `lapin::Channel`) and test harnesses (no
89// broker available). This mirrors the exact boxed-future seam
90// `extractors::auth::SessionValidator` already uses for the identical
91// reason. Mechanical call sites are unaffected: `state.mail_outbound_publisher
92// .publish(msg).await` reads identically to the old `mail_publisher.publish(msg).await`.
93
94/// Object-safe wrapper trait over [`axiam_core::repository::MailPublisher`].
95pub trait DynMailPublisher: Send + Sync {
96    fn publish<'a>(
97        &'a self,
98        msg: axiam_core::models::mail::OutboundMailMessage,
99    ) -> Pin<Box<dyn Future<Output = axiam_core::error::AxiamResult<()>> + Send + 'a>>;
100}
101
102impl<T> DynMailPublisher for T
103where
104    T: axiam_core::repository::MailPublisher + Send + Sync,
105{
106    fn publish<'a>(
107        &'a self,
108        msg: axiam_core::models::mail::OutboundMailMessage,
109    ) -> Pin<Box<dyn Future<Output = axiam_core::error::AxiamResult<()>> + Send + 'a>> {
110        Box::pin(axiam_core::repository::MailPublisher::publish(self, msg))
111    }
112}

114/// Always-succeeds no-op mail publisher for test harnesses that don't
115/// exercise a mail-sending route (the common case — see 29-03-SUMMARY.md).
116/// Mirrors the `AllowAllAuthzChecker` test-fixture precedent already

121    fn publish<'a>(
122        &'a self,
123        _msg: axiam_core::models::mail::OutboundMailMessage,

202/// Single composition root for every REST handler dependency (QUAL-01).
203///
204/// Generic bound mirrors `server.rs::register_api_v1_routes<C>` exactly, so
205/// the same `AppState<C>` instantiation backs both production
206/// (`AppState<axiam_db::DbClient>`) and every test harness
207/// (`AppState<surrealdb::engine::local::Db>`).
208///
209/// All fields are `pub` — this struct is an internal DI container, not a
210/// public API surface with invariants to protect; tests need to override
211/// individual fields (e.g. swap `authz_checker`... see note above: that one
212/// stays outside `AppState`, but e.g. a test overriding `health_checker`)
213/// after calling [`AppState::for_test`].

298impl<C: Connection + Clone> AppState<C> {
299    /// Dispatch a domain event to any webhooks subscribed to `event_type` in
300    /// `tenant_id` (CQ-B22). Best-effort: if no AMQP publisher is wired
301    /// (`webhook_publisher` is `None`, e.g. tests or AMQP disabled) this is a
302    /// no-op, and publish failures inside `emit` are logged, never propagated —
303    /// a webhook side effect must not fail the originating API request.
304    pub async fn emit_webhook(
305        &self,
306        tenant_id: uuid::Uuid,
307        event_type: &str,
308        payload: serde_json::Value,
309    ) {
310        if let Some(publisher) = &self.webhook_publisher {
311            self.webhook_delivery
312                .emit(publisher, tenant_id, event_type.to_string(), payload)
313                .await;
314        }
315    }

317    /// Build a fully-populated `AppState<C>` for test harnesses from just a
318    /// `db` handle and an `AuthConfig` — every field gets a working,
319    /// connection-backed default (repos constructed via `::new(db.clone())`,
320    /// services via their real constructors with a fresh crypto semaphore and
321    /// no PKI/webhook/email encryption key configured).
322    ///
323    /// Fields all being `pub` means any test needing to exercise a
324    /// non-default value (a real PKI encryption key, a mock
325    /// `health_checker`, a `RecordingMailPublisher`, ...) can do so with a
326    /// plain field-mutation after calling this:
327    /// ```ignore
