1//! Webhook AMQP consumer — drives `WebhookDeliveryService::deliver_once` for
2//! each queued delivery (CORR-03/D-06), schedules retries natively via the
3//! `axiam.webhook.retry` queue's per-message TTL + DLX (D-07/D-08 — no
4//! in-process wait tying up a consumer slot), and writes per-attempt +
5//! terminal audit records (D-09).
6//!
7//! Lives in `axiam-api-rest` (NOT `axiam-amqp`) per the 26-03 architecture
8//! note: `axiam-amqp` cannot depend on `axiam-federation`/`axiam-auth`/
9//! `axiam-api-rest` without introducing a dependency cycle, and
10//! `deliver_once` needs the shared SSRF guard (SEC-019/SECHRD-02) and
11//! webhook-secret decryption (SEC-031). It uses `lapin::Channel`,
12//! `axiam_amqp::connection::queues::{WEBHOOK,WEBHOOK_RETRY,WEBHOOK_DLQ}`,
13//! `axiam_amqp::WebhookMessage`, and `axiam_amqp::WebhookPublisher` — all
14//! already present in `axiam-amqp` (a dependency of this crate).
15

29// ---------------------------------------------------------------------------
30// Retry config (D-08/D-20)
31// ---------------------------------------------------------------------------
32
33/// Default maximum delivery attempts before a webhook delivery is
34/// dead-lettered to `WEBHOOK_DLQ`. Matches the mail-consumer's convention of
35/// a small, safe default (`mail_consumer::MAX_RETRIES = 3`) scaled slightly
36/// up for webhooks, whose receivers are external/third-party endpoints more
37/// prone to transient outages than the mail-consumer's SMTP relay.
38const DEFAULT_MAX_ATTEMPTS: u32 = 5;
39
40/// Default base backoff (milliseconds) applied to the first retry.
41const DEFAULT_BACKOFF_BASE_MS: u64 = 5_000; // 5s
42
43/// Default backoff ceiling (milliseconds) — no single retry TTL exceeds this.
44const DEFAULT_BACKOFF_CEILING_MS: u64 = 3_600_000; // 1h
45
46/// Exponential backoff multiplier applied per subsequent retry attempt.
47/// Mirrors `mail_consumer::MAIL_RETRY_BACKOFF_MULTIPLIER`.
48const BACKOFF_MULTIPLIER: f64 = 2.0;
49
50/// Config-driven webhook retry policy (D-20): `AXIAM__WEBHOOK__MAX_ATTEMPTS`,
51/// `AXIAM__WEBHOOK__BACKOFF_BASE_MS`, `AXIAM__WEBHOOK__BACKOFF_CEILING_MS`.
52/// Every field has a safe default and is fully overridable — nothing is
53/// mandatory for the server to boot with webhook delivery enabled.
54#[derive(Debug, Clone, Copy, PartialEq, Eq)]
55pub struct WebhookRetryConfig {
56    /// Maximum number of delivery attempts (first attempt counts as 1)
57    /// before a delivery is dead-lettered to `WEBHOOK_DLQ`.
58    pub max_attempts: u32,
59    /// Base backoff (milliseconds) used for the first retry.
60    pub backoff_base_ms: u64,
61    /// Upper bound (milliseconds) any single retry TTL can reach.
62    pub backoff_ceiling_ms: u64,
63}
64
65impl Default for WebhookRetryConfig {
66    fn default() -> Self {
67        Self {
68            max_attempts: DEFAULT_MAX_ATTEMPTS,
69            backoff_base_ms: DEFAULT_BACKOFF_BASE_MS,
70            backoff_ceiling_ms: DEFAULT_BACKOFF_CEILING_MS,
71        }
72    }
73}
74
75impl WebhookRetryConfig {
76    /// Reads `AXIAM__WEBHOOK__MAX_ATTEMPTS`, `AXIAM__WEBHOOK__BACKOFF_BASE_MS`,
77    /// `AXIAM__WEBHOOK__BACKOFF_CEILING_MS` via
78    /// `std::env::var(...).ok().and_then(parse).unwrap_or(default)`, mirroring
79    /// the existing `AXIAM__SECTION__KEY` env-config precedent
80    /// (`axiam-api-grpc::middleware::rate_limit::trusted_hops_from_env`).
81    pub fn from_env() -> Self {
82        let defaults = Self::default();
83        Self {
84            max_attempts: std::env::var("AXIAM__WEBHOOK__MAX_ATTEMPTS")
85                .ok()
86                .and_then(|v| v.parse().ok())
87                .unwrap_or(defaults.max_attempts),
88            backoff_base_ms: std::env::var("AXIAM__WEBHOOK__BACKOFF_BASE_MS")
89                .ok()
90                .and_then(|v| v.parse().ok())
91                .unwrap_or(defaults.backoff_base_ms),
92            backoff_ceiling_ms: std::env::var("AXIAM__WEBHOOK__BACKOFF_CEILING_MS")
93                .ok()
94                .and_then(|v| v.parse().ok())
95                .unwrap_or(defaults.backoff_ceiling_ms),
96        }
97    }
98}

100/// Bounded exponential backoff (D-08): `base_ms * multiplier^(attempt-1)`,
101/// clamped to `[0, ceiling_ms]`. The result becomes the retry queue's
102/// per-message TTL (`WebhookPublisher::publish_retry`'s `ttl_ms`), not an
103/// in-process sleep duration — RabbitMQ's native TTL + DLX schedules the
104/// delay so no consumer slot is held for its duration (D-07).
105///
106/// `attempt` is the *post-increment* attempt number about to be published
107/// (`1` for the first retry, `2` for the second, ...), mirroring
108/// `mail_consumer::backoff_delay_secs`'s convention.
109pub fn backoff_ttl_ms(attempt: u32, cfg: &WebhookRetryConfig) -> u64 {
110    let exponent = attempt.saturating_sub(1) as i32;
111    let delay_ms = cfg.backoff_base_ms as f64 * BACKOFF_MULTIPLIER.powi(exponent);
112    let ceiling_ms = cfg.backoff_ceiling_ms as f64;
113    delay_ms.clamp(0.0, ceiling_ms) as u64
114}

120/// Build the terminal/per-attempt audit entry for a webhook delivery outcome
121/// (D-09). `actor_id` uses the `Uuid::nil()` system-actor convention (no
122/// human/service-account initiated this delivery attempt — mirrors
123/// `mail_consumer`'s and `axiam-federation::secrets`'s existing
124/// `ActorType::System` + nil-actor-id pattern).
125fn build_audit_entry(
126    tenant_id: Uuid,
127    webhook_id: Uuid,
128    action: &str,
129    outcome: AuditOutcome,
130    metadata: serde_json::Value,
131) -> CreateAuditLogEntry {
132    CreateAuditLogEntry {
133        tenant_id,
134        actor_id: Uuid::nil(),
135        actor_type: ActorType::System,
136        action: action.into(),
137        resource_id: Some(webhook_id),
138        outcome,
139        ip_address: None,
140        metadata: Some(metadata),
141    }
142}

144/// Start consuming webhook deliveries from `queues::WEBHOOK` (D-06).
145///
146/// For each dequeued [`WebhookMessage`]:
147/// - Deserialize; a malformed payload is nacked `requeue:false` (bad
148///   payload, not retried forever — mirrors `mail_consumer`'s bad-payload
149///   handling).
150/// - Call `delivery_service.deliver_once` exactly once.
151/// - On a 2xx response: ack + write a terminal `webhook.delivery_succeeded`
152///   audit record.
153/// - On a non-2xx response or `WebhookError`, with `attempt+1 < max_attempts`:
154///   publish a copy (with incremented `attempt`) to `queues::WEBHOOK_RETRY`
155///   with `expiration = backoff_ttl_ms(attempt+1, cfg)`, write a per-attempt
156///   `webhook.delivery_attempt` audit record, then ack the ORIGINAL message
157///   (the retry copy re-enters `queues::WEBHOOK` via TTL+DLX once the delay
158///   expires — no in-process wait ties up this consumer's slot, D-07).
159/// - On exhaustion (`attempt+1 >= max_attempts`): nack `requeue:false` so the
160///   message dead-letters to `queues::WEBHOOK_DLQ` (replayable), and write a
161///   terminal `webhook.delivery_failed` audit record.
162///
163/// SEC-019/SECHRD-02: `deliver_once` is reused unchanged — the SSRF guard is
164/// preserved when delivery is driven from AMQP, not bypassed.
165pub async fn start_webhook_consumer<W, A>(
166    channel: Channel,
167    delivery_service: WebhookDeliveryService<W>,
168    publisher: WebhookPublisher,
169    audit_repo: A,
170    cfg: WebhookRetryConfig,
171) where
172    W: WebhookRepository + Clone + 'static,
173    A: AuditLogRepository + 'static,
174{
175    info!("Starting webhook AMQP consumer");
176
177    let mut consumer = match channel
178        .basic_consume(
179            queues::WEBHOOK.into(),
180            "axiam-webhook-consumer".into(),
181            BasicConsumeOptions::default(),
182            FieldTable::default(),
183        )
184        .await
185    {
186        Ok(c) => c,
187        Err(e) => {
188            error!(error = %e, "Failed to start webhook consumer");
189            return;
190        }
191    };
192
193    while let Some(delivery_result) = consumer.next().await {
194        let delivery = match delivery_result {
195            Ok(d) => d,
196            Err(e) => {
197                error!(error = %e, "Error receiving webhook delivery");
198                continue;
199            }
200        };
201
202        let tag = delivery.delivery_tag;
203
204        // Deserialize. Bad payload -> nack requeue:false (not re-deliverable).
205        let msg: WebhookMessage = match serde_json::from_slice(&delivery.data) {
206            Ok(m) => m,
207            Err(e) => {
208                warn!(
209                    error = %e,
210                    delivery_tag = tag,
211                    "Invalid webhook message payload, nacking"
212                );
213                let _ = delivery
214                    .acker
215                    .nack(BasicNackOptions {
216                        requeue: false,
217                        ..BasicNackOptions::default()
218                    })
219                    .await;
220                continue;
221            }
222        };
223
224        let result = delivery_service
225            .deliver_once(
226                msg.tenant_id,
227                msg.webhook_id,
228                msg.delivery_id,
229                &msg.event_type,
230                &msg.payload,
231            )
232            .await;
233
234        match result {
235            Ok(status) if status.is_success() => {
236                let entry = build_audit_entry(
237                    msg.tenant_id,
238                    msg.webhook_id,
239                    "webhook.delivery_succeeded",
240                    AuditOutcome::Success,
241                    serde_json::json!({
242                        "delivery_id": msg.delivery_id,
243                        "attempt": msg.attempt + 1,
244                        "status": status.as_u16(),
245                    }),
246                );
247                if let Err(e) = audit_repo.append(entry).await {
248                    error!(error = %e, "Failed to write webhook.delivery_succeeded audit event");
249                }
250                if let Err(e) = delivery.acker.ack(BasicAckOptions::default()).await {
251                    error!(error = %e, delivery_tag = tag, "Failed to ack webhook delivery");
252                }
253            }
254            Ok(status) => {
255                handle_delivery_failure(
256                    &msg,
257                    &publisher,
258                    &audit_repo,
259                    &cfg,
260                    format!("non-2xx status: {}", status.as_u16()),
261                    tag,
262                    &delivery.acker,
263                )
264                .await;
265            }
266            Err(e) => {
267                handle_delivery_failure(
268                    &msg,
269                    &publisher,
270                    &audit_repo,
271                    &cfg,
272                    e.to_string(),
273                    tag,
274                    &delivery.acker,
275                )
276                .await;
277            }
278        }
279    }
280
281    warn!("Webhook AMQP consumer stream ended");
282}

284/// Shared failure-path handling for both a non-2xx delivery response and a
285/// `WebhookError` (SSRF-blocked, secret-decrypt failure, lookup failure,
286/// etc.) — both are "this attempt did not succeed" and follow the same
287/// retry-or-exhaust decision (D-07/D-09).
288async fn handle_delivery_failure<A>(
289    msg: &WebhookMessage,
290    publisher: &WebhookPublisher,
291    audit_repo: &A,
292    cfg: &WebhookRetryConfig,
293    error_detail: String,
294    delivery_tag: DeliveryTag,
295    acker: &Acker,
296) where
297    A: AuditLogRepository + 'static,
298{
299    let next_attempt = msg.attempt + 1;
300
301    if next_attempt < cfg.max_attempts {
302        let ttl_ms = backoff_ttl_ms(next_attempt, cfg);
303        let mut retry_msg = msg.clone();
304        retry_msg.attempt = next_attempt;
305
306        // CQ-B49: if the retry copy fails to enqueue, the original message must
307        // NOT be acked — acking it would drop the delivery entirely (no retry,
308        // no DLQ) while the audit trail falsely claims a retry was scheduled.
309        // Nack with requeue so the broker redelivers the original for another
310        // attempt, and skip the "retry scheduled" audit record.
311        if let Err(e) = publisher.publish_retry(&retry_msg, ttl_ms).await {
312            error!(
313                error = %e,
314                webhook_id = %msg.webhook_id,
315                delivery_id = %msg.delivery_id,
316                delivery_tag,
317                "Failed to publish webhook retry — requeuing original instead of acking"
318            );
319            if let Err(nack_err) = acker
320                .nack(BasicNackOptions {
321                    requeue: true,
322                    ..BasicNackOptions::default()
323                })
324                .await
325            {
326                error!(
327                    error = %nack_err,
328                    delivery_tag,
329                    "Failed to nack original webhook delivery after retry-publish failure"
330                );
331            }
332            return;
333        }
334
335        let entry = build_audit_entry(
336            msg.tenant_id,
337            msg.webhook_id,
338            "webhook.delivery_attempt",
339            AuditOutcome::Failure,
340            serde_json::json!({
341                "delivery_id": msg.delivery_id,
342                "attempt": next_attempt,
343                "error": error_detail,
344                "next_retry_in_ms": ttl_ms,
345            }),
346        );
347        if let Err(e) = audit_repo.append(entry).await {
348            error!(error = %e, "Failed to write webhook.delivery_attempt audit event");
349        }
350
351        // Ack the ORIGINAL message — the retry copy has been queued onto
352        // WEBHOOK_RETRY and will re-enter WEBHOOK via TTL+DLX (D-07).
353        if let Err(e) = acker.ack(BasicAckOptions::default()).await {
354            error!(error = %e, delivery_tag, "Failed to ack original webhook delivery");
355        }
356    } else {
357        warn!(
358            webhook_id = %msg.webhook_id,
359            delivery_id = %msg.delivery_id,
360            attempt = next_attempt,
361            max_attempts = cfg.max_attempts,
362            error = %error_detail,
363            "Webhook delivery exhausted retries — dead-lettering"
364        );
365
366        let entry = build_audit_entry(
367            msg.tenant_id,
368            msg.webhook_id,
369            "webhook.delivery_failed",
370            AuditOutcome::Failure,
371            serde_json::json!({
372                "delivery_id": msg.delivery_id,
373                "attempt": next_attempt,
374                "error": error_detail,
375                "next_retry_in_ms": null,
376            }),
377        );
378        if let Err(e) = audit_repo.append(entry).await {
379            error!(error = %e, "Failed to write webhook.delivery_failed audit event");
380        }
381
382        // Terminal exhaustion -> nack requeue:false -> WEBHOOK's own DLX ->
383        // WEBHOOK_DLQ (replayable, per 26-03's topology).
384        let _ = acker
385            .nack(BasicNackOptions {
386                requeue: false,
387                ..BasicNackOptions::default()
388            })
389            .await;
390    }
391}

397#[cfg(test)]
398mod webhook_consumer_tests {
399    use super::*;
400
401    #[test]
402    fn backoff_ttl_ms_nonzero_at_attempt_1() {
403        let cfg = WebhookRetryConfig::default();
404        assert!(
405            backoff_ttl_ms(1, &cfg) > 0,
406            "first retry TTL must not be zero-delay"
407        );
408    }
409
410    #[test]
411    fn backoff_ttl_ms_increases_until_ceiling() {
412        let cfg = WebhookRetryConfig::default();
413        let first = backoff_ttl_ms(1, &cfg);
414        let second = backoff_ttl_ms(2, &cfg);
415        let third = backoff_ttl_ms(3, &cfg);
416        assert!(second > first, "backoff must increase between attempts");
417        assert!(third > second, "backoff must increase between attempts");
418    }
419
420    #[test]
421    fn backoff_ttl_ms_clamped_to_ceiling() {
422        let cfg = WebhookRetryConfig::default();
423        let delay = backoff_ttl_ms(1_000, &cfg);
424        assert!(
425            delay <= cfg.backoff_ceiling_ms,
426            "backoff TTL must never exceed the ceiling, got {delay}"
427        );
428    }
429
430    #[test]
431    fn backoff_ttl_ms_never_negative_defensively() {
432        let cfg = WebhookRetryConfig::default();
433        // attempt = 0 is defensive (the retry branch always passes attempt >= 1).
434        let delay = backoff_ttl_ms(0, &cfg);
435        // u64 cannot be negative; assert it is well-formed (no panic/overflow).
436        assert!(delay <= cfg.backoff_ceiling_ms);
437    }
438
439    #[test]
440    fn webhook_retry_config_defaults_resolve_when_env_unset() {
441        // AXIAM__WEBHOOK__* is unique to this module — no other test in this
442        // crate reads or writes these vars, so removing them here cannot
443        // race with unrelated tests running in parallel in the same binary.
444        unsafe {
445            std::env::remove_var("AXIAM__WEBHOOK__MAX_ATTEMPTS");
446            std::env::remove_var("AXIAM__WEBHOOK__BACKOFF_BASE_MS");
447            std::env::remove_var("AXIAM__WEBHOOK__BACKOFF_CEILING_MS");
448        }
449        let cfg = WebhookRetryConfig::from_env();
450        let defaults = WebhookRetryConfig::default();
451        assert_eq!(cfg, defaults);
452    }
