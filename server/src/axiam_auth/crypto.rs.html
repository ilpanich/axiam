1//! Domain-neutral AES-256-GCM encryption helpers.
2//!
3//! Provides two variants:
4//!
5//! - **Bundled** (`aes256gcm_encrypt` / `aes256gcm_decrypt`): encodes
6//!   `nonce || ciphertext || tag` as a single base64 string. Used by TOTP
7//!   secret storage.
8//! - **Split-output** (`encrypt_separate` / `decrypt_separate`): returns
9//!   nonce and ciphertext+tag as separate base64 strings. Used by federation
10//!   config storage (D-11) where each piece lives in its own DB column.
11//!
12//! Both variants use a fresh 12-byte nonce from `OsRng` and AES-256-GCM.
13
14use aes_gcm::aead::consts::U12;
15use aes_gcm::aead::{Aead, Generate, KeyInit};
16use aes_gcm::{Aes256Gcm, Key, Nonce};
17use base64::Engine;
18use base64::engine::general_purpose::STANDARD;
19use hmac::{Hmac, Mac};
20use sha2::Sha256;
21use uuid::Uuid;
22
23use crate::error::AuthError;
24
25type HmacSha256 = Hmac<Sha256>;

31fn build_cipher(key: &[u8; 32]) -> Aes256Gcm {
32    Aes256Gcm::new(&Key::<Aes256Gcm>::from(*key))
33}
34
35// ---------------------------------------------------------------------------
36// Bundled variant (TOTP wire format — MUST NOT change)
37// ---------------------------------------------------------------------------
38
39/// Encrypt `plaintext` with AES-256-GCM.
40///
41/// Returns `base64(nonce(12) || ciphertext || tag)`.
42///
43/// This is the bundled-format variant. The wire format is shared with
44/// TOTP secret storage — do **not** change the layout.
45pub fn aes256gcm_encrypt(key: &[u8; 32], plaintext: &[u8]) -> Result<String, AuthError> {
46    let cipher = build_cipher(key);
47    let nonce_bytes: [u8; 12] = Generate::generate();
48    let nonce = Nonce::<U12>::from(nonce_bytes);
49
50    let ciphertext = cipher
51        .encrypt(&nonce, plaintext)
52        .map_err(|e| AuthError::Crypto(format!("AES-GCM encrypt: {e}")))?;
53
54    let mut combined = nonce_bytes.to_vec();
55    combined.extend_from_slice(&ciphertext);
56    Ok(STANDARD.encode(combined))
57}
58
59/// Decrypt AES-256-GCM ciphertext in the bundled format.
60///
61/// Accepts `base64(nonce(12) || ciphertext || tag)`.
62pub fn aes256gcm_decrypt(key: &[u8; 32], encoded: &str) -> Result<Vec<u8>, AuthError> {
63    let combined = STANDARD
64        .decode(encoded)
65        .map_err(|e| AuthError::Crypto(format!("base64 decode: {e}")))?;
66
67    if combined.len() < 13 {
68        return Err(AuthError::Crypto("ciphertext too short".into()));
69    }
70
71    let (nonce_bytes, ciphertext) = combined.split_at(12);
72    let cipher = build_cipher(key);
73    let nonce = Nonce::<U12>::try_from(nonce_bytes)
74        .map_err(|_| AuthError::Crypto("invalid nonce length".into()))?;
75
76    cipher
77        .decrypt(&nonce, ciphertext)
78        .map_err(|e| AuthError::Crypto(format!("AES-GCM decrypt: {e}")))
79}

81// ---------------------------------------------------------------------------
82// Split-output variant (federation_config storage — D-11)
83// ---------------------------------------------------------------------------
84
85/// Encrypt `plaintext` with AES-256-GCM, returning nonce and ciphertext
86/// as **separate** base64 strings.
87///
88/// Returns `(nonce_b64, ciphertext_with_tag_b64)`. The ciphertext output
89/// is `ciphertext || tag` (no nonce prefix). Each output can be stored in
90/// its own DB column, enabling column-level key rotation without re-encoding.
91///
92/// # Separation from bundled format
93///
94/// Values encrypted with this function **cannot** be decrypted by
95/// [`aes256gcm_decrypt`] — the formats are intentionally incompatible.
96/// Always use matching encrypt/decrypt pairs.
97pub fn encrypt_separate(key: &[u8; 32], plaintext: &[u8]) -> Result<(String, String), AuthError> {
98    let cipher = build_cipher(key);
99    let nonce_bytes: [u8; 12] = Generate::generate();
100    let nonce = Nonce::<U12>::from(nonce_bytes);
101
102    let ciphertext = cipher
103        .encrypt(&nonce, plaintext)
104        .map_err(|e| AuthError::Crypto(format!("AES-GCM encrypt: {e}")))?;
105
106    let nonce_b64 = STANDARD.encode(nonce_bytes);
107    let ct_b64 = STANDARD.encode(&ciphertext);
108    Ok((nonce_b64, ct_b64))
109}
110
111/// Decrypt AES-256-GCM ciphertext from the split-output format.
112///
113/// Accepts separately base64-encoded `nonce_b64` and `ciphertext_b64`
114/// (where `ciphertext_b64` is `ciphertext || tag`, no nonce prefix).
115pub fn decrypt_separate(
116    key: &[u8; 32],
117    nonce_b64: &str,
118    ciphertext_b64: &str,
119) -> Result<Vec<u8>, AuthError> {
120    let nonce_bytes = STANDARD
121        .decode(nonce_b64)
122        .map_err(|e| AuthError::Crypto(format!("nonce base64 decode: {e}")))?;
123    let ciphertext = STANDARD
124        .decode(ciphertext_b64)
125        .map_err(|e| AuthError::Crypto(format!("ciphertext base64 decode: {e}")))?;
126
127    if nonce_bytes.len() != 12 {
128        return Err(AuthError::Crypto(format!(
129            "nonce must be 12 bytes, got {}",
130            nonce_bytes.len()
131        )));
132    }
133
134    let cipher = build_cipher(key);
135    let nonce = Nonce::<U12>::try_from(nonce_bytes.as_slice())
136        .map_err(|_| AuthError::Crypto("nonce must be 12 bytes".into()))?;
137
138    cipher
139        .decrypt(&nonce, ciphertext.as_slice())
140        .map_err(|e| AuthError::Crypto(format!("AES-GCM decrypt: {e}")))
141}

143// ---------------------------------------------------------------------------
144// GDPR pseudonym helper (D-02)
145// ---------------------------------------------------------------------------
146
147/// Compute a deterministic, keyed GDPR audit pseudonym for a deleted user.
148///
149/// Returns `"DELETED_USER_{16-char-hex}"` (64 bits from HMAC-SHA256).
150///
151/// - **Deterministic**: the same `(pepper, tenant_id, user_id)` triple always
152///   produces the same pseudonym, so all audit entries for a deleted user can
153///   still be correlated post-erasure without re-identifying the subject.
154/// - **Brute-force resistant**: the pepper must be secret. Without it, an
155///   attacker cannot map candidate `user_id` values back to a pseudonym.
156/// - **Per-tenant**: `tenant_id` is part of the HMAC input, so the same
157///   `user_id` in two different tenants produces different pseudonyms.
158pub fn gdpr_pseudonym(pepper: &[u8; 32], tenant_id: Uuid, user_id: Uuid) -> String {
159    let mut mac =
160        <HmacSha256 as hmac::KeyInit>::new_from_slice(pepper).expect("HMAC accepts any key length");
161    mac.update(tenant_id.as_bytes());
162    mac.update(user_id.as_bytes());
163    let tag = mac.finalize().into_bytes();
164    format!("DELETED_USER_{}", hex::encode(&tag[..8]))
165}

171#[cfg(test)]
172mod tests {
173    use super::*;
174
175    #[test]
176    fn bundled_round_trip() {
177        let key = [0x42u8; 32];
178        let plaintext = b"hello-axiam";
179        let enc = aes256gcm_encrypt(&key, plaintext).unwrap();
180        let dec = aes256gcm_decrypt(&key, &enc).unwrap();
181        assert_eq!(dec, plaintext);
182    }
183
184    #[test]
185    fn separate_round_trip() {
186        let key = [0x55u8; 32];
187        let plaintext = b"federation-secret";
188        let (nonce_b64, ct_b64) = encrypt_separate(&key, plaintext).unwrap();
189        assert!(!nonce_b64.is_empty());
190        assert!(!ct_b64.is_empty());
191        let dec = decrypt_separate(&key, &nonce_b64, &ct_b64).unwrap();
192        assert_eq!(dec, plaintext);
193    }
194
195    #[test]
196    fn separate_wrong_nonce_fails() {
197        let key = [0xAAu8; 32];
198        let plaintext = b"sensitive-data";
199        let (_, ct_b64) = encrypt_separate(&key, plaintext).unwrap();
200        // Generate a different nonce
201        let (other_nonce, _) = encrypt_separate(&key, b"other").unwrap();
202        let result = decrypt_separate(&key, &other_nonce, &ct_b64);
203        assert!(result.is_err(), "decryption with wrong nonce must fail");
204    }
205
206    #[test]
207    fn gdpr_pseudonym_deterministic() {
208        let pepper = [0x42u8; 32];
209        let tenant = Uuid::nil();
210        let user = Uuid::new_v4();
211        let p1 = gdpr_pseudonym(&pepper, tenant, user);
212        let p2 = gdpr_pseudonym(&pepper, tenant, user);
213        assert_eq!(p1, p2, "same inputs must produce the same pseudonym");
214        assert!(
215            p1.starts_with("DELETED_USER_"),
216            "format must match DELETED_USER_<hex>"
217        );
218        assert_eq!(
219            p1.len(),
220            "DELETED_USER_".len() + 16,
221            "must be 16 hex chars (8 bytes)"
222        );
223    }
224
225    #[test]
226    fn gdpr_pseudonym_differs_for_different_users() {
227        let pepper = [0x42u8; 32];
228        let tenant = Uuid::nil();
229        let user1 = Uuid::new_v4();
230        let user2 = Uuid::new_v4();
231        let p1 = gdpr_pseudonym(&pepper, tenant, user1);
232        let p2 = gdpr_pseudonym(&pepper, tenant, user2);
233        assert_ne!(
234            p1, p2,
235            "different user_ids must produce different pseudonyms"
236        );
237    }

239    #[test]
240    fn gdpr_pseudonym_differs_for_different_tenants() {
241        let pepper = [0x42u8; 32];
242        let user = Uuid::new_v4();
243        let t1 = Uuid::new_v4();
244        let t2 = Uuid::new_v4();
245        let p1 = gdpr_pseudonym(&pepper, t1, user);
246        let p2 = gdpr_pseudonym(&pepper, t2, user);
247        assert_ne!(
248            p1, p2,
249            "different tenant_ids must produce different pseudonyms"
250        );
251    }
252
253    #[test]
254    fn bundled_and_separate_are_independent_formats() {
255        let key = [0x11u8; 32];
256        let plaintext = b"format-check";
257        // Encrypt with bundled format: nonce || ct+tag all in one base64
258        let bundled = aes256gcm_encrypt(&key, plaintext).unwrap();
259        // Treat the entire bundled string as if it were the ct_b64 in split format.
260        // The nonce we pass is independent of the real nonce in the bundled blob.
261        // This MUST fail (wrong nonce / mangled ciphertext).
262        let fake_nonce = STANDARD.encode([0u8; 12]);
263        let result = decrypt_separate(&key, &fake_nonce, &bundled);
264        assert!(
265            result.is_err(),
266            "bundled format must not be decryptable by decrypt_separate with a zero nonce"
267        );
268    }
