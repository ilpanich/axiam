1//! Process-wide circuit breaker guarding [`crate::policy::check_hibp`]
2//! (PERF-01, D-01/D-02/D-03/D-04).
3//!
4//! `check_hibp` is ALREADY fail-open on every error branch — this module does
5//! NOT add fail-open behavior. Its sole purpose is burst-protection: once a
6//! sustained run of HIBP failures/timeouts is observed, subsequent calls
7//! short-circuit to `Ok(None)` immediately, skipping the outbound HTTP
8//! request (and its 5s timeout) entirely for the cooldown window. This
9//! prevents a credential-stuffing burst against a downed/rate-limiting HIBP
10//! endpoint from starving legitimate auth flows.
11
12use std::sync::{Mutex, OnceLock};
13use std::time::{Duration, Instant};
14
15/// Internal breaker state machine.
16///
17/// `std::sync::Mutex` (not `tokio::sync::Mutex`) is used because the
18/// critical section is a few field reads/writes with no `.await` inside it —
19/// matching the lock-type convention in `axiam-federation`'s `jwks_cache.rs`
20/// (choose lock type by hold-duration/await-need).
21#[derive(Debug, Clone, Copy)]
22enum BreakerState {
23    Closed { consecutive_failures: u32 },
24    Open { opened_at: Instant },
25}
26
27/// Hand-rolled circuit breaker (D-02 — no new circuit-breaker crate
28/// dependency) around `check_hibp`'s outbound HTTP call.
29pub struct HibpBreaker {
30    state: Mutex<BreakerState>,
31    threshold: u32,
32    cooldown: Duration,
33}
34
35impl HibpBreaker {
36    /// Construct a new breaker with the given threshold (consecutive
37    /// failures before tripping) and cooldown (seconds the breaker stays
38    /// open before allowing a half-open probe).
39    pub fn new(threshold: u32, cooldown_secs: u64) -> Self {
40        Self {
41            state: Mutex::new(BreakerState::Closed {
42                consecutive_failures: 0,
43            }),
44            threshold,
45            cooldown: Duration::from_secs(cooldown_secs),
46        }
47    }
48
49    /// Returns `true` if the call should proceed (breaker Closed, or Open but
50    /// the cooldown has elapsed — a single half-open probe). Returns `false`
51    /// if the caller should short-circuit to `Ok(None)` WITHOUT making the
52    /// HTTP call at all, saving the 5s timeout under a sustained failure
53    /// burst.
54    pub fn should_attempt(&self) -> bool {
55        let state = self.state.lock().unwrap();
56        match *state {
57            BreakerState::Closed { .. } => true,
58            BreakerState::Open { opened_at } => opened_at.elapsed() >= self.cooldown,
59        }
60    }
61
62    /// Record a successful HIBP check — re-closes the breaker.
63    pub fn record_success(&self) {
64        *self.state.lock().unwrap() = BreakerState::Closed {
65            consecutive_failures: 0,
66        };
67    }
68
69    /// Record a failed/timed-out HIBP check. Increments the consecutive
70    /// failure count while Closed, tripping to Open once the threshold is
71    /// reached. While already Open (i.e. a half-open probe just failed),
72    /// re-opens with a fresh `opened_at`, resetting the cooldown clock.
73    pub fn record_failure(&self) {
74        let mut state = self.state.lock().unwrap();
75        match *state {
76            BreakerState::Closed {
77                consecutive_failures,
78            } => {
79                let n = consecutive_failures + 1;
80                *state = if n >= self.threshold {
81                    BreakerState::Open {
82                        opened_at: Instant::now(),
83                    }
84                } else {
85                    BreakerState::Closed {
86                        consecutive_failures: n,
87                    }
88                };
89            }
90            BreakerState::Open { .. } => {
91                *state = BreakerState::Open {
92                    opened_at: Instant::now(),
93                };
94            }
95        }
96    }
97}
98
99/// Process-wide global breaker instance (D-01 — one global breaker, not
100/// per-tenant).
101static GLOBAL: OnceLock<HibpBreaker> = OnceLock::new();
102
103/// Idempotent global initializer. Should be called once at startup from
104/// `AuthConfig`'s resolved `hibp_breaker_threshold`/`hibp_breaker_cooldown_secs`.
105/// Calling this more than once has no effect after the first call
106/// (`OnceLock::get_or_init` semantics).
107pub fn init_global(threshold: u32, cooldown_secs: u64) {
108    GLOBAL.get_or_init(|| HibpBreaker::new(threshold, cooldown_secs));
109}
110
111/// Access the process-wide global breaker, lazily initializing with defaults
112/// (threshold 5, cooldown 30s) if `init_global` was never called.
113pub fn global() -> &'static HibpBreaker {
114    GLOBAL.get_or_init(|| HibpBreaker::new(5, 30))
115}

117#[cfg(test)]
118mod tests {
119    use super::*;
120
121    #[test]
122    fn closed_allows_attempts() {
123        let b = HibpBreaker::new(5, 30);
124        assert!(b.should_attempt());
125    }
126
127    #[test]
128    fn trips_after_exactly_threshold_failures() {
129        let b = HibpBreaker::new(3, 30);
130        // 2 failures: still closed, still allows attempts.
131        b.record_failure();
132        assert!(b.should_attempt());
133        b.record_failure();
134        assert!(b.should_attempt());
135        // 3rd failure trips the breaker (with a non-zero cooldown, so the
136        // just-tripped state is still within its cooldown window).
137        b.record_failure();
138        assert!(!b.should_attempt());
139    }
140
141    #[test]
142    fn short_circuits_within_cooldown() {
143        // Large cooldown so we can assert `should_attempt` stays false
144        // without any wall-clock sleep.
145        let b = HibpBreaker::new(1, 30);
146        b.record_failure(); // trips immediately (threshold 1)
147        assert!(!b.should_attempt());
148        assert!(!b.should_attempt());
149    }
150
151    #[test]
152    fn allows_one_probe_after_cooldown_elapsed() {
153        // Zero-second cooldown: elapsed() >= Duration::from_secs(0) is true
154        // on the very next check, with no sleep required.
155        let b = HibpBreaker::new(1, 0);
156        b.record_failure(); // trips
157        // Cooldown of 0 means the very next should_attempt() is the
158        // half-open probe.
159        assert!(b.should_attempt());
160        // Breaker remains Open (not yet re-closed) until record_success().
161        assert!(b.should_attempt());
162    }
163
164    #[test]
165    fn record_success_recloses_breaker() {
166        let b = HibpBreaker::new(1, 30);
167        b.record_failure(); // trips (Open, long cooldown)
168        assert!(!b.should_attempt());
169        b.record_success();
170        assert!(b.should_attempt());
171    }
172
173    #[test]
174    fn record_failure_while_open_resets_opened_at() {
175        // Long cooldown so the breaker stays Open (not a half-open probe)
176        // across both failures; directly inspect the private opened_at
177        // Instant (accessible from this submodule) to prove it advances
178        // rather than sleeping 30s.
179        let b = HibpBreaker::new(1, 30);
180        b.record_failure(); // trips -> Open { opened_at: t0 }
181        let opened_at_1 = match *b.state.lock().unwrap() {
182            BreakerState::Open { opened_at } => opened_at,
183            BreakerState::Closed { .. } => panic!("expected Open state"),
184        };
185
186        // Still within cooldown, so should_attempt() short-circuits and a
187        // "probe" isn't actually in flight yet — but record_failure() while
188        // Open must still reset the clock regardless of half-open status.
189        b.record_failure();
190        let opened_at_2 = match *b.state.lock().unwrap() {
191            BreakerState::Open { opened_at } => opened_at,
192            BreakerState::Closed { .. } => panic!("expected still Open state"),
193        };
194
195        assert!(
196            opened_at_2 >= opened_at_1,
197            "opened_at must be reset (monotonically non-decreasing) on failure while Open"
198        );
199        assert!(
200            !b.should_attempt(),
201            "still within the (long) cooldown window"
202        );
203    }
204
205    #[test]
206    fn default_threshold_and_cooldown() {
207        let b = HibpBreaker::new(5, 30);
208        assert_eq!(b.threshold, 5);
209        assert_eq!(b.cooldown, Duration::from_secs(30));
210    }
211
212    #[test]
213    fn global_lazily_initializes_with_defaults() {
214        // NOTE: GLOBAL is process-wide and shared across all tests in this
215        // binary; only assert non-panicking access here, not specific
216        // defaults, since another test in this module may have already
217        // called init_global.
218        let g = global();
219        assert!(g.should_attempt() || !g.should_attempt());
220    }
221
222    #[test]
223    fn init_global_is_idempotent() {
224        init_global(7, 42);
225        // Second call is a no-op — must not panic.
226        init_global(99, 999);
227    }
