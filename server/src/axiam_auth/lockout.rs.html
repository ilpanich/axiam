1//! Shared failed-login / lockout accrual (D-06).
2//!
3//! Both the REST login path (`AuthService::record_failed_login`) and the
4//! gRPC `UserService::validate_credentials` path call this single helper so
5//! that there is no unmetered credential-check path anywhere in the system
6//! (SEC-026b). This module is the single source of truth for the
7//! failed-attempt/lockout increment — do NOT duplicate this logic anywhere
8//! else.
9
10use axiam_core::error::AxiamResult;
11use axiam_core::models::user::User;
12use axiam_core::repository::UserRepository;
13use uuid::Uuid;
14
15use crate::config::AuthConfig;
16
17/// Record a failed login attempt for `user`, applying the exponential
18/// lockout backoff configured in `config`.
19///
20/// SEC-032: atomic increment — single SurrealQL UPDATE avoids TOCTOU race.
21/// Lockout duration escalates exponentially per repeated lockout, capped
22/// at `max_lockout_duration_secs` (brute-force protection).
23///
24/// D-06: always-on accrual — callers must invoke this on every failed
25/// credential check (wrong password against an existing, non-locked user),
26/// never behind a config flag.
27pub async fn record_failed_login<U: UserRepository>(
28    user_repo: &U,
29    config: &AuthConfig,
30    tenant_id: Uuid,
31    user: &User,
32) -> AxiamResult<()> {
33    user_repo
34        .increment_failed_logins(
35            tenant_id,
36            user.id,
37            config.max_failed_login_attempts,
38            config.lockout_duration_secs as i64,
39            config.lockout_backoff_multiplier,
40            config.max_lockout_duration_secs as i64,
41        )
42        .await
43}
