23/// Maximum password reset requests per user per day.
24const MAX_RESETS_PER_DAY: u64 = 3;

26/// Password reset service.
27///
28/// Handles reset token generation with rate limiting and prior-token
29/// invalidation, plus reset confirmation with password policy
30/// enforcement, fail2ban counter reset, and password history storage.
31///
32/// `S` is the session repository — used by `confirm_reset` to invalidate ALL
33/// active sessions after a successful password reset (D-16 — caller is
34/// unauthenticated so there is no current session to preserve).
35///
36/// `T` is the OAuth2 refresh-token repository — used by `confirm_reset` to
37/// also revoke OAuth2-flow refresh tokens (RESEARCH §4 "two chokepoints").
38///
39/// `Clone` (QUAL-07): this service is now a hoisted `AppState<C>` singleton
40/// constructed once at startup and cloned per Actix worker, rather than
41/// rebuilt per-request.
42#[derive(Clone)]
43pub struct PasswordResetService<U, R, F, H, S, T>
44where
45    U: UserRepository,
46    R: PasswordResetTokenRepository,
47    F: FederationLinkRepository,
48    H: PasswordHistoryRepository,
49    S: SessionRepository,
50    T: RefreshTokenRepository,
51{
52    user_repo: U,
53    token_repo: R,
54    federation_repo: F,
55    history_repo: H,
56    session_repo: S,
57    refresh_token_repo: T,
58    /// Bounding semaphore (CQ-B02, mirrors `AuthService`): limits concurrent
59    /// Argon2 operations (including the T-24-91 dummy-hash timing
60    /// equalization below and the T-24-92 current-password check) to
61    /// prevent CPU-bound crypto from starving the Tokio async runtime.
62    crypto_semaphore: Arc<Semaphore>,
63}
64

71    S: SessionRepository,
72    T: RefreshTokenRepository,

94    /// Constant-time dummy Argon2 verify (T-24-91).
95    ///
96    /// Mirrors the SEC-026 pattern already live in `AuthService::login`
97    /// exactly: acquire the shared `crypto_semaphore`, run a dummy
98    /// Argon2id verify against the relocated `DUMMY_HASH` constant inside
99    /// `spawn_blocking`, and discard the result. Called from BOTH of
100    /// `initiate_reset`'s `Ok(None)` branches (unknown email, federated
101    /// user) so those responses are time-indistinguishable from the
102    /// valid-account branch — no hand-tuned fixed-duration sleep.
103    async fn dummy_hash_wait(&self, pepper: Option<&str>) {
104        let _permit = self.crypto_semaphore.acquire().await.ok();
105        let pepper_owned = pepper.map(str::to_string);
106        let _ = tokio::task::spawn_blocking(move || {
107            password::verify_password("dummy", DUMMY_HASH, pepper_owned.as_deref())
108        })
109        .await;
110    }

112    /// Initiate a password reset for the given email.
113    ///
114    /// Returns `Ok(Some((raw_token, user_id, expires_at)))` on success,
115    /// `Ok(None)` if the email doesn't exist or user is federated
116    /// (to prevent user enumeration).
117    ///
118    /// Returns `Err(RateLimited)` if the daily limit is exceeded.
119    ///
120    /// `pepper` is threaded through so the unknown-email and federated-user
121    /// branches can run the T-24-91 constant-time dummy Argon2 verify with
122    /// the same pepper the real hashing/verification call sites use.
123    pub async fn initiate_reset(
124        &self,
125        tenant_id: Uuid,
126        email: &str,
127        expiry_hours: u32,
128        pepper: Option<&str>,
129    ) -> AxiamResult<Option<(String, Uuid, chrono::DateTime<chrono::Utc>)>> {
130        // Look up user — silently return None if not found.
131        let user = match self.user_repo.get_by_email(tenant_id, email).await {
132            Ok(u) => u,
133            Err(AxiamError::NotFound { .. }) => {
134                // T-24-91: constant-time dummy Argon2 verify so an unknown
135                // email is time-indistinguishable from a valid one.
136                self.dummy_hash_wait(pepper).await;
137                return Ok(None);
138            }
139            Err(e) => return Err(e),
140        };
141
142        // Federated users cannot reset passwords — return None to
143        // prevent enumeration.
144        let links = self
145            .federation_repo
146            .get_by_user_id(tenant_id, user.id)
147            .await?;
148        if !links.is_empty() {
149            // T-24-91: same constant-time treatment as the unknown-email
150            // branch above — a federated account must not resolve faster.
151            self.dummy_hash_wait(pepper).await;
152            return Ok(None);
153        }
154
155        // Rate limit: max resets per day.
156        let count = self.token_repo.count_today(tenant_id, user.id).await?;
157        if count >= MAX_RESETS_PER_DAY {
158            return Err(AxiamError::RateLimited);
159        }
160
161        // Invalidate any prior unconsumed tokens for this user so
162        // intercepted older emails cannot be used after a newer
163        // reset request.
164        self.token_repo
165            .delete_unconsumed_for_user(tenant_id, user.id)
166            .await?;
167
168        let raw_token = token::generate_refresh_token();
169        let token_hash = token::hash_refresh_token(&raw_token);
170        let expires_at = Utc::now() + Duration::hours(expiry_hours as i64);
171
172        self.token_repo
173            .create(CreatePasswordResetToken {
174                tenant_id,
175                user_id: user.id,
176                token_hash,
177                expires_at,
178            })
179            .await?;
180
181        Ok(Some((raw_token, user.id, expires_at)))
182    }

184    /// Confirm a password reset using the raw token and a new password.
185    ///
186    /// Validates the token, enforces password policy, hashes the new
187    /// password, updates the user (including resetting fail2ban
188    /// counters), and stores the old password in history.
189    pub async fn confirm_reset(
190        &self,
191        tenant_id: Uuid,
192        raw_token: &str,
193        new_password: &str,
194        policy: &PasswordPolicy,
195        pepper: Option<&str>,
196        http_client: Option<&reqwest::Client>,
197    ) -> AxiamResult<()> {
198        let token_hash = token::hash_refresh_token(raw_token);
199
200        // Atomically consume the token.
201        let consumed = self
202            .token_repo
203            .consume(tenant_id, &token_hash)
204            .await
205            .map_err(|e| match e {
206                AxiamError::NotFound { .. } => AxiamError::from(AuthError::ResetTokenInvalid),
207                other => other,
208            })?;
209
210        // Look up the user.
211        let user = self
212            .user_repo
213            .get_by_id(tenant_id, consumed.user_id)
214            .await?;
215
216        // Federated users cannot reset passwords.
217        let links = self
218            .federation_repo
219            .get_by_user_id(tenant_id, user.id)
220            .await?;
221        if !links.is_empty() {
222            return Err(AuthError::FederatedUserPasswordReset.into());
223        }
224
225        // T-24-92 / RESEARCH Pitfall 4: explicit current-password-reuse
226        // rejection, independent of and in addition to the
227        // password_history_count-based check below. The history check
228        // alone is insufficient because the pre-reset hash is not yet in
229        // `password_history` at check time (it's written further down),
230        // so a user with zero prior history rows could otherwise reset
231        // straight back to their own current password. CPU-bound Argon2,
232        // run under the crypto_semaphore-gated spawn_blocking path (A3)
233        // consistent with every other Argon2 call site in this codebase.
234        {
235            let _permit = self
236                .crypto_semaphore
237                .acquire()
238                .await
239                .map_err(|_| AxiamError::Internal("crypto semaphore closed".into()))?;
240            let new_pw_owned = new_password.to_string();
241            let current_hash_owned = user.password_hash.clone();
242            let pepper_owned = pepper.map(str::to_string);
243            let is_current_password = tokio::task::spawn_blocking(move || {
244                verify_password(&new_pw_owned, &current_hash_owned, pepper_owned.as_deref())
245            })
246            .await
247            .map_err(|e| AxiamError::Internal(format!("spawn_blocking join error: {e}")))?
248            .map_err(|e| AxiamError::Crypto(e.to_string()))?;
249            if is_current_password {
250                return Err(AuthError::PasswordReusedCurrent.into());
251            }
252        }
253
254        // Evaluate password policy, including the password_history_count
255        // check (which performs its own CPU-bound Argon2 verifies against
256        // recent history entries) and the HIBP breach check when the
257        // policy has it enabled and a client is provided. Held under the
258        // crypto_semaphore for the duration of the call so the history
259        // check's Argon2 work is CPU-isolated like every other crypto
260        // call site (A3) — `confirm_reset` previously performed this work
261        // fully ungated.
262        let check: PolicyCheckResult = {
263            let _permit = self
264                .crypto_semaphore
265                .acquire()
266                .await
267                .map_err(|_| AxiamError::Internal("crypto semaphore closed".into()))?;
268            evaluate_password(
269                new_password,
270                pepper,
271                policy,
272                tenant_id,
273                user.id,
274                &self.history_repo,
275                http_client,
276            )
277            .await?
278        };
279
280        if !check.is_ok() {
281            return Err(AxiamError::Validation {
282                message: check.error_message(),
283            });
284        }
285
286        // Hash the new password.
287        let new_hash = hash_password(new_password, pepper)?;
288
289        // Store the old password in history.
290        self.history_repo
291            .create(CreatePasswordHistoryEntry {
292                tenant_id,
293                user_id: user.id,
294                password_hash: user.password_hash.clone(),
295            })
296            .await?;
297
298        // Invalidate all active sessions for the user (D-16: password reset
299        // caller is unauthenticated, so there is no current session to preserve —
300        // ALL sessions die).  Both the session-flow tokens AND the OAuth2-flow
301        // refresh tokens must be revoked (RESEARCH §4 — D-18 "two chokepoints").
302        self.session_repo
303            .invalidate_user_sessions(tenant_id, user.id)
304            .await?;
305        self.refresh_token_repo
306            .revoke_all_for_user(tenant_id, user.id)
307            .await?;
308
309        // Update user: new password hash + reset fail2ban counters.
310        self.user_repo
311            .update(
312                tenant_id,
313                user.id,
314                UpdateUser {
315                    password_hash: Some(new_hash),
316                    failed_login_attempts: Some(0),
317                    locked_until: Some(None),
318                    last_failed_login_at: Some(None),
319                    ..Default::default()
320                },
321            )
322            .await?;
323
324        Ok(())
325    }

962    /// T-24-91 statistical timing test: sample the ineligible/unknown and
963    /// federated `Ok(None)` branches (both now run the constant-time
964    /// `dummy_hash_wait`) alongside the valid-account branch, and assert
965    /// the ineligible branches are never anomalously fast relative to the
966    /// valid branch — the exact enumeration side-channel this task closes
967    /// (an attacker timing responses to distinguish "unknown email" from
968    /// "valid email" by an near-instant vs. slower response).
969    ///
970    /// `#[ignore]`d because timing tests are inherently sensitive to host
971    /// scheduling noise; run explicitly with `-- --ignored` (24-RESEARCH
972    /// "Sampling Rate" / Wave 0 Gaps).
973    #[tokio::test]
974    #[ignore = "statistical timing test — run explicitly with `cargo test -- --ignored`"]
975    async fn reset_timing_indistinguishable() {
976        use std::time::{Duration as StdDuration, Instant};
977
978        const N: usize = 15;
979
980        let db = surrealdb::Surreal::new::<surrealdb::engine::local::Mem>(())
981            .await
982            .unwrap();
983        db.use_ns("test").use_db("test").await.unwrap();
984        run_migrations(&db).await.unwrap();
985        let (_org_id, tid) = create_org_tenant(&db).await;
986
987        let user_repo = SurrealUserRepository::new(db.clone());
988        let token_repo = SurrealPasswordResetTokenRepository::new(db.clone());
989        let fed_repo = SurrealFederationLinkRepository::new(db.clone());
990        let hist_repo = SurrealPasswordHistoryRepository::new(db.clone());
991        let session_repo = SurrealSessionRepository::new(db.clone());
992        let refresh_token_repo = SurrealRefreshTokenRepository::new(db.clone());
993
994        let svc = PasswordResetService::new(
995            user_repo.clone(),
996            token_repo,
997            fed_repo,
998            hist_repo,
999            session_repo,
1000            refresh_token_repo,
1001            Arc::new(Semaphore::new(4)),
1002        );
1003
1004        fn mean(samples: &[StdDuration]) -> StdDuration {
1005            let total: StdDuration = samples.iter().sum();
1006            total / (samples.len() as u32)
1007        }
1008
1009        // Sample 1: unknown-email branch (Ok(None), dummy_hash_wait).
1010        let mut unknown_samples = Vec::with_capacity(N);
1011        for i in 0..N {
1012            let email = format!("nonexistent-{i}@example.com");
1013            let start = Instant::now();
1014            let result = svc.initiate_reset(tid, &email, 1, None).await.unwrap();
1015            unknown_samples.push(start.elapsed());
1016            assert!(result.is_none());
1017        }
1018
1019        // Sample 2: valid-account branch — a fresh user per iteration so
1020        // the per-user MAX_RESETS_PER_DAY limit never trips.
1021        let mut valid_samples = Vec::with_capacity(N);
1022        for i in 0..N {
1023            let user = user_repo
1024                .create(CreateUser {
1025                    tenant_id: tid,
1026                    username: format!("timing-user-{i}"),
1027                    email: format!("timing-user-{i}@example.com"),
1028                    password: "OldPassw0rd!Strong".into(),
1029                    metadata: None,
1030                })
1031                .await
1032                .unwrap();
1033            let start = Instant::now();
1034            let result = svc.initiate_reset(tid, &user.email, 1, None).await.unwrap();
1035            valid_samples.push(start.elapsed());
1036            assert!(result.is_some());
1037        }
1038
1039        let mean_unknown = mean(&unknown_samples);
1040        let mean_valid = mean(&valid_samples);
1041
1042        tracing::info!(
1043            ?mean_unknown,
1044            ?mean_valid,
1045            "reset_timing_indistinguishable: sampled means"
1046        );
1047
1048        // T-24-91's regression target: BEFORE this fix, the unknown-email
1049        // branch returned `Ok(None)` almost immediately (no Argon2 work at
1050        // all) while the valid branch did real DB work — an attacker could
1051        // trivially distinguish "unknown" from "valid" by response time
1052        // alone. Now BOTH branches perform comparable dominant-cost work
1053        // (the valid branch's DB round-trips vs. the unknown branch's
1054        // dummy Argon2 verify), so the unknown branch must never resolve
1055        // in a small fraction of the valid branch's time. A generous
1056        // bound (rather than a tight one) is used deliberately — per
1057        // 24-RESEARCH Anti-Patterns, this must never regress into a
1058        // hand-tuned fixed-duration assertion; it only guards against the
1059        // original "near-instant" enumeration signal.
1060        let floor = mean_valid / 4;
1061        assert!(
1062            mean_unknown >= floor,
1063            "unknown-email branch resolved suspiciously fast \
1064             (mean_unknown={mean_unknown:?} vs mean_valid={mean_valid:?}, \
1065             expected mean_unknown >= {floor:?}) — dummy_hash_wait may not \
1066             be running"
1067        );
1068
1069        // The two Ok(None) branches (unknown + federated) execute the
1070        // identical dummy_hash_wait code path, so they should be tightly
1071        // overlapping with each other. Sample the federated branch too.
1072        let fed_user = user_repo
1073            .create(CreateUser {
1074                tenant_id: tid,
1075                username: "fed-timing-user".into(),
1076                email: "fed-timing-user@example.com".into(),
1077                password: "OldPassw0rd!Strong".into(),
1078                metadata: None,
1079            })
1080            .await
1081            .unwrap();
1082        let fed_config_repo = SurrealFederationConfigRepository::new(db.clone());
1083        let fed_config = fed_config_repo
1084            .create(CreateFederationConfig {
1085                tenant_id: tid,
1086                provider: "google".into(),
1087                protocol: FederationProtocol::OidcConnect,
1088                metadata_url: None,
1089                client_id: "google-client-id".into(),
1090                client_secret: "google-secret".into(),
1091                attribute_map: None,
1092                idp_signing_cert_pem: None,
1093                allowed_algorithms: None,
1094            })
1095            .await
1096            .unwrap();
1097        let fed_link_repo = SurrealFederationLinkRepository::new(db.clone());
1098        fed_link_repo
1099            .create(CreateFederationLink {
1100                tenant_id: tid,
1101                user_id: fed_user.id,
1102                federation_config_id: fed_config.id,
1103                external_subject: "ext-timing".into(),
1104                external_email: Some(fed_user.email.clone()),
1105            })
1106            .await
1107            .unwrap();
1108
1109        let mut fed_samples = Vec::with_capacity(N);
1110        for _ in 0..N {
1111            let start = Instant::now();
1112            let result = svc
1113                .initiate_reset(tid, &fed_user.email, 1, None)
1114                .await
1115                .unwrap();
1116            fed_samples.push(start.elapsed());
1117            assert!(result.is_none());
1118        }
1119        let mean_fed = mean(&fed_samples);
1120
1121        tracing::info!(?mean_fed, "reset_timing_indistinguishable: federated mean");
1122
1123        // Overlap check between the two structurally-identical dummy-hash
1124        // branches: neither should be more than 5x the other. This is a
1125        // loose bound tolerant of scheduler jitter while still catching a
1126        // gross asymmetry (e.g. dummy_hash_wait accidentally skipped on
1127        // one of the two Ok(None) branches).
1128        let (lo, hi) = if mean_unknown <= mean_fed {
1129            (mean_unknown, mean_fed)
1130        } else {
1131            (mean_fed, mean_unknown)
1132        };
1133        assert!(
1134            hi <= lo * 5,
1135            "unknown-email and federated-user branches diverged too far: \
1136             mean_unknown={mean_unknown:?}, mean_fed={mean_fed:?}"
1137        );
1138    }
