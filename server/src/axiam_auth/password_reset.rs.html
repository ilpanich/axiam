27/// Password reset service.
28///
29/// Handles reset token generation with rate limiting and prior-token
30/// invalidation, plus reset confirmation with password policy
31/// enforcement, fail2ban counter reset, and password history storage.
32///
33/// `S` is the session repository — used by `confirm_reset` to invalidate ALL
34/// active sessions after a successful password reset (D-16 — caller is
35/// unauthenticated so there is no current session to preserve).
36///
37/// `T` is the OAuth2 refresh-token repository — used by `confirm_reset` to
38/// also revoke OAuth2-flow refresh tokens (RESEARCH §4 "two chokepoints").
39///
40/// `Clone` (QUAL-07): this service is now a hoisted `AppState<C>` singleton
41/// constructed once at startup and cloned per Actix worker, rather than
42/// rebuilt per-request.
43#[derive(Clone)]
44pub struct PasswordResetService<U, R, F, H, S, T>
45where
46    U: UserRepository,
47    R: PasswordResetTokenRepository,
48    F: FederationLinkRepository,
49    H: PasswordHistoryRepository,
50    S: SessionRepository,
51    T: RefreshTokenRepository,
52{
53    user_repo: U,
54    token_repo: R,
55    federation_repo: F,
56    history_repo: H,
57    session_repo: S,
58    refresh_token_repo: T,
59    /// Bounding semaphore (CQ-B02, mirrors `AuthService`): limits concurrent
60    /// Argon2 operations (including the T-24-91 dummy-hash timing
61    /// equalization below and the T-24-92 current-password check) to
62    /// prevent CPU-bound crypto from starving the Tokio async runtime.
63    crypto_semaphore: Arc<Semaphore>,
64    /// B1: Seconds to wait for a `crypto_semaphore` permit before returning a
65    /// 503 backpressure error. Threaded from `AuthConfig::hash_acquire_timeout_secs`.
66    hash_acquire_timeout_secs: u64,
67}
68

70where
71    U: UserRepository,
72    R: PasswordResetTokenRepository,
73    F: FederationLinkRepository,
74    H: PasswordHistoryRepository,
75    S: SessionRepository,
76    T: RefreshTokenRepository,

100    /// Constant-time dummy Argon2 verify (T-24-91).
101    ///
102    /// Mirrors the SEC-026 pattern already live in `AuthService::login`
103    /// exactly: acquire the shared `crypto_semaphore`, run a dummy
104    /// Argon2id verify against the relocated `DUMMY_HASH` constant inside
105    /// `spawn_blocking`, and discard the result. Called from BOTH of
106    /// `initiate_reset`'s `Ok(None)` branches (unknown email, federated
107    /// user) so those responses are time-indistinguishable from the
108    /// valid-account branch — no hand-tuned fixed-duration sleep.
109    async fn dummy_hash_wait(&self, pepper: Option<&str>) {
110        // B1: bound this enumeration-defence dummy hash by the crypto
111        // semaphore (and its timeout) so it counts against the same
112        // concurrency budget as real verifies and cannot itself become an
113        // un-throttled arena allocator.
114        //
115        // Unlike the login not-found branch, we deliberately do NOT surface a
116        // backpressure error here: `initiate_reset`'s *valid-account* branch
117        // does no Argon2 work and never touches the semaphore, so if this path
118        // returned 503 under saturation while the valid path returned
119        // `Ok(None)`, that difference would itself be a user-enumeration oracle
120        // (D-15). Instead we bound the wait by the same timeout and, if it
121        // elapses, simply skip the dummy hash and return normally — the
122        // observable result (`Ok(None)`) is identical to the valid branch
123        // regardless of load. Memory stays bounded because the permit *count*
124        // still caps concurrent arenas even when this path skips hashing.
125        let acquired = tokio::time::timeout(
126            std::time::Duration::from_secs(self.hash_acquire_timeout_secs),
127            self.crypto_semaphore.acquire(),
128        )
129        .await;
130        if let Ok(Ok(_permit)) = acquired {
131            let pepper_owned = pepper.map(str::to_string);
132            let _ = tokio::task::spawn_blocking(move || {
133                password::verify_password("dummy", DUMMY_HASH, pepper_owned.as_deref())
134            })
135            .await;
136        }
137    }

139    /// Initiate a password reset for the given email.
140    ///
141    /// Returns `Ok(Some((raw_token, user_id, expires_at)))` on success,
142    /// `Ok(None)` if the email doesn't exist or user is federated
143    /// (to prevent user enumeration).
144    ///
145    /// Returns `Err(RateLimited)` if the daily limit is exceeded.
146    ///
147    /// `pepper` is threaded through so the unknown-email and federated-user
148    /// branches can run the T-24-91 constant-time dummy Argon2 verify with
149    /// the same pepper the real hashing/verification call sites use.
150    pub async fn initiate_reset(
151        &self,
152        tenant_id: Uuid,
153        email: &str,
154        expiry_hours: u32,
155        pepper: Option<&str>,
156    ) -> AxiamResult<Option<(String, Uuid, chrono::DateTime<chrono::Utc>)>> {
157        // Look up user — silently return None if not found.
158        let user = match self.user_repo.get_by_email(tenant_id, email).await {
159            Ok(u) => u,
160            Err(AxiamError::NotFound { .. }) => {
161                // T-24-91: constant-time dummy Argon2 verify so an unknown
162                // email is time-indistinguishable from a valid one.
163                self.dummy_hash_wait(pepper).await;
164                return Ok(None);
165            }
166            Err(e) => return Err(e),
167        };
168
169        // Federated users cannot reset passwords — return None to
170        // prevent enumeration.
171        let links = self
172            .federation_repo
173            .get_by_user_id(tenant_id, user.id)
174            .await?;
175        if !links.is_empty() {
176            // T-24-91: same constant-time treatment as the unknown-email
177            // branch above — a federated account must not resolve faster.
178            self.dummy_hash_wait(pepper).await;
179            return Ok(None);
180        }
181
182        // Rate limit: max resets per day.
183        let count = self.token_repo.count_today(tenant_id, user.id).await?;
184        if count >= MAX_RESETS_PER_DAY {
185            return Err(AxiamError::RateLimited);
186        }
187
188        // Invalidate any prior unconsumed tokens for this user so
189        // intercepted older emails cannot be used after a newer
190        // reset request.
191        self.token_repo
192            .delete_unconsumed_for_user(tenant_id, user.id)
193            .await?;
194
195        let raw_token = token::generate_refresh_token();
196        let token_hash = token::hash_refresh_token(&raw_token);
197        let expires_at = Utc::now() + Duration::hours(expiry_hours as i64);
198
199        self.token_repo
200            .create(CreatePasswordResetToken {
201                tenant_id,
202                user_id: user.id,
203                token_hash,
204                expires_at,
205            })
206            .await?;
207
208        Ok(Some((raw_token, user.id, expires_at)))
209    }

211    /// Confirm a password reset using the raw token and a new password.
212    ///
213    /// Validates the token, enforces password policy, hashes the new
214    /// password, updates the user (including resetting fail2ban
215    /// counters), and stores the old password in history.
216    pub async fn confirm_reset(
217        &self,
218        tenant_id: Uuid,
219        raw_token: &str,
220        new_password: &str,
221        policy: &PasswordPolicy,
222        pepper: Option<&str>,
223        http_client: Option<&reqwest::Client>,
224    ) -> AxiamResult<()> {
225        let token_hash = token::hash_refresh_token(raw_token);
226
227        // Atomically consume the token.
228        let consumed = self
229            .token_repo
230            .consume(tenant_id, &token_hash)
231            .await
232            .map_err(|e| match e {
233                AxiamError::NotFound { .. } => AxiamError::from(AuthError::ResetTokenInvalid),
234                other => other,
235            })?;
236
237        // Look up the user.
238        let user = self
239            .user_repo
240            .get_by_id(tenant_id, consumed.user_id)
241            .await?;
242
243        // Federated users cannot reset passwords.
244        let links = self
245            .federation_repo
246            .get_by_user_id(tenant_id, user.id)
247            .await?;
248        if !links.is_empty() {
249            return Err(AuthError::FederatedUserPasswordReset.into());
250        }
251
252        // T-24-92 / RESEARCH Pitfall 4: explicit current-password-reuse
253        // rejection, independent of and in addition to the
254        // password_history_count-based check below. The history check
255        // alone is insufficient because the pre-reset hash is not yet in
256        // `password_history` at check time (it's written further down),
257        // so a user with zero prior history rows could otherwise reset
258        // straight back to their own current password. CPU-bound Argon2,
259        // run under the crypto_semaphore-gated spawn_blocking path (A3)
260        // consistent with every other Argon2 call site in this codebase.
261        {
262            // B1: bounded acquire — 503 backpressure on timeout. Enumeration is
263            // not a concern here (the caller already holds a valid reset token),
264            // so surfacing the error is safe and correct.
265            let _permit = acquire_hash_permit(
266                &self.crypto_semaphore,
267                std::time::Duration::from_secs(self.hash_acquire_timeout_secs),
268            )
269            .await?;
270            let new_pw_owned = new_password.to_string();
271            let current_hash_owned = user.password_hash.clone();
272            let pepper_owned = pepper.map(str::to_string);
273            let is_current_password = tokio::task::spawn_blocking(move || {
274                verify_password(&new_pw_owned, &current_hash_owned, pepper_owned.as_deref())
275            })
276            .await
277            .map_err(|e| AxiamError::Internal(format!("spawn_blocking join error: {e}")))?
278            .map_err(|e| AxiamError::Crypto(e.to_string()))?;
279            if is_current_password {
280                return Err(AuthError::PasswordReusedCurrent.into());
281            }
282        }
283
284        // Evaluate password policy, including the password_history_count
285        // check (which performs its own CPU-bound Argon2 verifies against
286        // recent history entries) and the HIBP breach check when the
287        // policy has it enabled and a client is provided. Held under the
288        // crypto_semaphore for the duration of the call so the history
289        // check's Argon2 work is CPU-isolated like every other crypto
290        // call site (A3) — `confirm_reset` previously performed this work
291        // fully ungated.
292        let check: PolicyCheckResult = {
293            // B1: bounded acquire — held for the duration of the policy
294            // evaluation so the history check's Argon2 verifies run under one
295            // permit; 503 backpressure on timeout.
296            let _permit = acquire_hash_permit(
297                &self.crypto_semaphore,
298                std::time::Duration::from_secs(self.hash_acquire_timeout_secs),
299            )
300            .await?;
301            evaluate_password(
302                new_password,
303                pepper,
304                policy,
305                tenant_id,
306                user.id,
307                &self.history_repo,
308                http_client,
309            )
310            .await?
311        };
312
313        if !check.is_ok() {
314            return Err(AxiamError::Validation {
315                message: check.error_message(),
316            });
317        }
318
319        // Hash the new password.
320        let new_hash = hash_password(new_password, pepper)?;
321
322        // Store the old password in history.
323        self.history_repo
324            .create(CreatePasswordHistoryEntry {
325                tenant_id,
326                user_id: user.id,
327                password_hash: user.password_hash.clone(),
328            })
329            .await?;
330
331        // Invalidate all active sessions for the user (D-16: password reset
332        // caller is unauthenticated, so there is no current session to preserve —
333        // ALL sessions die).  Both the session-flow tokens AND the OAuth2-flow
334        // refresh tokens must be revoked (RESEARCH §4 — D-18 "two chokepoints").
335        self.session_repo
336            .invalidate_user_sessions(tenant_id, user.id)
337            .await?;
338        self.refresh_token_repo
339            .revoke_all_for_user(tenant_id, user.id)
340            .await?;
341
342        // Update user: new password hash + reset fail2ban counters.
343        self.user_repo
344            .update(
345                tenant_id,
346                user.id,
347                UpdateUser {
348                    password_hash: Some(new_hash),
349                    failed_login_attempts: Some(0),
350                    locked_until: Some(None),
351                    last_failed_login_at: Some(None),
352                    ..Default::default()
353                },
354            )
355            .await?;
356
357        Ok(())
358    }

1006    /// T-24-91 statistical timing test: sample the ineligible/unknown and
1007    /// federated `Ok(None)` branches (both now run the constant-time
1008    /// `dummy_hash_wait`) alongside the valid-account branch, and assert
1009    /// the ineligible branches are never anomalously fast relative to the
1010    /// valid branch — the exact enumeration side-channel this task closes
1011    /// (an attacker timing responses to distinguish "unknown email" from
1012    /// "valid email" by an near-instant vs. slower response).
1013    ///
1014    /// `#[ignore]`d because timing tests are inherently sensitive to host
1015    /// scheduling noise; run explicitly with `-- --ignored` (24-RESEARCH
1016    /// "Sampling Rate" / Wave 0 Gaps).
1017    #[tokio::test]
1018    #[ignore = "statistical timing test — run explicitly with `cargo test -- --ignored`"]
1019    async fn reset_timing_indistinguishable() {
1020        use std::time::{Duration as StdDuration, Instant};
1021
1022        const N: usize = 15;
1023
1024        let db = surrealdb::Surreal::new::<surrealdb::engine::local::Mem>(())
1025            .await
1026            .unwrap();
1027        db.use_ns("test").use_db("test").await.unwrap();
1028        run_migrations(&db).await.unwrap();
1029        let (_org_id, tid) = create_org_tenant(&db).await;
1030
1031        let user_repo = SurrealUserRepository::new(db.clone());
1032        let token_repo = SurrealPasswordResetTokenRepository::new(db.clone());
1033        let fed_repo = SurrealFederationLinkRepository::new(db.clone());
1034        let hist_repo = SurrealPasswordHistoryRepository::new(db.clone());
1035        let session_repo = SurrealSessionRepository::new(db.clone());
1036        let refresh_token_repo = SurrealRefreshTokenRepository::new(db.clone());
1037
1038        let svc = PasswordResetService::new(
1039            user_repo.clone(),
1040            token_repo,
1041            fed_repo,
1042            hist_repo,
1043            session_repo,
1044            refresh_token_repo,
1045            Arc::new(Semaphore::new(4)),
1046            5,
1047        );
1048
1049        fn mean(samples: &[StdDuration]) -> StdDuration {
1050            let total: StdDuration = samples.iter().sum();
1051            total / (samples.len() as u32)
1052        }
1053
1054        // Sample 1: unknown-email branch (Ok(None), dummy_hash_wait).
1055        let mut unknown_samples = Vec::with_capacity(N);
1056        for i in 0..N {
1057            let email = format!("nonexistent-{i}@example.com");
1058            let start = Instant::now();
1059            let result = svc.initiate_reset(tid, &email, 1, None).await.unwrap();
1060            unknown_samples.push(start.elapsed());
1061            assert!(result.is_none());
1062        }
1063
1064        // Sample 2: valid-account branch — a fresh user per iteration so
1065        // the per-user MAX_RESETS_PER_DAY limit never trips.
1066        let mut valid_samples = Vec::with_capacity(N);
1067        for i in 0..N {
1068            let user = user_repo
1069                .create(CreateUser {
1070                    tenant_id: tid,
1071                    username: format!("timing-user-{i}"),
1072                    email: format!("timing-user-{i}@example.com"),
1073                    password: "OldPassw0rd!Strong".into(),
1074                    metadata: None,
1075                })
1076                .await
1077                .unwrap();
1078            let start = Instant::now();
1079            let result = svc.initiate_reset(tid, &user.email, 1, None).await.unwrap();
1080            valid_samples.push(start.elapsed());
1081            assert!(result.is_some());
1082        }
1083
1084        let mean_unknown = mean(&unknown_samples);
1085        let mean_valid = mean(&valid_samples);
1086
1087        tracing::info!(
1088            ?mean_unknown,
1089            ?mean_valid,
1090            "reset_timing_indistinguishable: sampled means"
1091        );
1092
1093        // T-24-91's regression target: BEFORE this fix, the unknown-email
1094        // branch returned `Ok(None)` almost immediately (no Argon2 work at
1095        // all) while the valid branch did real DB work — an attacker could
1096        // trivially distinguish "unknown" from "valid" by response time
1097        // alone. Now BOTH branches perform comparable dominant-cost work
1098        // (the valid branch's DB round-trips vs. the unknown branch's
1099        // dummy Argon2 verify), so the unknown branch must never resolve
1100        // in a small fraction of the valid branch's time. A generous
1101        // bound (rather than a tight one) is used deliberately — per
1102        // 24-RESEARCH Anti-Patterns, this must never regress into a
1103        // hand-tuned fixed-duration assertion; it only guards against the
1104        // original "near-instant" enumeration signal.
1105        let floor = mean_valid / 4;
1106        assert!(
1107            mean_unknown >= floor,
1108            "unknown-email branch resolved suspiciously fast \
1109             (mean_unknown={mean_unknown:?} vs mean_valid={mean_valid:?}, \
1110             expected mean_unknown >= {floor:?}) — dummy_hash_wait may not \
1111             be running"
1112        );
1113
1114        // The two Ok(None) branches (unknown + federated) execute the
1115        // identical dummy_hash_wait code path, so they should be tightly
1116        // overlapping with each other. Sample the federated branch too.
1117        let fed_user = user_repo
1118            .create(CreateUser {
1119                tenant_id: tid,
1120                username: "fed-timing-user".into(),
1121                email: "fed-timing-user@example.com".into(),
1122                password: "OldPassw0rd!Strong".into(),
1123                metadata: None,
1124            })
1125            .await
1126            .unwrap();
1127        let fed_config_repo = SurrealFederationConfigRepository::new(db.clone());
1128        let fed_config = fed_config_repo
1129            .create(CreateFederationConfig {
1130                tenant_id: tid,
1131                provider: "google".into(),
1132                protocol: FederationProtocol::OidcConnect,
1133                metadata_url: None,
1134                client_id: "google-client-id".into(),
1135                client_secret: "google-secret".into(),
1136                attribute_map: None,
1137                idp_signing_cert_pem: None,
1138                allowed_algorithms: None,
1139            })
1140            .await
1141            .unwrap();
1142        let fed_link_repo = SurrealFederationLinkRepository::new(db.clone());
1143        fed_link_repo
1144            .create(CreateFederationLink {
1145                tenant_id: tid,
1146                user_id: fed_user.id,
1147                federation_config_id: fed_config.id,
1148                external_subject: "ext-timing".into(),
1149                external_email: Some(fed_user.email.clone()),
1150            })
1151            .await
1152            .unwrap();
1153
1154        let mut fed_samples = Vec::with_capacity(N);
1155        for _ in 0..N {
1156            let start = Instant::now();
1157            let result = svc
1158                .initiate_reset(tid, &fed_user.email, 1, None)
1159                .await
1160                .unwrap();
1161            fed_samples.push(start.elapsed());
1162            assert!(result.is_none());
1163        }
1164        let mean_fed = mean(&fed_samples);
1165
1166        tracing::info!(?mean_fed, "reset_timing_indistinguishable: federated mean");
1167
1168        // Overlap check between the two structurally-identical dummy-hash
1169        // branches: neither should be more than 5x the other. This is a
1170        // loose bound tolerant of scheduler jitter while still catching a
1171        // gross asymmetry (e.g. dummy_hash_wait accidentally skipped on
1172        // one of the two Ok(None) branches).
1173        let (lo, hi) = if mean_unknown <= mean_fed {
1174            (mean_unknown, mean_fed)
1175        } else {
1176            (mean_fed, mean_unknown)
1177        };
1178        assert!(
1179            hi <= lo * 5,
1180            "unknown-email and federated-user branches diverged too far: \
1181             mean_unknown={mean_unknown:?}, mean_fed={mean_fed:?}"
1182        );
1183    }
