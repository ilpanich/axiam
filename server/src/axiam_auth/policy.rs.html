1//! Password policy engine.
2//!
3//! Evaluates candidate passwords against the effective
4//! [`PasswordPolicy`] resolved from org + tenant settings.
5//!
6//! Four check layers:
7//! 1. **Complexity** — length, uppercase, lowercase, digit, symbol
8//! 2. **History** — reject reuse of the last N passwords
9//! 3. **HIBP breach** — k-Anonymity API (only 5-char SHA-1 prefix sent)
10//! 4. **Orchestrator** — combines all checks into a single result
11

14use axiam_core::repository::PasswordHistoryRepository;
15use sha1::{Digest, Sha1};
16use uuid::Uuid;

24/// A single password policy violation.
25#[derive(Debug, Clone, PartialEq, Eq)]
26pub enum PolicyViolation {
27    TooShort { min: u32, actual: usize },
28    MissingUppercase,
29    MissingLowercase,
30    MissingDigit,
31    MissingSymbol,
32    ReusedPassword { position: usize },
33    BreachedPassword { occurrences: u64 },
34}
35
36impl std::fmt::Display for PolicyViolation {
37    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
38        match self {
39            Self::TooShort { min, actual } => {
40                write!(f, "password too short: minimum {min}, got {actual}")
41            }
42            Self::MissingUppercase => {
43                write!(f, "password must contain an uppercase letter")
44            }
45            Self::MissingLowercase => {
46                write!(f, "password must contain a lowercase letter")
47            }
48            Self::MissingDigit => {
49                write!(f, "password must contain a digit")
50            }
51            Self::MissingSymbol => {
52                write!(f, "password must contain a symbol")
53            }
54            Self::ReusedPassword { position } => write!(
55                f,
56                "password was used recently (position {position} in history)"
57            ),
58            Self::BreachedPassword { occurrences } => {
59                write!(f, "password seen {occurrences} time(s) in known breaches")
60            }
61        }
62    }
63}

65/// Result of a full password policy evaluation.
66#[derive(Debug, Clone)]
67pub struct PolicyCheckResult {
68    pub violations: Vec<PolicyViolation>,
69}
70
71impl PolicyCheckResult {
72    pub fn is_ok(&self) -> bool {
73        self.violations.is_empty()
74    }
75
76    /// Format all violations into a single error message.
77    pub fn error_message(&self) -> String {
78        self.violations
79            .iter()
80            .map(|v| v.to_string())
81            .collect::<Vec<_>>()
82            .join("; ")
83    }
84}

90/// Check password against complexity requirements.
91///
92/// This is a pure function with no I/O — easy to unit test.
93pub fn check_complexity(password: &str, policy: &PasswordPolicy) -> Vec<PolicyViolation> {
94    // PERF-01/D-05: exactly 5 possible violations (TooShort, MissingUppercase,
95    // MissingLowercase, MissingDigit, MissingSymbol) — pre-size to avoid
96    // reallocation on the hot path. Behavior unchanged.
97    let mut violations = Vec::with_capacity(5);
98
99    let char_count = password.chars().count();
100    if (char_count as u32) < policy.min_length {
101        violations.push(PolicyViolation::TooShort {
102            min: policy.min_length,
103            actual: char_count,
104        });
105    }
106
107    if policy.require_uppercase && !password.chars().any(|c| c.is_uppercase()) {
108        violations.push(PolicyViolation::MissingUppercase);
109    }
110
111    if policy.require_lowercase && !password.chars().any(|c| c.is_lowercase()) {
112        violations.push(PolicyViolation::MissingLowercase);
113    }
114
115    if policy.require_digits && !password.chars().any(|c| c.is_ascii_digit()) {
116        violations.push(PolicyViolation::MissingDigit);
117    }
118
119    if policy.require_symbols
120        && !password
121            .chars()
122            .any(|c| !c.is_alphanumeric() && !c.is_whitespace())
123    {
124        violations.push(PolicyViolation::MissingSymbol);
125    }
126
127    violations
128}

134/// Compute the SHA-1 hash of a password and split into prefix + suffix
135/// for the k-Anonymity HIBP API.
136pub fn sha1_prefix_suffix(password: &str) -> (String, String) {
137    let hash = Sha1::digest(password.as_bytes());
138    let hex = hex::encode_upper(hash);
139    let prefix = hex[..5].to_string();
140    let suffix = hex[5..].to_string();
141    (prefix, suffix)
142}
143
144/// Parse the HIBP range response body and find the occurrence count
145/// for a given suffix.
146///
147/// Response format: one line per hash suffix, e.g.:
148/// ```text
149/// 003D68EB55068C33ACE09247EE4C639306B:3
150/// 1E4C9B93F3F0682250B6CF8331B7EE68FD8:5
151/// ```
152///
153/// Lines with count `0` are padding (added when `Add-Padding: true`).
154pub fn parse_hibp_response(body: &str, suffix: &str) -> Option<u64> {
155    for line in body.lines() {
156        let line = line.trim();
157        if let Some((line_suffix, count_str)) = line.split_once(':')
158            && line_suffix.eq_ignore_ascii_case(suffix)
159        {
160            if let Ok(count) = count_str.trim().parse::<u64>()
161                && count > 0
162            {
163                return Some(count);
164            }
165            return None;
166        }
167    }
168    None
169}

171/// Check a password against the HIBP Pwned Passwords API using
172/// k-Anonymity (only a 5-character SHA-1 prefix is sent).
173///
174/// Returns `Some(violation)` if breached, `None` if clean or on error.
175/// HIBP is best-effort — network failures never block the user.
176pub async fn check_hibp(
177    password: &str,
178    http_client: &reqwest::Client,
179) -> Result<Option<PolicyViolation>, AxiamError> {
180    // PERF-01/T-27-01: consult the process-wide circuit breaker BEFORE
181    // issuing the HTTP request. If it's open (sustained failures/timeouts),
182    // short-circuit to Ok(None) immediately — do NOT reach the 5s-timeout
183    // HTTP call. This is burst-protection only; check_hibp was ALREADY
184    // fail-open on every error branch below.
185    if !crate::hibp_breaker::global().should_attempt() {
186        tracing::warn!(
187            reason = "hibp_breaker_open",
188            "HIBP breaker open; skipping HTTP request, treating as not breached"
189        );
190        return Ok(None);
191    }
192
193    let (prefix, suffix) = sha1_prefix_suffix(password);
194
195    let url = format!("https://api.pwnedpasswords.com/range/{prefix}");
196
197    let response = match http_client
198        .get(&url)
199        .header("User-Agent", "AXIAM-IAM/0.1")
200        .header("Add-Padding", "true")
201        .timeout(std::time::Duration::from_secs(5))
202        .send()
203        .await
204    {
205        Ok(r) => r,
206        Err(e) => {
207            tracing::warn!(
208                error = %e,
209                "HIBP API request failed; treating as not breached"
210            );
211            crate::hibp_breaker::global().record_failure();
212            return Ok(None);
213        }
214    };
215
216    if !response.status().is_success() {
217        tracing::warn!(
218            status = %response.status(),
219            "HIBP API returned non-200; treating as not breached"
220        );
221        crate::hibp_breaker::global().record_failure();
222        return Ok(None);
223    }
224
225    let body = match response.text().await {
226        Ok(b) => b,
227        Err(e) => {
228            tracing::warn!(
229                error = %e,
230                "Failed to read HIBP response body"
231            );
232            return Ok(None);
233        }
234    };
235
236    crate::hibp_breaker::global().record_success();
237
238    Ok(parse_hibp_response(&body, &suffix)
239        .map(|occurrences| PolicyViolation::BreachedPassword { occurrences }))
240}

242// -----------------------------------------------------------------------
243// Password history check (async, DB)
244// -----------------------------------------------------------------------
245
246/// Check a candidate password against the user's recent password
247/// history.
248///
249/// For each stored historical hash, we use Argon2id verification.
250/// If `history_count` is 0, no check is performed.
251pub async fn check_history<R: PasswordHistoryRepository>(
252    password: &str,
253    pepper: Option<&str>,
254    tenant_id: Uuid,
255    user_id: Uuid,
256    history_count: u32,
257    repo: &R,
258) -> AxiamResult<Vec<PolicyViolation>> {
259    if history_count == 0 {
260        return Ok(Vec::new());
261    }
262
263    let entries = repo.get_recent(tenant_id, user_id, history_count).await?;
264
265    let mut violations = Vec::new();
266    for (idx, entry) in entries.iter().enumerate() {
267        // CPU-bound Argon2 verify runs in spawn_blocking (CQ-B02).
268        let pw_owned = password.to_string();
269        let hash_owned = entry.password_hash.clone();
270        let pepper_owned = pepper.map(str::to_string);
271        let entry_id = entry.id;
272        let result = tokio::task::spawn_blocking(move || {
273            verify_password(&pw_owned, &hash_owned, pepper_owned.as_deref())
274        })
275        .await
276        .map_err(|e| {
277            AxiamError::Internal(format!("spawn_blocking join error in history check: {e}"))
278        })?;
279        match result {
280            Ok(true) => {
281                violations.push(PolicyViolation::ReusedPassword { position: idx + 1 });
282                // One match is enough — stop early.
283                break;
284            }
285            Ok(false) => continue,
286            Err(e) => {
287                tracing::warn!(
288                    error = %e,
289                    %entry_id,
290                    "Error verifying password history entry; skipping"
291                );
292                continue;
293            }
294        }
295    }
296
297    Ok(violations)
298}

300// -----------------------------------------------------------------------
301// Full policy evaluation orchestrator
302// -----------------------------------------------------------------------
303
304/// Evaluate a candidate password against the full password policy.
305///
306/// Runs complexity checks (sync), then history + HIBP checks (async).
307/// The `http_client` is `Option` so that callers can skip HIBP
308/// (e.g., in unit tests or when the feature is disabled).
309pub async fn evaluate_password<R: PasswordHistoryRepository>(
310    password: &str,
311    pepper: Option<&str>,
312    policy: &PasswordPolicy,
313    tenant_id: Uuid,
314    user_id: Uuid,
315    history_repo: &R,
316    http_client: Option<&reqwest::Client>,
317) -> AxiamResult<PolicyCheckResult> {
318    let mut violations = check_complexity(password, policy);
319
320    // Short-circuit: if complexity checks already failed, skip the
321    // expensive history (Argon2) and HIBP (network) checks to avoid
322    // unnecessary CPU/network cost and DoS potential.
323    if !violations.is_empty() {
324        return Ok(PolicyCheckResult { violations });
325    }
326
327    // History check
328    if policy.password_history_count > 0 {
329        let history_violations = check_history(
330            password,
331            pepper,
332            tenant_id,
333            user_id,
334            policy.password_history_count,
335            history_repo,
336        )
337        .await?;
338        violations.extend(history_violations);
339    }
340
341    // HIBP breach check
342    if policy.hibp_check_enabled
343        && let Some(client) = http_client
344        && let Ok(Some(violation)) = check_hibp(password, client).await
345    {
346        violations.push(violation);
347    }
348
349    Ok(PolicyCheckResult { violations })
350}
