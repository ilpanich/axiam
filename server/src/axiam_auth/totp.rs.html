8/// Encrypt a TOTP secret with AES-256-GCM.
9///
10/// Returns `base64(nonce || ciphertext || tag)`.
11///
12/// Delegates to [`crate::crypto::aes256gcm_encrypt`] — the bundled format
13/// is shared; changing the wire format here would break existing TOTP secrets
14/// stored in the database.
15pub fn encrypt_secret(key: &[u8; 32], plaintext: &[u8]) -> Result<String, AuthError> {
16    crypto::aes256gcm_encrypt(key, plaintext)
17}
18
19/// Decrypt an AES-256-GCM encrypted TOTP secret.
20///
21/// Delegates to [`crate::crypto::aes256gcm_decrypt`].
22pub fn decrypt_secret(key: &[u8; 32], encoded: &str) -> Result<Vec<u8>, AuthError> {
23    crypto::aes256gcm_decrypt(key, encoded)
24}

26/// Generate a TOTP enrollment: secret + otpauth URI.
27///
28/// Returns `(base32_secret, otpauth_uri)`.
29pub fn generate_enrollment(issuer: &str, account: &str) -> Result<(String, String), AuthError> {
30    let secret = Secret::generate_secret();
31    let secret_bytes = secret
32        .to_bytes()
33        .map_err(|e| AuthError::Crypto(format!("secret bytes: {e}")))?;
34
35    let totp = TOTP::new(
36        Algorithm::SHA1, // RFC 6238 default
37        6,               // digits
38        1,               // skew (±1 step)
39        30,              // step seconds
40        secret_bytes,
41        Some(issuer.to_string()),
42        account.to_string(),
43    )
44    .map_err(|e| AuthError::Crypto(format!("TOTP init: {e}")))?;
45
46    let base32 = secret.to_encoded().to_string();
47    let uri = totp.get_url();
48
49    Ok((base32, uri))
50}

52/// Verify a TOTP code against a raw secret.
53pub fn verify_code(
54    secret_bytes: &[u8],
55    code: &str,
56    issuer: &str,
57    account: &str,
58) -> Result<bool, AuthError> {
59    let totp = TOTP::new(
60        Algorithm::SHA1,
61        6,
62        1,
63        30,
64        secret_bytes.to_vec(),
65        Some(issuer.to_string()),
66        account.to_string(),
67    )
68    .map_err(|e| AuthError::Crypto(format!("TOTP init: {e}")))?;
69
70    totp.check_current(code)
71        .map_err(|e| AuthError::Crypto(format!("TOTP check: {e}")))
72}

74/// TOTP step size in seconds (RFC 6238 default), shared by every `TOTP::new`
75/// call in this module.
76const TOTP_STEP_SECS: u64 = 30;
77
78/// TOTP skew tolerance (±1 step), shared by every `TOTP::new` call in this
79/// module.
80const TOTP_SKEW: u8 = 1;
81
82/// Verify a TOTP code with replay protection.
83///
84/// Computes the current time-step (`unix_timestamp / 30`) and, if the HMAC is
85/// valid within the tolerated ±1-step skew window, determines WHICH step
86/// actually matched (`current_step - 1`, `current_step`, or
87/// `current_step + 1`) rather than assuming it was always `current_step`.
88/// This matters because `totp-rs`'s `check()`/`check_current()` only report
89/// pass/fail for the whole skew window, not the matched step — recording
90/// `current_step` unconditionally would let a code accepted via the -1-skew
91/// step be replayed again once the wall clock advances past it (T-24-02).
92///
93/// The code is rejected (even though the HMAC is correct) unless the matched
94/// step is strictly greater than `last_used_step.unwrap_or(0)`.
95///
96/// Returns `Ok((valid, matched_step))` on success. The caller MUST persist
97/// `matched_step` via the atomic `user_repo.update_totp_step` CAS when
98/// `valid` is `true`, and MUST treat a lost CAS (`Ok(false)`) as an invalid
99/// code (SECHRD-01) — this function cannot see concurrent submissions by
100/// itself, so the persisted-step compare-and-set is the actual replay guard.
101///
102/// Per SEC-008 / SECHRD-01 (REQ-14 AC-5).
103pub fn verify_code_with_replay_check(
104    secret_bytes: &[u8],
105    code: &str,
106    issuer: &str,
107    account: &str,
108    last_used_step: Option<u64>,
109) -> Result<(bool, u64), AuthError> {
110    let totp = TOTP::new(
111        Algorithm::SHA1,
112        6,
113        TOTP_SKEW,
114        TOTP_STEP_SECS,
115        secret_bytes.to_vec(),
116        Some(issuer.to_string()),
117        account.to_string(),
118    )
119    .map_err(|e| AuthError::Crypto(format!("TOTP init: {e}")))?;
120
121    // Compute current step independently of totp-rs internals.
122    let current_step = std::time::SystemTime::now()
123        .duration_since(std::time::UNIX_EPOCH)
124        .map_err(|e| AuthError::Crypto(format!("system time error: {e}")))?
125        .as_secs()
126        / TOTP_STEP_SECS;
127
128    // Check the HMAC across the tolerated ±1-step skew window.
129    let hmac_valid = totp
130        .check_current(code)
131        .map_err(|e| AuthError::Crypto(format!("TOTP check: {e}")))?;
132
133    if !hmac_valid {
134        return Ok((false, current_step));
135    }
136
137    // Determine which candidate step actually matched, probing in the same
138    // order totp-rs's own `check()` does for skew=1: current_step - 1,
139    // current_step, current_step + 1. The code has already passed the
140    // constant-time HMAC check above, so re-deriving `generate()` here to
141    // identify the matched candidate does not introduce a new secret-timing
142    // side channel — it only distinguishes between three already-known-valid
143    // outcomes.
144    let matched_step = [
145        current_step.checked_sub(1),
146        Some(current_step),
147        current_step.checked_add(1),
148    ]
149    .into_iter()
150    .flatten()
151    .find(|&candidate_step| totp.generate(candidate_step * TOTP_STEP_SECS) == code)
152    .unwrap_or(current_step);
153
154    // Replay check: reject unless the ACTUAL matched step (incl. -1 skew) is
155    // strictly greater than the last-used step.
156    let last = last_used_step.unwrap_or(0);
157    if matched_step <= last {
158        return Ok((false, matched_step));
159    }
160
161    Ok((true, matched_step))
162}

168    #[test]
169    fn encrypt_decrypt_roundtrip() {
170        let key = [42u8; 32];
171        let plaintext = b"totp-secret-bytes";
172        let encrypted = encrypt_secret(&key, plaintext).unwrap();
173        let decrypted = decrypt_secret(&key, &encrypted).unwrap();
174        assert_eq!(decrypted, plaintext);
175    }

243    /// T-24-02: a code that only validates against the -1 skew step must have
244    /// `current_step - 1` (not `current_step`) recorded as the matched step,
245    /// and once persisted, the SAME code must not be replayable — even though
246    /// it still falls inside the ±1 skew window relative to a later call.
247    #[test]
248    fn totp_skew_step_recorded() {
249        let secret = Secret::generate_secret();
250        let secret_bytes = secret.to_bytes().unwrap();
251
252        let totp = TOTP::new(
253            Algorithm::SHA1,
254            6,
255            1,
256            30,
257            secret_bytes.clone(),
258            Some("AXIAM".into()),
259            "skew@test.com".into(),
260        )
261        .unwrap();
262
263        let current_step = std::time::SystemTime::now()
264            .duration_since(std::time::UNIX_EPOCH)
265            .unwrap()
266            .as_secs()
267            / 30;
268        let prev_step = current_step - 1;
269
270        // Hand-generate a code for the PREVIOUS step (simulates a client
271        // that entered a code right at a step boundary) — this is the -1
272        // skew case `TOTP::check` tolerates.
273        let prev_code = totp.generate(prev_step * 30);
274
275        let (valid, matched_step) = verify_code_with_replay_check(
276            &secret_bytes,
277            &prev_code,
278            "AXIAM",
279            "skew@test.com",
280            None,
281        )
282        .unwrap();
283        assert!(valid, "code from the -1 skew step should be accepted");
284        assert_eq!(
285            matched_step, prev_step,
286            "matched step must be current_step - 1 (the step that actually \
287             matched), not always current_step"
288        );
289
290        // Persist `matched_step` (as the real caller does via
291        // `update_totp_step`) and resubmit the SAME code. It is still inside
292        // the ±1 skew window relative to `check()`, but must now be rejected
293        // because its matched step is not strictly greater than the
294        // persisted last-used step.
295        let (replay_valid, _) = verify_code_with_replay_check(
296            &secret_bytes,
297            &prev_code,
298            "AXIAM",
299            "skew@test.com",
300            Some(matched_step),
301        )
302        .unwrap();
303        assert!(
304            !replay_valid,
305            "a skew-accepted code must not be replayable at a later step"
306        );
307    }
