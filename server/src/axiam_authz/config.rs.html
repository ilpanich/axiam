10/// Strategy for evaluating a `BatchCheckAccess` call (REST + gRPC).
11///
12/// Both strategies produce **byte-identical decisions in the same order** —
13/// they differ only in how the DB work is scheduled, so switching is a pure
14/// performance choice with no authorization-semantics risk.
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Deserialize)]
16#[serde(rename_all = "snake_case")]
17pub enum BatchStrategy {
18    /// **D1 path.** Coalesce the shared role-assignment, ancestor and grant
19    /// lookups across same-subject/same-resource items into a minimal set of DB
20    /// round-trips (3 for the benchmark's 5-item shape). Minimizes round-trips,
21    /// but the whole batch resolves on a single task: the run-2 benchmark showed
22    /// this serializes on the database (DB pinned at ~1 core while idle, ~1 s
23    /// p50) and did **not** beat repeated single checks.
24    Coalesced,
25    /// **D10 path (default).** Evaluate each item as an independent
26    /// `check_access`, **concurrently**, bounded by `batch_max_concurrency`.
27    /// Issues more DB round-trips (one set per item) but recovers the
28    /// parallelism that lets single checks saturate the DB — the run-2 data
29    /// showed single checks pegging the DB at 745 req/s while the coalesced
30    /// batch left it idle. Decisions and result order are identical to the
31    /// coalesced path (the cache is consulted per item, exactly as for a plain
32    /// `check_access`).
33    Concurrent,
34}
35
36/// Configuration for the authorization engine.
37#[derive(Debug, Clone, Deserialize)]
38#[serde(default)]
39pub struct AuthzConfig {
40    /// Bound on concurrent `check_access` evaluations within a single
41    /// `BatchCheckAccess` call (gRPC and REST). Kept well under the
42    /// ~30-connection SurrealDB handle budget so a batch cannot self-DoS
43    /// the connection pool (D-07). Enforced by the `Concurrent`
44    /// [`BatchStrategy`]; ignored by `Coalesced` (which issues a fixed, small
45    /// number of round-trips regardless of batch size).
46    ///
47    /// Configure via `AXIAM__AUTHZ__BATCH_MAX_CONCURRENCY` env var.
48    pub batch_max_concurrency: usize,
49
50    /// How a `BatchCheckAccess` is evaluated (D10). Defaults to
51    /// [`BatchStrategy::Concurrent`] — per-item parallel evaluation, which the
52    /// run-2 benchmark analysis identified as the fix for the coalesced path's
53    /// DB serialization. Set to `coalesced` to restore the D1 round-trip-
54    /// minimizing behavior for an apples-to-apples laptop A/B.
55    ///
56    /// Configure via `AXIAM__AUTHZ__BATCH_STRATEGY` (`concurrent` | `coalesced`).
57    pub batch_strategy: BatchStrategy,
58
59    /// Enable the per-tenant authorization **decision cache** (D7). When
60    /// `false` (the default) the engine issues its usual DB round-trips on
61    /// every check — behaviour is byte-for-byte identical to a build without
62    /// the cache. Only when `true` does `axiam-server` attach a
63    /// [`DecisionCache`] to the authorization engines.
64    ///
65    /// Configure via `AXIAM__AUTHZ__DECISION_CACHE_ENABLED` (default `false`).
66    ///
67    /// SECURITY: the cache is safe under AXIAM's additive allow-wins /
68    /// default-deny model *only because* every access-narrowing mutation
69    /// invalidates the affected entries immediately (see
70    /// `decision_cache` module docs). A stale allow can outlive a revocation
71    /// by at most `decision_cache_ttl_secs` even if an invalidation is missed.
72    pub decision_cache_enabled: bool,
73
74    /// TTL, in seconds, for a cached decision (D7). Bounds worst-case
75    /// revocation latency if an invalidation event is ever missed. Short by
76    /// design.
77    ///
78    /// Configure via `AXIAM__AUTHZ__DECISION_CACHE_TTL_SECS` (default `5`).
79    pub decision_cache_ttl_secs: u64,
80
81    /// Maximum cached decisions retained **per tenant** before FIFO eviction
82    /// (D7). Bounds memory.
83    ///
84    /// Configure via `AXIAM__AUTHZ__DECISION_CACHE_MAX_ENTRIES` (default
85    /// `10000`).
86    pub decision_cache_max_entries: usize,
87}
88
89impl Default for AuthzConfig {
90    fn default() -> Self {
91        Self {
92            batch_max_concurrency: 16,
93            batch_strategy: BatchStrategy::Concurrent,
94            decision_cache_enabled: false,
95            decision_cache_ttl_secs: 5,
96            decision_cache_max_entries: 10_000,
97        }
98    }
99}
100
101impl AuthzConfig {
102    /// Build the shared [`DecisionCache`] iff caching is enabled, returning it
103    /// as an `Arc` ready to attach to every authorization engine
104    /// (`AuthorizationEngine::with_decision_cache`). Returns `None` when the
105    /// feature flag is off — the caller then constructs engines exactly as
106    /// before (no cache, no behaviour change).
107    ///
108    /// The *same* `Arc<DecisionCache>` must be shared across the REST, gRPC and
109    /// AMQP engines so that an invalidation triggered from a REST mutation
110    /// handler is observed by every read path. (All role/permission/resource
111    /// mutations are REST endpoints today.)
112    pub fn build_decision_cache(&self) -> Option<Arc<DecisionCache>> {
113        if !self.decision_cache_enabled {
114            return None;
115        }
116        Some(Arc::new(DecisionCache::new(DecisionCacheConfig {
117            ttl: Duration::from_secs(self.decision_cache_ttl_secs),
118            max_entries_per_tenant: self.decision_cache_max_entries,
119        })))
120    }
121}

127    #[test]
128    fn default_batch_max_concurrency_is_16() {
129        let cfg = AuthzConfig::default();

133    #[test]

135        let cfg: AuthzConfig = serde_json::from_str("{}").expect("empty object must deserialize");
136        assert_eq!(cfg.batch_max_concurrency, 16);
137    }
138
139    #[test]
140    fn deserializes_explicit_override() {
141        let cfg: AuthzConfig = serde_json::from_str(r#"{"batch_max_concurrency": 4}"#)
142            .expect("explicit override must deserialize");
143        assert_eq!(cfg.batch_max_concurrency, 4);
144    }
145
146    #[test]
147    fn default_batch_strategy_is_concurrent() {
148        // D10: per-item parallel evaluation is the shipped default (the fix for
149        // the coalesced path's DB serialization); `coalesced` is opt-in.
150        let cfg = AuthzConfig::default();
151        assert_eq!(cfg.batch_strategy, BatchStrategy::Concurrent);
152    }
153
154    #[test]
155    fn deserializes_batch_strategy_override() {
156        let cfg: AuthzConfig = serde_json::from_str(r#"{"batch_strategy": "coalesced"}"#)
157            .expect("batch_strategy override must deserialize");
158        assert_eq!(cfg.batch_strategy, BatchStrategy::Coalesced);
159        let cfg: AuthzConfig = serde_json::from_str(r#"{"batch_strategy": "concurrent"}"#)
160            .expect("batch_strategy override must deserialize");
161        assert_eq!(cfg.batch_strategy, BatchStrategy::Concurrent);
162    }

164    #[test]

166        let cfg: AuthzConfig = serde_json::from_str("{}").expect("empty object must deserialize");
167        assert_eq!(cfg.batch_strategy, BatchStrategy::Concurrent);
