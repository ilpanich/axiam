18/// Permission evaluation engine.
19///
20/// Implements the design-doc algorithm:
21/// 1. Fetch subject's role assignments (direct + group, with resource scope)
22/// 2. Filter applicable roles (global, scoped to target resource, or ancestor)
23/// 3. Collect permissions from applicable roles
24/// 4. Check if requested action matches any permission
25/// 5. Validate scope if specified
26/// 6. Default deny
27pub struct AuthorizationEngine<R, P, Res, S, G>
28where
29    R: RoleRepository,
30    P: PermissionRepository,
31    Res: ResourceRepository,
32    S: ScopeRepository,
33    G: GroupRepository,
34{
35    role_repo: R,
36    permission_repo: P,
37    resource_repo: Res,
38    scope_repo: S,
39    #[allow(dead_code)]
40    group_repo: G,
41    /// Optional per-tenant decision cache (D7). `None` unless
42    /// `with_decision_cache` was called (feature-flagged in `axiam-server`).
43    /// When `None`, `check_access` / `check_access_batch` behave exactly as a
44    /// build without the cache.
45    decision_cache: Option<Arc<DecisionCache>>,
46}
47

49// Pure evaluation helpers (no I/O) — shared by the single-check and coalesced
50// batch paths so their decision/deny-reason semantics can never diverge.

53/// Filter a subject's role assignments down to those applicable to
54/// `resource_id`, returning the deduplicated list of applicable role IDs in
55/// first-seen order.
56///
57/// A role applies when it is global, scoped directly to the target resource,
58/// or scoped to any ancestor of the target resource (hierarchy inheritance).
59fn applicable_role_ids(
60    assignments: &[RoleAssignment],
61    resource_id: Uuid,
62    ancestor_ids: &HashSet<Uuid>,
63) -> Vec<Uuid> {
64    let mut seen = HashSet::new();
65    assignments
66        .iter()
67        .filter(|a| {
68            a.role.is_global
69                || a.resource_id == Some(resource_id)
70                || a.resource_id
71                    .map(|rid| ancestor_ids.contains(&rid))
72                    .unwrap_or(false)
73        })
74        .map(|a| a.role.id)
75        .filter(|role_id| seen.insert(*role_id))
76        .collect()
77}

79/// Given the applicable role IDs and the (shared) grants-by-role map, decide
80/// whether any grant matches `action` under the optional `requested_scope_id`.
81///
82/// Wildcard grants (empty `scope_ids`) match any requested scope; otherwise the
83/// requested scope must be present in the grant's scope list. When no scope is
84/// requested, an action match is sufficient.
85fn grants_allow(
86    unique_role_ids: &[Uuid],
87    grants_by_role: &HashMap<Uuid, Vec<PermissionGrant>>,
88    action: &str,
89    requested_scope_id: Option<Uuid>,
90) -> bool {
91    for role_id in unique_role_ids {
92        let Some(grants) = grants_by_role.get(role_id) else {
93            continue;
94        };
95        for grant in grants {
96            if grant.permission.action != action {
97                continue;
98            }
99            match requested_scope_id {
100                Some(scope_id) => {
101                    if grant.scope_ids.is_empty() || grant.scope_ids.contains(&scope_id) {
102                        return true;
103                    }
104                }
105                None => return true,
106            }
107        }
108    }
109    false
110}

137    /// Attach a shared [`DecisionCache`] to this engine (D7). Consumed builder
138    /// style so existing `new(..)` call sites are unaffected. `axiam-server`
139    /// calls this only when `AXIAM__AUTHZ__DECISION_CACHE_ENABLED=true`, and
140    /// passes the **same** `Arc<DecisionCache>` to every engine (REST, gRPC,
141    /// AMQP) so a REST-triggered invalidation is seen by all read paths.
142    #[must_use]
143    pub fn with_decision_cache(mut self, cache: Arc<DecisionCache>) -> Self {
144        self.decision_cache = Some(cache);
145        self
146    }

148    /// Immediately drop every cached decision for `tenant_id`. No-op when no
149    /// cache is attached. Called by the REST mutation handlers for coarse,
150    /// access-narrowing changes whose affected-subject set isn't known without
151    /// a query (grant revoke, role/permission delete or update, group-role
152    /// unassignment, resource reparent/delete).
153    pub fn invalidate_tenant(&self, tenant_id: Uuid) {
154        if let Some(cache) = self.decision_cache.as_ref() {
155            cache.invalidate_tenant(tenant_id);
156        }
157    }
158
159    /// Immediately drop every cached decision for a single `subject_id` within
160    /// `tenant_id`. No-op when no cache is attached. Called by the REST mutation
161    /// handlers when exactly one subject's effective permissions change (user
162    /// role unassign/assign, group membership add/remove).
163    pub fn invalidate_subject(&self, tenant_id: Uuid, subject_id: Uuid) {
164        if let Some(cache) = self.decision_cache.as_ref() {
165            cache.invalidate_subject(tenant_id, subject_id);
166        }
167    }

169    /// Evaluate whether the subject can perform the requested action.
170    ///
171    /// Issues 3–4 sequential DB round-trips (role assignments, ancestors,
172    /// optional scope lookup, batched grants). For batches that repeat the same
173    /// subject/resource, prefer [`Self::check_access_batch`], which resolves the
174    /// shared lookups once.
175    #[tracing::instrument(
176        name = "authz.check_access",
177        skip(self, request),
178        fields(
179            tenant_id = %request.tenant_id,
180            subject_id = %request.subject_id,
181            resource_id = %request.resource_id,
182            action = %request.action,
183            scope = request.scope.as_deref().unwrap_or("")
184        )
185    )]
186    pub async fn check_access(&self, request: &AccessRequest) -> AxiamResult<AccessDecision> {
187        // D7 fast path: when a decision cache is attached (feature-flagged),
188        // a fresh cached decision skips the DB round-trips entirely. When no
189        // cache is attached this whole block is absent and the call is
190        // identical to a build without the cache.
191        if let Some(cache) = self.decision_cache.as_ref()
192            && let Some(decision) = cache.get(request)
193        {
194            return Ok(decision);
195        }
196        let decision = self.evaluate(request).await?;
197        if let Some(cache) = self.decision_cache.as_ref() {
198            // Cache the FULL decision (allow *or* deny-with-reason), so a hit
199            // is byte-identical to a miss.
200            cache.insert(request, decision.clone());
201        }
202        Ok(decision)
203    }
204
205    /// Uncached RBAC evaluation — the actual algorithm (3–4 sequential DB
206    /// round-trips). Kept as a separate method so the [`Self::check_access`]
207    /// cache wrapper can never alter decision or deny-reason semantics: a cache
208    /// hit returns exactly what this would have produced.
209    async fn evaluate(&self, request: &AccessRequest) -> AxiamResult<AccessDecision> {
210        // 1. Fetch all role assignments (direct + group) with resource scope.
211        let assignments = self
212            .role_repo
213            .get_user_role_assignments(request.tenant_id, request.subject_id)
214            .instrument(tracing::debug_span!("db.get_user_role_assignments"))
215            .await?;
216
217        if assignments.is_empty() {
218            return Ok(AccessDecision::Deny("no roles assigned".into()));
219        }
220
221        // 2. Build the set of ancestor resource IDs for hierarchy inheritance.
222        let ancestors = self
223            .resource_repo
224            .get_ancestors(request.tenant_id, request.resource_id)
225            .instrument(tracing::debug_span!("db.get_ancestors"))
226            .await?;
227        let ancestor_ids: HashSet<Uuid> = ancestors.iter().map(|r| r.id).collect();
228
229        // 3. Filter applicable roles (global / target-scoped / ancestor-scoped),
230        //    deduplicated in first-seen order.
231        let unique_role_ids = applicable_role_ids(&assignments, request.resource_id, &ancestor_ids);
232
233        if unique_role_ids.is_empty() {
234            return Ok(AccessDecision::Deny(
235                "no applicable roles for this resource".into(),
236            ));
237        }
238
239        // 4. Resolve the requested scope to an ID (if specified).
240        let requested_scope_id = match self.resolve_scope(request).await? {
241            ScopeResolution::None => None,
242            ScopeResolution::Resolved(id) => Some(id),
243            ScopeResolution::NotFound(deny) => return Ok(deny),
244        };
245
246        // 5. Fetch every applicable role's grants in a SINGLE batched query
247        //    (CQ-B13 N+1 fix), then evaluate action + scope constraints.
248        let grants_by_role = self
249            .permission_repo
250            .get_role_permission_grants_for_roles(request.tenant_id, &unique_role_ids)
251            .instrument(tracing::debug_span!(
252                "db.get_role_permission_grants_for_roles"
253            ))
254            .await?;
255
256        if grants_allow(
257            &unique_role_ids,
258            &grants_by_role,
259            &request.action,
260            requested_scope_id,
261        ) {
262            Ok(AccessDecision::Allow)
263        } else {
264            Ok(AccessDecision::Deny(format!(
265                "no permission grants action '{}'",
266                request.action
267            )))
268        }
269    }

271    /// Resolve the request's optional scope name to a scope ID on the target
272    /// resource. Shared by [`Self::check_access`] and the batch path.
273    async fn resolve_scope(&self, request: &AccessRequest) -> AxiamResult<ScopeResolution> {
274        let Some(ref scope_name) = request.scope else {
275            return Ok(ScopeResolution::None);
276        };
277        let scopes = self
278            .scope_repo
279            .list_by_resource(request.tenant_id, request.resource_id)
280            .instrument(tracing::debug_span!("db.list_by_resource"))
281            .await?;
282        match scopes.iter().find(|s| s.name == *scope_name) {
283            Some(s) => Ok(ScopeResolution::Resolved(s.id)),
284            None => Ok(ScopeResolution::NotFound(AccessDecision::Deny(format!(
285                "scope '{}' not found on resource",
286                scope_name
287            )))),
288        }
289    }

291    /// Evaluate an ordered batch of authorization checks, coalescing the shared
292    /// DB lookups so that items repeating the same subject or resource resolve
293    /// each lookup **once** instead of once per item.
294    ///
295    /// The returned `Vec` has the same length and order as `requests`; result
296    /// `i` is the decision for `requests[i]` and is byte-identical to what
297    /// [`Self::check_access`] would return for that request in isolation.
298    ///
299    /// Round-trip reduction for the benchmark shape (5 items, one shared subject
300    /// and resource, no scope): **15 round-trips → 3** — one
301    /// `get_user_role_assignments`, one `get_ancestors`, and one batched
302    /// `get_role_permission_grants_for_roles` across every applicable role in
303    /// the batch.
304    pub async fn check_access_batch(
305        &self,
306        requests: &[AccessRequest],
307    ) -> AxiamResult<Vec<AccessDecision>> {
308        // D7: with no cache attached, this is exactly the D1 coalesced path —
309        // zero behaviour change.
310        let Some(cache) = self.decision_cache.as_ref() else {
311            return self.evaluate_batch(requests).await;
312        };
313
314        // Cache-enabled path: serve hits from the cache, evaluate only the
315        // misses (still via the coalesced batch path so shared lookups are
316        // resolved once), then backfill the cache. Input order is preserved.
317        let mut results: Vec<Option<AccessDecision>> =
318            requests.iter().map(|r| cache.get(r)).collect();
319
320        let miss_indices: Vec<usize> = results
321            .iter()
322            .enumerate()
323            .filter_map(|(i, slot)| slot.is_none().then_some(i))
324            .collect();
325
326        if !miss_indices.is_empty() {
327            let miss_requests: Vec<AccessRequest> =
328                miss_indices.iter().map(|&i| requests[i].clone()).collect();
329            let miss_decisions = self.evaluate_batch(&miss_requests).await?;
330            for (slot, &i) in miss_indices.iter().enumerate() {
331                let decision = miss_decisions[slot].clone();
332                cache.insert(&requests[i], decision.clone());
333                results[i] = Some(decision);
334            }
335        }
336
337        // Every slot is now filled (hit or freshly evaluated).
338        Ok(results.into_iter().map(|d| d.expect("filled")).collect())
339    }
340
341    /// Uncached coalesced batch evaluation (the D1 fast path). Separated from
342    /// [`Self::check_access_batch`] so the D7 cache wrapper cannot alter
343    /// decision/deny-reason semantics.
344    #[tracing::instrument(
345        name = "authz.check_access_batch",
346        skip(self, requests),
347        fields(batch_size = requests.len())
348    )]
349    async fn evaluate_batch(&self, requests: &[AccessRequest]) -> AxiamResult<Vec<AccessDecision>> {
350        if requests.is_empty() {
351            return Ok(Vec::new());
352        }
353
354        // --- Coalesce round-trip 1: role assignments, once per (tenant, subject).
355        let mut assignments_by_subject: HashMap<(Uuid, Uuid), Vec<RoleAssignment>> = HashMap::new();
356        for req in requests {
357            let key = (req.tenant_id, req.subject_id);
358            if let std::collections::hash_map::Entry::Vacant(slot) =
359                assignments_by_subject.entry(key)
360            {
361                let assignments = self
362                    .role_repo
363                    .get_user_role_assignments(req.tenant_id, req.subject_id)
364                    .instrument(tracing::debug_span!(
365                        "db.get_user_role_assignments",
366                        tenant_id = %req.tenant_id,
367                        subject_id = %req.subject_id
368                    ))
369                    .await?;
370                slot.insert(assignments);
371            }
372        }
373
374        // Only items whose subject actually has role assignments proceed past
375        // the first gate — so ancestors/scopes/grants are fetched ONLY for
376        // resources those items target. This keeps the batch's round-trip
377        // profile identical to per-item `check_access` (no wasted ancestor walk
378        // for a subject that would deny at "no roles assigned").
379        let has_roles = |req: &AccessRequest| {
380            assignments_by_subject
381                .get(&(req.tenant_id, req.subject_id))
382                .map(|a| !a.is_empty())
383                .unwrap_or(false)
384        };
385
386        // --- Coalesce round-trip 2: ancestors, once per (tenant, resource).
387        let mut ancestors_by_resource: HashMap<(Uuid, Uuid), HashSet<Uuid>> = HashMap::new();
388        for req in requests.iter().filter(|r| has_roles(r)) {
389            let key = (req.tenant_id, req.resource_id);
390            if let std::collections::hash_map::Entry::Vacant(slot) =
391                ancestors_by_resource.entry(key)
392            {
393                let ancestors = self
394                    .resource_repo
395                    .get_ancestors(req.tenant_id, req.resource_id)
396                    .instrument(tracing::debug_span!(
397                        "db.get_ancestors",
398                        tenant_id = %req.tenant_id,
399                        resource_id = %req.resource_id
400                    ))
401                    .await?;
402                slot.insert(ancestors.iter().map(|r| r.id).collect());
403            }
404        }
405
406        // --- Coalesce round-trip 3: scope lists, once per (tenant, resource)
407        //     that any scoped, role-bearing item targets.
408        let mut scopes_by_resource: HashMap<(Uuid, Uuid), Vec<axiam_core::models::scope::Scope>> =
409            HashMap::new();
410        for req in requests
411            .iter()
412            .filter(|r| r.scope.is_some() && has_roles(r))
413        {
414            let key = (req.tenant_id, req.resource_id);
415            if let std::collections::hash_map::Entry::Vacant(slot) = scopes_by_resource.entry(key) {
416                let scopes = self
417                    .scope_repo
418                    .list_by_resource(req.tenant_id, req.resource_id)
419                    .instrument(tracing::debug_span!(
420                        "db.list_by_resource",
421                        tenant_id = %req.tenant_id,
422                        resource_id = %req.resource_id
423                    ))
424                    .await?;
425                slot.insert(scopes);
426            }
427        }
428
429        // --- Per-item: compute applicable roles from the shared lookups, and
430        //     accumulate the union of role IDs so grants can be fetched ONCE.
431        //     `PreDecision` short-circuits (deny) are recorded here so the final
432        //     pass need not re-derive them.
433        enum PreDecision {
434            Decided(AccessDecision),
435            NeedsGrants {
436                unique_role_ids: Vec<Uuid>,
437                requested_scope_id: Option<Uuid>,
438            },
439        }
440
441        let mut pre: Vec<PreDecision> = Vec::with_capacity(requests.len());
442        // Union of (tenant, role_id) that we must fetch grants for, deduped.
443        let mut grant_role_ids: HashMap<Uuid, Vec<Uuid>> = HashMap::new();
444        let mut grant_role_seen: HashSet<(Uuid, Uuid)> = HashSet::new();
445
446        for (i, req) in requests.iter().enumerate() {
447            let _item_span = tracing::debug_span!(
448                "authz.batch.item",
449                index = i,
450                resource_id = %req.resource_id,
451                action = %req.action
452            )
453            .entered();
454
455            let assignments = assignments_by_subject
456                .get(&(req.tenant_id, req.subject_id))
457                .map(Vec::as_slice)
458                .unwrap_or(&[]);
459            if assignments.is_empty() {
460                pre.push(PreDecision::Decided(AccessDecision::Deny(
461                    "no roles assigned".into(),
462                )));
463                continue;
464            }
465
466            let empty_ancestors = HashSet::new();
467            let ancestor_ids = ancestors_by_resource
468                .get(&(req.tenant_id, req.resource_id))
469                .unwrap_or(&empty_ancestors);
470
471            let unique_role_ids = applicable_role_ids(assignments, req.resource_id, ancestor_ids);
472            if unique_role_ids.is_empty() {
473                pre.push(PreDecision::Decided(AccessDecision::Deny(
474                    "no applicable roles for this resource".into(),
475                )));
476                continue;
477            }
478
479            // Resolve scope against the coalesced scope list (no I/O here).
480            let requested_scope_id = if let Some(ref scope_name) = req.scope {
481                let scopes = scopes_by_resource
482                    .get(&(req.tenant_id, req.resource_id))
483                    .map(Vec::as_slice)
484                    .unwrap_or(&[]);
485                match scopes.iter().find(|s| s.name == *scope_name) {
486                    Some(s) => Some(s.id),
487                    None => {
488                        pre.push(PreDecision::Decided(AccessDecision::Deny(format!(
489                            "scope '{}' not found on resource",
490                            scope_name
491                        ))));
492                        continue;
493                    }
494                }
495            } else {
496                None
497            };
498
499            for rid in &unique_role_ids {
500                if grant_role_seen.insert((req.tenant_id, *rid)) {
501                    grant_role_ids.entry(req.tenant_id).or_default().push(*rid);
502                }
503            }
504
505            pre.push(PreDecision::NeedsGrants {
506                unique_role_ids,
507                requested_scope_id,
508            });
509        }
510
511        // --- Coalesce round-trip 4: fetch grants for every applicable role in
512        //     the batch, one batched query per tenant.
513        let mut grants_by_role: HashMap<Uuid, Vec<PermissionGrant>> = HashMap::new();
514        for (tenant_id, role_ids) in &grant_role_ids {
515            let map = self
516                .permission_repo
517                .get_role_permission_grants_for_roles(*tenant_id, role_ids)
518                .instrument(tracing::debug_span!(
519                    "db.get_role_permission_grants_for_roles",
520                    tenant_id = %tenant_id,
521                    role_count = role_ids.len()
522                ))
523                .await?;
524            grants_by_role.extend(map);
525        }
526
527        // --- Final pass: evaluate each item against the shared grants map,
528        //     preserving input order.
529        let mut decisions = Vec::with_capacity(requests.len());
530        for (item, req) in pre.into_iter().zip(requests.iter()) {
531            match item {
532                PreDecision::Decided(d) => decisions.push(d),
533                PreDecision::NeedsGrants {
534                    unique_role_ids,
535                    requested_scope_id,
536                } => {
537                    if grants_allow(
538                        &unique_role_ids,
539                        &grants_by_role,
540                        &req.action,
541                        requested_scope_id,
542                    ) {
543                        decisions.push(AccessDecision::Allow);
544                    } else {
545                        decisions.push(AccessDecision::Deny(format!(
546                            "no permission grants action '{}'",
547                            req.action
548                        )));
549                    }
550                }
551            }
552        }
553
554        Ok(decisions)
555    }
556}

558/// Outcome of resolving an optional scope name to a scope ID.
559enum ScopeResolution {
560    /// No scope was requested.
561    None,
562    /// Scope name resolved to this ID.
563    Resolved(Uuid),
564    /// Scope name was requested but does not exist on the resource — carries
565    /// the ready-made deny decision.
566    NotFound(AccessDecision),
