19/// Permission evaluation engine.
20///
21/// Implements the design-doc algorithm:
22/// 1. Fetch subject's role assignments (direct + group, with resource scope)
23/// 2. Filter applicable roles (global, scoped to target resource, or ancestor)
24/// 3. Collect permissions from applicable roles
25/// 4. Check if requested action matches any permission
26/// 5. Validate scope if specified
27/// 6. Default deny
28pub struct AuthorizationEngine<R, P, Res, S, G>
29where
30    R: RoleRepository,
31    P: PermissionRepository,
32    Res: ResourceRepository,
33    S: ScopeRepository,
34    G: GroupRepository,
35{
36    role_repo: R,
37    permission_repo: P,
38    resource_repo: Res,
39    scope_repo: S,
40    #[allow(dead_code)]
41    group_repo: G,
42    /// Optional per-tenant decision cache (D7). `None` unless
43    /// `with_decision_cache` was called (feature-flagged in `axiam-server`).
44    /// When `None`, `check_access` / `check_access_batch` behave exactly as a
45    /// build without the cache.
46    decision_cache: Option<Arc<DecisionCache>>,
47    /// How `check_access_batch` schedules its work (D10). Defaults to
48    /// [`BatchStrategy::Concurrent`]; `axiam-server` overrides from
49    /// `AXIAM__AUTHZ__BATCH_STRATEGY`. Never affects decisions or their order.
50    batch_strategy: BatchStrategy,
51    /// Bound on in-flight per-item evaluations under
52    /// [`BatchStrategy::Concurrent`] (D-07 pool-safety). Ignored by
53    /// `Coalesced`. Defaults to 16.
54    batch_max_concurrency: usize,
55}
56

59// batch paths so their decision/deny-reason semantics can never diverge.

62/// Filter a subject's role assignments down to those applicable to
63/// `resource_id`, returning the deduplicated list of applicable role IDs in
64/// first-seen order.
65///
66/// A role applies when it is global, scoped directly to the target resource,
67/// or scoped to any ancestor of the target resource (hierarchy inheritance).
68fn applicable_role_ids(
69    assignments: &[RoleAssignment],
70    resource_id: Uuid,
71    ancestor_ids: &HashSet<Uuid>,
72) -> Vec<Uuid> {
73    let mut seen = HashSet::new();
74    assignments
75        .iter()
76        .filter(|a| {
77            a.role.is_global
78                || a.resource_id == Some(resource_id)
79                || a.resource_id
80                    .map(|rid| ancestor_ids.contains(&rid))
81                    .unwrap_or(false)
82        })
83        .map(|a| a.role.id)
84        .filter(|role_id| seen.insert(*role_id))
85        .collect()
86}

88/// Given the applicable role IDs and the (shared) grants-by-role map, decide
89/// whether any grant matches `action` under the optional `requested_scope_id`.
90///
91/// Wildcard grants (empty `scope_ids`) match any requested scope; otherwise the
92/// requested scope must be present in the grant's scope list. When no scope is
93/// requested, an action match is sufficient.
94fn grants_allow(
95    unique_role_ids: &[Uuid],
96    grants_by_role: &HashMap<Uuid, Vec<PermissionGrant>>,
97    action: &str,
98    requested_scope_id: Option<Uuid>,
99) -> bool {
100    for role_id in unique_role_ids {
101        let Some(grants) = grants_by_role.get(role_id) else {
102            continue;
103        };
104        for grant in grants {
105            if grant.permission.action != action {
106                continue;
107            }
108            match requested_scope_id {
109                Some(scope_id) => {
110                    if grant.scope_ids.is_empty() || grant.scope_ids.contains(&scope_id) {
111                        return true;
112                    }
113                }
114                None => return true,
115            }
116        }
117    }
118    false
119}

174    /// Immediately drop every cached decision for `tenant_id`. No-op when no
175    /// cache is attached. Called by the REST mutation handlers for coarse,
176    /// access-narrowing changes whose affected-subject set isn't known without
177    /// a query (grant revoke, role/permission delete or update, group-role
178    /// unassignment, resource reparent/delete).
179    pub fn invalidate_tenant(&self, tenant_id: Uuid) {

182        }
183    }

186    /// `tenant_id`. No-op when no cache is attached. Called by the REST mutation

189    pub fn invalidate_subject(&self, tenant_id: Uuid, subject_id: Uuid) {

192        }
193    }

195    /// Evaluate whether the subject can perform the requested action.
196    ///
197    /// Issues 3–4 sequential DB round-trips (role assignments, ancestors,
198    /// optional scope lookup, batched grants). For batches that repeat the same
199    /// subject/resource, prefer [`Self::check_access_batch`], which resolves the
200    /// shared lookups once.
201    #[tracing::instrument(
202        name = "authz.check_access",
203        skip(self, request),
204        fields(
205            tenant_id = %request.tenant_id,
206            subject_id = %request.subject_id,
207            resource_id = %request.resource_id,
208            action = %request.action,
209            scope = request.scope.as_deref().unwrap_or("")
210        )
211    )]
212    pub async fn check_access(&self, request: &AccessRequest) -> AxiamResult<AccessDecision> {
213        // D7 fast path: when a decision cache is attached (feature-flagged),
214        // a fresh cached decision skips the DB round-trips entirely. When no
215        // cache is attached this whole block is absent and the call is
216        // identical to a build without the cache.
217        if let Some(cache) = self.decision_cache.as_ref()
218            && let Some(decision) = cache.get(request)
219        {
220            return Ok(decision);
221        }
222        let decision = self.evaluate(request).await?;
223        if let Some(cache) = self.decision_cache.as_ref() {
224            // Cache the FULL decision (allow *or* deny-with-reason), so a hit
225            // is byte-identical to a miss.
226            cache.insert(request, decision.clone());
227        }
228        Ok(decision)
229    }
230
231    /// Uncached RBAC evaluation — the actual algorithm (3–4 sequential DB
232    /// round-trips). Kept as a separate method so the [`Self::check_access`]
233    /// cache wrapper can never alter decision or deny-reason semantics: a cache
234    /// hit returns exactly what this would have produced.
235    async fn evaluate(&self, request: &AccessRequest) -> AxiamResult<AccessDecision> {
236        // 1. Fetch all role assignments (direct + group) with resource scope.
237        let assignments = self
238            .role_repo
239            .get_user_role_assignments(request.tenant_id, request.subject_id)
240            .instrument(tracing::debug_span!("db.get_user_role_assignments"))
241            .await?;
242
243        if assignments.is_empty() {
244            return Ok(AccessDecision::Deny("no roles assigned".into()));
245        }
246
247        // 2. Build the set of ancestor resource IDs for hierarchy inheritance.
248        let ancestors = self
249            .resource_repo
250            .get_ancestors(request.tenant_id, request.resource_id)
251            .instrument(tracing::debug_span!("db.get_ancestors"))
252            .await?;
253        let ancestor_ids: HashSet<Uuid> = ancestors.iter().map(|r| r.id).collect();
254
255        // 3. Filter applicable roles (global / target-scoped / ancestor-scoped),
256        //    deduplicated in first-seen order.
257        let unique_role_ids = applicable_role_ids(&assignments, request.resource_id, &ancestor_ids);
258
259        if unique_role_ids.is_empty() {
260            return Ok(AccessDecision::Deny(
261                "no applicable roles for this resource".into(),
262            ));
263        }
264
265        // 4. Resolve the requested scope to an ID (if specified).
266        let requested_scope_id = match self.resolve_scope(request).await? {
267            ScopeResolution::None => None,
268            ScopeResolution::Resolved(id) => Some(id),
269            ScopeResolution::NotFound(deny) => return Ok(deny),
270        };
271
272        // 5. Fetch every applicable role's grants in a SINGLE batched query
273        //    (CQ-B13 N+1 fix), then evaluate action + scope constraints.
274        let grants_by_role = self
275            .permission_repo
276            .get_role_permission_grants_for_roles(request.tenant_id, &unique_role_ids)
277            .instrument(tracing::debug_span!(
278                "db.get_role_permission_grants_for_roles"
279            ))
280            .await?;
281
282        if grants_allow(
283            &unique_role_ids,
284            &grants_by_role,
285            &request.action,
286            requested_scope_id,
287        ) {
288            Ok(AccessDecision::Allow)
289        } else {
290            Ok(AccessDecision::Deny(format!(
291                "no permission grants action '{}'",
292                request.action
293            )))
294        }
295    }

297    /// Resolve the request's optional scope name to a scope ID on the target
298    /// resource. Shared by [`Self::check_access`] and the batch path.
299    async fn resolve_scope(&self, request: &AccessRequest) -> AxiamResult<ScopeResolution> {
300        let Some(ref scope_name) = request.scope else {
301            return Ok(ScopeResolution::None);
302        };
303        let scopes = self
304            .scope_repo
305            .list_by_resource(request.tenant_id, request.resource_id)
306            .instrument(tracing::debug_span!("db.list_by_resource"))
307            .await?;
308        match scopes.iter().find(|s| s.name == *scope_name) {
309            Some(s) => Ok(ScopeResolution::Resolved(s.id)),
310            None => Ok(ScopeResolution::NotFound(AccessDecision::Deny(format!(
311                "scope '{}' not found on resource",
312                scope_name
313            )))),
314        }
315    }

317    /// Evaluate an ordered batch of authorization checks.
318    ///
319    /// The returned `Vec` has the same length and order as `requests`; result
320    /// `i` is the decision for `requests[i]` and is byte-identical to what
321    /// [`Self::check_access`] would return for that request in isolation — this
322    /// holds for **both** [`BatchStrategy`] variants, which differ only in how
323    /// the DB work is scheduled:
324    ///
325    /// - [`BatchStrategy::Concurrent`] (default, D10) — each item is an
326    ///   independent cache-aware [`Self::check_access`], run concurrently
327    ///   (bounded by `batch_max_concurrency`). Recovers per-item DB parallelism.
328    /// - [`BatchStrategy::Coalesced`] (D1) — shared role-assignment/ancestor/
329    ///   grant lookups resolved once per (tenant, subject|resource): 15
330    ///   round-trips → 3 for the benchmark's 5-item shape. Fewer round-trips,
331    ///   but serializes on the DB under load (see run-2 analysis).
332    pub async fn check_access_batch(
333        &self,
334        requests: &[AccessRequest],
335    ) -> AxiamResult<Vec<AccessDecision>> {
336        match self.batch_strategy {
337            BatchStrategy::Concurrent => self.evaluate_concurrent(requests).await,
338            BatchStrategy::Coalesced => self.evaluate_coalesced_cached(requests).await,
339        }
340    }
341
342    /// **D10 default** — evaluate each item as an independent, cache-aware
343    /// [`Self::check_access`], run **concurrently** with a bound of
344    /// `batch_max_concurrency` in-flight, preserving input order.
345    ///
346    /// Why this is the default (run-2 evidence): the coalesced path
347    /// ([`Self::evaluate_batch`]) minimizes round-trips but resolves the whole
348    /// batch on a single task, which serialized on the database (DB pinned at
349    /// ~1 core, ~1 s p50, everything else idle) and did not beat repeated
350    /// single checks. Single `check_access` calls, by contrast, saturated the
351    /// DB (2 cores, 745 req/s) precisely because 50 of them ran concurrently.
352    /// Evaluating a batch's items the same way recovers that parallelism.
353    /// `check_access` is reused verbatim, so every item's decision, deny reason,
354    /// cache lookup and cache insert are identical to a standalone call —
355    /// `buffered` keeps results in request order.
356    async fn evaluate_concurrent(
357        &self,
358        requests: &[AccessRequest],
359    ) -> AxiamResult<Vec<AccessDecision>> {
360        use futures::stream::{StreamExt, TryStreamExt};
361
362        // Build the per-item futures EAGERLY into a Vec before handing them to
363        // the stream. Mapping `requests.iter()` lazily inside `stream::iter`
364        // would store the `|req| self.check_access(req)` closure in the stream
365        // adapter, and when the whole `check_access_batch` future is erased
366        // behind the `AuthzChecker` trait object (`Pin<Box<dyn Future + Send>>`
367        // in axiam-api-rest / the gRPC async-trait), the borrow checker cannot
368        // prove that closure is `for<'a> FnMut(&'a AccessRequest) -> …`
369        // ("implementation of `FnOnce` is not general enough"). Collecting
370        // first applies the closure in this concrete-lifetime scope, so the
371        // boxed future holds only already-built item futures — no HRTB closure.
372        let item_futures: Vec<_> = requests.iter().map(|req| self.check_access(req)).collect();
373
374        futures::stream::iter(item_futures)
375            .buffered(self.batch_max_concurrency.max(1))
376            .try_collect()
377            .await
378    }

380    /// **D1 path** (opt-in via `AXIAM__AUTHZ__BATCH_STRATEGY=coalesced`) —
381    /// coalesced evaluation with the D7 decision cache layered on top: serve
382    /// hits from the cache, evaluate only the misses through the shared-lookup
383    /// batch path, then backfill. Input order is preserved.
384    async fn evaluate_coalesced_cached(
385        &self,
386        requests: &[AccessRequest],
387    ) -> AxiamResult<Vec<AccessDecision>> {
388        // With no cache attached, this is exactly the D1 coalesced path —
389        // zero behaviour change.
390        let Some(cache) = self.decision_cache.as_ref() else {
391            return self.evaluate_batch(requests).await;
392        };
393
394        // Cache-enabled path: serve hits from the cache, evaluate only the
395        // misses (still via the coalesced batch path so shared lookups are
396        // resolved once), then backfill the cache. Input order is preserved.
397        let mut results: Vec<Option<AccessDecision>> =
398            requests.iter().map(|r| cache.get(r)).collect();
399
400        let miss_indices: Vec<usize> = results
401            .iter()
402            .enumerate()
403            .filter_map(|(i, slot)| slot.is_none().then_some(i))
404            .collect();
405
406        if !miss_indices.is_empty() {
407            let miss_requests: Vec<AccessRequest> =
408                miss_indices.iter().map(|&i| requests[i].clone()).collect();
409            let miss_decisions = self.evaluate_batch(&miss_requests).await?;
410            for (slot, &i) in miss_indices.iter().enumerate() {
411                let decision = miss_decisions[slot].clone();
412                cache.insert(&requests[i], decision.clone());
413                results[i] = Some(decision);
414            }
415        }
416
417        // Every slot is now filled (hit or freshly evaluated).
418        Ok(results.into_iter().map(|d| d.expect("filled")).collect())
419    }
420
421    /// Uncached coalesced batch evaluation (the D1 fast path). Separated from
422    /// [`Self::check_access_batch`] so the D7 cache wrapper cannot alter
423    /// decision/deny-reason semantics.
424    #[tracing::instrument(
425        name = "authz.check_access_batch",
426        skip(self, requests),
427        fields(batch_size = requests.len())
428    )]
429    async fn evaluate_batch(&self, requests: &[AccessRequest]) -> AxiamResult<Vec<AccessDecision>> {
430        if requests.is_empty() {
431            return Ok(Vec::new());
432        }
433
434        // --- Coalesce round-trip 1: role assignments, once per (tenant, subject).
435        let mut assignments_by_subject: HashMap<(Uuid, Uuid), Vec<RoleAssignment>> = HashMap::new();
436        for req in requests {
437            let key = (req.tenant_id, req.subject_id);
438            if let std::collections::hash_map::Entry::Vacant(slot) =
439                assignments_by_subject.entry(key)
440            {
441                let assignments = self
442                    .role_repo
443                    .get_user_role_assignments(req.tenant_id, req.subject_id)
444                    .instrument(tracing::debug_span!(
445                        "db.get_user_role_assignments",
446                        tenant_id = %req.tenant_id,
447                        subject_id = %req.subject_id
448                    ))
449                    .await?;
450                slot.insert(assignments);
451            }
452        }
453
454        // Only items whose subject actually has role assignments proceed past
455        // the first gate — so ancestors/scopes/grants are fetched ONLY for
456        // resources those items target. This keeps the batch's round-trip
457        // profile identical to per-item `check_access` (no wasted ancestor walk
458        // for a subject that would deny at "no roles assigned").
459        let has_roles = |req: &AccessRequest| {
460            assignments_by_subject
461                .get(&(req.tenant_id, req.subject_id))
462                .map(|a| !a.is_empty())
463                .unwrap_or(false)
464        };
465
466        // --- Coalesce round-trip 2: ancestors, once per (tenant, resource).
467        let mut ancestors_by_resource: HashMap<(Uuid, Uuid), HashSet<Uuid>> = HashMap::new();
468        for req in requests.iter().filter(|r| has_roles(r)) {
469            let key = (req.tenant_id, req.resource_id);
470            if let std::collections::hash_map::Entry::Vacant(slot) =
471                ancestors_by_resource.entry(key)
472            {
473                let ancestors = self
474                    .resource_repo
475                    .get_ancestors(req.tenant_id, req.resource_id)
476                    .instrument(tracing::debug_span!(
477                        "db.get_ancestors",
478                        tenant_id = %req.tenant_id,
479                        resource_id = %req.resource_id
480                    ))
481                    .await?;
482                slot.insert(ancestors.iter().map(|r| r.id).collect());
483            }
484        }
485
486        // --- Coalesce round-trip 3: scope lists, once per (tenant, resource)
487        //     that any scoped, role-bearing item targets.
488        let mut scopes_by_resource: HashMap<(Uuid, Uuid), Vec<axiam_core::models::scope::Scope>> =
489            HashMap::new();
490        for req in requests
491            .iter()
492            .filter(|r| r.scope.is_some() && has_roles(r))
493        {
494            let key = (req.tenant_id, req.resource_id);
495            if let std::collections::hash_map::Entry::Vacant(slot) = scopes_by_resource.entry(key) {
496                let scopes = self
497                    .scope_repo
498                    .list_by_resource(req.tenant_id, req.resource_id)
499                    .instrument(tracing::debug_span!(
500                        "db.list_by_resource",
501                        tenant_id = %req.tenant_id,
502                        resource_id = %req.resource_id
503                    ))
504                    .await?;
505                slot.insert(scopes);
506            }
507        }
508
509        // --- Per-item: compute applicable roles from the shared lookups, and
510        //     accumulate the union of role IDs so grants can be fetched ONCE.
511        //     `PreDecision` short-circuits (deny) are recorded here so the final
512        //     pass need not re-derive them.
513        enum PreDecision {
514            Decided(AccessDecision),
515            NeedsGrants {
516                unique_role_ids: Vec<Uuid>,
517                requested_scope_id: Option<Uuid>,
518            },
519        }
520
521        let mut pre: Vec<PreDecision> = Vec::with_capacity(requests.len());
522        // Union of (tenant, role_id) that we must fetch grants for, deduped.
523        let mut grant_role_ids: HashMap<Uuid, Vec<Uuid>> = HashMap::new();
524        let mut grant_role_seen: HashSet<(Uuid, Uuid)> = HashSet::new();
525
526        for (i, req) in requests.iter().enumerate() {
527            let _item_span = tracing::debug_span!(
528                "authz.batch.item",
529                index = i,
530                resource_id = %req.resource_id,
531                action = %req.action
532            )
533            .entered();
534
535            let assignments = assignments_by_subject
536                .get(&(req.tenant_id, req.subject_id))
537                .map(Vec::as_slice)
538                .unwrap_or(&[]);
539            if assignments.is_empty() {
540                pre.push(PreDecision::Decided(AccessDecision::Deny(
541                    "no roles assigned".into(),
542                )));
543                continue;
544            }
545
546            let empty_ancestors = HashSet::new();
547            let ancestor_ids = ancestors_by_resource
548                .get(&(req.tenant_id, req.resource_id))
549                .unwrap_or(&empty_ancestors);
550
551            let unique_role_ids = applicable_role_ids(assignments, req.resource_id, ancestor_ids);
552            if unique_role_ids.is_empty() {
553                pre.push(PreDecision::Decided(AccessDecision::Deny(
554                    "no applicable roles for this resource".into(),
555                )));
556                continue;
557            }
558
559            // Resolve scope against the coalesced scope list (no I/O here).
560            let requested_scope_id = if let Some(ref scope_name) = req.scope {
561                let scopes = scopes_by_resource
562                    .get(&(req.tenant_id, req.resource_id))
563                    .map(Vec::as_slice)
564                    .unwrap_or(&[]);
565                match scopes.iter().find(|s| s.name == *scope_name) {
566                    Some(s) => Some(s.id),
567                    None => {
568                        pre.push(PreDecision::Decided(AccessDecision::Deny(format!(
569                            "scope '{}' not found on resource",
570                            scope_name
571                        ))));
572                        continue;
573                    }
574                }
575            } else {
576                None
577            };
578
579            for rid in &unique_role_ids {
580                if grant_role_seen.insert((req.tenant_id, *rid)) {
581                    grant_role_ids.entry(req.tenant_id).or_default().push(*rid);
582                }
583            }
584
585            pre.push(PreDecision::NeedsGrants {
586                unique_role_ids,
587                requested_scope_id,
588            });
589        }
590
591        // --- Coalesce round-trip 4: fetch grants for every applicable role in
592        //     the batch, one batched query per tenant.
593        let mut grants_by_role: HashMap<Uuid, Vec<PermissionGrant>> = HashMap::new();
594        for (tenant_id, role_ids) in &grant_role_ids {
595            let map = self
596                .permission_repo
597                .get_role_permission_grants_for_roles(*tenant_id, role_ids)
598                .instrument(tracing::debug_span!(
599                    "db.get_role_permission_grants_for_roles",
600                    tenant_id = %tenant_id,
601                    role_count = role_ids.len()
602                ))
603                .await?;
604            grants_by_role.extend(map);
605        }
606
607        // --- Final pass: evaluate each item against the shared grants map,
608        //     preserving input order.
609        let mut decisions = Vec::with_capacity(requests.len());
610        for (item, req) in pre.into_iter().zip(requests.iter()) {
611            match item {
612                PreDecision::Decided(d) => decisions.push(d),
613                PreDecision::NeedsGrants {
614                    unique_role_ids,
615                    requested_scope_id,
616                } => {
617                    if grants_allow(
618                        &unique_role_ids,
619                        &grants_by_role,
620                        &req.action,
621                        requested_scope_id,
622                    ) {
623                        decisions.push(AccessDecision::Allow);
624                    } else {
625                        decisions.push(AccessDecision::Deny(format!(
626                            "no permission grants action '{}'",
627                            req.action
628                        )));
629                    }
630                }
631            }
632        }
633
634        Ok(decisions)
635    }
636}
637
638/// Outcome of resolving an optional scope name to a scope ID.
639enum ScopeResolution {
640    /// No scope was requested.
641    None,
642    /// Scope name resolved to this ID.
643    Resolved(Uuid),
644    /// Scope name was requested but does not exist on the resource — carries
645    /// the ready-made deny decision.
646    NotFound(AccessDecision),
647}
