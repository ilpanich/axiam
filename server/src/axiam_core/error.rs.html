1//! Error types for the AXIAM system.
2
3use thiserror::Error;
4
5#[derive(Debug, Error)]
6pub enum AxiamError {
7    #[error("Entity not found: {entity} with id {id}")]
8    NotFound { entity: String, id: String },
9
10    #[error("Entity already exists: {entity}")]
11    AlreadyExists { entity: String },
12
13    #[error("Authentication failed: {reason}")]
14    AuthenticationFailed { reason: String },
15
16    #[error("Authorization denied: {reason}")]
17    AuthorizationDenied {
18        reason: String,
19        /// The action being checked (e.g. `"users:create"`), when known at
20        /// the denial site. Surfaced to clients so SDKs can parse it from
21        /// the 403 response body (SDK-Q02).
22        action: Option<String>,
23        /// The resource id being checked, when known and not the
24        /// "global" nil-UUID sentinel. Surfaced to clients alongside
25        /// `action` (SDK-Q02).
26        resource_id: Option<String>,
27    },
28
29    #[error("Validation error: {message}")]
30    Validation { message: String },
31
32    #[error("Password policy violation: {message}")]
33    PasswordPolicy { message: String },
34
35    #[error("Database error: {0}")]
36    Database(String),
37
38    #[error("Certificate error: {0}")]
39    Certificate(String),
40
41    #[error("Cryptography error: {0}")]
42    Crypto(String),
43
44    #[error("Email delivery failed: {0}")]
45    EmailDelivery(String),
46
47    #[error("Email configuration error: {0}")]
48    EmailConfig(String),
49
50    #[error("Webhook delivery failed: {0}")]
51    WebhookDelivery(String),
52
53    #[error("Tenant context missing or invalid")]
54    TenantContext,
55
56    #[error("Rate limit exceeded")]
57    RateLimited,
58
59    #[error("Service unavailable: {0}")]
60    ServiceUnavailable(String),
61
62    #[error("SAML assertion replay detected")]
63    ReplayDetected,
64
65    #[error("Internal error: {0}")]
66    Internal(String),
67}
68
69pub type AxiamResult<T> = Result<T, AxiamError>;
