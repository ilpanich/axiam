10/// Status of a certificate in its lifecycle.
11#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, utoipa::ToSchema)]
12pub enum CertificateStatus {
13    Active,
14    Revoked,
15    Expired,
16}

18/// The type of key algorithm used for a certificate.
19#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, utoipa::ToSchema)]
20pub enum KeyAlgorithm {
21    Rsa4096,
22    Ed25519,
23}
24
25/// The purpose for which a certificate was issued.
26#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, utoipa::ToSchema)]
27pub enum CertificateType {
28    /// Certificate for authenticating a user.
29    User,
30    /// Certificate for authenticating a service or application.
31    Service,
32    /// Certificate for authenticating an IoT device.
33    Device,
34}

36/// A CA (Certificate Authority) certificate at the organization level.
37///
38/// CA certificates are the root of trust for all tenant certificates
39/// within the organization. Private keys for signing CAs are encrypted
40/// with AES-256-GCM and stored separately; non-signing CAs only store
41/// the public certificate.
42#[derive(Clone, Serialize, Deserialize, utoipa::ToSchema)]
43pub struct CaCertificate {
44    pub id: Uuid,
45    /// The organization this CA belongs to.
46    pub organization_id: Uuid,
47    /// The certificate subject (e.g., `CN=ACME Corp Root CA`).
48    pub subject: String,
49    /// PEM-encoded public certificate.
50    pub public_cert_pem: String,
51    /// SHA-256 fingerprint of the certificate.
52    pub fingerprint: String,
53    pub key_algorithm: KeyAlgorithm,
54    /// Validity start.
55    pub not_before: DateTime<Utc>,
56    /// Validity end.
57    pub not_after: DateTime<Utc>,
58    pub status: CertificateStatus,
59    /// AES-256-GCM encrypted private key (only for signing CAs).
60    /// `None` for uploaded CAs where the private key is not stored.
61    #[serde(skip_serializing)]
62    #[schema(read_only)]
63    pub encrypted_private_key: Option<Vec<u8>>,
64    pub created_at: DateTime<Utc>,
65}
66
67/// Manual `Debug` impl (SECHRD-09 / D-06): `#[serde(skip_serializing)]` only
68/// affects `Serialize`, not `{:?}` — this closes that residual leak by
69/// redacting `encrypted_private_key` while keeping other fields readable.
70impl std::fmt::Debug for CaCertificate {
71    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
72        f.debug_struct("CaCertificate")
73            .field("id", &self.id)
74            .field("organization_id", &self.organization_id)
75            .field("subject", &self.subject)
76            .field("public_cert_pem", &self.public_cert_pem)
77            .field("fingerprint", &self.fingerprint)
78            .field("key_algorithm", &self.key_algorithm)
79            .field("not_before", &self.not_before)
80            .field("not_after", &self.not_after)
81            .field("status", &self.status)
82            .field(
83                "encrypted_private_key",
84                &self.encrypted_private_key.as_ref().map(|_| "[REDACTED]"),
85            )
86            .field("created_at", &self.created_at)
87            .finish()
88    }
89}

117/// Response returned when a CA certificate is generated.
118///
119/// Includes the private key PEM, which is returned **once** and never
120/// stored or retrievable again.
121#[derive(Clone, Serialize, Deserialize, utoipa::ToSchema)]
122pub struct GeneratedCaCertificate {
123    #[serde(flatten)]
124    pub certificate: CaCertificate,
125    /// PEM-encoded private key — returned only on generation.
126    pub private_key_pem: String,
127}

129/// Manual `Debug` impl (SECHRD-09 / D-06): redacts `private_key_pem` (raw
130/// key material returned only once on generation); delegates to
131/// `CaCertificate`'s own redacting `Debug` for the nested `certificate` field.
132impl std::fmt::Debug for GeneratedCaCertificate {
133    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
134        f.debug_struct("GeneratedCaCertificate")
135            .field("certificate", &self.certificate)
136            .field("private_key_pem", &"[REDACTED]")
137            .finish()
138    }
139}

141/// A tenant-level certificate for users, services, or IoT devices.
142///
143/// Certificates are signed by the organization's CA. The private key is
144/// returned once on generation and never stored by AXIAM.
145#[derive(Debug, Clone, Serialize, Deserialize, utoipa::ToSchema)]
146pub struct Certificate {
147    pub id: Uuid,
148    /// The tenant this certificate belongs to.
149    pub tenant_id: Uuid,
150    /// The CA certificate that signed this certificate.
151    pub issuer_ca_id: Uuid,
152    /// The certificate subject (e.g., `CN=device-001`).
153    pub subject: String,
154    /// PEM-encoded public certificate.
155    pub public_cert_pem: String,
156    /// SHA-256 fingerprint of the certificate.
157    pub fingerprint: String,
158    pub cert_type: CertificateType,
159    pub key_algorithm: KeyAlgorithm,
160    /// Validity start.
161    pub not_before: DateTime<Utc>,
162    /// Validity end.
163    pub not_after: DateTime<Utc>,
164    pub status: CertificateStatus,
165    /// Arbitrary key-value metadata (e.g., device serial, user ID binding).
166    pub metadata: serde_json::Value,
167    pub created_at: DateTime<Utc>,
168}
