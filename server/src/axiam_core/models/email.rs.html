18/// Which email backend to use.
19#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, utoipa::ToSchema)]
20#[serde(rename_all = "snake_case")]
21pub enum EmailProviderKind {
22    Smtp,
23    SendGrid,
24    Postmark,
25    Resend,
26    Brevo,
27}

29/// SMTP-specific configuration.
30///
31/// `password` is write-only (D-01): `#[serde(skip_serializing)]` means it is
32/// never emitted in a GET/serialized response. On the write path (D-02),
33/// `#[serde(default)]` lets a caller omit the field entirely (deserializing
34/// to `""`); an empty string is the sentinel for "no new secret supplied —
35/// preserve whatever is already stored" (see `SurrealEmailConfigRepository::
36/// set_org_config`). A non-empty value is a real secret to encrypt+replace.
37#[derive(Clone, PartialEq, Serialize, Deserialize, utoipa::ToSchema)]
38pub struct SmtpConfig {
39    pub host: String,
40    pub port: u16,
41    pub username: String,
42    /// Stored encrypted at rest by the DB layer. Write-only + omit-preserving;
43    /// see struct docs (D-01/D-02).
44    #[serde(default, skip_serializing)]
45    pub password: String,
46    /// Use STARTTLS (true) or implicit TLS (false).
47    pub starttls: bool,
48}
49
50/// Manual `Debug` impl (D-01, SECHRD-09 precedent): redacts `password` so
51/// `{:?}` never prints the plaintext secret while keeping every other field
52/// human-readable for logs/traces.
53impl std::fmt::Debug for SmtpConfig {
54    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
55        f.debug_struct("SmtpConfig")
56            .field("host", &self.host)
57            .field("port", &self.port)
58            .field("username", &self.username)
59            .field("password", &"[REDACTED]")
60            .field("starttls", &self.starttls)
61            .finish()
62    }
63}

65/// API-based provider configuration (SendGrid, Postmark, Resend, Brevo).
66///
67/// `api_key` follows the same write-only + omit-preserving contract as
68/// [`SmtpConfig::password`] (D-01/D-02).
69#[derive(Clone, PartialEq, Serialize, Deserialize, utoipa::ToSchema)]
70pub struct ApiProviderConfig {
71    /// Stored encrypted at rest by the DB layer. Write-only + omit-preserving;
72    /// see struct docs (D-01/D-02).
73    #[serde(default, skip_serializing)]
74    pub api_key: String,
75    /// Override base URL (useful for testing / self-hosted instances).
76    pub api_url: Option<String>,
77}
78
79/// Manual `Debug` impl (D-01, SECHRD-09 precedent): redacts `api_key` so
80/// `{:?}` never prints the plaintext secret.
81impl std::fmt::Debug for ApiProviderConfig {
82    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
83        f.debug_struct("ApiProviderConfig")
84            .field("api_key", &"[REDACTED]")
85            .field("api_url", &self.api_url)
86            .finish()
87    }
88}

90/// Provider-specific connection details.
91#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, utoipa::ToSchema)]
92#[serde(tag = "kind", rename_all = "snake_case")]
93pub enum ProviderConfig {
94    Smtp(SmtpConfig),
95    SendGrid(ApiProviderConfig),
96    Postmark(ApiProviderConfig),
97    Resend(ApiProviderConfig),
98    Brevo(ApiProviderConfig),
99}

101impl ProviderConfig {
102    pub fn kind(&self) -> EmailProviderKind {
103        match self {
104            Self::Smtp(_) => EmailProviderKind::Smtp,
105            Self::SendGrid(_) => EmailProviderKind::SendGrid,
106            Self::Postmark(_) => EmailProviderKind::Postmark,
107            Self::Resend(_) => EmailProviderKind::Resend,
108            Self::Brevo(_) => EmailProviderKind::Brevo,
109        }
110    }
111}

137/// Partial tenant overrides for email configuration.

178/// Validate an org-level email configuration.
179pub fn validate_email_config(input: &SetOrgEmailConfig) -> AxiamResult<()> {
180    let mut violations = Vec::new();
181
182    if input.from_email.is_empty() || !input.from_email.contains('@') {
183        violations.push("from_email must be a valid email address".to_string());
184    }
185
186    if input.from_name.is_empty() {
187        violations.push("from_name must not be empty".to_string());
188    }
189
190    if let Some(ref reply_to) = input.reply_to
191        && (reply_to.is_empty() || !reply_to.contains('@'))
192    {
193        violations.push("reply_to must be a valid email address if provided".to_string());
194    }
195
196    match &input.provider {
197        ProviderConfig::Smtp(smtp) => {
198            if smtp.host.is_empty() {
199                violations.push("SMTP host must not be empty".to_string());
200            }
201            if smtp.port == 0 {
202                violations.push("SMTP port must be > 0".to_string());
203            }
204        }
205        ProviderConfig::SendGrid(_)
206        | ProviderConfig::Postmark(_)
207        | ProviderConfig::Resend(_)
208        | ProviderConfig::Brevo(_) => {
209            // D-02: an empty api_key on the write path is the "no new
210            // secret supplied — preserve the stored ciphertext" sentinel,
211            // not a structural violation (mirrors SmtpConfig.password,
212            // which was never validated for emptiness either). Whether a
213            // secret is required at all (e.g. first-time creation with no
214            // prior stored value) is a repository-layer concern, not a
215            // structural validation concern.
216        }
217    }
218
219    if violations.is_empty() {
220        Ok(())
221    } else {
222        Err(AxiamError::Validation {
223            message: format!("Invalid email config: {}", violations.join("; ")),
224        })
225    }
226}

232/// Merge org email config with tenant overrides to produce effective
233/// config.
234pub fn effective_email_config(
235    org: &EmailConfig,
236    tenant_override: &EmailConfigOverride,
237    tenant_id: Uuid,
238    result_id: Uuid,
239) -> EmailConfig {
240    EmailConfig {
241        id: result_id,
242        scope: SettingsScope::Tenant,
243        scope_id: tenant_id,
244        enabled: tenant_override.enabled.unwrap_or(org.enabled),
245        from_name: tenant_override
246            .from_name
247            .clone()
248            .unwrap_or_else(|| org.from_name.clone()),
249        from_email: tenant_override
250            .from_email
251            .clone()
252            .unwrap_or_else(|| org.from_email.clone()),
253        reply_to: match &tenant_override.reply_to {
254            Some(val) => val.clone(),
255            None => org.reply_to.clone(),
256        },
257        provider: tenant_override
258            .provider
259            .clone()
260            .unwrap_or_else(|| org.provider.clone()),
261        created_at: Utc::now(),
262        updated_at: Utc::now(),
263    }
264}

266/// Build an `EmailConfig` from a `SetOrgEmailConfig` input.
267pub fn email_config_from_org_input(
268    id: Uuid,
269    org_id: Uuid,
270    input: &SetOrgEmailConfig,
271) -> EmailConfig {
272    let now = Utc::now();
273    EmailConfig {
274        id,
275        scope: SettingsScope::Org,
276        scope_id: org_id,
277        enabled: input.enabled,
278        from_name: input.from_name.clone(),
279        from_email: input.from_email.clone(),
280        reply_to: input.reply_to.clone(),
281        provider: input.provider.clone(),
282        created_at: now,
283        updated_at: now,
284    }
285}
