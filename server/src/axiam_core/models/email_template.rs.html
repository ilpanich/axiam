18/// Which email template this is.
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize, utoipa::ToSchema)]
20#[serde(rename_all = "snake_case")]
21pub enum TemplateKind {
22    Activation,
23    PasswordReset,
24    MfaSetupReminder,
25    AdminNotification,
26    /// Sent when an account deletion is scheduled (includes cancel link). D-09.
27    DeletionScheduled,
28    /// Sent when a data-export file is ready for download. D-12.
29    ExportReady,
30}
31
32impl TemplateKind {
33    /// All defined template kinds.
34    pub const ALL: &[Self] = &[
35        Self::Activation,
36        Self::PasswordReset,
37        Self::MfaSetupReminder,
38        Self::AdminNotification,
39        Self::DeletionScheduled,
40        Self::ExportReady,
41    ];
42
43    /// Return a built-in default template for this kind.
44    ///
45    /// Placeholder tokens follow the `{{name}}` syntax understood by the
46    /// template renderer.  Available tokens per kind:
47    /// - `DeletionScheduled`: `username`, `tenant_name`, `action_url` (cancel link), `expiry_time`
48    /// - `ExportReady`: `username`, `tenant_name`, `action_url` (download link), `expiry_time`
49    pub fn builtin_template(&self) -> (&'static str, &'static str, &'static str) {
50        // (subject, html_body, text_body)
51        match self {
52            Self::DeletionScheduled => (
53                "Your account deletion has been scheduled",
54                "<p>Hi {{username}},</p>\
55                 <p>Your account on <strong>{{tenant_name}}</strong> is scheduled for \
56                 deletion on <strong>{{expiry_time}}</strong>.</p>\
57                 <p>If you did not request this, \
58                 <a href=\"{{action_url}}\">click here to cancel</a> before that date.</p>",
59                "Hi {{username}},\n\nYour account on {{tenant_name}} is scheduled for \
60                 deletion on {{expiry_time}}.\n\nTo cancel, visit: {{action_url}}",
61            ),
62            Self::ExportReady => (
63                "Your data export is ready",
64                "<p>Hi {{username}},</p>\
65                 <p>Your data export for <strong>{{tenant_name}}</strong> is ready. \
66                 <a href=\"{{action_url}}\">Download it here</a> — the link expires at \
67                 <strong>{{expiry_time}}</strong> and can only be used once.</p>",
68                "Hi {{username}},\n\nYour data export for {{tenant_name}} is ready.\n\
69                 Download: {{action_url}}\nExpires: {{expiry_time}}\nSingle-use link.",
70            ),
71            _ => ("", "", ""),
72        }
73    }
74}

76impl std::fmt::Display for TemplateKind {
77    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
78        match self {
79            Self::Activation => write!(f, "activation"),
80            Self::PasswordReset => write!(f, "password_reset"),
81            Self::MfaSetupReminder => write!(f, "mfa_setup_reminder"),
82            Self::AdminNotification => write!(f, "admin_notification"),
83            Self::DeletionScheduled => write!(f, "deletion_scheduled"),
84            Self::ExportReady => write!(f, "export_ready"),
85        }
86    }
87}

89impl std::str::FromStr for TemplateKind {
90    type Err = String;
91
92    fn from_str(s: &str) -> Result<Self, Self::Err> {
93        match s {
94            "activation" => Ok(Self::Activation),
95            "password_reset" => Ok(Self::PasswordReset),
96            "mfa_setup_reminder" => Ok(Self::MfaSetupReminder),
97            "admin_notification" => Ok(Self::AdminNotification),
98            "deletion_scheduled" => Ok(Self::DeletionScheduled),
99            "export_ready" => Ok(Self::ExportReady),
100            other => Err(format!("invalid template kind: {other}")),
101        }
102    }
103}

144/// Validate a template input.
145pub fn validate_email_template(input: &SetEmailTemplate) -> AxiamResult<()> {
146    let mut violations = Vec::new();
147
148    if input.subject.trim().is_empty() {
149        violations.push("subject must not be empty".to_string());
150    }
151    if input.html_body.trim().is_empty() {
152        violations.push("html_body must not be empty".to_string());
153    }
154    if input.text_body.trim().is_empty() {
155        violations.push("text_body must not be empty".to_string());
156    }
157
158    if violations.is_empty() {
159        Ok(())
160    } else {
161        Err(AxiamError::Validation {
162            message: format!("Invalid email template: {}", violations.join("; ")),
163        })
164    }
165}
