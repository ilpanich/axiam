1//! Outbound mail message types for async delivery (D-14).
2//!
3//! These types are defined in `axiam-core` so that both `axiam-audit`
4//! (which cannot depend on `axiam-amqp` due to the circular-dep constraint)
5//! and `axiam-api-rest` can build outbound mail messages without coupling to
6//! the AMQP infrastructure layer.  `axiam-amqp` re-exports these types and
7//! provides the concrete AMQP publishing implementation.
8
9use chrono::{DateTime, Utc};
10use serde::{Deserialize, Serialize};
11use uuid::Uuid;
12
13/// Mail type variants for async outbound delivery (D-14).
14#[derive(Debug, Clone, Serialize, Deserialize)]
15#[serde(rename_all = "snake_case")]
16pub enum MailType {
17    PasswordReset,
18    EmailVerification,
19    Notification,
20    DeletionCancel,
21    ExportReady,
22}
23
24/// Outbound mail message published to `axiam.mail.outbound` (D-14).
25///
26/// All fields except `to_address` are safe to include in audit metadata.
27/// `to_address` is present for delivery only — **MUST NOT** be logged in
28/// audit events (D-16).  Consumers MUST exclude it from any audit records
29/// they produce.
30#[derive(Debug, Clone, Serialize, Deserialize)]
31pub struct OutboundMailMessage {
32    pub mail_type: MailType,
33    pub tenant_id: Uuid,
34    pub org_id: Uuid,
35    pub user_id: Uuid,
36    /// Delivery address — present for SMTP delivery only.
37    /// **MUST NOT** appear in audit log metadata (D-16).
38    pub to_address: String,
39    pub template_context: serde_json::Value,
40    pub attempt_count: u32,
41    pub enqueued_at: DateTime<Utc>,
42}
