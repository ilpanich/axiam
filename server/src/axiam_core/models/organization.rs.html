1//! Organization domain model.
2//!
3//! Organizations are the top-level entity in AXIAM's multi-tenancy hierarchy.
4//! They contain tenants and hold CA certificates for signing tenant certificates.
5
6use chrono::{DateTime, Utc};
7use serde::{Deserialize, Serialize};
8use uuid::Uuid;
9
10/// An organization groups multiple tenants under a single administrative entity.
11///
12/// Organizations represent companies, departments, or business units.
13/// CA certificates are registered at the organization level, enabling
14/// a hierarchical trust model across all tenants.
15#[derive(Debug, Clone, Serialize, Deserialize, utoipa::ToSchema)]
16pub struct Organization {
17    pub id: Uuid,
18    /// Human-readable name.
19    pub name: String,
20    /// URL-safe unique identifier (e.g., `acme-corp`).
21    pub slug: String,
22    /// Arbitrary key-value metadata.
23    pub metadata: serde_json::Value,
24    pub created_at: DateTime<Utc>,
25    pub updated_at: DateTime<Utc>,
26}
27
28/// Fields required to create a new organization.
29#[derive(Debug, Clone, Serialize, Deserialize, utoipa::ToSchema)]
30pub struct CreateOrganization {
31    pub name: String,
32    pub slug: String,
33    pub metadata: Option<serde_json::Value>,
34}
35
36/// Fields that can be updated on an existing organization.
37#[derive(Debug, Clone, Serialize, Deserialize, Default, utoipa::ToSchema)]
38pub struct UpdateOrganization {
39    pub name: Option<String>,
40    pub slug: Option<String>,
41    pub metadata: Option<serde_json::Value>,
42}
