1//! User domain model.
2
3use chrono::{DateTime, Utc};
4use serde::{Deserialize, Serialize};
5use uuid::Uuid;
6
7#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, utoipa::ToSchema)]
8pub enum UserStatus {
9    Active,
10    Inactive,
11    Locked,
12    PendingVerification,
13    /// User has been anonymized in-place following Art. 17 erasure (D-05).
14    Anonymized,
15}
16
17#[derive(Debug, Clone, Serialize, Deserialize, utoipa::ToSchema)]
18pub struct User {
19    pub id: Uuid,
20    pub tenant_id: Uuid,
21    pub username: String,
22    pub email: String,
23    pub password_hash: String,
24    pub status: UserStatus,
25    pub mfa_enabled: bool,
26    /// AES-256-GCM encrypted TOTP secret (if MFA is enrolled).
27    pub mfa_secret: Option<String>,
28    /// The TOTP time-step counter value last successfully verified.
29    ///
30    /// Used to prevent replay attacks: a code whose step ≤ this value is
31    /// rejected even if the HMAC is correct (SEC-008).
32    pub totp_last_used_step: Option<u64>,
33    pub failed_login_attempts: u32,
34    pub last_failed_login_at: Option<DateTime<Utc>>,
35    pub locked_until: Option<DateTime<Utc>>,
36    pub email_verified_at: Option<DateTime<Utc>>,
37    /// GDPR Art. 17 — set when user requests account deletion (D-08).
38    pub deletion_pending: bool,
39    /// Scheduled purge date when `deletion_pending` is true (D-08).
40    pub scheduled_purge_at: Option<DateTime<Utc>>,
41    pub metadata: serde_json::Value,
42    pub created_at: DateTime<Utc>,
43    pub updated_at: DateTime<Utc>,
44}
45
46#[derive(Debug, Clone, Serialize, Deserialize, utoipa::ToSchema)]
47pub struct CreateUser {
48    pub tenant_id: Uuid,
49    pub username: String,
50    pub email: String,
51    /// Raw password (will be hashed with Argon2id before storage).
52    pub password: String,
53    pub metadata: Option<serde_json::Value>,
54}
55
56#[derive(Debug, Clone, Serialize, Deserialize, Default, utoipa::ToSchema)]
57pub struct UpdateUser {
58    pub username: Option<String>,
59    pub email: Option<String>,
60    /// Internal-only field set programmatically after Argon2id hashing.
61    /// Never accepted from or exposed to API consumers.
62    #[serde(skip)]
63    #[schema(ignore = true)]
64    pub password_hash: Option<String>,
65    pub status: Option<UserStatus>,
66    pub metadata: Option<serde_json::Value>,
67    pub mfa_enabled: Option<bool>,
68    /// `Some(Some(val))` = set, `Some(None)` = clear, `None` = no change.
69    pub mfa_secret: Option<Option<String>>,
70    /// `Some(Some(val))` = set, `Some(None)` = clear, `None` = no change.
71    pub totp_last_used_step: Option<Option<u64>>,
72    pub failed_login_attempts: Option<u32>,
73    pub last_failed_login_at: Option<Option<DateTime<Utc>>>,
74    pub locked_until: Option<Option<DateTime<Utc>>>,
75    /// `Some(Some(val))` = set, `Some(None)` = clear, `None` = no change.
76    pub email_verified_at: Option<Option<DateTime<Utc>>>,
77}
