1//! Database-specific error types and conversions.
2
3use axiam_core::error::AxiamError;
4
5/// Database-layer error type.
6#[derive(Debug, thiserror::Error)]
7pub enum DbError {
8    #[error("SurrealDB error: {0}")]
9    Surreal(#[from] surrealdb::Error),
10
11    /// Authentication is expired/unrecoverable (root token expiry or a
12    /// genuinely revoked/invalid credential — CORR-02/D-05). Distinct from
13    /// [`DbError::Surreal`] so a readiness probe can alarm on this
14    /// specifically rather than treating it as an ordinary, possibly
15    /// transient query error.
16    #[error("SurrealDB authentication unhealthy: {0}")]
17    Unhealthy(String),
18
19    #[error("Migration failed: {0}")]
20    Migration(String),
21
22    #[error("Record not found: {entity} with id {id}")]
23    NotFound { entity: String, id: String },
24
25    #[error("Record already exists: {entity}")]
26    AlreadyExists { entity: String },
27
28    /// A value read back from SurrealDB could not be deserialized/parsed into
29    /// its expected Rust type (e.g. a corrupt UUID column). Distinct from
30    /// [`DbError::Migration`] so this class of error is never mislabeled as a
31    /// schema-migration failure (QUAL-03/D-10). Falls through the same
32    /// `other => AxiamError::Database` catch-all below, so the observable
33    /// HTTP status is unchanged (still 5xx) — this is a log-clarity fix only.
34    #[error("Data serialization error: {0}")]
35    Serialization(String),
36}
37
38impl From<DbError> for AxiamError {
39    fn from(err: DbError) -> Self {
40        match err {
41            DbError::NotFound { entity, id } => AxiamError::NotFound { entity, id },
42            DbError::AlreadyExists { entity } => AxiamError::AlreadyExists { entity },
43            other => AxiamError::Database(other.to_string()),
44        }
45    }
46}
