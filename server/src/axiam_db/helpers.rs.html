23// ---------------------------------------------------------------------------
24// parse_uuid — typed parse, no longer misusing DbError::Migration (CQ-B11)
25// ---------------------------------------------------------------------------
26
27/// Parse a UUID string coming out of SurrealDB, embedding the field name in
28/// the error message for easier debugging.
29///
30/// Unlike the inline `Uuid::parse_str(..).map_err(|e| DbError::Migration(…))`
31/// pattern scattered across ~25 repos, this function names the offending
32/// field and reports a corrupt-data read as `DbError::Serialization` — a
33/// malformed value coming back out of a read is not a schema-migration
34/// failure (QUAL-03/D-10).
35pub fn parse_uuid(s: &str, field: &str) -> Result<Uuid, DbError> {
36    s.parse::<Uuid>()
37        .map_err(|e| DbError::Serialization(format!("invalid {field} UUID: {e}")))
38}

40// ---------------------------------------------------------------------------
41// classify_write_error — centralized index-violation detection (QUAL-03/D-09)
42// ---------------------------------------------------------------------------
43
44/// Classify a write-path error (from a `CREATE`/`RELATE` statement's
45/// `.check()` or an upstream mapped error) into the correct [`DbError`]
46/// variant.
47///
48/// Reuses the exact marker-string set already proven correct against this
49/// codebase's SurrealDB version by `saml_replay.rs`, `federation_login_state.rs`,
50/// and `seeder.rs`: a genuine unique/index violation surfaces one of
51/// `"already contains"`, `"already exists"`, or `"unique"` in the error text.
52/// Those map to [`DbError::AlreadyExists`] (→ HTTP 409). Anything else —
53/// crucially including a DB outage/connection error, which contains none of
54/// these markers — falls through to [`DbError::Migration`] (→ 5xx). A DB
55/// outage must never be misclassified as a false 409.
56///
57/// Generic over `E: Display` (rather than a single concrete error type) so
58/// the same centralized detector can be called uniformly from every
59/// `.map_err` site on the write paths this phase routes through it,
60/// regardless of whether the site's error came from `Response::check()`
61/// (`surrealdb::Error`) or another fallible step in the same write path.
62///
63/// Per D-09, this is the ONLY place that inspects error text for these
64/// markers — call sites must not add their own inline `contains(...)` checks.
65pub fn classify_write_error<E: std::fmt::Display>(err: E, entity: &str) -> DbError {
66    let msg = err.to_string();
67    if msg.contains("already contains") || msg.contains("already exists") || msg.contains("unique")
68    {
69        DbError::AlreadyExists {
70            entity: entity.to_string(),
71        }
72    } else {
73        DbError::Migration(msg)
74    }
75}

77// ---------------------------------------------------------------------------
78// take_first_or_not_found — unified "0 rows → NotFound" helper
79// ---------------------------------------------------------------------------
80
81/// Take the first element of `items` or return `DbError::NotFound`.
82///
83/// Replaces the `into_iter().next().ok_or_else(|| DbError::NotFound{…})`
84/// pattern repeated in every `get_by_id` / find method.
85pub fn take_first_or_not_found<T>(items: Vec<T>, entity: &str, id: &str) -> Result<T, DbError> {
86    items.into_iter().next().ok_or_else(|| DbError::NotFound {
87        entity: entity.to_string(),
88        id: id.to_string(),
89    })
90}

92// ---------------------------------------------------------------------------
93// paginate — shared PaginatedResult<T> construction (QUAL-02/CQ-B10)
94// ---------------------------------------------------------------------------
95
96/// Assemble a [`PaginatedResult<T>`] from already-fetched `items`, the
97/// `count_rows` from a `SELECT count() AS total ... GROUP ALL` query, and the
98/// `pagination` window that was applied to the data query.
99///
100/// Replaces the `let total = count_rows.first().map(|r| r.total).unwrap_or(0);
101/// Ok(PaginatedResult { items, total, offset, limit })` tail duplicated across
102/// every repo's list method (CQ-B10).
103pub fn paginate<T>(
104    items: Vec<T>,
105    count_rows: Vec<CountRow>,
106    pagination: &Pagination,
107) -> PaginatedResult<T> {
108    let total = count_rows.first().map(|r| r.total).unwrap_or(0);
109    PaginatedResult {
110        items,
111        total,
112        offset: pagination.offset,
113        limit: pagination.limit,
114    }
115}
