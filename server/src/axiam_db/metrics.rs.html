1//! Zero-behavior-change instrumentation for the `axiam-db` boundary (F1).
2//!
3//! ## Why instrument here and not inside `surrealdb`
4//!
5//! The pinned `surrealdb` 3.2.1 HTTP engine spawns a tokio task **per request**
6//! over one shared `reqwest::Client`, and every `Surreal::clone()` shares the
7//! same `Arc<inner>` — so the whole process funnels DB traffic through a single
8//! router dispatch task with **no upper bound** on concurrency (see
9//! `connection.rs` module docs and `claude_dev/db-pool-design.md`). The crate
10//! exposes no hook into that router channel, so we cannot measure its internal
11//! queue depth directly. Instead we instrument **our own** call path — the point
12//! at which `axiam-db` hands a query to the SurrealDB client — which is the
13//! boundary a `DbPool` (F2) will own.
14//!
15//! ## What this module provides
16//!
17//! * [`db_in_flight`] — a process-wide gauge of DB requests currently awaiting a
18//!   response through the `axiam-db` boundary. Under the current single-funnel
19//!   architecture this is the count of requests contending for the one shared
20//!   dispatcher; after F2 it becomes the pool's aggregate in-flight count that
21//!   the `AXIAM__DB__POOL_MAX_IN_FLIGHT` semaphore bounds.
22//! * [`db_handle_checkouts`] — a monotonic counter of how many `Surreal<Client>`
23//!   handles have been handed out via [`DbManager::client_cloned`]. Today every
24//!   one of these is a clone that shares the single router — so this counter
25//!   directly quantifies the "single funnel" fan-in (~30 repositories at
26//!   startup). After F2 the equivalent counter measures per-handle checkouts.
27//! * [`instrument_query`] — a **transparent passthrough** wrapper that measures
28//!   spawn→response latency and maintains the in-flight gauge around a query
29//!   future, emitting a structured `tracing` event on completion.
30//!
31//! ## Zero behavior change (invariant)
32//!
33//! Nothing here introduces a new `.await` that alters ordering or latency
34//! semantics of the wrapped work: [`instrument_query`] awaits exactly the future
35//! it was given and returns its output unchanged; the gauge is a relaxed atomic
36//! inc/dec; the emitted `tracing` event is a no-op when no subscriber is
37//! attached. A gauge/counter read never blocks. F2 will route the repository
38//! query path through [`instrument_query`]; F1 only adds the machinery and wires
39//! the two boundary choke points that already exist today ([`DbManager::client_cloned`]
40//! and [`DbManager::health_check`]).
41
42use std::future::IntoFuture;
43use std::sync::atomic::{AtomicI64, AtomicU64, Ordering};
44use std::time::Instant;
45
46use tracing::trace;
47
48/// Process-wide count of DB requests currently in flight through the
49/// `axiam-db` boundary (incremented on entry to [`instrument_query`],
50/// decremented when its future resolves — even on error/panic, via the RAII
51/// guard). Relaxed ordering is sufficient: this is an observability gauge, not
52/// a synchronization primitive.
53static DB_IN_FLIGHT: AtomicI64 = AtomicI64::new(0);
54
55/// Monotonic count of `Surreal<Client>` handles handed out by
56/// [`DbManager::client_cloned`]. See module docs — under today's
57/// single-router architecture this measures the single-funnel fan-in.
58static DB_HANDLE_CHECKOUTS: AtomicU64 = AtomicU64::new(0);
59
60/// Current number of DB requests in flight through the `axiam-db` boundary.
61/// Public so tests (and F2's pool) can assert the gauge tracks entry/exit.
62pub fn db_in_flight() -> i64 {
63    DB_IN_FLIGHT.load(Ordering::Relaxed)
64}
65
66/// Total handles checked out via [`DbManager::client_cloned`] since process
67/// start. Public for tests and for a future readiness/metrics endpoint.
68pub fn db_handle_checkouts() -> u64 {
69    DB_HANDLE_CHECKOUTS.load(Ordering::Relaxed)
70}
71
72/// Record that a client handle was checked out (called from
73/// [`DbManager::client_cloned`]). Zero behavior change — a single relaxed
74/// atomic increment plus a `trace` event.
75pub(crate) fn record_handle_checkout() {
76    let n = DB_HANDLE_CHECKOUTS.fetch_add(1, Ordering::Relaxed) + 1;
77    trace!(
78        target: "axiam_db::metrics",
79        handle_checkouts_total = n,
80        "axiam-db client handle checked out (single-router funnel until F2 pool lands)"
81    );
82}

84/// RAII gauge guard: increments [`DB_IN_FLIGHT`] on construction and
85/// decrements it on drop, so the gauge is correct even if the guarded future
86/// is cancelled or panics. Not exported directly — callers use
87/// [`instrument_query`], but F2 may reuse this pattern for the pool's own
88/// checkout guard.
89struct InFlightGuard {
90    op: &'static str,
91    started: Instant,
92    peak: i64,
93}
94
95impl InFlightGuard {
96    fn enter(op: &'static str) -> Self {
97        let peak = DB_IN_FLIGHT.fetch_add(1, Ordering::Relaxed) + 1;
98        Self {
99            op,
100            started: Instant::now(),
101            peak,
102        }
103    }
104}
105
106impl Drop for InFlightGuard {
107    fn drop(&mut self) {
108        DB_IN_FLIGHT.fetch_sub(1, Ordering::Relaxed);
109        // Emit spawn→response latency at TRACE so it is free when no subscriber
110        // is attached and opt-in when the operator wants DB-boundary timing.
111        trace!(
112            target: "axiam_db::metrics",
113            op = self.op,
114            latency_us = self.started.elapsed().as_micros() as u64,
115            in_flight_at_entry = self.peak,
116            "axiam-db query completed"
117        );
118    }
119}

121/// Wrap a DB query future so the in-flight gauge and spawn→response latency are
122/// observed at the `axiam-db` boundary.
123///
124/// **Transparent passthrough:** awaits exactly `fut` and returns its output
125/// unchanged — no added await points, no reordering, no altered error
126/// semantics. Safe to drop onto the hot path (F2 routes the repository query
127/// path through here); F1 wires only the choke points that exist today.
128pub async fn instrument_query<F, T>(op: &'static str, fut: F) -> T
129where
130    // `IntoFuture` (not `Future`) so this accepts SurrealDB's `Query`/`Select`/…
131    // request builders directly — they implement `IntoFuture`, not `Future`, and
132    // `.await` drives them through it. Plain futures also satisfy this bound.
133    F: IntoFuture<Output = T>,
134{
135    let _guard = InFlightGuard::enter(op);
136    fut.await
137}

139#[cfg(test)]
140mod tests {
141    use tokio::sync::Mutex;
142
143    use super::*;
144
145    /// `DB_IN_FLIGHT` is a single process-wide static, and `cargo test` runs
146    /// `#[tokio::test]` functions in parallel across threads within the same
147    /// test binary — so two tests that both read/assert the gauge's absolute
148    /// value can race (one test's `baseline` capture landing between another
149    /// concurrently-running test's increment and decrement), causing an
150    /// intermittent, non-deterministic failure even though the gauge logic
151    /// itself is correct (observed once in CI: `left: 0, right: 1`). Every
152    /// test below that touches `DB_IN_FLIGHT` (directly or via
153    /// `instrument_query`) holds this lock for its full body so the shared
154    /// static's mutations never interleave across tests.
155    ///
156    /// `tokio::sync::Mutex` (not `std::sync::Mutex`) deliberately — the lock
157    /// is held across an `.await` point below, and holding a std mutex guard
158    /// across an await is an anti-pattern clippy correctly flags
159    /// (`await_holding_lock`): it isn't designed to be released by a
160    /// different task/thread than the one that acquired it.
161    static GAUGE_TEST_LOCK: Mutex<()> = Mutex::const_new(());
162
163    #[test]
164    fn handle_checkout_counter_increments_monotonically() {
165        // DB_HANDLE_CHECKOUTS is monotonic-only (never decremented), so a
166        // delta assertion is race-safe without the gauge lock even under
167        // parallel test execution.
168        let before = db_handle_checkouts();
169        record_handle_checkout();
170        record_handle_checkout();
171        assert_eq!(
172            db_handle_checkouts(),
173            before + 2,
174            "checkout counter must increment by exactly the number of checkouts"
175        );
176    }
177
178    #[tokio::test]
179    async fn in_flight_gauge_is_incremented_during_and_restored_after() {
180        let _serialize = GAUGE_TEST_LOCK.lock().await;
181        let baseline = db_in_flight();
182        // The gauge must read baseline+1 *inside* the instrumented future and
183        // return to baseline once it resolves.
184        let observed_inside = instrument_query("test.op", async { db_in_flight() }).await;
185        assert_eq!(
186            observed_inside,
187            baseline + 1,
188            "gauge must reflect the in-flight request while the future runs"
189        );
190        assert_eq!(
191            db_in_flight(),
192            baseline,
193            "gauge must return to baseline after the future resolves"
194        );
195    }
196
197    #[tokio::test]
198    async fn instrument_query_returns_inner_output_unchanged() {
199        let _serialize = GAUGE_TEST_LOCK.lock().await;
200        // Passthrough invariant: the wrapper must yield exactly the inner value.
201        let out = instrument_query("test.passthrough", async { 40 + 2 }).await;
202        assert_eq!(out, 42);
203    }
204
205    #[tokio::test]
206    async fn gauge_restored_even_when_future_errors() {
207        let _serialize = GAUGE_TEST_LOCK.lock().await;
208        let baseline = db_in_flight();
209        let r: Result<(), &str> = instrument_query("test.err", async { Err("boom") }).await;
210        assert!(r.is_err());
211        assert_eq!(
212            db_in_flight(),
213            baseline,
214            "gauge must be decremented on the error path too (RAII drop)"
215        );
216    }
217}
