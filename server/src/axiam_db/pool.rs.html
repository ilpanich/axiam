1//! Database connection pool (F2) — N independent `Surreal<Client>` handles.
2//!
3//! See `claude_dev/db-pool-design.md` for the full design rationale. In short:
4//!
5//! * The pinned `surrealdb` 3.2.1 HTTP engine offers **no** native connection
6//!   pool, and every `Surreal::clone()` shares one `Arc<inner>` (one router
7//!   task + one `reqwest` client). So the only way to get genuine dispatch
8//!   parallelism, N independent TCP pools, and independently-renewable sessions
9//!   is N separate `Surreal::new::<Http>(..) + signin` connections — exactly
10//!   what [`DbManager::connect_handle`] builds. [`DbPool`] holds N of them.
11//! * Each pooled handle carries its OWN proactive re-signin + reconnect loop
12//!   (the D-04/PERF-04 machinery previously applied only to the manager's
13//!   handle), so **CQ-B48 is closed**: no request can reach an un-renewed
14//!   snapshot session, and the former restart-only ~4-week token outage is gone.
15//! * A process-wide `tokio::sync::Semaphore` optionally bounds total concurrent
16//!   in-flight DB ops (the stampede fix, analogous to B1's Argon2id gate). On
17//!   acquire-timeout the checkout returns the **existing** overload error,
18//!   [`AxiamError::ServiceUnavailable`] → HTTP 503 — reusing B1's taxonomy
19//!   exactly (see [`acquire_db_permit`]).
20//!
21//! ## Safe-rollout invariant (`pool_size = 1`, cap disabled — the default)
22//!
23//! With the default [`DbConfig`] (`pool_size = 1`, `pool_max_in_flight = 0`)
24//! the pool is observably identical to today: exactly one handle (so
25//! least-in-flight selection and round-robin are no-ops), and **no** semaphore
26//! is built — [`DbPool::checkout`] never acquires a permit and the
27//! acquire-timeout path can never fire, preserving today's unbounded DB
28//! concurrency byte-for-byte. `pool_size > 1` / a positive cap is opt-in.
29

45/// One pooled handle: an independent `Surreal<C>` connection behind the same
46/// swappable `Arc<RwLock<..>>` [`DbManager`] uses (D-12, so its own reconnect
47/// loop can evict a poisoned handle), plus this handle's in-flight count
48/// (least-in-flight checkout input) and its two owned background tasks.
49struct PooledHandle<C: Connection> {
50    /// Swappable so the per-handle reconnect loop can atomically evict a
51    /// poisoned handle (D-12) — identical to `DbManager::db`.
52    db: Arc<RwLock<Surreal<C>>>,
53    /// In-flight count for THIS handle — the least-in-flight selection input.
54    /// Incremented by [`DbPool::checkout`], decremented when the returned
55    /// [`DbCheckout`] guard drops.
56    in_flight: Arc<AtomicUsize>,
57    /// Proactive re-signin task (D-04) for this handle. `None` for test-only
58    /// pools built over the embedded `kv-mem` engine, whose tokens never expire.
59    refresh_handle: Option<JoinHandle<()>>,
60    /// Reconnect loop (PERF-04) for this handle. `None` for `kv-mem` test pools.
61    reconnect_handle: Option<JoinHandle<()>>,
62}

64/// A pool of N independent, individually-renewable SurrealDB handles.
65///
66/// Generic over the connection type so the checkout/selection logic can be
67/// unit-tested against the embedded `kv-mem` engine (`DbPool<Db>`) where a live
68/// server would otherwise be required; production always uses the default
69/// `DbPool<Client>` (the HTTP engine) built via [`DbPool::connect`].
70pub struct DbPool<C: Connection = Client> {
71    handles: Vec<PooledHandle<C>>,
72    /// `None` when the in-flight cap is disabled (`pool_max_in_flight == 0`,
73    /// the default) — checkout then never acquires a permit, so today's
74    /// unbounded concurrency is preserved exactly.
75    semaphore: Option<Arc<Semaphore>>,
76    /// How long [`DbPool::checkout`] waits for a permit before returning the
77    /// overload error. Only consulted when `semaphore` is `Some`.
78    acquire_timeout: Duration,
79    /// Round-robin cursor for [`DbPool::handle_for_repo`] construction-time
80    /// binding (spreads repositories evenly across handles at startup).
81    rr: AtomicUsize,
82}
83
84impl DbPool<Client> {
85    /// Connect and build the pool using the fixed production
86    /// [`ROOT_TOKEN_DURATION`]. Mirrors [`DbManager::connect`].
87    pub async fn connect(config: &DbConfig) -> Result<Self, surrealdb::Error> {
88        Self::connect_with_ttl(config, ROOT_TOKEN_DURATION).await
89    }
90
91    /// Build a pool of `config.pool_size` INDEPENDENT handles (each via the
92    /// shared [`DbManager::connect_handle`] path — its own router task, its own
93    /// `reqwest` client, its own proactive re-signin + reconnect loop). The
94    /// root-token duration is extended once up-front (the DEFINE USER is
95    /// idempotent), then each handle signs in fresh and minting its own
96    /// long-lived token.
97    ///
98    /// `ttl` is a test hook (short TTLs prove per-handle session renewal without
99    /// waiting four weeks); production passes [`ROOT_TOKEN_DURATION`] via
100    /// [`connect`](Self::connect).
101    pub async fn connect_with_ttl(
102        config: &DbConfig,
103        ttl: Duration,
104    ) -> Result<Self, surrealdb::Error> {
105        // Guard against a misconfigured `pool_size = 0`: at least one handle.
106        let size = config.pool_size.max(1);
107        tracing::info!(
108            pool_size = size,
109            pool_max_in_flight = config.pool_max_in_flight,
110            "Building SurrealDB connection pool"
111        );
112
113        // Idempotent, once per process even for N handles (D-04/CORR-02).
114        DbManager::extend_root_token_duration(config, ttl).await;
115
116        let mut handles = Vec::with_capacity(size);
117        for _ in 0..size {
118            let (db, refresh, reconnect) = DbManager::connect_handle(config, ttl).await?;
119            handles.push(PooledHandle {
120                db,
121                in_flight: Arc::new(AtomicUsize::new(0)),
122                refresh_handle: Some(refresh),
123                reconnect_handle: Some(reconnect),
124            });
125        }
126
127        tracing::info!(pool_size = size, "SurrealDB connection pool ready");
128        Ok(Self::assemble(
129            handles,
130            config.pool_max_in_flight,
131            Duration::from_secs(config.pool_acquire_timeout_secs),
132        ))
133    }
134
135    /// Assemble the pool struct from already-built handles. The semaphore is
136    /// built ONLY when `max_in_flight > 0` — a `0` (default) cap means no
137    /// semaphore at all, so the acquire path is entirely bypassed.
138    fn assemble(
139        handles: Vec<PooledHandle<Client>>,
140        max_in_flight: usize,
141        acquire_timeout: Duration,
142    ) -> Self {
143        Self {
144            handles,
145            semaphore: (max_in_flight > 0).then(|| Arc::new(Semaphore::new(max_in_flight))),
146            acquire_timeout,
147            rr: AtomicUsize::new(0),
148        }
149    }
150}

152impl<C: Connection> DbPool<C> {
153    /// Number of pooled handles.
154    pub fn size(&self) -> usize {
155        self.handles.len()
156    }
157
158    /// Bind a handle to a repository at CONSTRUCTION time — the minimal-churn
159    /// repo seam (design §6, Option B). Returns an owned `Surreal<C>` bound to
160    /// one pooled handle, chosen round-robin so repositories spread evenly
161    /// across handles at startup. The repository then calls `self.db.query(..)`
162    /// exactly as before — repo bodies are untouched.
163    ///
164    /// This replaces the ~48 `db.client_cloned().await` composition sites. At
165    /// `pool_size = 1` it hands out clones of the single pooled handle exactly
166    /// as `DbManager::client_cloned()` did — byte-for-byte identical routing —
167    /// the only difference being that the handle is now proactively re-signed-in
168    /// (a strict improvement over the frozen snapshot it replaces).
169    pub async fn handle_for_repo(&self) -> Surreal<C> {
170        // Preserve the F1 checkout counter semantics (handles handed to repos).
171        metrics::record_handle_checkout();
172        let idx = self.rr.fetch_add(1, Ordering::Relaxed) % self.handles.len();
173        self.handles[idx].db.read().await.clone()
174    }
175
176    /// Per-operation checkout with least-in-flight handle selection and, when
177    /// the cap is enabled, a process-wide concurrency bound.
178    ///
179    /// 1. If the in-flight cap is enabled, acquire a semaphore permit within
180    ///    `acquire_timeout`; on timeout return [`AxiamError::ServiceUnavailable`]
181    ///    (HTTP 503, B1's overload taxonomy — see [`acquire_db_permit`]). When
182    ///    the cap is disabled (default) this step is skipped entirely.
183    /// 2. Pick the handle with the smallest current in-flight count (ties → the
184    ///    lowest index); this is load-aware and steers work away from a
185    ///    momentarily-congested handle. It is a no-op at `pool_size = 1`.
186    /// 3. Increment that handle's in-flight count and return a [`DbCheckout`]
187    ///    RAII guard that holds the permit and decrements the count on drop.
188    pub async fn checkout(&self) -> Result<DbCheckout<C>, AxiamError> {
189        let permit = match &self.semaphore {
190            Some(sem) => Some(acquire_db_permit(sem, self.acquire_timeout).await?),
191            None => None,
192        };
193
194        let idx = self.select_least_in_flight();
195        let slot = &self.handles[idx];
196        slot.in_flight.fetch_add(1, Ordering::Relaxed);
197        let handle = slot.db.read().await.clone();
198
199        Ok(DbCheckout {
200            handle,
201            in_flight: Arc::clone(&slot.in_flight),
202            _permit: permit,
203        })
204    }
205
206    /// Index of the least-in-flight handle (ties broken by lowest index).
207    /// Reads each count `Relaxed` — an approximate hint is fine, so no lock is
208    /// taken on the hot path (design §3.1).
209    fn select_least_in_flight(&self) -> usize {
210        let mut best = 0usize;
211        let mut best_load = usize::MAX;
212        for (i, h) in self.handles.iter().enumerate() {
213            let load = h.in_flight.load(Ordering::Relaxed);
214            if load < best_load {
215                best_load = load;
216                best = i;
217            }
218        }
219        best
220    }
221
222    /// Readiness probe: query EVERY pooled handle so readiness reflects the
223    /// whole pool (an auth-expired or poisoned handle anywhere trips the gate).
224    /// Each probe is routed through the F1 [`metrics::instrument_query`] gauge.
225    /// An auth failure classifies as [`DbError::Unhealthy`] (D-05) so `/ready`
226    /// alarms rather than treating it as a transient query error.
227    pub async fn health_check(&self) -> Result<(), DbError> {
228        for slot in &self.handles {
229            let guard = slot.db.read().await;
230            let result = metrics::instrument_query("health_check.pool", guard.query("RETURN 1"))
231                .await
232                .map_err(DbManager::classify_query_error)?;
233            result.check().map_err(DbManager::classify_query_error)?;
234        }
235        Ok(())
236    }
237
238    /// Test-only constructor: build a pool directly over already-open handles
239    /// (e.g. embedded `kv-mem` instances) with no background renewal tasks, so
240    /// the checkout/selection/cap logic can be exercised without a live server.
241    #[cfg(test)]
242    fn from_handles(raw: Vec<Surreal<C>>, max_in_flight: usize, acquire_timeout: Duration) -> Self {
243        let handles = raw
244            .into_iter()
245            .map(|db| PooledHandle {
246                db: Arc::new(RwLock::new(db)),
247                in_flight: Arc::new(AtomicUsize::new(0)),
248                refresh_handle: None,
249                reconnect_handle: None,
250            })
251            .collect();
252        Self {
253            handles,
254            semaphore: (max_in_flight > 0).then(|| Arc::new(Semaphore::new(max_in_flight))),
255            acquire_timeout,
256            rr: AtomicUsize::new(0),
257        }
258    }
259}

261impl<C: Connection> Drop for DbPool<C> {
262    fn drop(&mut self) {
263        // Explicitly stop each handle's background tasks (mirrors
264        // `DbManager::drop`) so a dropped pool never leaks detached tasks.
265        for slot in &self.handles {
266            if let Some(t) = &slot.refresh_handle {
267                t.abort();
268            }
269            if let Some(t) = &slot.reconnect_handle {
270                t.abort();
271            }
272        }
273    }
274}

276/// RAII checkout guard. `Deref`s to the chosen `Surreal<C>` so callers use it
277/// exactly like the owned handle they hold today (`checkout().query(..)`).
278/// Holds the semaphore permit (released on drop) and decrements the chosen
279/// handle's in-flight count on drop.
280pub struct DbCheckout<C: Connection = Client> {
281    /// Owned clone of the chosen pooled handle, read once out of its `RwLock`
282    /// at checkout time (a later reconnect-loop swap is observed by the NEXT
283    /// checkout, not this in-flight one).
284    handle: Surreal<C>,
285    in_flight: Arc<AtomicUsize>,
286    /// `None` when the cap is disabled; `Some` holds a permit released on drop.
287    _permit: Option<OwnedSemaphorePermit>,
288}
289
290impl<C: Connection> Deref for DbCheckout<C> {
291    type Target = Surreal<C>;
292    fn deref(&self) -> &Self::Target {
293        &self.handle
294    }
295}
296
297impl<C: Connection> Drop for DbCheckout<C> {
298    fn drop(&mut self) {
299        self.in_flight.fetch_sub(1, Ordering::Relaxed);
300        // `_permit` (if any) is released as the field drops.
301    }
302}

304/// Acquire a permit from the pool's in-flight semaphore, bounded by `timeout`.
305///
306/// Mirrors `axiam_auth::crypto_gate::acquire_hash_permit` EXACTLY so the DB
307/// pool and the Argon2id gate (B1) share ONE overload taxonomy:
308///
309/// * success → the held [`OwnedSemaphorePermit`] (drop to release),
310/// * acquire-timeout → [`AxiamError::ServiceUnavailable`] (the REST layer maps
311///   this to **HTTP 503** — a transient server-capacity condition, not a
312///   per-client rate-limit),
313/// * a closed semaphore (never happens — the pool never closes it) →
314///   [`AxiamError::Internal`].
315///
316/// `DbError` intentionally grows NO `ServiceUnavailable` variant — surfacing the
317/// existing `AxiamError` here keeps a single overload shape across B1 and F2
318/// (design §3.2).
319async fn acquire_db_permit(
320    sem: &Arc<Semaphore>,
321    timeout: Duration,
322) -> Result<OwnedSemaphorePermit, AxiamError> {
323    match tokio::time::timeout(timeout, Arc::clone(sem).acquire_owned()).await {
324        Ok(Ok(permit)) => Ok(permit),
325        Ok(Err(_closed)) => Err(AxiamError::Internal("db pool semaphore closed".into())),
326        Err(_elapsed) => Err(AxiamError::ServiceUnavailable(
327            "database is at capacity; please retry shortly".into(),
328        )),
329    }
330}

364    /// Safe-rollout invariant: `pool_size = 1` with the DISABLED cap (the

367    /// checkouts are held concurrently.
368    #[tokio::test]
369    async fn pool_size_one_default_is_a_single_uncapped_handle() {
370        let pool = DbPool::from_handles(vec![new_mem().await], 0, Duration::from_millis(20));
371        assert_eq!(pool.size(), 1);
372        assert!(
373            pool.semaphore.is_none(),
374            "disabled cap (0) must build NO semaphore — today's unbounded behavior"
375        );

377        // Hold several checkouts at once; with the cap disabled none blocks or
378        // times out — the acquire-timeout path cannot fire under the default.
379        let a = pool.checkout().await.expect("checkout 1");
380        let b = pool.checkout().await.expect("checkout 2");
381        let c = pool
382            .checkout()
383            .await

401    /// Least-in-flight spreads concurrent load evenly across handles and starves

456    /// Peak concurrency never exceeds the permit count (memory/stampede bound).
457    /// Mirrors `crypto_gate::concurrency_is_bounded_to_permit_count`.
458    #[tokio::test(flavor = "multi_thread", worker_threads = 4)]
459    async fn concurrency_is_bounded_to_permit_count() {
460        const PERMITS: usize = 2;
461        const TASKS: usize = 8;
462        let pool = Arc::new(DbPool::from_handles(
463            vec![new_mem().await],
464            PERMITS,
465            Duration::from_secs(5),
466        ));
467        let in_flight = Arc::new(AtomicUsize::new(0));
468        let max_observed = Arc::new(AtomicUsize::new(0));
469
470        let mut tasks = Vec::new();
471        for _ in 0..TASKS {
472            let pool = Arc::clone(&pool);
473            let in_flight = Arc::clone(&in_flight);
474            let max_observed = Arc::clone(&max_observed);
475            tasks.push(tokio::spawn(async move {
476                let _guard = pool.checkout().await.expect("checkout within timeout");
477                let now = in_flight.fetch_add(1, Ordering::SeqCst) + 1;
478                max_observed.fetch_max(now, Ordering::SeqCst);
479                tokio::time::sleep(Duration::from_millis(25)).await;
480                in_flight.fetch_sub(1, Ordering::SeqCst);
481            }));
482        }
483        for t in tasks {
484            t.await.unwrap();
485        }
486        assert!(
487            max_observed.load(Ordering::SeqCst) <= PERMITS,
488            "observed concurrency {} exceeded permit bound {PERMITS}",
489            max_observed.load(Ordering::SeqCst)
490        );
491    }

493    /// Per-handle poisoned-handle eviction (D-12) — mirrors
494    /// `connection.rs::poisoned_handle_is_evicted_and_never_returned_after_swap`
495    /// but proves the swap is isolated to ONE pooled handle: swapping handle 0's
496    /// `Arc<RwLock<Surreal<C>>>` under the write guard leaves the OTHER pooled
497    /// handles untouched.
498    #[tokio::test]
499    async fn poisoned_handle_evicted_per_handle_others_unaffected() {
500        let pool = DbPool::from_handles(
501            vec![new_marked("old-0").await, new_marked("keep-1").await],
502            0,
503            Duration::from_millis(20),
504        );
505
506        // Pre-swap: handle 0 is the "old" (poisoned) one.
507        {
508            let pre = pool.handles[0].db.read().await.clone();
509            assert_eq!(read_marker(&pre).await, vec!["old-0".to_string()]);

511
512        // Evict handle 0 by swapping a fresh handle in under the write guard,
513        // exactly as spawn_reconnect_loop does (`*db.write().await = fresh`).
514        let fresh = new_marked("new-0").await;
515        *pool.handles[0].db.write().await = fresh;
516
517        // Handle 0 now observes ONLY the new handle...
518        let post0 = pool.handles[0].db.read().await.clone();
519        assert_eq!(
520            read_marker(&post0).await,
521            vec!["new-0".to_string()],
522            "the poisoned handle must never be observed after the swap (D-12)"

524        // ...and the sibling handle is entirely unaffected by that swap.
525        let post1 = pool.handles[1].db.read().await.clone();
526        assert_eq!(
527            read_marker(&post1).await,
528            vec!["keep-1".to_string()],
529            "swapping one pooled handle must not disturb the others"
530        );
531    }
