1//! SurrealDB implementation of [`AmqpNonceRepository`] (NEW-4).
2//!
3//! Provides insert-or-conflict semantics for AMQP message nonces. Inserting a
4//! `(tenant_id, nonce)` pair that already exists returns
5//! `Err(AxiamError::ReplayDetected)` so the authz/audit AMQP consumers can
6//! reject replayed messages (NEW-4). Mirrors [`super::saml_replay`].
7
8use axiam_core::error::{AxiamError, AxiamResult};
9use axiam_core::repository::AmqpNonceRepository;
10use chrono::{DateTime, Utc};
11use surrealdb::{Connection, Surreal};
12use uuid::Uuid;
13
14use crate::error::DbError;
15use crate::helpers::CountRow;
16
17// ---------------------------------------------------------------------------
18// Repository
19// ---------------------------------------------------------------------------

21/// SurrealDB implementation of the AMQP nonce replay repository (NEW-4).
22pub struct SurrealAmqpNonceRepository<C: Connection> {
23    db: Surreal<C>,
24}
25
26// Manual Clone impl (not derive): avoids the spurious `C: Clone` bound that
27// blocks cloning under generic `C: Connection` callers. Matches
28// SurrealAssertionReplayRepository.
29impl<C: Connection> Clone for SurrealAmqpNonceRepository<C> {
30    fn clone(&self) -> Self {
31        Self {
32            db: self.db.clone(),
33        }
34    }
35}
36
37impl<C: Connection> SurrealAmqpNonceRepository<C> {
38    pub fn new(db: Surreal<C>) -> Self {
39        Self { db }
40    }
41}
42
43impl<C: Connection> AmqpNonceRepository for SurrealAmqpNonceRepository<C> {
44    async fn insert_nonce(
45        &self,
46        tenant_id: Uuid,
47        nonce: Uuid,
48        expires_at: DateTime<Utc>,
49    ) -> AxiamResult<()> {
50        let row_id = Uuid::new_v4().to_string();
51
52        let result = self
53            .db
54            .query(
55                "CREATE type::record('amqp_nonce_replay', $row_id) SET \
56                 tenant_id = $tenant_id, \
57                 nonce = $nonce, \
58                 expires_at = $expires_at",
59            )
60            .bind(("row_id", row_id))
61            .bind(("tenant_id", tenant_id.to_string()))
62            .bind(("nonce", nonce.to_string()))
63            .bind(("expires_at", expires_at))
64            .await
65            .map_err(DbError::from)?;
66
67        result
68            .check()
69            .map_err(|e| {
70                let msg = e.to_string();
71                // SurrealDB v3 UNIQUE index violation message contains
72                // "already contains" (e.g. "Database index
73                // `idx_amqp_nonce_uniq` already contains [...]"). Also match
74                // "already exists" and "unique" as fallback patterns.
75                if msg.contains("already contains")
76                    || msg.contains("already exists")
77                    || msg.contains("unique")
78                {
79                    AxiamError::ReplayDetected
80                } else {
81                    AxiamError::Database(msg)
82                }
83            })
84            .map(|_| ())
85    }
86
87    async fn cleanup_expired(&self) -> AxiamResult<u64> {
88        // Count expired rows first, then delete.
89        let mut count_result = self
90            .db
91            .query(
92                "SELECT count() AS total FROM amqp_nonce_replay \
93                 WHERE expires_at < time::now() GROUP ALL",
94            )
95            .await
96            .map_err(DbError::from)?;
97
98        let count_rows: Vec<CountRow> = count_result.take(0).map_err(DbError::from)?;
99        let total = count_rows.first().map(|r| r.total).unwrap_or(0);
100
101        self.db
102            .query("DELETE amqp_nonce_replay WHERE expires_at < time::now()")
103            .await
104            .map_err(DbError::from)?;
105
106        Ok(total)
107    }
108}
