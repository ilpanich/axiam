62fn parse_status(s: &str) -> Result<CertificateStatus, DbError> {
63    match s {
64        "Active" => Ok(CertificateStatus::Active),
65        "Revoked" => Ok(CertificateStatus::Revoked),
66        "Expired" => Ok(CertificateStatus::Expired),
67        other => Err(DbError::Migration(format!(
68            "unknown certificate status: {other}"
69        ))),
70    }
71}
72
73fn status_str(s: &CertificateStatus) -> &'static str {
74    match s {
75        CertificateStatus::Active => "Active",
76        CertificateStatus::Revoked => "Revoked",
77        CertificateStatus::Expired => "Expired",
78    }
79}
80
81fn parse_key_algorithm(s: &str) -> Result<KeyAlgorithm, DbError> {
82    match s {
83        "Rsa4096" => Ok(KeyAlgorithm::Rsa4096),
84        "Ed25519" => Ok(KeyAlgorithm::Ed25519),
85        other => Err(DbError::Migration(format!(
86            "unknown key algorithm: {other}"
87        ))),
88    }
89}
90
91fn key_algorithm_str(k: &KeyAlgorithm) -> &'static str {
92    match k {
93        KeyAlgorithm::Rsa4096 => "Rsa4096",
94        KeyAlgorithm::Ed25519 => "Ed25519",
95    }
96}

121impl CertificateRow {
122    fn into_entry(self, id: Uuid) -> Result<Certificate, DbError> {
123        let tenant_id = Uuid::parse_str(&self.tenant_id)
124            .map_err(|e| DbError::Migration(format!("invalid tenant UUID: {e}")))?;
125        let issuer_ca_id = Uuid::parse_str(&self.issuer_ca_id)
126            .map_err(|e| DbError::Migration(format!("invalid issuer CA UUID: {e}")))?;
127        Ok(Certificate {
128            id,
129            tenant_id,
130            issuer_ca_id,
131            subject: self.subject,
132            public_cert_pem: self.public_cert_pem,
133            fingerprint: self.fingerprint,
134            cert_type: parse_cert_type(&self.cert_type)?,
135            key_algorithm: parse_key_algorithm(&self.key_algorithm)?,
136            not_before: self.not_before,
137            not_after: self.not_after,
138            status: parse_status(&self.status)?,
139            metadata: self.metadata,
140            created_at: self.created_at,
141        })
142    }
143}

186impl<C: Connection> CertificateRepository for SurrealCertificateRepository<C> {
187    async fn create(&self, input: StoreCertificate) -> AxiamResult<Certificate> {
188        let id = Uuid::new_v4();
189        let id_str = id.to_string();
190
191        let result = self
192            .db
193            .query(
194                "CREATE type::record('certificate', $id) SET \
195                 tenant_id = $tenant_id, \
196                 issuer_ca_id = $issuer_ca_id, \
197                 subject = $subject, \
198                 public_cert_pem = $public_cert_pem, \
199                 fingerprint = $fingerprint, \
200                 cert_type = $cert_type, \
201                 key_algorithm = $key_algorithm, \
202                 not_before = $not_before, \
203                 not_after = $not_after, \
204                 status = $status, \
205                 metadata = $metadata",
206            )
207            .bind(("id", id_str))
208            .bind(("tenant_id", input.tenant_id.to_string()))
209            .bind(("issuer_ca_id", input.issuer_ca_id.to_string()))
210            .bind(("subject", input.subject))
211            .bind(("public_cert_pem", input.public_cert_pem))
212            .bind(("fingerprint", input.fingerprint))
213            .bind(("cert_type", cert_type_str(&input.cert_type)))
214            .bind(("key_algorithm", key_algorithm_str(&input.key_algorithm)))
215            .bind(("not_before", input.not_before))
216            .bind(("not_after", input.not_after))
217            .bind(("status", status_str(&CertificateStatus::Active)))
218            .bind(("metadata", input.metadata))
219            .await
220            .map_err(DbError::from)?;
221
222        let mut result = result
223            .check()
224            .map_err(|e| DbError::Migration(e.to_string()))?;
225        let row: Option<CertificateRow> = result.take(0).map_err(DbError::from)?;
226        let row = row.ok_or_else(|| DbError::NotFound {
227            entity: "certificate".into(),
228            id: id.to_string(),
229        })?;
230
231        // Create the signed_by edge (RELATE doesn't accept type::record()).
232        let relate_sql = format!(
233            "RELATE certificate:`{}`->signed_by->ca_certificate:`{}`",
234            id, input.issuer_ca_id,
235        );
236        self.db
237            .query(&relate_sql)
238            .await
239            .map_err(DbError::from)?
240            .check()
241            .map_err(|e| DbError::Migration(e.to_string()))?;
242
243        Ok(row.into_entry(id)?)
244    }

288            .map_err(|e| DbError::Migration(e.to_string()))?;

295    async fn revoke(&self, tenant_id: Uuid, id: Uuid) -> AxiamResult<()> {
296        let result = self
297            .db
298            .query(
299                "UPDATE type::record('certificate', $id) SET \
300                 status = $status \
301                 WHERE tenant_id = $tenant_id",
302            )
303            .bind(("id", id.to_string()))
304            .bind(("tenant_id", tenant_id.to_string()))
305            .bind(("status", status_str(&CertificateStatus::Revoked)))
306            .await
307            .map_err(DbError::from)?;
308
309        let mut result = result
310            .check()
311            .map_err(|e| DbError::Migration(e.to_string()))?;
312        let row: Option<CertificateRow> = result.take(0).map_err(DbError::from)?;
313        if row.is_none() {
314            return Err(DbError::NotFound {
315                entity: "certificate".into(),
316                id: id.to_string(),
317            }
318            .into());
319        }
320
321        Ok(())
322    }

324    async fn list(
325        &self,
326        tenant_id: Uuid,
327        pagination: Pagination,
328    ) -> AxiamResult<PaginatedResult<Certificate>> {
329        let tenant_id_str = tenant_id.to_string();
330
331        let count_sql = "SELECT count() AS total FROM certificate \
332                         WHERE tenant_id = $tenant_id GROUP ALL";
333        let count_result = self
334            .db
335            .query(count_sql)
336            .bind(("tenant_id", tenant_id_str.clone()))
337            .await
338            .map_err(DbError::from)?;
339        let mut count_result = count_result
340            .check()
341            .map_err(|e| DbError::Migration(e.to_string()))?;
342        let count_rows: Vec<CountRow> = count_result.take(0).map_err(DbError::from)?;
343
344        let data_sql = "SELECT meta::id(id) AS record_id, * FROM certificate \
345                        WHERE tenant_id = $tenant_id \
346                        ORDER BY created_at DESC \
347                        LIMIT $limit START $offset";
348        let data_result = self
349            .db
350            .query(data_sql)
351            .bind(("tenant_id", tenant_id_str))
352            .bind(("limit", pagination.limit))
353            .bind(("offset", pagination.offset))
354            .await
355            .map_err(DbError::from)?;
356        let mut data_result = data_result
357            .check()
358            .map_err(|e| DbError::Migration(e.to_string()))?;
359        let rows: Vec<CertificateRowWithId> = data_result.take(0).map_err(DbError::from)?;
360
361        let items: Vec<Certificate> = rows
362            .into_iter()
363            .map(|r| r.try_into_entry())
364            .collect::<Result<_, _>>()?;
365
366        Ok(paginate(items, count_rows, &pagination))
367    }

389    async fn bind_to_service_account(
390        &self,
391        tenant_id: Uuid,
392        cert_id: Uuid,
393        sa_id: Uuid,
394    ) -> AxiamResult<()> {
395        // Verify both certificate and service account belong to the same tenant
396        // before creating the binding.
397        let verify_sql = format!(
398            "LET $cert = (SELECT tenant_id FROM certificate:`{cert_id}` WHERE tenant_id = $tid);\
399             LET $sa = (SELECT tenant_id FROM service_account:`{sa_id}` WHERE tenant_id = $tid);\
400             IF array::len($cert) = 0 OR array::len($sa) = 0 {{ \
401                 THROW 'cross-tenant binding denied'; \
402             }};\
403             RELATE certificate:`{cert_id}`->cert_bound_to->service_account:`{sa_id}`",
404        );
405        let result = self
406            .db
407            .query(&verify_sql)
408            .bind(("tid", tenant_id.to_string()))
409            .await
410            .map_err(DbError::from)?;
411
412        if let Err(e) = result.check() {
413            let msg = e.to_string();
414            if msg.contains("cross-tenant binding denied") {
415                return Err(AxiamError::AuthorizationDenied {
416                    reason: "cross-tenant certificate-to-service-account binding denied".into(),
417                    action: None,
418                    resource_id: None,
419                });
420            }
421            return Err(classify_write_error(msg, "certificate_binding").into());
422        }
423        Ok(())
424    }

426    async fn get_bound_service_account(&self, cert_id: Uuid) -> AxiamResult<Option<Uuid>> {
427        // Use a subquery to extract the service_account ID as a string
428        let sql = format!(
429            "SELECT meta::id(out) AS sa_id FROM cert_bound_to \
430             WHERE in = certificate:`{}`",
431            cert_id,
432        );
433        let result = self.db.query(&sql).await.map_err(DbError::from)?;
434        let mut result = result
435            .check()
436            .map_err(|e| DbError::Migration(e.to_string()))?;
437
438        let rows: Vec<BoundTargetRow> = result.take(0).map_err(DbError::from)?;
439        let Some(row) = rows.into_iter().next() else {
440            return Ok(None);
441        };
442
443        let sa_id = Uuid::parse_str(&row.sa_id)
444            .map_err(|e| DbError::Migration(format!("invalid service account UUID: {e}")))?;
445        Ok(Some(sa_id))
446    }
