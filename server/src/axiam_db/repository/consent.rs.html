28#[derive(Debug, SurrealValue)]
29struct ConsentRowWithId {
30    record_id: String,
31    tenant_id: String,
32    user_id: String,
33    consent_type: String,
34    version: String,
35    accepted_at: DateTime<Utc>,
36    ip_address: Option<String>,
37    user_agent: Option<String>,
38}
39
40impl ConsentRowWithId {
41    fn try_into_domain(self) -> Result<Consent, DbError> {
42        let id = Uuid::parse_str(&self.record_id)
43            .map_err(|e| DbError::Migration(format!("invalid UUID: {e}")))?;
44        let tenant_id = Uuid::parse_str(&self.tenant_id)
45            .map_err(|e| DbError::Migration(format!("invalid tenant UUID: {e}")))?;
46        let user_id = Uuid::parse_str(&self.user_id)
47            .map_err(|e| DbError::Migration(format!("invalid user UUID: {e}")))?;
48        Ok(Consent {
49            id,
50            tenant_id,
51            user_id,
52            consent_type: self.consent_type,
53            version: self.version,
54            accepted_at: self.accepted_at,
55            ip_address: self.ip_address,
56            user_agent: self.user_agent,
57        })
58    }
59}

83impl<C: Connection> ConsentRepository for SurrealConsentRepository<C> {
84    async fn create(&self, input: CreateConsent) -> AxiamResult<Consent> {
85        let id = Uuid::new_v4();
86        let result = self
87            .db
88            .query(
89                "CREATE type::record('consent', $id) SET \
90                 tenant_id = $tenant_id, \
91                 user_id = $user_id, \
92                 consent_type = $consent_type, \
93                 version = $version, \
94                 accepted_at = time::now(), \
95                 ip_address = $ip_address, \
96                 user_agent = $user_agent",
97            )
98            .bind(("id", id.to_string()))
99            .bind(("tenant_id", input.tenant_id.to_string()))
100            .bind(("user_id", input.user_id.to_string()))
101            .bind(("consent_type", input.consent_type))
102            .bind(("version", input.version))
103            .bind(("ip_address", input.ip_address))
104            .bind(("user_agent", input.user_agent))
105            .await
106            .map_err(DbError::from)?;
107
108        let mut result = result
109            .check()
110            .map_err(|e| DbError::Migration(e.to_string()))?;
111        let rows: Vec<ConsentRow> = result.take(0).map_err(DbError::from)?;
112        let row = rows.into_iter().next().ok_or_else(|| DbError::NotFound {
113            entity: "consent".into(),
114            id: id.to_string(),
115        })?;
116
117        // Build domain object from row + known id.
118        Ok(Consent {
119            id,
120            tenant_id: input.tenant_id,
121            user_id: input.user_id,
122            consent_type: row.consent_type,
123            version: row.version,
124            accepted_at: row.accepted_at,
125            ip_address: row.ip_address,
126            user_agent: row.user_agent,
127        })
128    }
129
130    async fn list_by_user(&self, tenant_id: Uuid, user_id: Uuid) -> AxiamResult<Vec<Consent>> {
131        let mut result = self
132            .db
133            .query(
134                "SELECT meta::id(id) AS record_id, * FROM consent \
135                 WHERE tenant_id = $tenant_id AND user_id = $user_id \
136                 ORDER BY accepted_at ASC",
137            )
138            .bind(("tenant_id", tenant_id.to_string()))
139            .bind(("user_id", user_id.to_string()))
140            .await
141            .map_err(DbError::from)?
142            .check()
143            .map_err(|e| DbError::Migration(e.to_string()))?;
144
145        let rows: Vec<ConsentRowWithId> = result.take(0).map_err(DbError::from)?;
146        rows.into_iter()
147            .map(|r| r.try_into_domain().map_err(Into::into))
148            .collect()
149    }
150}

156#[cfg(test)]
157mod tests {
158    use super::*;
159    use surrealdb::Surreal;
160    use surrealdb::engine::local::Mem;
161
162    async fn setup_db() -> Surreal<surrealdb::engine::local::Db> {
163        let db = Surreal::new::<Mem>(()).await.unwrap();
164        db.use_ns("test").use_db("test").await.unwrap();
165        crate::schema::run_migrations(&db).await.unwrap();
166        db
167    }
168
169    #[tokio::test]
170    async fn consent_round_trip() {
171        let db = setup_db().await;
172        let repo = SurrealConsentRepository::new(db);
173        let tenant_id = Uuid::new_v4();
174        let user_id = Uuid::new_v4();
175
176        let consent = repo
177            .create(CreateConsent {
178                tenant_id,
179                user_id,
180                consent_type: "terms_of_service".into(),
181                version: "2026-01-01".into(),
182                ip_address: Some("127.0.0.1".into()),
183                user_agent: None,
184            })
185            .await
186            .unwrap();
187
188        assert_eq!(consent.consent_type, "terms_of_service");
189        assert_eq!(consent.tenant_id, tenant_id);
190
191        let list = repo.list_by_user(tenant_id, user_id).await.unwrap();
192        assert_eq!(list.len(), 1);
193        assert_eq!(list[0].version, "2026-01-01");
194    }
195}
