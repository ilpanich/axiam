16#[derive(Debug, SurrealValue)]
17struct TokenRow {
18    tenant_id: String,
19    user_id: String,
20    token_hash: String,
21    expires_at: DateTime<Utc>,
22    consumed_at: Option<DateTime<Utc>>,

26#[derive(Debug, SurrealValue)]
27struct TokenRowWithId {

37impl TokenRow {
38    fn into_token(self, id: Uuid) -> Result<EmailVerificationToken, DbError> {
39        let tenant_id = Uuid::parse_str(&self.tenant_id)
40            .map_err(|e| DbError::Migration(format!("invalid tenant UUID: {e}")))?;
41        let user_id = Uuid::parse_str(&self.user_id)
42            .map_err(|e| DbError::Migration(format!("invalid user UUID: {e}")))?;
43        Ok(EmailVerificationToken {
44            id,
45            tenant_id,
46            user_id,
47            token_hash: self.token_hash,
48            expires_at: self.expires_at,
49            consumed_at: self.consumed_at,
50            created_at: self.created_at,
51        })
52    }
53}

94impl<C: Connection> EmailVerificationTokenRepository
95    for SurrealEmailVerificationTokenRepository<C>
96{
97    async fn create(
98        &self,
99        input: CreateEmailVerificationToken,
100    ) -> AxiamResult<EmailVerificationToken> {
101        let id = Uuid::new_v4();
102        let id_str = id.to_string();
103
104        let result = self
105            .db
106            .query(
107                "CREATE type::record('email_verification_token', $id) SET \
108                 tenant_id = $tenant_id, \
109                 user_id = $user_id, \
110                 token_hash = $token_hash, \
111                 expires_at = $expires_at, \
112                 consumed_at = NONE",
113            )
114            .bind(("id", id_str.clone()))
115            .bind(("tenant_id", input.tenant_id.to_string()))
116            .bind(("user_id", input.user_id.to_string()))
117            .bind(("token_hash", input.token_hash))
118            .bind(("expires_at", input.expires_at))
119            .await
120            .map_err(DbError::from)?;
121
122        let mut result = result
123            .check()
124            .map_err(|e| DbError::Migration(e.to_string()))?;
125
126        let rows: Vec<TokenRow> = result.take(0).map_err(DbError::from)?;
127        let row = take_first_or_not_found(rows, "email_verification_token", &id_str)?;
128
129        Ok(row.into_token(id)?)
130    }

132    async fn get_by_token_hash(
133        &self,
134        tenant_id: Uuid,
135        token_hash: &str,
136    ) -> AxiamResult<EmailVerificationToken> {
137        let result = self
138            .db
139            .query(
140                "SELECT meta::id(id) AS record_id, * \
141                 FROM email_verification_token \
142                 WHERE tenant_id = $tenant_id \
143                   AND token_hash = $token_hash \
144                   AND consumed_at IS NONE \
145                   AND expires_at > time::now()",
146            )
147            .bind(("tenant_id", tenant_id.to_string()))
148            .bind(("token_hash", token_hash.to_string()))
149            .await
150            .map_err(DbError::from)?;
151
152        let mut result = result
153            .check()
154            .map_err(|e| DbError::Migration(e.to_string()))?;
155
156        let rows: Vec<TokenRowWithId> = result.take(0).map_err(DbError::from)?;
157        let row = take_first_or_not_found(
158            rows,
159            "email_verification_token",
160            &format!("token_hash={token_hash}"),
161        )?;
162
163        Ok(row.try_into_token()?)
164    }

166    async fn consume(
167        &self,
168        tenant_id: Uuid,
169        token_hash: &str,
170    ) -> AxiamResult<EmailVerificationToken> {
171        // Use a single UPDATE + SELECT meta::id pattern to atomically
172        // consume and return the full record with its ID.
173        let result = self
174            .db
175            .query(
176                "UPDATE email_verification_token SET \
177                 consumed_at = time::now() \
178                 WHERE tenant_id = $tenant_id \
179                   AND token_hash = $token_hash \
180                   AND consumed_at IS NONE \
181                   AND expires_at > time::now(); \
182                 SELECT meta::id(id) AS record_id, * \
183                 FROM email_verification_token \
184                 WHERE tenant_id = $tenant_id \
185                   AND token_hash = $token_hash \
186                   AND consumed_at IS NOT NONE",
187            )
188            .bind(("tenant_id", tenant_id.to_string()))
189            .bind(("token_hash", token_hash.to_string()))
190            .await
191            .map_err(DbError::from)?;
192
193        let mut result = result
194            .check()
195            .map_err(|e| DbError::Migration(e.to_string()))?;
196
197        // Statement 0 is the UPDATE — if no rows were affected, the
198        // token was already consumed, invalidated, or expired.
199        let updated: Vec<TokenRow> = result.take(0).map_err(DbError::from)?;
200        if updated.is_empty() {
201            return Err(DbError::NotFound {
202                entity: "email_verification_token".into(),
203                id: format!("token_hash={token_hash}"),
204            }
205            .into());
206        }
207
208        // Statement 1 is the SELECT with full record_id.
209        let rows: Vec<TokenRowWithId> = result.take(1).map_err(DbError::from)?;
210        let row = take_first_or_not_found(
211            rows,
212            "email_verification_token",
213            &format!("token_hash={token_hash}"),
214        )?;
215
216        Ok(row.try_into_token()?)
217    }

219    async fn count_today(&self, tenant_id: Uuid, user_id: Uuid) -> AxiamResult<u64> {
220        let today_start = Utc::now()
221            .date_naive()
222            .and_hms_opt(0, 0, 0)
223            .expect("midnight is always valid")
224            .and_utc();
225
226        let result = self
227            .db
228            .query(
229                "SELECT count() AS total \
230                 FROM email_verification_token \
231                 WHERE tenant_id = $tenant_id \
232                   AND user_id = $user_id \
233                   AND created_at >= $today_start \
234                 GROUP ALL",
235            )
236            .bind(("tenant_id", tenant_id.to_string()))
237            .bind(("user_id", user_id.to_string()))
238            .bind(("today_start", today_start))
239            .await
240            .map_err(DbError::from)?;
241
242        let mut result = result
243            .check()
244            .map_err(|e| DbError::Migration(e.to_string()))?;
245
246        let rows: Vec<CountRow> = result.take(0).map_err(DbError::from)?;
247        Ok(rows.first().map(|r| r.total).unwrap_or(0))
248    }

250    async fn delete_expired(&self) -> AxiamResult<u64> {
251        let result = self
252            .db
253            .query(
254                "DELETE FROM email_verification_token \
255                 WHERE expires_at < time::now() \
256                    OR consumed_at IS NOT NONE \
257                 RETURN BEFORE",
258            )
259            .await
260            .map_err(DbError::from)?;
261
262        let mut result = result
263            .check()
264            .map_err(|e| DbError::Migration(e.to_string()))?;
265
266        let rows: Vec<TokenRow> = result.take(0).map_err(DbError::from)?;
267        Ok(rows.len() as u64)
268    }
