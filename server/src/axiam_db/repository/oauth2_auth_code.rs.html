47impl AuthCodeRowWithId {
48    fn try_into_auth_code(self) -> Result<AuthorizationCode, DbError> {
49        let id = Uuid::parse_str(&self.record_id)
50            .map_err(|e| DbError::Migration(format!("invalid UUID: {e}")))?;
51        let tenant_id = Uuid::parse_str(&self.tenant_id)
52            .map_err(|e| DbError::Migration(format!("invalid tenant UUID: {e}")))?;
53        let user_id = Uuid::parse_str(&self.user_id)
54            .map_err(|e| DbError::Migration(format!("invalid user UUID: {e}")))?;
55        Ok(AuthorizationCode {
56            id,
57            tenant_id,
58            client_id: self.client_id,
59            user_id,
60            code_hash: self.code_hash,
61            redirect_uri: self.redirect_uri,
62            scopes: self.scopes,
63            code_challenge: self.code_challenge,
64            code_challenge_method: self.code_challenge_method,
65            nonce: self.nonce,
66            expires_at: self.expires_at,
67            used: self.used,
68            created_at: self.created_at,
69        })
70    }

85impl<C: Connection> AuthorizationCodeRepository for SurrealAuthorizationCodeRepository<C> {
86    async fn create(&self, input: CreateAuthorizationCode) -> AxiamResult<AuthorizationCode> {
87        let id = Uuid::new_v4();
88        let id_str = id.to_string();
89
90        let result = self
91            .db
92            .query(
93                "CREATE type::record('oauth2_auth_code', $id) SET \
94                 tenant_id = $tenant_id, \
95                 client_id = $client_id, \
96                 user_id = $user_id, \
97                 code_hash = $code_hash, \
98                 redirect_uri = $redirect_uri, \
99                 scopes = $scopes, \
100                 code_challenge = $code_challenge, \
101                 code_challenge_method = $code_challenge_method, \
102                 nonce = $nonce, \
103                 expires_at = $expires_at, \
104                 used = false",
105            )
106            .bind(("id", id_str.clone()))
107            .bind(("tenant_id", input.tenant_id.to_string()))
108            .bind(("client_id", input.client_id))
109            .bind(("user_id", input.user_id.to_string()))
110            .bind(("code_hash", input.code_hash))
111            .bind(("redirect_uri", input.redirect_uri))
112            .bind(("scopes", input.scopes))
113            .bind(("code_challenge", input.code_challenge))
114            .bind(("code_challenge_method", input.code_challenge_method))
115            .bind(("nonce", input.nonce))
116            .bind(("expires_at", input.expires_at))
117            .await
118            .map_err(DbError::from)?;
119
120        let mut result = result
121            .check()
122            .map_err(|e| DbError::Migration(e.to_string()))?;
123
124        let rows: Vec<AuthCodeRow> = result.take(0).map_err(DbError::from)?;
125        let row = take_first_or_not_found(rows, "oauth2_auth_code", &id_str)?;
126
127        let tenant_id = Uuid::parse_str(&row.tenant_id)
128            .map_err(|e| DbError::Migration(format!("invalid tenant UUID: {e}")))?;
129        let user_id = Uuid::parse_str(&row.user_id)
130            .map_err(|e| DbError::Migration(format!("invalid user UUID: {e}")))?;
131
132        Ok(AuthorizationCode {
133            id,
134            tenant_id,
135            client_id: row.client_id,
136            user_id,
137            code_hash: row.code_hash,
138            redirect_uri: row.redirect_uri,
139            scopes: row.scopes,
140            code_challenge: row.code_challenge,
141            code_challenge_method: row.code_challenge_method,
142            nonce: row.nonce,
143            expires_at: row.expires_at,
144            used: row.used,
145            created_at: row.created_at,
146        })
147    }

149    async fn get_by_hash(
150        &self,
151        tenant_id: Uuid,
152        code_hash: &str,
153        client_id: &str,
154        redirect_uri: &str,
155    ) -> AxiamResult<AuthorizationCode> {
156        let code_hash_owned = code_hash.to_string();
157        let tenant_id_str = tenant_id.to_string();
158
159        let result = self
160            .db
161            .query(
162                "SELECT meta::id(id) AS record_id, * \
163                 FROM oauth2_auth_code \
164                 WHERE tenant_id = $tenant_id \
165                   AND code_hash = $code_hash \
166                   AND client_id = $client_id \
167                   AND redirect_uri = $redirect_uri \
168                   AND used = false \
169                   AND expires_at > time::now()",
170            )
171            .bind(("tenant_id", tenant_id_str))
172            .bind(("code_hash", code_hash_owned.clone()))
173            .bind(("client_id", client_id.to_string()))
174            .bind(("redirect_uri", redirect_uri.to_string()))
175            .await
176            .map_err(DbError::from)?;
177
178        let mut result = result
179            .check()
180            .map_err(|e| DbError::Migration(e.to_string()))?;
181
182        let rows: Vec<AuthCodeRowWithId> = result.take(0).map_err(DbError::from)?;
183        let row = take_first_or_not_found(
184            rows,
185            "oauth2_auth_code",
186            &format!("code_hash={code_hash_owned}"),
187        )?;
188
189        row.try_into_auth_code().map_err(Into::into)
190    }

192    async fn consume(
193        &self,
194        tenant_id: Uuid,
195        code_hash: &str,
196        client_id: &str,
197        redirect_uri: &str,
198    ) -> AxiamResult<AuthorizationCode> {
199        let code_hash_owned = code_hash.to_string();
200        let tenant_id_str = tenant_id.to_string();
201
202        // Single atomic UPDATE with WHERE guards: only one concurrent
203        // caller can match `used = false`, eliminating the race
204        // condition of a separate SELECT + UPDATE. client_id and
205        // redirect_uri are verified atomically to prevent
206        // code-burning attacks.
207        let result = self
208            .db
209            .query(
210                "SELECT meta::id(id) AS record_id, * FROM \
211                 (UPDATE oauth2_auth_code SET used = true \
212                  WHERE tenant_id = $tenant_id \
213                    AND code_hash = $code_hash \
214                    AND client_id = $client_id \
215                    AND redirect_uri = $redirect_uri \
216                    AND used = false \
217                    AND expires_at > time::now())",
218            )
219            .bind(("tenant_id", tenant_id_str))
220            .bind(("code_hash", code_hash_owned.clone()))
221            .bind(("client_id", client_id.to_string()))
222            .bind(("redirect_uri", redirect_uri.to_string()))
223            .await
224            .map_err(DbError::from)?;
225
226        let mut result = result
227            .check()
228            .map_err(|e| DbError::Migration(e.to_string()))?;
229
230        let rows: Vec<AuthCodeRowWithId> = result.take(0).map_err(DbError::from)?;
231        let row = take_first_or_not_found(
232            rows,
233            "oauth2_auth_code",
234            &format!("code_hash={code_hash_owned}"),
235        )?;
236
237        row.try_into_auth_code().map_err(Into::into)
238    }

240    async fn delete_expired(&self) -> AxiamResult<u64> {
241        let mut result = self
242            .db
243            .query(
244                "SELECT count() AS total FROM oauth2_auth_code \
245                 WHERE expires_at < time::now() OR used = true GROUP ALL",
246            )
247            .await
248            .map_err(DbError::from)?;
249
250        let count_rows: Vec<CountRow> = result.take(0).map_err(DbError::from)?;
251        let count = count_rows.first().map(|r| r.total).unwrap_or(0);
252
253        self.db
254            .query(
255                "DELETE FROM oauth2_auth_code \
256                 WHERE expires_at < time::now() OR used = true",
257            )
258            .await
259            .map_err(DbError::from)?;
260
261        Ok(count)
262    }
