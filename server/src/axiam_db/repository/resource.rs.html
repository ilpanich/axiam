25#[derive(Debug, SurrealValue)]
26struct ResourceRowWithId {
27    record_id: String,
28    tenant_id: String,
29    name: String,
30    resource_type: String,
31    parent_id: Option<String>,
32    metadata: serde_json::Value,
33    created_at: DateTime<Utc>,

37impl ResourceRowWithId {
38    fn try_into_resource(self) -> Result<Resource, DbError> {
39        let id = Uuid::parse_str(&self.record_id)
40            .map_err(|e| DbError::Migration(format!("invalid UUID: {e}")))?;
41        let tenant_id = Uuid::parse_str(&self.tenant_id)
42            .map_err(|e| DbError::Migration(format!("invalid tenant UUID: {e}")))?;
43        let parent_id = self
44            .parent_id
45            .map(|p| Uuid::parse_str(&p))
46            .transpose()
47            .map_err(|e| DbError::Migration(format!("invalid parent UUID: {e}")))?;
48        Ok(Resource {
49            id,
50            tenant_id,
51            name: self.name,
52            resource_type: self.resource_type,
53            parent_id,
54            metadata: self.metadata,
55            created_at: self.created_at,
56            updated_at: self.updated_at,
57        })
58    }
59}

61fn row_to_resource(row: ResourceRow, id: Uuid) -> Result<Resource, DbError> {
62    let tenant_id = Uuid::parse_str(&row.tenant_id)
63        .map_err(|e| DbError::Migration(format!("invalid tenant UUID: {e}")))?;
64    let parent_id = row
65        .parent_id
66        .map(|p| Uuid::parse_str(&p))
67        .transpose()
68        .map_err(|e| DbError::Migration(format!("invalid parent UUID: {e}")))?;
69    Ok(Resource {
70        id,
71        tenant_id,
72        name: row.name,
73        resource_type: row.resource_type,
74        parent_id,
75        metadata: row.metadata,
76        created_at: row.created_at,
77        updated_at: row.updated_at,
78    })
79}

97    async fn create(&self, input: CreateResource) -> AxiamResult<Resource> {
98        let id = Uuid::new_v4();
99        let id_str = id.to_string();
100        let tenant_id_str = input.tenant_id.to_string();
101        let parent_id_str = input.parent_id.map(|p| p.to_string());
102
103        let metadata = input
104            .metadata
105            .unwrap_or(serde_json::Value::Object(Default::default()));
106
107        // Create resource + optional child_of edge in one query.
108        let base = "CREATE type::record('resource', $id) SET \
109                    tenant_id = $tenant_id, \
110                    name = $name, resource_type = $resource_type, \
111                    parent_id = $parent_id, \
112                    metadata = $metadata";
113
114        let query = if let Some(ref pid) = parent_id_str {
115            format!("{base}; RELATE resource:`{id_str}` -> child_of -> resource:`{pid}`;")
116        } else {
117            base.to_string()
118        };
119
120        let result = self
121            .db
122            .query(query)
123            .bind(("id", id_str.clone()))
124            .bind(("tenant_id", tenant_id_str))
125            .bind(("name", input.name))
126            .bind(("resource_type", input.resource_type))
127            .bind(("parent_id", parent_id_str))
128            .bind(("metadata", metadata))
129            .await
130            .map_err(DbError::from)?;
131
132        let mut result = result
133            .check()
134            .map_err(|e| DbError::Migration(e.to_string()))?;
135
136        let rows: Vec<ResourceRow> = result.take(0).map_err(DbError::from)?;
137        let row = take_first_or_not_found(rows, "resource", &id_str)?;
138
139        row_to_resource(row, id).map_err(Into::into)
140    }

162    async fn update(
163        &self,
164        tenant_id: Uuid,
165        id: Uuid,
166        input: UpdateResource,
167    ) -> AxiamResult<Resource> {
168        let id_str = id.to_string();
169        let tenant_id_str = tenant_id.to_string();
170
171        let mut sets = Vec::new();
172        if input.name.is_some() {
173            sets.push("name = $name");
174        }
175        if input.resource_type.is_some() {
176            sets.push("resource_type = $resource_type");
177        }
178        if input.parent_id.is_some() {
179            sets.push("parent_id = $parent_id");
180        }
181        if input.metadata.is_some() {
182            sets.push("metadata = $metadata");
183        }
184        sets.push("updated_at = time::now()");
185
186        let mut query = format!(
187            "UPDATE type::record('resource', $id) SET {} \
188             WHERE tenant_id = $tenant_id",
189            sets.join(", ")
190        );
191
192        // If parent_id is being changed, update the child_of edge.
193        let parent_id_changed = input.parent_id.is_some();
194        if parent_id_changed {
195            // CQ-B08: cycle check — walk ancestors of the new parent;
196            // if we visit the current resource id, this re-parent would form a cycle.
197            if let Some(Some(new_parent)) = &input.parent_id {
198                // Self-parent is always a cycle.
199                if *new_parent == id {
200                    return Err(AxiamError::Validation {
201                        message: "cycle detected in resource hierarchy".into(),
202                    });
203                }
204                // Validate the new parent exists within the caller's tenant, so
205                // a re-parent cannot attach this resource under another tenant's
206                // node (get_by_id is tenant-scoped and returns NotFound
207                // otherwise).
208                self.get_by_id(tenant_id, *new_parent).await?;
209                // Walk the new parent's ancestors: if any equals `id`, it's a cycle.
210                let ancestors = self.get_ancestors(tenant_id, *new_parent).await?;
211                if ancestors.iter().any(|a| a.id == id) {
212                    return Err(AxiamError::Validation {
213                        message: "cycle detected in resource hierarchy".into(),
214                    });
215                }
216            }
217
218            // Re-parenting mutates two-to-three statements (DELETE the old
219            // child_of edge, UPDATE the record, optionally RELATE the new
220            // edge). Wrap them in one transaction so a failure after the
221            // DELETE (e.g. the RELATE hitting a UNIQUE violation) cannot leave
222            // the resource orphaned with no rollback — the same class the
223            // resource.delete transaction closed. The child_of edge carries no
224            // flat tenant_id column, so the DELETE is tenant-scoped via a
225            // node-tenant guard on its `in` endpoint (in.tenant_id); without
226            // it a foreign-tenant resource id could strip another tenant's
227            // parent edge.
228            //
229            // Result slots become: BEGIN=0, DELETE child_of=1, UPDATE=2,
230            // [RELATE=3], COMMIT=last — so the UPDATE row lands at slot 2.
231            query = format!(
232                "BEGIN TRANSACTION; \
233                 DELETE child_of WHERE in = resource:`{id_str}` AND in.tenant_id = $tenant_id; \
234                 {query}"
235            );
236
237            // If new parent is Some, create new child_of edge.
238            if let Some(Some(new_parent)) = &input.parent_id {
239                let new_parent_str = new_parent.to_string();
240                query = format!(
241                    "{query}; RELATE resource:`{id_str}` -> child_of -> resource:`{new_parent_str}`"
242                );
243            }
244
245            query = format!("{query}; COMMIT TRANSACTION");
246        }
247
248        let parent_id_str: Option<String> = input
249            .parent_id
250            .map(|opt| opt.map(|u| u.to_string()))
251            .unwrap_or(None);
252
253        let mut builder = self
254            .db
255            .query(&query)
256            .bind(("id", id_str.clone()))
257            .bind(("tenant_id", tenant_id_str));
258
259        if let Some(name) = input.name {
260            builder = builder.bind(("name", name));
261        }
262        if let Some(resource_type) = input.resource_type {
263            builder = builder.bind(("resource_type", resource_type));
264        }
265        if input.parent_id.is_some() {
266            builder = builder.bind(("parent_id", parent_id_str));
267        }
268        if let Some(metadata) = input.metadata {
269            builder = builder.bind(("metadata", metadata));
270        }
271
272        let result = builder.await.map_err(DbError::from)?;
273        let mut result = result
274            .check()
275            .map_err(|e| DbError::Migration(e.to_string()))?;
276
277        // The UPDATE statement index depends on whether we wrapped the
278        // re-parent in a transaction: BEGIN=0, DELETE child_of=1, UPDATE=2.
279        let stmt_idx = if parent_id_changed { 2 } else { 0 };
280        let rows: Vec<ResourceRow> = result.take(stmt_idx).map_err(DbError::from)?;
281        let row = take_first_or_not_found(rows, "resource", &id_str)?;
282
283        row_to_resource(row, id).map_err(Into::into)
284    }

286    async fn delete(&self, tenant_id: Uuid, id: Uuid) -> AxiamResult<()> {
287        let id_str = id.to_string();
288        let tenant_id_str = tenant_id.to_string();
289
290        // D-13/CQ-B46: fold the child-count guard into the SAME
291        // transaction as the deletes via a LET-capture (mirroring
292        // federation_login_state.rs's `LET $row = (...)` idiom). Before
293        // this fix, the guard ran as a separate `.query()` round-trip
294        // BEFORE the deletes — a classic TOCTOU: a concurrent child-create
295        // could commit in the gap between the count-check and the delete,
296        // leaving both a successful parent delete AND an orphaned child.
297        // Now the read-then-decide-then-delete happens atomically: either
298        // the transaction observes children and aborts before any DELETE
299        // runs, or it observes none and every DELETE below commits
300        // together.
301        //
302        // D-13/CQ-B07/SEC-058: child_of/on_resource edge tables carry no
303        // tenant_id field of their own (schema v19 defines
304        // idx_child_of_unique / idx_on_resource_unique on (in, out) only),
305        // so tenant scoping on those two DELETEs is expressed as a
306        // node-tenant subquery guard on the edge's in/out endpoint
307        // (in.tenant_id / out.tenant_id via graph-traversal dereference),
308        // not a flat WHERE clause. `scope` and the resource record itself
309        // DO carry their own tenant_id field, so those two keep their
310        // pre-existing flat predicates.
311        //
312        // Result slots: BEGIN=0, LET children=1, IF/THROW=2,
313        // DELETE child_of=3, DELETE on_resource=4, DELETE scope=5,
314        // DELETE resource=6, COMMIT=7. delete() returns Ok(()) — no row
315        // data to extract, .check() alone proves the transaction
316        // committed (or surfaces the child-guard THROW).
317        let query = format!(
318            "BEGIN TRANSACTION; \
319             LET $children = (SELECT VALUE id FROM child_of WHERE out = resource:`{id_str}`); \
320             IF array::len($children) > 0 {{ \
321                 THROW 'cannot delete resource with children'; \
322             }}; \
323             DELETE child_of WHERE \
324                 (in = resource:`{id_str}` AND in.tenant_id = $tenant_id) OR \
325                 (out = resource:`{id_str}` AND out.tenant_id = $tenant_id); \
326             DELETE on_resource WHERE out = resource:`{id_str}` AND out.tenant_id = $tenant_id; \
327             DELETE scope WHERE resource_id = $resource_id AND tenant_id = $tenant_id; \
328             DELETE type::record('resource', $id) WHERE tenant_id = $tenant_id; \
329             COMMIT TRANSACTION"
330        );
331
332        let mut result = self
333            .db
334            .query(query)
335            .bind(("id", id_str.clone()))
336            .bind(("resource_id", id_str))
337            .bind(("tenant_id", tenant_id_str))
338            .await
339            .map_err(DbError::from)?;
340
341        // The child-guard THROW fires on its own statement slot, but the
342        // trailing DELETE/COMMIT slots report the generic "not executed due
343        // to a failed transaction" error, and Response::check() can surface
344        // one of those instead of the THROW text — which silently downgraded
345        // the child-guard rejection to an opaque Migration error
346        // (req14_tenant_isolation_test::resource_delete_with_children_rejected
347        // regressed when 29-02 moved the guard from a Rust pre-check into the
348        // transaction). Scan every statement error so the THROW message is
349        // reliably detected regardless of which slot check() would return.
350        let errors = result.take_errors();
351        if !errors.is_empty() {
352            let combined = errors
353                .into_values()
354                .map(|e| e.to_string())
355                .collect::<Vec<_>>()
356                .join("; ");
357            if combined.contains("cannot delete resource with children") {
358                return Err(AxiamError::Validation {
359                    message: "cannot delete resource with children".into(),
360                });
361            }
362            return Err(DbError::Migration(combined).into());
363        }
364
365        Ok(())
366    }

368    async fn list(
369        &self,
370        tenant_id: Uuid,
371        pagination: Pagination,
372    ) -> AxiamResult<PaginatedResult<Resource>> {
373        let tenant_id_str = tenant_id.to_string();
374
375        let mut count_result = self
376            .db
377            .query(
378                "SELECT count() AS total FROM resource \
379                 WHERE tenant_id = $tenant_id GROUP ALL",
380            )
381            .bind(("tenant_id", tenant_id_str.clone()))
382            .await
383            .map_err(DbError::from)?;
384        let count_rows: Vec<CountRow> = count_result.take(0).map_err(DbError::from)?;
385
386        let mut result = self
387            .db
388            .query(
389                "SELECT meta::id(id) AS record_id, * FROM resource \
390                 WHERE tenant_id = $tenant_id \
391                 ORDER BY created_at ASC \
392                 LIMIT $limit START $offset",
393            )
394            .bind(("tenant_id", tenant_id_str))
395            .bind(("limit", pagination.limit))
396            .bind(("offset", pagination.offset))
397            .await
398            .map_err(DbError::from)?;
399
400        let rows: Vec<ResourceRowWithId> = result.take(0).map_err(DbError::from)?;
401
402        let items = rows
403            .into_iter()
404            .map(|row| row.try_into_resource())
405            .collect::<Result<Vec<_>, DbError>>()?;
406
407        Ok(paginate(items, count_rows, &pagination))
408    }

433    async fn get_ancestors(&self, tenant_id: Uuid, id: Uuid) -> AxiamResult<Vec<Resource>> {
434        let tenant_id_str = tenant_id.to_string();
435        let id_str = id.to_string();
436
437        // CQ-B13: walk the hierarchy in a SINGLE recursive graph query instead of
438        // one SELECT per level. The `child_of` edge points child -> parent, so
439        // `->child_of->resource` climbs one level; `.{..N+path}` follows that edge
440        // up to N times and yields the ordered path of ancestor records
441        // (nearest-first, EXCLUDING the starting resource — matching the previous
442        // loop's output). The `(resource WHERE tenant_id = $tenant_id)` node filter
443        // enforces tenant scoping: an ancestor in another tenant terminates the
444        // walk, exactly like the old per-level `WHERE tenant_id` guard.
445        //
446        // CQ-B08 preserved: `+path` does NOT deduplicate, so both a
447        // pathologically deep chain AND a cycle traverse until the depth cap and
448        // return exactly MAX_ANCESTOR_DEPTH records. The original loop errored as
449        // soon as it would step onto the MAX_ANCESTOR_DEPTH-th ancestor (i.e. when
450        // there are >= MAX_ANCESTOR_DEPTH ancestors), so we recurse to
451        // MAX_ANCESTOR_DEPTH and treat a result of that length as an
452        // overflow/cycle — surfacing the same error rather than silently
453        // truncating.
454        let query = format!(
455            "SELECT meta::id(id) AS record_id, tenant_id, name, resource_type, \
456                    parent_id, metadata, created_at, updated_at \
457             FROM array::flatten(\
458                 type::record('resource', $id).{{..{depth}+path}}(\
459                     ->child_of->(resource WHERE tenant_id = $tenant_id)));",
460            depth = MAX_ANCESTOR_DEPTH,
461        );
462
463        let mut result = self
464            .db
465            .query(query)
466            .bind(("id", id_str))
467            .bind(("tenant_id", tenant_id_str))
468            .await
469            .map_err(DbError::from)?;
470
471        let rows: Vec<ResourceRowWithId> = result.take(0).map_err(DbError::from)?;
472
473        if rows.len() >= MAX_ANCESTOR_DEPTH {
474            return Err(DbError::Migration(
475                "resource hierarchy exceeds maximum depth — possible cycle".into(),
476            )
477            .into());
478        }
479
480        rows.into_iter()
481            .map(|row| row.try_into_resource())
482            .collect::<Result<Vec<_>, DbError>>()
483            .map_err(Into::into)
484    }
