37fn row_to_session(row: SessionRow, id: Uuid) -> Result<Session, DbError> {
38    let tenant_id = Uuid::parse_str(&row.tenant_id)
39        .map_err(|e| DbError::Migration(format!("invalid tenant UUID: {e}")))?;
40    let user_id = Uuid::parse_str(&row.user_id)
41        .map_err(|e| DbError::Migration(format!("invalid user UUID: {e}")))?;
42    Ok(Session {
43        id,
44        tenant_id,
45        user_id,
46        token_hash: row.token_hash,
47        ip_address: row.ip_address,
48        user_agent: row.user_agent,
49        expires_at: row.expires_at,
50        created_at: row.created_at,
51    })
52}

67            ip_address: self.ip_address,

75/// SurrealDB implementation of the Session repository.
76pub struct SurrealSessionRepository<C: Connection> {
77    db: Surreal<C>,
78}
79
80// Manual Clone impl (not derive): `#[derive(Clone)]` would add a `C: Clone`
81// bound that generic `C: Connection` callers (e.g. REST handlers) cannot
82// satisfy, silently cloning a `&Self` instead. Matches SurrealUserRepository.
83impl<C: Connection> Clone for SurrealSessionRepository<C> {
84    fn clone(&self) -> Self {
85        Self {
86            db: self.db.clone(),
87        }
88    }
89}
90
91impl<C: Connection> SurrealSessionRepository<C> {
92    pub fn new(db: Surreal<C>) -> Self {
93        Self { db }
94    }
95}

97impl<C: Connection> SessionRepository for SurrealSessionRepository<C> {
98    async fn create(&self, input: CreateSession) -> AxiamResult<Session> {
99        let id = Uuid::new_v4();
100        let id_str = id.to_string();
101
102        let result = self
103            .db
104            .query(
105                "CREATE type::record('session', $id) SET \
106                 tenant_id = $tenant_id, \
107                 user_id = $user_id, \
108                 token_hash = $token_hash, \
109                 ip_address = $ip_address, \
110                 user_agent = $user_agent, \
111                 expires_at = $expires_at",
112            )
113            .bind(("id", id_str.clone()))
114            .bind(("tenant_id", input.tenant_id.to_string()))
115            .bind(("user_id", input.user_id.to_string()))
116            .bind(("token_hash", input.token_hash))
117            .bind(("ip_address", input.ip_address))
118            .bind(("user_agent", input.user_agent))
119            .bind(("expires_at", input.expires_at))
120            .await
121            .map_err(DbError::from)?;
122
123        let mut result = result
124            .check()
125            .map_err(|e| DbError::Migration(e.to_string()))?;
126
127        let rows: Vec<SessionRow> = result.take(0).map_err(DbError::from)?;
128        let row = take_first_or_not_found(rows, "session", &id_str)?;
129
130        row_to_session(row, id).map_err(Into::into)
131    }
132
133    async fn get_by_id(&self, tenant_id: Uuid, id: Uuid) -> AxiamResult<Session> {
134        let id_str = id.to_string();
135
136        let mut result = self
137            .db
138            .query(
139                "SELECT * FROM type::record('session', $id) \
140                 WHERE tenant_id = $tenant_id",
141            )
142            .bind(("id", id_str.clone()))
143            .bind(("tenant_id", tenant_id.to_string()))
144            .await
145            .map_err(DbError::from)?;
146
147        let rows: Vec<SessionRow> = result.take(0).map_err(DbError::from)?;
148        let row = take_first_or_not_found(rows, "session", &id_str)?;
149
150        row_to_session(row, id).map_err(Into::into)
151    }
152
153    async fn get_by_token_hash(&self, tenant_id: Uuid, token_hash: &str) -> AxiamResult<Session> {
154        let token_hash_owned = token_hash.to_string();
155
156        let mut result = self
157            .db
158            .query(
159                "SELECT meta::id(id) AS record_id, * FROM session \
160                 WHERE tenant_id = $tenant_id AND token_hash = $token_hash",
161            )
162            .bind(("tenant_id", tenant_id.to_string()))
163            .bind(("token_hash", token_hash_owned.clone()))
164            .await
165            .map_err(DbError::from)?;
166
167        let rows: Vec<SessionRowWithId> = result.take(0).map_err(DbError::from)?;
168        let row =
169            take_first_or_not_found(rows, "session", &format!("token_hash={token_hash_owned}"))?;
170
171        row.try_into_session().map_err(Into::into)
172    }
173
174    async fn invalidate(&self, tenant_id: Uuid, id: Uuid) -> AxiamResult<()> {
175        self.db
176            .query(
177                "DELETE type::record('session', $id) \
178                 WHERE tenant_id = $tenant_id",
179            )
180            .bind(("id", id.to_string()))
181            .bind(("tenant_id", tenant_id.to_string()))
182            .await
183            .map_err(DbError::from)?;
184
185        Ok(())
186    }
187
188    async fn consume(&self, tenant_id: Uuid, id: Uuid) -> AxiamResult<bool> {
189        // NEW-3: `RETURN BEFORE` yields the pre-delete row; a non-empty result
190        // means THIS statement removed the session. SurrealDB serializes two
191        // concurrent DELETEs on the same record, so exactly one caller sees a
192        // non-empty BEFORE image — the single-use gate for refresh rotation.
193        let mut result = self
194            .db
195            .query(
196                "DELETE type::record('session', $id) \
197                 WHERE tenant_id = $tenant_id RETURN BEFORE",
198            )
199            .bind(("id", id.to_string()))
200            .bind(("tenant_id", tenant_id.to_string()))
201            .await
202            .map_err(DbError::from)?;
203
204        let deleted: Vec<SessionRow> = result.take(0).map_err(DbError::from)?;
205        Ok(!deleted.is_empty())
206    }
207
208    async fn invalidate_user_sessions(&self, tenant_id: Uuid, user_id: Uuid) -> AxiamResult<()> {
209        self.db
210            .query("DELETE session WHERE tenant_id = $tenant_id AND user_id = $user_id")
211            .bind(("tenant_id", tenant_id.to_string()))
212            .bind(("user_id", user_id.to_string()))
213            .await
214            .map_err(DbError::from)?;
215
216        Ok(())
217    }
218
219    async fn invalidate_user_sessions_except(
220        &self,
221        tenant_id: Uuid,
222        user_id: Uuid,
223        current_session_id: Uuid,
224    ) -> AxiamResult<u64> {
225        // DELETE all sessions for this user in this tenant EXCEPT the
226        // current one (identified by its record ID). RETURN BEFORE gives us
227        // the deleted rows so we can count them.
228        let mut result = self
229            .db
230            .query(
231                "DELETE session \
232                 WHERE tenant_id = $tenant_id \
233                   AND user_id = $user_id \
234                   AND id != type::record('session', $current_session_id) \
235                 RETURN BEFORE",
236            )
237            .bind(("tenant_id", tenant_id.to_string()))
238            .bind(("user_id", user_id.to_string()))
239            .bind(("current_session_id", current_session_id.to_string()))
240            .await
241            .map_err(DbError::from)?;
242
243        let deleted: Vec<SessionRow> = result.take(0).map_err(DbError::from)?;
244        Ok(deleted.len() as u64)
245    }
246
247    async fn list_by_user(&self, tenant_id: Uuid, user_id: Uuid) -> AxiamResult<Vec<Session>> {
248        let mut result = self
249            .db
250            .query(
251                "SELECT meta::id(id) AS record_id, * FROM session \
252                 WHERE tenant_id = $tenant_id AND user_id = $user_id",
253            )
254            .bind(("tenant_id", tenant_id.to_string()))
255            .bind(("user_id", user_id.to_string()))
256            .await
257            .map_err(DbError::from)?;
258
259        let rows: Vec<SessionRowWithId> = result.take(0).map_err(DbError::from)?;
260        rows.into_iter()
261            .map(|r| r.try_into_session().map_err(Into::into))
262            .collect()
263    }
264
265    async fn cleanup_expired(&self, tenant_id: Uuid) -> AxiamResult<u64> {
266        // Count expired sessions first, then delete.
267        let mut count_result = self
268            .db
269            .query(
270                "SELECT count() AS total FROM session \
271                 WHERE tenant_id = $tenant_id AND expires_at < time::now() \
272                 GROUP ALL",
273            )
274            .bind(("tenant_id", tenant_id.to_string()))
275            .await
276            .map_err(DbError::from)?;
277        let count_rows: Vec<CountRow> = count_result.take(0).map_err(DbError::from)?;
278        let total = count_rows.first().map(|r| r.total).unwrap_or(0);
279
280        self.db
281            .query("DELETE session WHERE tenant_id = $tenant_id AND expires_at < time::now()")
282            .bind(("tenant_id", tenant_id.to_string()))
283            .await
284            .map_err(DbError::from)?;
285
286        Ok(total)
287    }

305    #[tokio::test]
306    async fn list_by_user_returns_only_target_users_sessions() {
307        let db = setup_db().await;
308        let repo = SurrealSessionRepository::new(db);
309        let tenant_id = Uuid::new_v4();
310        let other_tenant_id = Uuid::new_v4();
311        let target_user_id = Uuid::new_v4();
312        let other_user_id = Uuid::new_v4();
313
314        let expires = Utc::now() + Duration::hours(1);
315
316        // 2 sessions for the target user in the target tenant.
317        for i in 0..2 {
318            repo.create(CreateSession {
319                tenant_id,
320                user_id: target_user_id,
321                token_hash: format!("target-token-hash-{i}"),
322                ip_address: Some("127.0.0.1".into()),
323                user_agent: Some("test-agent".into()),
324                expires_at: expires,
325            })
326            .await
327            .unwrap();
328        }
329
330        // 1 session for a different user in the same tenant.
331        repo.create(CreateSession {
332            tenant_id,
333            user_id: other_user_id,
334            token_hash: "other-user-token-hash".into(),
335            ip_address: Some("127.0.0.1".into()),
336            user_agent: Some("test-agent".into()),
337            expires_at: expires,
338        })
339        .await
340        .unwrap();
341
342        // 1 session for the target user but in a DIFFERENT tenant — must be
343        // excluded (tenant isolation).
344        repo.create(CreateSession {
345            tenant_id: other_tenant_id,
346            user_id: target_user_id,
347            token_hash: "cross-tenant-token-hash".into(),
348            ip_address: Some("127.0.0.1".into()),
349            user_agent: Some("test-agent".into()),
350            expires_at: expires,
351        })
352        .await
353        .unwrap();
354
355        let sessions = repo.list_by_user(tenant_id, target_user_id).await.unwrap();
356        assert_eq!(
357            sessions.len(),
358            2,
359            "expected exactly the target user's 2 sessions"
360        );
361        for session in &sessions {
362            assert_eq!(session.tenant_id, tenant_id);
363            assert_eq!(session.user_id, target_user_id);
364        }
365    }
