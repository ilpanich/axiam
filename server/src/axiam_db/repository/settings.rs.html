320    /// Fetch a settings row by scope and scope_id.
321    async fn fetch_row(
322        &self,
323        scope: &str,
324        scope_id: &str,
325    ) -> Result<Option<SecuritySettings>, DbError> {
326        let mut result = self
327            .db
328            .query(SELECT_WITH_ID)
329            .bind(("scope", scope.to_string()))
330            .bind(("scope_id", scope_id.to_string()))
331            .await

334        let rows: Vec<SettingsRowWithId> = result.take(0).map_err(DbError::from)?;
335        match rows.into_iter().next() {
336            Some(row) => Ok(Some(row.try_into_security_settings()?)),
337            None => Ok(None),

366    /// Upsert a complete settings row. Reuses the existing record ID
367    /// and preserves `created_at` when a row already exists (UPDATE).
368    /// Creates a new row otherwise (CREATE).
369    ///
370    /// `overrides_json` is `Some(json)` for tenant rows (sparse override mask,
371    /// CQ-B03) and `None` for org rows.
372    async fn upsert(
373        &self,
374        settings: &SecuritySettings,
375        overrides_json: Option<String>,
376    ) -> Result<SecuritySettings, DbError> {
377        // Use a deterministic record ID derived from (scope, scope_id) so
378        // that concurrent requests for the same key converge on the same
379        // row.  UPDATE on a deterministic ID is a single-statement upsert
380        // that is safe under concurrency (no read-then-create race).
381        let scope_str = settings.scope.to_string();
382        let scope_id_str = settings.scope_id.to_string();
383        let deterministic_id = Uuid::new_v5(
384            &Uuid::NAMESPACE_OID,
385            format!("{scope_str}:{scope_id_str}").as_bytes(),
386        );
387        let id_str = deterministic_id.to_string();
388
389        // UPSERT: single statement — creates the row if it doesn't
390        // exist, updates it otherwise. Timestamps are handled inline:
391        // `created_at` is only set on initial creation via
392        // `time::now()` default, and `updated_at` is always refreshed.
393        let query = format!(
394            "UPSERT type::record('security_settings', $id) SET \
395             {SETTINGS_FIELDS}, \
396             created_at = created_at OR time::now(), \
397             updated_at = time::now()",
398        );
399
400        let bindings = self.bind_settings(settings, overrides_json);
401        let mut builder = self.db.query(&query).bind(("id", id_str.clone()));
402        for (name, value) in bindings {
403            builder = match value {
404                BindValue::Str(v) => builder.bind((name, v)),
405                BindValue::OptionStr(v) => builder.bind((name, v)),
406                BindValue::Bool(v) => builder.bind((name, v)),
407                BindValue::U32(v) => builder.bind((name, v)),
408                BindValue::U64(v) => builder.bind((name, v)),
409                BindValue::F64(v) => builder.bind((name, v)),
410            };
411        }
412
413        let result = builder.await.map_err(DbError::from)?;
414        let mut result = result
415            .check()
416            .map_err(|e| DbError::Migration(e.to_string()))?;
417
418        let rows: Vec<SettingsRow> = result.take(0).map_err(DbError::from)?;
419        let row = rows.into_iter().next().ok_or_else(|| DbError::NotFound {
420            entity: "security_settings".into(),
421            id: id_str,
422        })?;
423
424        Ok(SecuritySettings {
425            id: deterministic_id,
426            scope: settings.scope,
427            scope_id: settings.scope_id,
428            password: PasswordPolicy {
429                min_length: row.pw_min_length,
430                require_uppercase: row.pw_require_uppercase,
431                require_lowercase: row.pw_require_lowercase,
432                require_digits: row.pw_require_digits,
433                require_symbols: row.pw_require_symbols,
434                password_history_count: row.pw_history_count,
435                hibp_check_enabled: row.pw_hibp_check,
436            },
437            mfa: MfaPolicy {
438                mfa_enforced: row.mfa_enforced,
439                mfa_challenge_lifetime_secs: row.mfa_challenge_lifetime,
440            },
441            lockout: LockoutPolicy {
442                max_failed_login_attempts: row.lockout_max_attempts,
443                lockout_duration_secs: row.lockout_duration,
444                lockout_backoff_multiplier: row.lockout_backoff,
445                max_lockout_duration_secs: row.lockout_max_duration,
446            },
447            token: TokenPolicy {
448                access_token_lifetime_secs: row.token_access_lifetime,
449                refresh_token_lifetime_secs: row.token_refresh_lifetime,
450            },
451            email: EmailVerificationPolicy {
452                email_verification_required: row.email_verification_required,
453                email_verification_grace_period_hours: row.email_grace_period_hours,
454            },
455            certificate: CertificatePolicy {
456                default_cert_validity_days: row.cert_default_validity,
457                max_cert_validity_days: row.cert_max_validity,

459            notification: NotificationPolicy {
460                admin_notifications_enabled: row.notif_admin_enabled,
461            },
462            created_at: row.created_at,
463            updated_at: row.updated_at,
464        })
465    }

484impl<C: Connection> SettingsRepository for SurrealSettingsRepository<C> {
485    async fn get_org_settings(&self, org_id: Uuid) -> AxiamResult<SecuritySettings> {
486        match self.fetch_row("org", &org_id.to_string()).await? {
487            Some(s) => Ok(s),
488            None => {
489                // Return system defaults with a deterministic ID and
490                // stable timestamps so clients see consistent values
491                // across reads (no row persisted yet).
492                let defaults = system_defaults();
493                let deterministic_id =
494                    Uuid::new_v5(&Uuid::NAMESPACE_OID, format!("org:{org_id}").as_bytes());
495                let mut settings = settings_from_org_input(deterministic_id, org_id, &defaults);
496                settings.created_at = DateTime::<Utc>::UNIX_EPOCH;
497                settings.updated_at = DateTime::<Utc>::UNIX_EPOCH;
498                Ok(settings)
499            }
500        }
501    }
502
503    async fn set_org_settings(
504        &self,
505        org_id: Uuid,
506        input: SetOrgSettings,
507    ) -> AxiamResult<SecuritySettings> {
508        // ID will be determined by upsert (reuses existing or generates new).
509        // Org rows carry no overrides_json (None = org-scope row).
510        let settings = settings_from_org_input(Uuid::nil(), org_id, &input);
511        let result = self.upsert(&settings, None).await?;
512        Ok(result)
513    }

515    async fn get_tenant_override(
516        &self,
517        tenant_id: Uuid,
518    ) -> AxiamResult<Option<TenantSettingsOverride>> {
519        // Always look up the org so we can fall back to diff-against-org
520        // for pre-V16 rows (no overrides_json present).
521        let org_id = self.lookup_org_id(tenant_id).await?;
522        let org = self.get_org_settings(org_id).await?;
523
524        match self
525            .fetch_row_with_overrides("tenant", &tenant_id.to_string())
526            .await?
527        {
528            None => Ok(None),
529            Some((tenant_settings, overrides_json_opt)) => {
530                let overrides = if let Some(ref json) = overrides_json_opt {
531                    // V16+ row: return the stored sparse mask directly.
532                    serde_json::from_str::<TenantSettingsOverride>(json)
533                        .unwrap_or_else(|_| diff_against_org(&org, &tenant_settings))
534                } else {
535                    // Pre-V16 row: compute from diff.
536                    diff_against_org(&org, &tenant_settings)
537                };
538                Ok(Some(overrides))
539            }
540        }
541    }

543    /// Store tenant settings as a sparse override (CQ-B03/SEC-033).
544    ///
545    /// Diffs the supplied merged `settings` against the current org baseline
546    /// so that only genuinely overridden fields are persisted in the
547    /// `overrides_json` column.  `get_effective_settings` reads from
548    /// `overrides_json` at read time, so a later org baseline change
549    /// propagates to any field the tenant did not explicitly override.
550    async fn store_effective_tenant_settings(
551        &self,
552        tenant_id: Uuid,
553        settings: SecuritySettings,
554    ) -> AxiamResult<SecuritySettings> {
555        // Look up the current org baseline so we can isolate the true overrides.
556        let org_id = self.lookup_org_id(tenant_id).await?;
557        let org = self.get_org_settings(org_id).await?;
558
559        // Compute only the fields that genuinely differ from the org baseline.
560        let sparse_overrides = diff_against_org(&org, &settings);
561
562        // Persist the sparse override mask alongside the merged row.
563        let overrides_json = serde_json::to_string(&sparse_overrides)
564            .map_err(|e| DbError::Migration(format!("failed to serialize overrides: {e}")))?;
565
566        // Re-merge to produce the canonical stored row.  The `overrides_json`
567        // column preserves which fields were explicitly set so that
568        // `get_effective_settings` can distinguish intent from inheritance.
569        let merged = effective_settings(&org, &sparse_overrides, tenant_id, settings.id);
570
571        let result = self.upsert(&merged, Some(overrides_json)).await?;
572        Ok(result)
573    }

575    async fn set_tenant_override(
576        &self,
577        tenant_id: Uuid,
578        input: SetTenantOverride,
579    ) -> AxiamResult<TenantSettingsOverride> {
580        // Look up the tenant's org_id to fetch the org baseline.
581        let org_id = self.lookup_org_id(tenant_id).await?;
582        let org = self.get_org_settings(org_id).await?;
583
584        // Persist the sparse override mask (CQ-B03) so that a later org
585        // baseline change propagates to fields the tenant did not override.
586        let overrides_json = serde_json::to_string(&input)
587            .map_err(|e| DbError::Migration(format!("failed to serialize overrides: {e}")))?;
588
589        // Merge org baseline + overrides into a complete tenant row to be
590        // stored; the full merged values populate all required schema fields.
591        let merged = effective_settings(&org, &input, tenant_id, Uuid::nil());
592
593        self.upsert(&merged, Some(overrides_json)).await?;
594        // Return the sparse diff against the org baseline so the caller sees
595        // only the explicitly-set overrides (not inherited values).
596        Ok(diff_against_org(&org, &merged))
597    }

611    async fn get_effective_settings(
612        &self,
613        org_id: Uuid,
614        tenant_id: Uuid,
615    ) -> AxiamResult<SecuritySettings> {
616        let org = self.get_org_settings(org_id).await?;
617
618        // If there is a stored tenant row, resolve effective settings by
619        // merging the current org baseline with the tenant's sparse overrides.
620        // The sparse overrides are stored in `overrides_json` (V16, CQ-B03).
621        // For rows written before V16 we fall back to diff-against-org for
622        // backward compatibility.
623        if let Some((tenant_row, overrides_json_opt)) = self
624            .fetch_row_with_overrides("tenant", &tenant_id.to_string())
625            .await?
626        {
627            let overrides: TenantSettingsOverride = if let Some(ref json) = overrides_json_opt {
628                // V16+ row: use the stored sparse override mask directly so
629                // that fields the tenant did NOT override pick up the current
630                // org baseline (CQ-B03).
631                serde_json::from_str(json).unwrap_or_else(|_| diff_against_org(&org, &tenant_row))
632            } else {
633                // Pre-V16 row: fall back to diff-against-org.  This may
634                // produce stale results until the row is next written through
635                // `set_tenant_override` or `store_effective_tenant_settings`.
636                diff_against_org(&org, &tenant_row)
637            };
638            let mut merged = effective_settings(&org, &overrides, tenant_id, tenant_row.id);
639            // Preserve the persisted timestamps rather than using
640            // Utc::now() from the merge function.
641            merged.created_at = tenant_row.created_at;
642            merged.updated_at = tenant_row.updated_at;
643            return Ok(merged);
644        }
645
646        // No tenant row — org settings (or system defaults) apply.
647        // Re-scope as tenant so callers always receive a tenant-scoped
648        // result with a stable deterministic ID and sentinel timestamps
649        // (epoch = "never explicitly configured").
650        let deterministic_id = Uuid::new_v5(
651            &Uuid::NAMESPACE_OID,
652            format!("tenant:{tenant_id}").as_bytes(),
653        );
654        Ok(SecuritySettings {
655            id: deterministic_id,
656            scope: SettingsScope::Tenant,
657            scope_id: tenant_id,
658            created_at: DateTime::<Utc>::UNIX_EPOCH,
659            updated_at: DateTime::<Utc>::UNIX_EPOCH,
660            ..org
661        })
662    }
663}
