452-- =======================================================================
453-- AMQP Nonce Replay Table (NEW-4)
454-- =======================================================================
455-- Durable per-tenant nonce store for AMQP message replay protection. The
456-- authz/audit consumers insert (tenant_id, nonce) after a valid HMAC + fresh
457-- issued_at; the UNIQUE index turns a replayed nonce into a conflict that the
458-- repository maps to ReplayDetected. Rows expire at issued_at + skew and are
459-- swept by the periodic cleanup task (no native SurrealDB TTL, RESEARCH §7).
460DEFINE TABLE IF NOT EXISTS amqp_nonce_replay SCHEMAFULL;
461DEFINE FIELD IF NOT EXISTS tenant_id ON TABLE amqp_nonce_replay TYPE string;
462DEFINE FIELD IF NOT EXISTS nonce ON TABLE amqp_nonce_replay TYPE string;
463DEFINE FIELD IF NOT EXISTS expires_at ON TABLE amqp_nonce_replay TYPE datetime;
464DEFINE FIELD IF NOT EXISTS created_at ON TABLE amqp_nonce_replay TYPE datetime \
465    DEFAULT time::now();
466DEFINE INDEX IF NOT EXISTS idx_amqp_nonce_uniq ON TABLE amqp_nonce_replay \
467    COLUMNS tenant_id, nonce UNIQUE;
468DEFINE INDEX IF NOT EXISTS idx_amqp_nonce_expires_at ON TABLE amqp_nonce_replay \
469    COLUMNS expires_at;

1161// -----------------------------------------------------------------------
1162// Schema v22 — bootstrap atomicity + mandatory gate (SECHRD-04 / SEC-049)
1163// -----------------------------------------------------------------------
1164//
1165// Three additive tables backing the atomic single-super-admin invariant
1166// and the mandatory first-run gate:
1167//
1168// - `bootstrap_lock`: record ID = tenant_id. The bootstrap transaction
1169//   CREATEs a lock record for the target tenant; a concurrent second
1170//   request racing on the SAME tenant_id hits a UNIQUE-index violation on
1171//   this CREATE (the record ID itself IS the uniqueness constraint) and
1172//   its whole transaction rolls back — no partial admin, no orphan role
1173//   RELATE (D-03c).
1174// - `bootstrap_setup_token`: record ID = sha256(token) hex. Stores ONLY
1175//   the token hash — the plaintext token is never persisted, only logged
1176//   once at first boot (D-03b).
1177// - `bootstrap_setup_token_consumed`: record ID = sha256(token) hex.
1178//   Consumption-by-existence: the bootstrap transaction CREATEs this
1179//   record in the SAME transaction as the admin user, so a replay of the
1180//   same token also loses to the same UNIQUE-index violation.
1181
1182const SCHEMA_V22: &str = "\
1183-- =======================================================================
1184-- Bootstrap atomicity + mandatory gate (SECHRD-04 / SEC-049)
1185-- =======================================================================
1186DEFINE TABLE IF NOT EXISTS bootstrap_lock SCHEMAFULL TYPE NORMAL;
1187DEFINE FIELD IF NOT EXISTS locked_at ON TABLE bootstrap_lock TYPE datetime
1188    DEFAULT time::now();
1189
1190DEFINE TABLE IF NOT EXISTS bootstrap_setup_token SCHEMAFULL TYPE NORMAL;
1191DEFINE FIELD IF NOT EXISTS created_at ON TABLE bootstrap_setup_token TYPE datetime
1192    DEFAULT time::now();
1193
1194DEFINE TABLE IF NOT EXISTS bootstrap_setup_token_consumed SCHEMAFULL TYPE NORMAL;
1195DEFINE FIELD IF NOT EXISTS consumed_at ON TABLE bootstrap_setup_token_consumed TYPE datetime
1196    DEFAULT time::now();
1197";

1242DEFINE FIELD IF NOT EXISTS status ON TABLE tenant TYPE string
1243    ASSERT $value IN ['Active', 'Suspended'] DEFAULT 'Active';
1244UPDATE tenant SET status = 'Active' WHERE status = NONE;
1245-- Service account optional description (no backfill: option<string> reads
1246-- back as NONE when absent).
1247DEFINE FIELD IF NOT EXISTS description ON TABLE service_account TYPE option<string>;

1254/// Run all pending migrations against the given SurrealDB client.
1255///
1256/// Creates `_migration` and `_migration_lock` tracking tables on first run,
1257/// then applies each pending migration atomically: the schema DDL and the
1258/// version-record INSERT are wrapped in a single `BEGIN TRANSACTION …
1259/// COMMIT TRANSACTION` so that a mid-step failure leaves the row
1260/// re-selectable (CQ-B06 / REQ-14 AC-5).
1261///
1262/// The `_migration_lock` record (`CREATE IF NOT EXISTS`) guards concurrent
1263/// startup — only one process may hold the lock at a time.  The record is
1264/// removed once migrations complete.
1265pub async fn run_migrations<C: Connection>(db: &Surreal<C>) -> Result<(), DbError> {
1266    // Ensure migration-tracking tables exist (all DDL is IF NOT EXISTS).
1267    db.query(MIGRATION_TABLE_DDL)
1268        .await?
1269        .check()
1270        .map_err(|e| DbError::Migration(e.to_string()))?;
1271
1272    // Acquire a startup lock so that concurrent instances do not race on
1273    // schema application.  UPSERT with a deterministic record ID is
1274    // idempotent — if a lock record already exists this is a no-op.
1275    db.query(
1276        "BEGIN TRANSACTION; \
1277         UPSERT _migration_lock:`startup` SET locked_at = time::now(); \
1278         COMMIT TRANSACTION",
1279    )
1280    .await?
1281    .check()
1282    .map_err(|e| DbError::Migration(format!("failed to acquire _migration_lock: {e}")))?;
1283
1284    // Determine current schema version.
1285    let mut result = db
1286        .query("SELECT * FROM _migration ORDER BY version DESC LIMIT 1")
1287        .await?;
1288    let records: Vec<MigrationRecord> = result.take(0)?;
1289    let current_version = records.first().map(|m| m.version).unwrap_or(0);
1290
1291    for migration in MIGRATIONS {
1292        if migration.version > current_version {
1293            info!(
1294                version = migration.version,
1295                name = migration.name,
1296                "Applying migration"
1297            );
1298
1299            // Wrap the schema DDL and the version-record INSERT in a single
1300            // transaction (CQ-B06).  If the DDL succeeds but the INSERT
1301            // fails, the whole block is rolled back and the migration will
1302            // be retried on the next startup.
1303            //
1304            // SurrealDB v3 transaction slot offset: BEGIN = slot 0, first
1305            // statement = slot 1, … (MEMORY.md).  We don't take() from the
1306            // result; we only call .check() to surface errors.
1307            let txn = format!(
1308                "BEGIN TRANSACTION;\n\
1309                 {};\n\
1310                 CREATE _migration SET version = {v}, name = '{n}';\n\
1311                 COMMIT TRANSACTION",
1312                migration.sql,
1313                v = migration.version,
1314                n = migration.name,
1315            );
1316            db.query(&txn).await?.check().map_err(|e| {
1317                DbError::Migration(format!(
1318                    "Migration v{} '{}' failed (transaction rolled back): {}",
1319                    migration.version, migration.name, e,
1320                ))
1321            })?;
1322
1323            info!(
1324                version = migration.version,
1325                "Migration applied successfully"
1326            );
1327        }
1328    }
1329
1330    // Release the startup lock now that migrations are complete.
1331    db.query("DELETE _migration_lock:`startup`")
1332        .await?
1333        .check()
1334        .map_err(|e| DbError::Migration(format!("failed to release migration lock: {e}")))?;
1335
1336    Ok(())
1337}
