1//! Resend email provider (REST API).
2
3use std::future::Future;
4use std::pin::Pin;
5
6use axiam_core::error::{AxiamError, AxiamResult};
7use axiam_core::models::email::ApiProviderConfig;
8use reqwest::Client;
9
10use super::build_http_client;
11use crate::message::EmailMessage;
12use crate::provider::{EmailProvider, SendResult};
13
14const DEFAULT_API_URL: &str = "https://api.resend.com/emails";
15
16pub struct ResendProvider {
17    client: Client,
18    api_key: String,
19    api_url: String,
20}
21
22impl ResendProvider {
23    pub fn new(config: &ApiProviderConfig) -> AxiamResult<Self> {
24        Ok(Self {
25            client: build_http_client()?,
26            api_key: config.api_key.clone(),
27            api_url: config
28                .api_url
29                .clone()
30                .unwrap_or_else(|| DEFAULT_API_URL.to_string()),
31        })
32    }
33}
34
35impl EmailProvider for ResendProvider {
36    fn send(
37        &self,
38        from_name: &str,
39        from_email: &str,
40        reply_to: Option<&str>,
41        message: &EmailMessage,
42    ) -> Pin<Box<dyn Future<Output = AxiamResult<SendResult>> + Send + '_>> {
43        let from = format!("{from_name} <{from_email}>");
44        let reply_to = reply_to.map(str::to_string);
45        let message = message.clone();
46
47        Box::pin(async move {
48            let mut body = serde_json::json!({
49                "from": from,
50                "to": [message.to],
51                "subject": message.subject,
52            });
53
54            if let Some(ref html) = message.html_body {
55                body["html"] = serde_json::Value::String(html.clone());
56            }
57            if let Some(ref text) = message.text_body {
58                body["text"] = serde_json::Value::String(text.clone());
59            }
60            if let Some(rt) = reply_to {
61                body["reply_to"] = serde_json::Value::String(rt);
62            }
63
64            let resp = self
65                .client
66                .post(&self.api_url)
67                .bearer_auth(&self.api_key)
68                .json(&body)
69                .send()
70                .await
71                .map_err(|e| AxiamError::EmailDelivery(format!("Resend request failed: {e}")))?;
72
73            if !resp.status().is_success() {
74                let status = resp.status();
75                let text = resp.text().await.unwrap_or_else(|_| "unknown".into());
76                return Err(AxiamError::EmailDelivery(format!(
77                    "Resend returned {status}: {text}"
78                )));
79            }
80
81            let resp_json: serde_json::Value = resp.json().await.unwrap_or_default();
82            let message_id = resp_json["id"].as_str().map(str::to_string);
83
84            Ok(SendResult { message_id })
85        })
86    }
87
88    fn provider_name(&self) -> &'static str {
89        "resend"
90    }
91}
