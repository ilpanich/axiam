1//! High-level email service that resolves the correct provider
2//! from configuration and sends messages.
3
4use axiam_core::error::{AxiamError, AxiamResult};
5use axiam_core::models::email::EmailConfig;
6
7use crate::message::EmailMessage;
8use crate::provider::{EmailProvider, SendResult};
9use crate::providers;
10
11/// High-level email service — wraps a resolved provider and sender
12/// identity.
13///
14/// Manual `Debug` impl because `Box<dyn EmailProvider>` is not `Debug`.
15pub struct EmailService {
16    provider: Box<dyn EmailProvider>,
17    from_name: String,
18    from_email: String,
19    reply_to: Option<String>,
20}
21
22impl std::fmt::Debug for EmailService {
23    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
24        f.debug_struct("EmailService")
25            .field("provider", &self.provider.provider_name())
26            .field("from_name", &self.from_name)
27            .field("from_email", &self.from_email)
28            .field("reply_to", &self.reply_to)
29            .finish()
30    }
31}
32
33impl EmailService {
34    /// Build an `EmailService` from a resolved `EmailConfig`.
35    ///
36    /// Returns an error if the provider cannot be constructed (e.g.,
37    /// invalid SMTP host) or if email is disabled.
38    pub fn from_config(config: &EmailConfig) -> AxiamResult<Self> {
39        if !config.enabled {
40            return Err(AxiamError::EmailConfig(
41                "email is disabled for this scope".into(),
42            ));
43        }
44
45        let provider = providers::build_provider(&config.provider)?;
46
47        Ok(Self {
48            provider,
49            from_name: config.from_name.clone(),
50            from_email: config.from_email.clone(),
51            reply_to: config.reply_to.clone(),
52        })
53    }
54
55    /// Build from an explicit provider (useful for testing).
56    pub fn with_provider(
57        provider: Box<dyn EmailProvider>,
58        from_name: String,
59        from_email: String,
60        reply_to: Option<String>,
61    ) -> Self {
62        Self {
63            provider,
64            from_name,
65            from_email,
66            reply_to,
67        }
68    }
69
70    /// Send an email using the configured provider.
71    pub async fn send(&self, message: &EmailMessage) -> AxiamResult<SendResult> {
72        if !message.has_body() {
73            return Err(AxiamError::EmailDelivery(
74                "email has no body (html or text)".into(),
75            ));
76        }
77
78        // Log at debug level to avoid exposing PII (recipient, subject)
79        // in production logs. Info level only shows the provider name.
80        tracing::info!(provider = self.provider.provider_name(), "sending email");
81        tracing::debug!(
82            to = %message.to,
83            subject = %message.subject,
84            "email details"
85        );
86
87        self.provider
88            .send(
89                &self.from_name,
90                &self.from_email,
91                self.reply_to.as_deref(),
92                message,
93            )
94            .await
95    }
96
97    /// Returns the name of the underlying provider.
98    pub fn provider_name(&self) -> &'static str {
99        self.provider.provider_name()
100    }
101}
