27pub type TemplateContext = HashMap<String, String>;

29// -----------------------------------------------------------------------
30// Rendering
31// -----------------------------------------------------------------------
32
33/// Replace all `{{key}}` placeholders with values from the context.
34///
35/// Uses single-pass rendering to prevent template injection: values
36/// inserted from the context are never re-processed for further
37/// placeholder expansion.  Unknown placeholders are left as-is.
38///
39/// When `html_escape` is `true`, context values are HTML-escaped
40/// (`<`, `>`, `&`, `"`, `'`) before insertion. This MUST be used
41/// for HTML body rendering to prevent XSS/injection from
42/// user-controlled values like usernames.
43pub fn render(template: &str, context: &TemplateContext) -> String {
44    render_inner(template, context, false)
45}
46
47/// Like [`render`], but HTML-escapes all substituted values.
48pub fn render_html(template: &str, context: &TemplateContext) -> String {
49    render_inner(template, context, true)
50}
51
52fn render_inner(template: &str, context: &TemplateContext, html_escape: bool) -> String {
53    let mut output = String::with_capacity(template.len());
54    let mut rest = template;
55
56    while let Some(start) = rest.find("{{") {
57        output.push_str(&rest[..start]);
58        let after_open = &rest[start + 2..];
59        if let Some(end) = after_open.find("}}") {
60            let key = &after_open[..end];
61            if let Some(value) = context.get(key) {
62                if html_escape {
63                    output.push_str(&escape_html(value));
64                } else {
65                    output.push_str(value);
66                }
67            } else {
68                // Unknown placeholder — preserve verbatim.
69                output.push_str("{{");
70                output.push_str(key);
71                output.push_str("}}");
72            }
73            rest = &after_open[end + 2..];
74        } else {
75            // Unclosed `{{` — emit as literal text.
76            output.push_str("{{");
77            rest = after_open;
78        }
79    }
80    output.push_str(rest);
81    output
82}
83
84/// Escape the five HTML-significant characters to their entity form.
85fn escape_html(s: &str) -> String {
86    let mut out = String::with_capacity(s.len());
87    for ch in s.chars() {
88        match ch {
89            '&' => out.push_str("&amp;"),
90            '<' => out.push_str("&lt;"),
91            '>' => out.push_str("&gt;"),
92            '"' => out.push_str("&quot;"),
93            '\'' => out.push_str("&#x27;"),
94            _ => out.push(ch),
95        }
96    }
97    out
98}

100/// Strip CR and LF characters from a rendered email header value.
101///
102/// Email headers are line-delimited; injecting `\r` or `\n` into a
103/// subject (or other single-line header) can cause header injection
104/// attacks.  This function removes those characters entirely.
105fn sanitize_header(value: &str) -> String {
106    value.chars().filter(|&c| c != '\r' && c != '\n').collect()
107}
108
109/// Render a resolved template into an `EmailMessage` ready for delivery.
110///
111/// HTML body values are HTML-escaped to prevent injection from
112/// user-controlled context values. Text body and subject are
113/// rendered without escaping (plain-text has no injection risk).
114/// The subject is additionally sanitized to strip CR/LF characters,
115/// preventing email header injection via user-controlled values.
116pub fn render_email(template: &EmailTemplate, to: &str, context: &TemplateContext) -> EmailMessage {
117    EmailMessage {
118        to: to.to_string(),
119        subject: sanitize_header(&render(&template.subject, context)),
120        html_body: Some(render_html(&template.html_body, context)),
121        text_body: Some(render(&template.text_body, context)),
122    }
123}

129/// Resolve the effective template: tenant → org → built-in default.
130pub fn resolve_template(
131    kind: TemplateKind,
132    org_template: Option<&EmailTemplate>,
133    tenant_template: Option<&EmailTemplate>,
134) -> EmailTemplate {
135    if let Some(t) = tenant_template {
136        return t.clone();
137    }
138    if let Some(t) = org_template {
139        return t.clone();
140    }
141    builtin_template(kind)
142}
