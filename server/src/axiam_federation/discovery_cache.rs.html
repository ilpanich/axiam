1//! OIDC discovery-document cache (plan A4 / CQ-B23).
2//!
3//! - 1-hour TTL — return the cached document if `fetched_at + 1h > now`.
4//! - 24-hour stale-while-revalidate — if the discovery endpoint is
5//!   unreachable but the cache entry is no older than 24 h past TTL, serve
6//!   the stale document with a WARN log rather than surfacing an error.
7//!
8//! Modeled directly on [`crate::jwks_cache::JwksCache`]: same
9//! `Arc<RwLock<map>>` shape, the same TTL/stale-window constants, and the
10//! same `allow_private_networks` SSRF test seam (mirrored, not shared,
11//! because `OidcFederationService::discover` needs its own cache instance
12//! distinct from the JWKS cache — see `oidc.rs`).
13//!
14//! Keyed by `metadata_url` rather than `(tenant_id, federation_config_id)`:
15//! unlike JWKS lookups (which key by the per-config tuple passed in
16//! explicitly by `verify_id_token`), an OIDC discovery document is a pure
17//! function of the metadata_url alone, so keying by URL is both sufficient
18//! and naturally dedupes configs/tenants that happen to share a
19//! metadata_url.
20

31// ---------------------------------------------------------------------------
32// Constants
33// ---------------------------------------------------------------------------
34
35/// Discovery-document cache TTL (1 hour) — mirrors `jwks_cache::TTL`.
36pub const TTL: Duration = Duration::from_secs(3600);
37
38/// Stale-while-revalidate window (24 h past TTL) — mirrors
39/// `jwks_cache::STALE_WINDOW`.
40pub const STALE_WINDOW: Duration = Duration::from_secs(24 * 3600);
41
42/// Maximum discovery document response body size: 256 KiB.
43///
44/// Prevents a malicious or misconfigured metadata endpoint from sending an
45/// unbounded response that exhausts server memory. A legitimate discovery
46/// document is typically well under 10 KiB. Enforced via a streaming,
47/// running-byte-count cap (see [`fetch_discovery_document`]) rather than
48/// buffer-then-check, so a lying/chunked response can never force full
49/// buffering before the cap is noticed.
50const MAX_DISCOVERY_SIZE: usize = 256 * 1024;
51

56#[derive(Debug, Clone)]
57pub struct DiscoveryCacheEntry {
58    pub doc: OidcDiscoveryDocument,
59    /// When the document was last successfully fetched.
60    pub fetched_at: DateTime<Utc>,
61}

73/// Process-wide OIDC discovery-document cache keyed by `metadata_url`.
74///
75/// Uses a [`tokio::sync::RwLock`] so it can be shared across async handlers
76/// without blocking the executor — identical shape to
77/// [`crate::jwks_cache::JwksCache`].
78///
79/// The second field is the SEC-054 SSRF policy: when `false` (the default
80/// and the only value used in production), metadata URLs that resolve to a
81/// private/loopback IP are rejected. It is `true` only for integration tests
82/// that serve discovery documents from a loopback mock server.
83#[derive(Clone)]
84pub struct DiscoveryCache(pub(crate) Arc<RwLock<DiscoveryCacheMap>>, pub(crate) bool);
85
86impl DiscoveryCache {
87    pub fn new() -> Self {
88        Self(Arc::new(RwLock::new(HashMap::new())), false)
89    }
90
91    /// Test-only seam: construct a cache that permits metadata URLs
92    /// resolving to private/loopback IPs, bypassing the SEC-054 SSRF guard.
93    ///
94    /// This exists solely so cross-crate integration tests can point the
95    /// discovery fetcher at a loopback mock server (e.g. wiremock on
96    /// `127.0.0.1`). It MUST NOT be used in production code — production
97    /// always constructs via [`DiscoveryCache::new`], which keeps the SSRF
98    /// guard active.
99    #[doc(hidden)]
100    pub fn new_allow_private_networks() -> Self {
101        Self(Arc::new(RwLock::new(HashMap::new())), true)
102    }
103
104    /// Test-only seam: insert a cache entry directly so integration tests can
105    /// seed stale entries (e.g. to exercise stale-while-revalidate). Keeps
106    /// the inner map encapsulated (`pub(crate)`) while allowing cross-crate
107    /// tests.
108    #[doc(hidden)]
109    pub async fn insert_for_test(&self, key: String, entry: DiscoveryCacheEntry) {
110        self.0.write().await.insert(key, entry);
111    }
112
113    /// Return the discovery document for the given metadata URL, fetching
114    /// if needed.
115    ///
116    /// 1. Cache hit within 1-h TTL → return cached document (no HTTP).
117    /// 2. Cache miss or TTL expired → attempt fetch.
118    ///    - On success: update entry, return document.
119    ///    - On failure AND entry is within 24-h stale window → WARN + return stale document.
120    ///    - On failure AND no usable cached entry → propagate error.
121    pub async fn get_or_fetch(
122        &self,
123        http: &reqwest::Client,
124        metadata_url: &str,
125    ) -> Result<OidcDiscoveryDocument, FederationError> {
126        let now = Utc::now();
127
128        // --- Fast path: cache hit within TTL ---
129        {
130            let guard = self.0.read().await;
131            if let Some(entry) = guard.get(metadata_url) {
132                // Compare expiry directly to handle sub-millisecond clock jitter
133                // (to_std() returns Err for negative durations — fetched_at in the
134                // future by a tiny amount — which would falsely miss the cache).
135                let ttl_chrono = chrono::Duration::from_std(TTL).unwrap_or_default();
136                if entry.fetched_at + ttl_chrono > now {
137                    return Ok(entry.doc.clone());
138                }
139            }
140        }
141
142        // --- Slow path: attempt a fresh fetch ---
143        match fetch_discovery_document(http, metadata_url, self.1).await {
144            Ok(doc) => {
145                let mut guard = self.0.write().await;
146                let entry =
147                    guard
148                        .entry(metadata_url.to_string())
149                        .or_insert_with(|| DiscoveryCacheEntry {
150                            doc: doc.clone(),
151                            fetched_at: now,
152                        });
153                entry.doc = doc.clone();
154                entry.fetched_at = now;
155                Ok(doc)
156            }
157            Err(fetch_err) => {
158                // Check for stale-while-revalidate.
159                let guard = self.0.read().await;
160                if let Some(entry) = guard.get(metadata_url) {
161                    let stale_window_chrono =
162                        chrono::Duration::from_std(STALE_WINDOW).unwrap_or_default();
163                    if entry.fetched_at + stale_window_chrono > now {
164                        tracing::warn!(
165                            metadata_url,
166                            "serving stale OIDC discovery document while IdP unreachable"
167                        );
168                        return Ok(entry.doc.clone());
169                    }
170                }
171                Err(fetch_err)
172            }
173        }
174    }
175}
176
177impl Default for DiscoveryCache {
178    fn default() -> Self {
179        Self::new()
180    }
181}

187/// Fetch and parse the OIDC discovery document from the provider.
188///
189/// Moved out of `oidc.rs::discover` (plan A4 / CQ-B23) so the fetch itself
190/// lives alongside the cache that fronts it, mirroring
191/// `jwks_cache::fetch_jwks`.
192async fn fetch_discovery_document(
193    // Retained for API stability of `get_or_fetch` — no longer used directly:
194    // `ssrf::guarded_fetch` builds its own fresh, IP-pinned client per
195    // request rather than reusing an injected pooled client (see
196    // `jwks_cache::fetch_jwks`'s identical comment).
197    _http: &reqwest::Client,
198    metadata_url: &str,
199    allow_private_networks: bool,
200) -> Result<OidcDiscoveryDocument, FederationError> {
201    // SECHRD-02: route the discovery-document GET through the shared,
202    // IP-pinning SSRF guard (D-01a/b/c). Production always fails closed
203    // against private/loopback/link-local addresses and internal redirect
204    // targets — `allow_private_networks=false`.
205    let response =
206        crate::ssrf::guarded_fetch(metadata_url, allow_private_networks, |c, u| c.get(u))
207            .await
208            .map_err(|e| FederationError::DiscoveryFailed(e.to_string()))?;
209
210    if !response.status().is_success() {
211        return Err(FederationError::DiscoveryFailed(format!(
212            "HTTP {} from discovery endpoint",
213            response.status()
214        )));
215    }
216
217    // CQ-B23: stream the body with a running-byte-count cap instead of
218    // buffering the whole response and checking `.len()` afterward — a
219    // malicious endpoint that lies about (or omits) `Content-Length` could
220    // otherwise force full buffering before the existing check ever ran.
221    let bytes = crate::ssrf::read_capped_body(response, MAX_DISCOVERY_SIZE)
222        .await
223        .map_err(|e| match e {
224            crate::ssrf::SsrfError::ResponseTooLarge(cap) => FederationError::DiscoveryFailed(
225                format!("Discovery document too large (max {cap} bytes)"),
226            ),
227            other => {
228                FederationError::DiscoveryFailed(format!("Failed to read response body: {other}"))
229            }
230        })?;
231
232    let doc: OidcDiscoveryDocument = serde_json::from_slice(&bytes).map_err(|e| {
233        FederationError::DiscoveryFailed(format!("Failed to parse discovery document: {e}"))
234    })?;
235
236    // Validate that critical endpoints in the discovery document use HTTPS.
237    // A compromised/malicious discovery endpoint could return http:// URLs,
238    // leaking client_secret during token exchange. Skipped under the same
239    // test-only `allow_private_networks` seam as above — a loopback wiremock
240    // IdP serves plain HTTP.
241    for (name, url) in [
242        ("authorization_endpoint", &doc.authorization_endpoint),
243        ("token_endpoint", &doc.token_endpoint),
244        ("jwks_uri", &doc.jwks_uri),
245    ] {
246        let parsed = url::Url::parse(url).map_err(|e| {
247            FederationError::DiscoveryFailed(format!("{name} is not a valid URL: {e}"))
248        })?;
249        if !allow_private_networks && parsed.scheme() != "https" {
250            return Err(FederationError::DiscoveryFailed(format!(
251                "{name} must use HTTPS, got: {url}"
252            )));
253        }
254    }
255
256    Ok(doc)
257}

263#[cfg(test)]
264mod tests {
265    use super::*;
266    use chrono::Duration as CDuration;
267
268    fn test_doc() -> OidcDiscoveryDocument {
269        OidcDiscoveryDocument {
270            issuer: "https://idp.example.com".to_string(),
271            authorization_endpoint: "https://idp.example.com/auth".to_string(),
272            token_endpoint: "https://idp.example.com/token".to_string(),
273            userinfo_endpoint: None,
274            jwks_uri: "https://idp.example.com/jwks".to_string(),
275        }
276    }
277
278    /// Insert a cache entry with a custom `fetched_at`.
279    async fn insert_entry(
280        cache: &DiscoveryCache,
281        key: &str,
282        fetched_at: DateTime<Utc>,
283        doc: OidcDiscoveryDocument,
284    ) {
285        let mut guard = cache.0.write().await;
286        guard.insert(key.to_string(), DiscoveryCacheEntry { doc, fetched_at });
287    }
288
289    /// CQ-B23: a second `get_or_fetch` for the same metadata_url within the
290    /// 1-h TTL must be served from cache — no HTTP call is made. Mirrors
291    /// `jwks_cache::tests::cache_hit_within_ttl` (the cache key here IS the
292    /// fetch URL, so pointing it at an unreachable address and still getting
293    /// `Ok` proves the fast path never attempted a real fetch).
294    #[tokio::test]
295    async fn cache_hit_within_ttl() {
296        let cache = DiscoveryCache::new();
297        // Use a URL that would fail if actually called (port 0 → connection refused).
298        let key = "http://127.0.0.1:0/discovery-unreachable";
299
300        // Insert an entry fresh right now.
301        insert_entry(&cache, key, Utc::now(), test_doc()).await;
302
303        let http = reqwest::Client::new();
304        let result = cache.get_or_fetch(&http, key).await;
305
306        // Must return the cached entry without making an HTTP call.
307        assert!(
308            result.is_ok(),
309            "expected cached result, got error: {result:?}"
310        );
311        assert_eq!(result.unwrap().issuer, test_doc().issuer);
312    }
313
314    #[tokio::test]
315    async fn stale_while_revalidate_on_fetch_err() {
316        let cache = DiscoveryCache::new();
317        let key = "http://127.0.0.1:0/discovery-unreachable";
318
319        // Insert an entry that is past 1-h TTL but within 24-h stale window.
320        let fetched_at = Utc::now() - CDuration::minutes(90);
321        insert_entry(&cache, key, fetched_at, test_doc()).await;
322
323        let http = reqwest::Client::new();
324        // The fetch will fail because the URL is unreachable.
325        let result = cache.get_or_fetch(&http, key).await;
326
327        // Should return cached (stale) document rather than propagating the error.
328        assert!(
329            result.is_ok(),
330            "expected stale document to be served, got: {result:?}"
331        );
332    }
333
334    #[tokio::test]
335    async fn cache_miss_beyond_stale_window_propagates_error() {
336        let cache = DiscoveryCache::new();
337        let key = "http://127.0.0.1:0/discovery-unreachable";
338
339        // Insert an entry that is past both the TTL AND the 24-h stale window.
340        let fetched_at = Utc::now() - CDuration::hours(25);
341        insert_entry(&cache, key, fetched_at, test_doc()).await;
342
343        let http = reqwest::Client::new();
344        let result = cache.get_or_fetch(&http, key).await;
345
346        assert!(
347            result.is_err(),
348            "expected fetch error to propagate once stale window has also elapsed: {result:?}"
349        );
350    }
