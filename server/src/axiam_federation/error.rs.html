1//! Error types for federation operations (OIDC and SAML).

3/// Errors that can occur during federation operations.
4#[derive(Debug, thiserror::Error)]
5pub enum FederationError {
6    #[error("Federation config not found: {0}")]
7    ConfigNotFound(String),
8
9    #[error("Federation config is disabled")]
10    ConfigDisabled,
11
12    #[error("Federation config is incomplete (missing required credentials)")]
13    ConfigIncomplete,
14
15    #[error("Protocol mismatch: {0}")]
16    ProtocolMismatch(String),
17
18    #[error("Invalid metadata URL: {0}")]
19    InvalidMetadataUrl(String),
20
21    #[error("OIDC discovery failed: {0}")]
22    DiscoveryFailed(String),
23
24    #[error("Token exchange failed: {0}")]
25    TokenExchangeFailed(String),
26
27    #[error("ID token validation failed: {0}")]
28    IdTokenValidationFailed(String),
29
30    #[error("SAML metadata fetch/parse failed: {0}")]
31    SamlMetadataFailed(String),
32
33    #[error("SAML response validation failed: {0}")]
34    SamlResponseFailed(String),
35
36    #[error("User provisioning failed: {0}")]
37    ProvisioningFailed(String),
38
39    // ------------------------------------------------------------------
40    // JWKS / signature verification errors (plan 04-02)
41    // ------------------------------------------------------------------
42    /// JWKS endpoint returned an error or was unreachable (and no valid
43    /// stale-while-revalidate cache entry exists).
44    #[error("JWKS fetch failed: {0}")]
45    JwksFetchFailed(String),
46
47    /// The JWT header contained a `kid` not found in the JWKS (after
48    /// rate-limited forced refetch).
49    #[error("Unknown key id (kid) in ID token — key not in JWKS")]
50    JwksKidUnknown,
51
52    /// JWT signature verification failed.
53    #[error("JWT signature invalid")]
54    JwtSignatureInvalid,
55
56    /// A required JWT claim was missing or did not match the expected value
57    /// (iss, aud, exp, nonce, etc.).
58    #[error("JWT claim rejected: {0}")]
59    JwtClaimRejected(String),
60
61    /// The JWT algorithm is not in the per-config allow-list, or is "none"
62    /// (always rejected at code level regardless of configuration).
63    #[error("Algorithm not allowed: {0}")]
64    AlgorithmNotAllowed(String),
65
66    // ------------------------------------------------------------------
67    // Crypto error (plan 04-02 secrets module)
68    // ------------------------------------------------------------------
69    #[error("Cryptography error: {0}")]
70    CryptoError(String),
71
72    #[error("Internal error: {0}")]
73    Internal(String),
74
75    // ------------------------------------------------------------------
76    // SAML XML signature / replay errors (plan 04-03)
77    // ------------------------------------------------------------------
78    /// The IdP signing certificate PEM is invalid (rejected at upload or at
79    /// verify time if a stored cert becomes corrupt).
80    #[error("Invalid IdP certificate: {0}")]
81    InvalidIdpCert(String),
82
83    /// The SAML XML signature is missing, structurally invalid, or does not
84    /// verify against the configured IdP certificate.
85    #[error("SAML signature invalid: {0}")]
86    SamlSignatureInvalid(String),
87
88    /// A SAML assertion with this ID has already been consumed by this tenant
89    /// (replay attack detected).
90    #[error("Assertion replay detected")]
91    AssertionReplay,
92}

94impl From<FederationError> for axiam_core::error::AxiamError {
95    fn from(err: FederationError) -> Self {
96        match err {
97            FederationError::ConfigNotFound(id) => axiam_core::error::AxiamError::NotFound {
98                entity: "federation_config".into(),
99                id,
100            },
101            FederationError::ConfigDisabled
102            | FederationError::ProtocolMismatch(_)
103            | FederationError::InvalidMetadataUrl(_) => axiam_core::error::AxiamError::Validation {
104                message: err.to_string(),
105            },
106            FederationError::IdTokenValidationFailed(reason)
107            | FederationError::SamlResponseFailed(reason) => {
108                axiam_core::error::AxiamError::AuthenticationFailed { reason }
109            }
110            // OIDC signature / claim errors → 401
111            FederationError::JwtSignatureInvalid
112            | FederationError::JwtClaimRejected(_)
113            | FederationError::AlgorithmNotAllowed(_)
114            | FederationError::JwksKidUnknown => {
115                axiam_core::error::AxiamError::AuthenticationFailed {
116                    reason: err.to_string(),
117                }
118            }
119            // SAML signature / replay errors → 401
120            FederationError::SamlSignatureInvalid(reason) => {
121                axiam_core::error::AxiamError::AuthenticationFailed { reason }
122            }
123            FederationError::AssertionReplay => {
124                axiam_core::error::AxiamError::AuthenticationFailed {
125                    reason: "assertion replay".into(),
126                }
127            }
128            // IdP cert validation error → admin-facing 400 (Validation)
129            FederationError::InvalidIdpCert(msg) => {
130                axiam_core::error::AxiamError::Validation { message: msg }
131            }
132            // CQ-B23: upstream-IdP failures (discovery fetch/parse, token
133            // exchange, JWKS fetch) are NOT our bug — the local server did
134            // nothing wrong, but the external IdP is unreachable, timed out,
135            // misconfigured, or returned a malformed/erroring response.
136            // Mapping these to the generic `Internal` (500) catch-all
137            // mislabels an upstream dependency failure as a server bug.
138            // `AxiamError` has no dedicated 502 Bad Gateway variant; the
139            // closest existing variant is `ServiceUnavailable` (503), which
140            // — like a 502 — signals "an upstream dependency failed" rather
141            // than "the server itself is broken", so it is used here instead
142            // of inventing a new status/variant.
143            FederationError::DiscoveryFailed(msg)
144            | FederationError::TokenExchangeFailed(msg)
145            | FederationError::JwksFetchFailed(msg) => {
146                axiam_core::error::AxiamError::ServiceUnavailable(msg)
147            }
148            other => axiam_core::error::AxiamError::Internal(other.to_string()),
149        }
150    }
151}
