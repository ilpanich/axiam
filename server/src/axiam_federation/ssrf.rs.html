1//! Shared SSRF guard — resolve-once-and-pin outbound fetch helper.
2//!
3//! Generalizes the byte-identical guard logic previously duplicated in
4//! `jwks_cache::is_private_jwks_ip`/`validate_jwks_url` and
5//! `axiam-api-rest::webhook::is_private_ip`/`resolve_and_validate_host`
6//! (SECHRD-02 / D-01a DRY) into one reusable module, and — critically —
7//! adds IP **pinning**, which neither prior guard did: both validated the
8//! resolved `IpAddr` and then let `reqwest` re-resolve DNS independently at
9//! send time, leaving a DNS-rebind TOCTOU window open between validation and
10//! connect (D-01c closes this).
11//!
12//! Use [`guarded_fetch`] for every outbound fetch to an admin/IdP-supplied
13//! URL (JWKS, OIDC discovery, OIDC token exchange, SAML metadata, webhook
14//! delivery). It:
15//!
16//! 1. Resolves the host (A + AAAA) fresh — no cross-request DNS caching
17//!    (D-01c).
18//! 2. Rejects the fetch if ANY resolved address is
19//!    loopback/private/link-local/ULA/unspecified (D-01a) — unless the
20//!    caller opted into the `allow_private` test seam (see below).
21//! 3. Pins the exact validated `IpAddr` into a fresh, single-use
22//!    `reqwest::Client` via `ClientBuilder::resolve()`, so the socket that
23//!    is actually opened is the one that was validated — not a second,
24//!    independently-resolved address (D-01c).
25//! 4. Disables `reqwest`'s automatic redirect following and instead
26//!    manually re-runs the FULL guard (resolve → validate → pin → send)
27//!    against the `Location` target, bounded to [`MAX_HOPS`] hops (D-01b).
28//!
29//! ## The `allow_private` test seam only applies to the first hop
30//!
31//! `allow_private` exists solely so integration tests can point a guarded
32//! fetch at a loopback mock server (mirrors the pre-existing
33//! `JwksCache::new_allow_private_networks` seam). It is honored **only for
34//! the very first hop** of [`guarded_fetch`] — every redirect hop after that
35//! is always validated with the strict (production) check, regardless of
36//! `allow_private`. A `Location` header is attacker-influenced response
37//! data, not the admin-configured URL the caller opted to trust; holding it
38//! to the same relaxed standard as the test seam would silently defeat the
39//! redirect-bypass defense this module exists to provide (D-01b: "re-run
40//! the full SSRF guard against the redirect target").
41
42use std::net::{IpAddr, SocketAddr};
43use std::time::Duration;
44
45/// Maximum number of redirect hops [`guarded_fetch`] will follow before
46/// giving up. Each hop re-runs the full guard against the `Location` target.
47const MAX_HOPS: u8 = 3;
48
49/// Errors produced by the shared SSRF guard.
50#[derive(Debug, thiserror::Error)]
51pub enum SsrfError {
52    #[error("invalid URL")]
53    InvalidUrl,
54    #[error("failed to resolve host")]
55    ResolveFailed,
56    #[error("SSRF blocked: resolved IP is private/loopback/link-local/unspecified")]
57    Blocked,
58    #[error("failed to build HTTP client")]
59    ClientBuildFailed,
60    #[error("HTTP request failed: {0}")]
61    RequestFailed(String),
62    #[error("too many redirects")]
63    TooManyRedirects,
64    #[error("SSRF blocked: non-HTTPS scheme not permitted for IdP fetches")]
65    InsecureScheme,
66    #[error("SSRF blocked: response body exceeds the {0}-byte cap")]
67    ResponseTooLarge(usize),
68}
69
70/// Maximum acceptable `Content-Length` for a guarded IdP response (SEC-069).
71/// Discovery/metadata/token/JWKS documents are small; a multi-GB body is a
72/// memory-exhaustion DoS vector. JWKS additionally applies its own 512 KiB
73/// read cap downstream; this is the coarse first line of defence for all four
74/// federation fetch types.
75const MAX_RESPONSE_BYTES: usize = 5 * 1024 * 1024;
76
77/// Returns `true` for IP addresses that must never be contacted from a
78/// server-side outbound fetch to an admin/IdP-supplied URL.
79///
80/// Covers: RFC1918 private ranges, loopback (127/8 and ::1), link-local
81/// (169.254/16 and fe80::/10), broadcast (255.255.255.255), and unspecified
82/// (0.0.0.0 / ::). Lifted byte-identical from the two pre-existing
83/// duplicate copies (`jwks_cache::is_private_jwks_ip`,
84/// `webhook::is_private_ip`) — this is the D-01a dedup target.
85pub fn is_disallowed_ip(ip: IpAddr) -> bool {
86    match ip {
87        IpAddr::V4(v4) => {
88            v4.is_loopback()
89                || v4.is_private()
90                || v4.is_link_local()
91                || v4.is_broadcast()
92                || v4.is_unspecified()
93        }
94        IpAddr::V6(v6) => {
95            v6.is_loopback()
96                || v6.is_unspecified()
97                || (v6.segments()[0] & 0xffc0 == 0xfe80) // link-local fe80::/10
98                || (v6.segments()[0] & 0xfe00 == 0xfc00) // unique-local fc00::/7
99        }
100    }
101}

103/// Resolve `host:port` (A + AAAA), reject if ANY resolved address is
104/// disallowed, and return ONE validated address to pin into the connection.
105///
106/// `allow_private`, when `true`, skips the disallow check entirely. This
107/// exists solely to preserve the pre-existing loopback mock-server
108/// integration-test seam (mirrors `JwksCache::new_allow_private_networks`);
109/// it MUST be `false` in production code paths.
110pub async fn resolve_and_pick(
111    host: &str,
112    port: u16,
113    allow_private: bool,
114) -> Result<IpAddr, SsrfError> {
115    let addrs: Vec<IpAddr> = tokio::net::lookup_host((host, port))
116        .await
117        .map_err(|_| SsrfError::ResolveFailed)?
118        .map(|a| a.ip())
119        .collect();
120
121    if addrs.is_empty() {
122        return Err(SsrfError::ResolveFailed);
123    }
124
125    if !allow_private && addrs.iter().any(|ip| is_disallowed_ip(*ip)) {
126        return Err(SsrfError::Blocked);
127    }
128
129    Ok(addrs[0])
130}

132/// Build a fresh, single-use client pinned to `ip` for `host`.
133///
134/// No connection pooling/caching across requests — a new `Client` is built
135/// per guarded fetch (D-01c: "fresh per request", so a rebind between two
136/// calls minutes apart can never reuse a stale pinned connection). Automatic
137/// redirect following is disabled (D-01b) — [`guarded_fetch`] re-validates
138/// and re-issues each hop explicitly instead of trusting `reqwest` to follow
139/// a `Location` header unchecked.
140pub fn pinned_client(host: &str, ip: IpAddr, port: u16) -> Result<reqwest::Client, SsrfError> {
141    reqwest::Client::builder()
142        .resolve(host, SocketAddr::new(ip, port))
143        .redirect(reqwest::redirect::Policy::none())
144        .timeout(Duration::from_secs(10))
145        .build()
146        .map_err(|_| SsrfError::ClientBuildFailed)
147}

149/// Orchestrates resolve + pin + fetch + bounded manual redirect
150/// re-validation (D-01b).
151///
152/// `allow_private` is honored only for the first hop — see the module docs
153/// for why redirect targets are always strictly validated regardless of the
154/// caller's test-seam opt-in.
155///
156/// `build_request` builds the actual request (e.g. `|c, u| c.get(u)` or
157/// `|c, u| c.post(u).form(&params)`) against the freshly pinned client for
158/// the current hop's URL.
159pub async fn guarded_fetch(
160    url: &str,
161    allow_private: bool,
162    build_request: impl Fn(&reqwest::Client, &str) -> reqwest::RequestBuilder,
163) -> Result<reqwest::Response, SsrfError> {
164    let mut current = url.to_string();
165
166    for hop in 0..MAX_HOPS {
167        let parsed = url::Url::parse(&current).map_err(|_| SsrfError::InvalidUrl)?;
168        let host = parsed.host_str().ok_or(SsrfError::InvalidUrl)?.to_string();
169        let port = parsed.port_or_known_default().unwrap_or(443);
170
171        // Only the first hop honors the caller's test seam; every redirect
172        // hop thereafter is always strictly validated (module docs above).
173        let hop_allow_private = allow_private && hop == 0;
174
175        // SEC-069: enforce HTTPS for every hop. A plaintext `http://` IdP
176        // endpoint (admin-misconfigured, or an attacker-supplied redirect)
177        // would carry the decrypted client_secret / bearer material in the
178        // clear. `http` is tolerated only behind the private-network test seam
179        // (loopback/dev), exactly like the address checks.
180        if parsed.scheme() != "https" && !hop_allow_private {
181            return Err(SsrfError::InsecureScheme);
182        }
183
184        let ip = resolve_and_pick(&host, port, hop_allow_private).await?;
185        let client = pinned_client(&host, ip, port)?;
186
187        let resp = build_request(&client, &current)
188            .send()
189            .await
190            .map_err(|e| SsrfError::RequestFailed(e.to_string()))?;
191
192        if resp.status().is_redirection() {
193            let location = resp
194                .headers()
195                .get("location")
196                .and_then(|v| v.to_str().ok())
197                .ok_or(SsrfError::InvalidUrl)?
198                .to_string();
199            current = parsed
200                .join(&location)
201                .map_err(|_| SsrfError::InvalidUrl)?
202                .to_string();
203            continue;
204        }
205
206        // SEC-069: reject an over-large advertised body before the caller
207        // buffers it. This is the coarse Content-Length gate; body readers that
208        // need a hard guarantee against a lying/chunked response still apply
209        // their own streaming cap (JWKS: 512 KiB).
210        if let Some(len) = resp.content_length()
211            && len > MAX_RESPONSE_BYTES as u64
212        {
213            return Err(SsrfError::ResponseTooLarge(MAX_RESPONSE_BYTES));
214        }
215
216        return Ok(resp);
217    }
218
219    Err(SsrfError::TooManyRedirects)
220}

222/// Read a response body with a hard streaming cap, aborting as soon as `cap`
223/// is exceeded — WITHOUT buffering the rest of the body first (CQ-B23).
224///
225/// This replaces the previous "buffer the whole body via `.bytes()`, then
226/// check `.len()` against the cap" pattern used by discovery/token-exchange
227/// reads. That pattern still let a malicious or misconfigured endpoint force
228/// full in-memory buffering of an arbitrarily large response before the
229/// existing size check ever ran (the coarse `Content-Length`-based check in
230/// [`guarded_fetch`] above only catches endpoints that both send the header
231/// AND tell the truth about it — a chunked, no-`Content-Length` response
232/// bypasses it entirely). Reading chunk-by-chunk with a running byte count
233/// bounds peak memory use to ~`cap` bytes regardless.
234///
235/// Uses `reqwest::Response::chunk()` (always available, unlike
236/// `bytes_stream()` which needs the `stream` cargo feature this workspace
237/// does not enable) so no new dependency/feature is required.
238pub async fn read_capped_body(
239    mut response: reqwest::Response,
240    cap: usize,
241) -> Result<Vec<u8>, SsrfError> {
242    let mut buf = Vec::with_capacity(cap.min(64 * 1024));
243    while let Some(chunk) = response
244        .chunk()
245        .await
246        .map_err(|e| SsrfError::RequestFailed(e.to_string()))?
247    {
248        buf.extend_from_slice(&chunk);
249        if buf.len() > cap {
250            return Err(SsrfError::ResponseTooLarge(cap));
251        }
252    }
253    Ok(buf)
254}

401    /// no-`Content-Length` response so the ONLY thing that can catch it is
402    /// the streaming running-byte-count check, not the coarse
403    /// `Content-Length` gate in [`guarded_fetch`] (which this test bypasses
404    /// by calling `read_capped_body` directly on a plain `reqwest` response).

406    async fn read_capped_body_rejects_body_over_cap() {
407        use tokio::io::{AsyncReadExt, AsyncWriteExt};
408        use tokio::net::TcpListener;
409
410        const CAP: usize = 16;
411
412        let listener = TcpListener::bind("127.0.0.1:0")
413            .await
414            .expect("bind mock server");
415        let addr = listener.local_addr().expect("local_addr");
416
417        tokio::spawn(async move {
418            if let Ok((mut stream, _)) = listener.accept().await {
419                let mut buf = [0u8; 1024];
420                let _ = stream.read(&mut buf).await;
421                // Chunked transfer-encoding, no Content-Length: a body far

425                    .write_all(

439        let response = reqwest::Client::new().get(&url).send().await.unwrap();
