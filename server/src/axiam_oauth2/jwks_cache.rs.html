1//! In-process JWKS cache with RFC 7232 conditional-GET (ETag) support (B3).
2//!
3//! ## Baseline finding this module addresses
4//!
5//! Before this module existed, `GET /oauth2/jwks` called
6//! [`crate::oidc::build_jwks`] fresh on every single request: PEM strip →
7//! base64 decode → SHA-256(kid) → JSON-serialize, with zero HTTP caching
8//! headers. The benchmark's low 0.062 cpu·ms/req for that endpoint is
9//! simply because that computation is cheap (a 44-byte Ed25519 SPKI, no
10//! I/O) — it was never evidence of caching. This module adds a real
11//! in-process cache plus `ETag`/`If-None-Match`/304 support so repeat
12//! requests (the overwhelming majority in practice — relying parties are
13//! expected to poll `jwks_uri` on an interval, not per-request) skip the
14//! recompute and the response body entirely.
15//!
16//! ## Not the same thing as `axiam_federation::jwks_cache::JwksCache`
17//!
18//! That cache stores REMOTE identity providers' JWKS documents fetched
19//! during OIDC federation login (keyed by `(tenant_id, federation_config_id)`,
20//! TTL-based, with stale-while-revalidate). This module caches AXIAM's OWN
21//! signing-key JWKS — the document AXIAM *serves* at `GET /oauth2/jwks` —
22//! and invalidates itself based on the input key material rather than a
23//! wall-clock TTL. The two are unrelated and live in different crates.
24//!
25//! ## Known limitations (documented per B3, not fixed here)
26//!
27//! - **No key-rotation mechanism exists today.** `AuthConfig` loads
28//!   `jwt_public_key_pem` once at process startup from config/env and it
29//!   never changes for the life of the process — there is no reachable
30//!   code path that swaps it at runtime. In practice this cache is
31//!   therefore built exactly once per process. It is still keyed by a hash
32//!   of the input PEM (rather than being unconditionally "build once and
33//!   never touch again") so that IF a future rotation mechanism starts
34//!   swapping the PEM string held by the caller between calls, this cache
35//!   picks the change up automatically on the very next [`JwksCache::get`]
36//!   call — no code in that future rotation path needs to know this cache
37//!   exists or call anything special. [`JwksCache::invalidate`] is also
38//!   provided for a caller that wants to force an eager rebuild instead of
39//!   waiting for the input hash to differ.
40//! - **The endpoint is not actually per-tenant.** `GET /oauth2/jwks` serves
41//!   ONE global key set sourced from the server-wide `AuthConfig`, despite
42//!   living under the tenant-aware API surface. Every tenant is served the
43//!   same signing key today; this cache mirrors that reality (it is a
44//!   single-slot cache, not a per-tenant map).
45

61/// Configuration for the `GET /oauth2/jwks` HTTP caching headers (B3).
62///
63/// Mirrors the `#[serde(default)]` style used by `axiam_auth::config::AuthConfig`
64/// so it composes into the same `config`-crate-driven env var plumbing.
65///
66/// The field is named `max_age_secs` in Rust for readability at call sites
67/// (`config.max_age_secs`), but is deserialized under the key
68/// `jwks_cache_max_age_secs` so that, nested one level under the server's
69/// `oauth2` config section, it is reachable via the env var
70/// `AXIAM__OAUTH2__JWKS_CACHE_MAX_AGE_SECS`.
71#[derive(Debug, Clone, Deserialize)]
72#[serde(default)]
73pub struct JwksCacheConfig {
74    /// `max-age` value (seconds) advertised in the `Cache-Control` header on
75    /// `GET /oauth2/jwks` responses. Default: 300 (5 minutes). Override via
76    /// `AXIAM__OAUTH2__JWKS_CACHE_MAX_AGE_SECS`.
77    #[serde(rename = "jwks_cache_max_age_secs")]
78    pub max_age_secs: u64,
79}
80
81impl Default for JwksCacheConfig {
82    fn default() -> Self {
83        Self {
84            max_age_secs: default_max_age_secs(),
85        }
86    }
87}
88
89impl JwksCacheConfig {
90    /// Render the `Cache-Control` header value for this config, e.g.
91    /// `"public, max-age=300"`.
92    pub fn cache_control_header(&self) -> String {
93        format!("public, max-age={}", self.max_age_secs)
94    }
95}

101/// A cached, pre-serialized JWKS response plus the hash of the PEM it was
102/// built from (the self-invalidation key).
103struct CacheEntry {
104    /// SHA-256 of the source PEM this entry was built from. When a `get()`
105    /// call is made with a PEM that hashes differently, the entry is stale
106    /// and gets rebuilt.
107    pem_hash: [u8; 32],
108    /// Pre-serialized JSON body of the `JwksDocument`, cached verbatim so
109    /// repeat requests skip `serde_json` re-serialization too.
110    body: String,
111    /// Strong, quoted ETag: `"` + hex(SHA-256(body))[..16 bytes as hex] + `"`.
112    etag: String,
113}
114
115/// Outcome of a [`JwksCache::get`] call.
116#[derive(Debug, Clone, PartialEq, Eq)]
117pub enum JwksCacheResponse {
118    /// No conditional match (or no `If-None-Match` sent): the caller should
119    /// respond `200 OK` with this body and this `ETag`.
120    Fresh { body: String, etag: String },
121    /// The client's `If-None-Match` matched the current ETag: the caller
122    /// should respond `304 Not Modified` with this `ETag` and an empty body.
123    NotModified { etag: String },
124}
125
126/// Thread-safe, single-slot, in-process cache for AXIAM's own JWKS
127/// response (see module docs for the distinction from
128/// `axiam_federation::jwks_cache::JwksCache`).
129///
130/// Self-invalidates when the input PEM changes (compared by SHA-256 hash,
131/// not by string equality, to keep the stored key small and avoid retaining
132/// PEM copies longer than necessary). [`JwksCache::invalidate`] forces an
133/// eager rebuild on the next `get()` regardless of whether the PEM changed.
134pub struct JwksCache {
135    inner: RwLock<Option<CacheEntry>>,
136}
137
138impl Default for JwksCache {
139    fn default() -> Self {
140        Self::new()
141    }
142}
143
144impl JwksCache {
145    pub fn new() -> Self {
146        Self {
147            inner: RwLock::new(None),
148        }
149    }
150
151    /// Drop the cached entry unconditionally. The next [`JwksCache::get`]
152    /// call will rebuild from scratch (still hitting the fast path again on
153    /// every subsequent call, as long as the PEM does not change).
154    pub fn invalidate(&self) {
155        let mut guard = self.inner.write().unwrap_or_else(|e| e.into_inner());
156        *guard = None;
157    }
158
159    /// Get the (possibly cached) JWKS response for `pem`, honoring an
160    /// optional `If-None-Match` request header value.
161    ///
162    /// - Cache hit (PEM hash unchanged): served from the cached, already
163    ///   serialized body — no re-parsing, no re-serialization.
164    /// - Cache miss (first call, PEM changed, or after [`JwksCache::invalidate`]):
165    ///   rebuilds via [`build_jwks`], serializes once, computes the ETag,
166    ///   and stores the new entry. On a build error (malformed PEM), the
167    ///   error is returned and the existing cache entry (if any) is left
168    ///   untouched — a bad key never evicts a previously good one.
169    ///
170    /// `if_none_match` is compared per RFC 7232 §3.2: a `*` always matches;
171    /// a comma-separated list of (optionally `W/`-prefixed) entity-tags
172    /// matches if any entry equals the current strong ETag once its `W/`
173    /// prefix (if any) is stripped.
174    pub fn get(&self, pem: &str, if_none_match: Option<&str>) -> Result<JwksCacheResponse, String> {
175        let pem_hash = sha256(pem.as_bytes());
176
177        // Fast path: an existing entry built from the same PEM.
178        {
179            let guard = self.inner.read().unwrap_or_else(|e| e.into_inner());
180            if let Some(entry) = guard.as_ref()
181                && entry.pem_hash == pem_hash
182            {
183                return Ok(respond(entry, if_none_match));
184            }
185        }
186
187        // Slow path: (re)build. Do NOT touch the cache until this succeeds,
188        // so a malformed key never evicts a previously-good cached entry.
189        let doc = build_jwks(pem)?;
190        let body = serde_json::to_string(&doc).map_err(|e| format!("JWKS serialize: {e}"))?;
191        let etag = format!("\"{}\"", hex::encode(&sha256(body.as_bytes())[..16]));
192
193        let entry = CacheEntry {
194            pem_hash,
195            body,
196            etag,
197        };
198        let response = respond(&entry, if_none_match);
199
200        let mut guard = self.inner.write().unwrap_or_else(|e| e.into_inner());
201        *guard = Some(entry);
202
203        Ok(response)
204    }
205}
206
207fn respond(entry: &CacheEntry, if_none_match: Option<&str>) -> JwksCacheResponse {
208    if let Some(header) = if_none_match
209        && if_none_match_matches(header, &entry.etag)
210    {
211        return JwksCacheResponse::NotModified {
212            etag: entry.etag.clone(),
213        };
214    }
215    JwksCacheResponse::Fresh {
216        body: entry.body.clone(),
217        etag: entry.etag.clone(),
218    }
219}
220
221/// RFC 7232 §3.2 `If-None-Match` comparison (weak comparison, as required
222/// for use with `GET`): `*` matches unconditionally; otherwise the header
223/// is a comma-separated list of entity-tags, each optionally prefixed with
224/// `W/` for a weak validator. A match is found if any listed tag — with its
225/// `W/` prefix stripped, if present — equals `current_etag` exactly.
226fn if_none_match_matches(header: &str, current_etag: &str) -> bool {
227    let header = header.trim();
228    if header == "*" {
229        return true;
230    }
231    header.split(',').any(|raw| {
232        let candidate = raw.trim();
233        let stripped = candidate.strip_prefix("W/").unwrap_or(candidate);
234        stripped == current_etag
235    })
236}
237
238fn sha256(bytes: &[u8]) -> [u8; 32] {
239    let mut hasher = Sha256::new();
240    hasher.update(bytes);
241    hasher.finalize().into()
242}

277    #[test]
278    fn fresh_response_has_body_and_etag_with_no_if_none_match() {
279        let cache = JwksCache::new();
280        let resp = cache.get(PEM_A, None).unwrap();
281        match resp {
282            JwksCacheResponse::Fresh { body, etag } => {
283                assert!(body.contains("\"kty\":\"OKP\""));
284                assert!(etag.starts_with('"') && etag.ends_with('"'));
285                // 16 bytes of SHA-256 hex-encoded = 32 hex chars, plus 2 quotes.
286                assert_eq!(etag.len(), 34);
287            }
288            other => panic!("expected Fresh, got {other:?}"),
289        }

342    #[test]
343    fn etag_changes_on_key_rotation_and_stale_etag_no_longer_304s() {
344        let cache = JwksCache::new();
345        let etag_a = match cache.get(PEM_A, None).unwrap() {
346            JwksCacheResponse::Fresh { etag, .. } => etag,
347            _ => unreachable!(),
348        };
349
350        // Simulate rotation: caller now passes a different PEM.
351        let etag_b = match cache.get(PEM_B, None).unwrap() {
352            JwksCacheResponse::Fresh { etag, .. } => etag,
353            _ => unreachable!(),
354        };
355
356        assert_ne!(
357            etag_a, etag_b,
358            "ETag must change when the key material changes"
359        );
360
361        // The pre-rotation ETag must no longer produce a 304 against the
362        // now-current (post-rotation) cache state.
363        let resp = cache.get(PEM_B, Some(&etag_a)).unwrap();
364        assert!(matches!(resp, JwksCacheResponse::Fresh { .. }));
365
366        // But the fresh, post-rotation ETag still 304s.
367        let resp = cache.get(PEM_B, Some(&etag_b)).unwrap();
368        assert_eq!(resp, JwksCacheResponse::NotModified { etag: etag_b });
369    }

392    }
