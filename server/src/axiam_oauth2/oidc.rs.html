31/// Build a fully-populated OIDC discovery document for the given issuer URL.
32pub fn build_discovery_document(issuer: &str) -> OidcDiscoveryDocument {
33    let issuer = issuer.trim_end_matches('/');
34    OidcDiscoveryDocument {
35        issuer: issuer.to_string(),
36        authorization_endpoint: format!("{issuer}/oauth2/authorize"),
37        token_endpoint: format!("{issuer}/oauth2/token"),
38        userinfo_endpoint: format!("{issuer}/oauth2/userinfo"),
39        jwks_uri: format!("{issuer}/oauth2/jwks"),
40        revocation_endpoint: format!("{issuer}/oauth2/revoke"),
41        introspection_endpoint: format!("{issuer}/oauth2/introspect"),
42        response_types_supported: vec!["code".into()],
43        subject_types_supported: vec!["public".into()],
44        id_token_signing_alg_values_supported: vec!["EdDSA".into()],
45        scopes_supported: vec!["openid".into(), "profile".into(), "email".into()],
46        token_endpoint_auth_methods_supported: vec!["client_secret_post".into()],
47        claims_supported: vec![
48            "sub".into(),
49            "iss".into(),
50            "aud".into(),
51            "exp".into(),
52            "iat".into(),
53            "nonce".into(),
54            "email".into(),
55            "preferred_username".into(),
56            "tenant_id".into(),
57            "org_id".into(),
58        ],
59        grant_types_supported: vec![
60            "authorization_code".into(),
61            "client_credentials".into(),
62            "refresh_token".into(),
63        ],
64    }
65}

83/// JSON Web Key Set document.

87}

89/// Build a JWKS document from an Ed25519 public key in PEM format.
90///
91/// The PEM must contain a SubjectPublicKeyInfo structure (44 bytes
92/// when DER-decoded: 12-byte OID header + 32-byte raw Ed25519 key).
93/// The `kid` is derived deterministically as the first 16 hex
94/// characters (64 bits) of the SHA-256 hash of the raw public key bytes.
95pub fn build_jwks(public_key_pem: &str) -> Result<JwksDocument, String> {
96    // Strip PEM headers, trim whitespace (handles CRLF / trailing
97    // spaces from env vars or Windows-formatted PEMs), and decode.
98    let b64: String = public_key_pem
99        .lines()
100        .map(str::trim)
101        .filter(|l| !l.starts_with("-----"))
102        .collect();
103    let der = STANDARD
104        .decode(&b64)
105        .map_err(|e| format!("PEM decode: {e}"))?;
106
107    // Ed25519 SubjectPublicKeyInfo is exactly 44 bytes:
108    // 12-byte ASN.1/OID header + 32-byte raw public key.
109    if der.len() != 44 {
110        return Err(format!("expected 44-byte Ed25519 SPKI, got {}", der.len()));
111    }
112    let raw_key = &der[12..44];
113
114    // Base64url-encode the raw key bytes for the JWK `x` parameter.
115    let x = URL_SAFE_NO_PAD.encode(raw_key);
116
117    // Deterministic kid: first 16 hex chars of SHA-256(raw_key).
118    let kid = {
119        let mut h = Sha256::new();
120        h.update(raw_key);
121        hex::encode(h.finalize())[..16].to_string()
122    };
123
124    Ok(JwksDocument {
125        keys: vec![Jwk {
126            kty: "OKP".into(),
127            crv: "Ed25519".into(),
128            x,
129            kid,
130            use_: "sig".into(),
131            alg: "EdDSA".into(),
132        }],
133    })
134}

156#[cfg(test)]
157mod tests {
158    use super::*;
159
160    #[test]
161    fn discovery_document_has_required_fields() {
162        let doc = build_discovery_document("https://auth.example.com");
163        assert_eq!(doc.issuer, "https://auth.example.com");
164        assert_eq!(
165            doc.authorization_endpoint,
166            "https://auth.example.com/oauth2/authorize"
167        );
168        assert!(doc.response_types_supported.contains(&"code".into()));
169        assert!(doc.scopes_supported.contains(&"openid".into()));
170    }
171
172    #[test]
173    fn jwks_parses_ed25519_pem() {
174        let pem = "\
175-----BEGIN PUBLIC KEY-----
176MCowBQYDK2VwAyEAcweT2rPwpUxadO56wIhW1XBoMF63aWOE2UMAVsRudhs=
177-----END PUBLIC KEY-----";
178        let jwks = build_jwks(pem).unwrap();
179        assert_eq!(jwks.keys.len(), 1);
180        assert_eq!(jwks.keys[0].kty, "OKP");
181        assert_eq!(jwks.keys[0].crv, "Ed25519");
182        assert_eq!(jwks.keys[0].alg, "EdDSA");
183        assert_eq!(jwks.keys[0].use_, "sig");
184        assert!(!jwks.keys[0].x.is_empty());
185        assert!(!jwks.keys[0].kid.is_empty());
186    }
187
188    #[test]
189    fn jwk_kid_is_deterministic() {
190        let pem = "\
191-----BEGIN PUBLIC KEY-----
192MCowBQYDK2VwAyEAcweT2rPwpUxadO56wIhW1XBoMF63aWOE2UMAVsRudhs=
193-----END PUBLIC KEY-----";
194        let jwks1 = build_jwks(pem).unwrap();
195        let jwks2 = build_jwks(pem).unwrap();
196        assert_eq!(jwks1.keys[0].kid, jwks2.keys[0].kid);
197    }
