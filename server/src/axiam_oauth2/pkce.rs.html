1//! PKCE (Proof Key for Code Exchange) verification per RFC 7636.
2
3use base64::Engine;
4use base64::engine::general_purpose::URL_SAFE_NO_PAD;
5use sha2::{Digest, Sha256};
6use subtle::ConstantTimeEq;
7
8/// Verify a PKCE code verifier against a stored code challenge (S256).
9///
10/// Validates the verifier format per RFC 7636 §4.1 (43–128 chars from
11/// `[A-Z] / [a-z] / [0-9] / "-" / "." / "_" / "~"`) and uses
12/// constant-time comparison to prevent timing side-channels.
13pub fn verify_pkce(code_verifier: &str, code_challenge: &str) -> bool {
14    // RFC 7636 §4.1: code_verifier must be 43–128 characters
15    if !(43..=128).contains(&code_verifier.len()) {
16        return false;
17    }
18
19    // RFC 7636 §4.1: restricted character set
20    let valid_chars = code_verifier
21        .bytes()
22        .all(|b| b.is_ascii_alphanumeric() || b == b'-' || b == b'.' || b == b'_' || b == b'~');
23    if !valid_chars {
24        return false;
25    }
26
27    let mut hasher = Sha256::new();
28    hasher.update(code_verifier.as_bytes());
29    let computed = URL_SAFE_NO_PAD.encode(hasher.finalize());
30
31    // Constant-time comparison to prevent timing attacks
32    computed.as_bytes().ct_eq(code_challenge.as_bytes()).into()
33}

35#[cfg(test)]
36mod tests {
37    use super::*;
38
39    #[test]
40    fn pkce_s256_rfc7636_example() {
41        // RFC 7636 Appendix B test vector
42        let verifier = "dBjftJeZ4CVP-mB92K27uhbUJU1p1r_wW1gFWFOEjXk";
43        let challenge = "E9Melhoa2OwvFrEMTJguCHaoeK1t8URWbuGJSstw-cM";
44        assert!(verify_pkce(verifier, challenge));
45    }
46
47    #[test]
48    fn pkce_s256_mismatch() {
49        // Use a verifier that meets the 43-char minimum
50        let verifier = "wrong-verifier-padded-to-forty-three-chars1";
51        assert!(!verify_pkce(
52            verifier,
53            "E9Melhoa2OwvFrEMTJguCHaoeK1t8URWbuGJSstw-cM"
54        ));
55    }
56
57    #[test]
58    fn pkce_s256_roundtrip() {
59        let verifier = "my-custom-code-verifier-12345-padded-to-min";
60        let mut hasher = Sha256::new();
61        hasher.update(verifier.as_bytes());
62        let challenge = URL_SAFE_NO_PAD.encode(hasher.finalize());
63        assert!(verify_pkce(verifier, &challenge));
64    }
65
66    #[test]
67    fn pkce_rejects_short_verifier() {
68        assert!(!verify_pkce("too-short", "anything"));
69    }
70
71    #[test]
72    fn pkce_rejects_long_verifier() {
73        let long = "a".repeat(129);
74        assert!(!verify_pkce(&long, "anything"));
75    }
76
77    #[test]
78    fn pkce_rejects_invalid_chars() {
79        // Space is not in the allowed set
80        let verifier = "has spaces in it padded to forty three char";
81        assert!(!verify_pkce(verifier, "anything"));
82    }
