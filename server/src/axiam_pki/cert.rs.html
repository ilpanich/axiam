20/// Hard cap for leaf certificate validity: 825 days (~27 months).
21///
22/// Aligns with CA/Browser Forum Baseline Requirements and Apple/Mozilla
23/// root program policies. Internal PKI may allow up to this limit;
24/// tenants can configure a lower per-tenant maximum via the
25/// `max_certificate_validity_days` key in their metadata.
26pub const MAX_LEAF_CERT_VALIDITY_DAYS: u32 = 825;
27
28/// Default leaf certificate validity when no tenant override is set: 365 days.
29pub const DEFAULT_LEAF_CERT_VALIDITY_DAYS: u32 = 365;
30
31/// Service for tenant-level certificate operations.
32#[derive(Clone)]
33pub struct CertService<CA, CR> {
34    ca_repo: CA,
35    cert_repo: CR,
36    config: PkiConfig,
37    /// Shared bounding semaphore for CPU-bound crypto (CQ-B02).
38    crypto_semaphore: Arc<Semaphore>,
39}
40

56    /// Generate a new certificate signed by an organization CA.
57    ///
58    /// `org_id` is required to look up the CA certificate and its encrypted
59    /// private key for signing.
60    ///
61    /// `max_validity_days` is the tenant-level cap (from tenant metadata).
62    /// Pass `None` to use the default ([`DEFAULT_LEAF_CERT_VALIDITY_DAYS`]).
63    /// The hard cap ([`MAX_LEAF_CERT_VALIDITY_DAYS`], 825 days) is always
64    /// enforced per CA/Browser Forum Baseline Requirements.
65    pub async fn generate(
66        &self,
67        org_id: Uuid,
68        input: CreateCertificate,
69        max_validity_days: Option<u32>,
70    ) -> AxiamResult<GeneratedCertificate> {
71        // Enforce validity_days bounds: > 0 and <= tenant/hard cap
72        let effective_max = max_validity_days
73            .unwrap_or(DEFAULT_LEAF_CERT_VALIDITY_DAYS)
74            .min(MAX_LEAF_CERT_VALIDITY_DAYS);
75        if input.validity_days == 0 || input.validity_days > effective_max {
76            return Err(AxiamError::Validation {
77                message: format!(
78                    "validity_days must be between 1 and {effective_max} \
79                     (CA/Browser Forum BR hard cap: {MAX_LEAF_CERT_VALIDITY_DAYS} days)"
80                ),
81            });
82        }
83
84        // Fetch the CA cert to get the encrypted private key.
85        let ca_cert = self.ca_repo.get_by_id(org_id, input.issuer_ca_id).await?;
86
87        if ca_cert.status != CertificateStatus::Active {
88            return Err(AxiamError::Certificate(
89                "CA certificate is not active".into(),
90            ));
91        }
92
93        // Validate CA certificate validity window
94        let now = Utc::now();
95        if now < ca_cert.not_before || now > ca_cert.not_after {
96            return Err(AxiamError::Certificate(
97                "CA certificate is expired or not yet valid".into(),
98            ));
99        }
100
101        let encrypted_key = ca_cert.encrypted_private_key.ok_or_else(|| {
102            AxiamError::Certificate("CA certificate has no stored private key".into())
103        })?;
104
105        // Decrypt the CA private key — encryption key must be present (SEC-012).
106        let enc_key = self.config.encryption_key.ok_or_else(|| {
107            AxiamError::Internal(
108                "AXIAM__PKI__ENCRYPTION_KEY not set — CA/cert key encryption unavailable".into(),
109            )
110        })?;
111        let mut ca_private_key_pem = decrypt_ca_key_pem(&encrypted_key, &enc_key)?;
112
113        let not_before = now;
114        let requested_not_after = now
115            .checked_add_signed(Duration::days(i64::from(input.validity_days)))
116            .ok_or_else(|| AxiamError::Validation {
117                message: "validity_days produces a date out of range".into(),
118            })?;
119        // Cap leaf validity to CA validity window
120        let not_after = std::cmp::min(requested_not_after, ca_cert.not_after);
121
122        // CPU-bound: key generation + certificate signing run in spawn_blocking behind semaphore (CQ-B02).
123        let _permit = self
124            .crypto_semaphore
125            .acquire()
126            .await
127            .map_err(|_| AxiamError::Internal("crypto semaphore closed".into()))?;
128
129        let ca_cert_pem = ca_cert.public_cert_pem.clone();
130        let ee_subject = input.subject.clone();
131        let key_algorithm = input.key_algorithm.clone();
132        let not_before_ts = not_before.timestamp();
133        let not_after_ts = not_after.timestamp();
134
135        let (private_key_pem, public_cert_pem, fingerprint) =
136            tokio::task::spawn_blocking(move || -> AxiamResult<(String, String, String)> {
137                let ca_key_pair = KeyPair::from_pem(&ca_private_key_pem)
138                    .map_err(|e| AxiamError::Certificate(format!("invalid CA private key: {e}")))?;
139                // Scrub the decrypted CA private-key PEM from memory as soon as
140                // the KeyPair is parsed — it is not needed past this point and
141                // must not linger in the heap buffer (defense-in-depth).
142                ca_private_key_pem.zeroize();
143
144                // Reconstruct the signing CA issuer from its real, stored certificate
145                // PEM — NOT from the (mutable) `subject` field — so the issuer DN
146                // embedded in every leaf cert can never drift from the CA's actual
147                // Subject DN (QUAL-05/D-08, T-29-11). rcgen 0.14 moved
148                // `from_ca_cert_pem` onto `Issuer`, which now owns the signing key and
149                // is passed directly to `signed_by`.
150                let ca_issuer =
151                    Issuer::from_ca_cert_pem(&ca_cert_pem, ca_key_pair).map_err(|e| {
152                        AxiamError::Certificate(format!("invalid CA certificate PEM: {e}"))
153                    })?;
154
155                // Generate end-entity key pair.
156                let ee_key_pair = generate_keypair(&key_algorithm)?;
157                let private_key_pem = ee_key_pair.serialize_pem();
158
159                // Build end-entity certificate request.
160                let mut ee_params = CertificateParams::new(Vec::<String>::new())
161                    .map_err(|e| AxiamError::Certificate(e.to_string()))?;
162                ee_params
163                    .distinguished_name
164                    .push(DnType::CommonName, &ee_subject);
165                ee_params.is_ca = IsCa::NoCa;
166                ee_params.not_before = time::OffsetDateTime::from_unix_timestamp(not_before_ts)
167                    .expect("valid timestamp");
168                ee_params.not_after = time::OffsetDateTime::from_unix_timestamp(not_after_ts)
169                    .expect("valid timestamp");
170
171                let cert = ee_params.signed_by(&ee_key_pair, &ca_issuer).map_err(|e| {
172                    AxiamError::Certificate(format!("certificate signing failed: {e}"))
173                })?;
174
175                let public_cert_pem = cert.pem();
176                let fingerprint = compute_fingerprint(cert.der());
177                Ok((private_key_pem, public_cert_pem, fingerprint))
178            })
179            .await
180            .map_err(|e| AxiamError::Internal(format!("spawn_blocking join error: {e}")))??;
181
182        let store = StoreCertificate {
183            tenant_id: input.tenant_id,
184            issuer_ca_id: input.issuer_ca_id,
185            subject: input.subject,
186            public_cert_pem,
187            fingerprint,
188            cert_type: input.cert_type,
189            key_algorithm: input.key_algorithm,
190            not_before,
191            not_after,
192            metadata: input.metadata.unwrap_or(serde_json::json!({})),
193        };
194
195        let certificate = self.cert_repo.create(store).await?;
196
197        Ok(GeneratedCertificate {
198            certificate,
199            private_key_pem,
200        })
201    }

203    pub async fn get(&self, tenant_id: Uuid, id: Uuid) -> AxiamResult<Certificate> {
204        self.cert_repo.get_by_id(tenant_id, id).await
205    }

207    pub async fn get_by_fingerprint(

230/// Decrypt an AES-256-GCM encrypted private key PEM and validate it is UTF-8.
231///
232/// X.509-specific wrapper around the shared [`decrypt_secret`] — the CA/leaf
233/// private key material is always stored as UTF-8 PEM text.
234fn decrypt_ca_key_pem(data: &[u8], key_bytes: &[u8; 32]) -> AxiamResult<String> {
235    let plaintext = decrypt_secret(data, key_bytes)?;
236    String::from_utf8(plaintext)
237        .map_err(|e| AxiamError::Crypto(format!("decrypted key is not valid UTF-8: {e}")))
238}
