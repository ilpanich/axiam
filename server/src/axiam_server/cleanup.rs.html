1//! Periodic cleanup task for expired federation rows, GDPR purges, and export jobs.
2//!
3//! SurrealDB v3 does not support native TTL on rows (RESEARCH §7), so this task
4//! periodically sweeps various tables:
5//! - `saml_assertion_replay` and `federation_login_state`: expired rows
6//! - `user`: accounts past their scheduled purge date (D-05/D-06/D-08)
7//! - `export_job`: queued jobs waiting to have their encrypted blob generated (D-12)
8//!
9//! The task shuts down cleanly when the caller sends `true` through the watch
10//! channel (D-09, D-24).

95/// Run the GDPR erasure pipeline for a single user: pseudonymize audit actor
96/// references, anonymize the user row, then write the erasure proof
97/// STRICTLY LAST.
98///
99/// Extracted as a free function generic over the three repo traits it needs
100/// (rather than a `CleanupTask` method) so a unit test can inject a
101/// synthetic failing `AuditLogRepository` double without depending on
102/// `CleanupTask`'s concrete `Arc<SurrealXxxRepository<C>>` fields — `pub` so
103/// `axiam-server`'s integration tests (which link this crate's library
104/// target) can call it directly.
105///
106/// Ordering is a hard security invariant (D-03a, SECHRD-06 / T-25-13):
107/// - `pseudonymize_actor` is now FATAL (`?`, no swallow-and-continue). A
108///   failed audit-actor scrub must abort the erasure — no PII-bearing step
109///   may ever be silently skipped (RESEARCH Pitfall 2).
110/// - `anonymize_user` runs BEFORE the proof (not after). It is the ONLY
111///   step that clears the user's `deletion_pending` flag that
112///   `find_due_for_purge` selects on, so if it never runs (an earlier step
113///   failed and aborted via `?`), the user remains re-selectable for a
114///   retry (RESEARCH Assumption A3).
115/// - `erasure_proof_repo.create` is the LITERAL LAST statement. It only
116///   fires once every PII-bearing step above has succeeded — a proof must
117///   never certify an erasure that did not fully happen (Pitfall 3). The DB
118///   UNIQUE index on `(tenant_id, user_id)` (plan 25-04) makes a retried
119///   erasure's duplicate proof insert an idempotent rejection (D-03b), not
120///   a silent overwrite.
121pub async fn run_erasure_pipeline<A, EP, U>(
122    audit_repo: &A,
123    erasure_proof_repo: &EP,
124    user_repo: &U,
125    tenant_id: Uuid,
126    user_id: Uuid,
127    pseudonym: &str,
128    email_hash: &str,
129) -> Result<(), AxiamError>
130where
131    A: AuditLogRepository,
132    EP: ErasureProofRepository,
133    U: UserRepository,
134{
135    // FATAL now (was: `if let Err(e) = ... { tracing::warn!(...) }`).
136    audit_repo
137        .pseudonymize_actor(tenant_id, user_id, pseudonym)
138        .await?;
139
140    // Clears `deletion_pending` — the re-selection anchor. Must run before
141    // the proof so a failure here (unreachable in practice since this is
142    // now the second of three fallible steps) still leaves the user due.
143    user_repo
144        .anonymize_user(tenant_id, user_id, email_hash, pseudonym)
145        .await?;
146
147    // Written STRICTLY LAST — only reached once every step above succeeded.
148    erasure_proof_repo
149        .create(CreateErasureProof {
150            pseudonym: pseudonym.to_string(),
151            tenant_id,
152            user_id,
153            erased_at: Utc::now(),
154        })
155        .await?;
156
157    Ok(())
158}

213    /// Run the cleanup loop until a shutdown signal is received.
214    ///
215    /// Never returns `Err` — all sweep errors are logged at `warn` level and
216    /// the loop continues (T-04-36).
217    pub async fn run(mut self) -> Result<(), AxiamError> {
218        let mut ticker = tokio::time::interval(self.interval);
219        // Skip ticks that were missed while the sweep was running to prevent
220        // catch-up storms after a pause (T-04-35).
221        ticker.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
222
223        loop {
224            tokio::select! {
225                _ = ticker.tick() => {
226                    // Existing federation cleanup sweeps.
227                    match self.replay_repo.cleanup_expired().await {
228                        Ok(n) if n > 0 => {
229                            tracing::debug!(deleted = n, "saml_assertion_replay cleanup");
230                        }
231                        Ok(_) => {}
232                        Err(e) => {
233                            tracing::warn!(error = ?e, "saml_assertion_replay cleanup failed");
234                        }
235                    }
236
237                    match self.state_repo.cleanup_expired().await {
238                        Ok(n) if n > 0 => {
239                            tracing::debug!(deleted = n, "federation_login_state cleanup");
240                        }
241                        Ok(_) => {}
242                        Err(e) => {
243                            tracing::warn!(error = ?e, "federation_login_state cleanup failed");
244                        }
245                    }
246
247                    // NEW-4: sweep expired AMQP replay-protection nonces.
248                    match self.amqp_nonce_repo.cleanup_expired().await {
249                        Ok(n) if n > 0 => {
250                            tracing::debug!(deleted = n, "amqp_nonce_replay cleanup");
251                        }
252                        Ok(_) => {}
253                        Err(e) => {
254                            tracing::warn!(error = ?e, "amqp_nonce_replay cleanup failed");
255                        }
256                    }
257
258                    // GDPR purge sweep (D-01..D-06, D-08).
259                    match self.sweep_pending_purges().await {
260                        Ok(n) if n > 0 => {
261                            tracing::info!(purged = n, "GDPR purge sweep completed");
262                        }
263                        Ok(_) => {}
264                        Err(e) => {
265                            tracing::warn!(error = ?e, "GDPR purge sweep failed");
266                        }
267                    }
268
269                    // GDPR export sweep (D-10..D-12).
270                    match self.sweep_pending_exports().await {
271                        Ok(n) if n > 0 => {
272                            tracing::info!(exported = n, "GDPR export sweep completed");
273                        }
274                        Ok(_) => {}
275                        Err(e) => {
276                            tracing::warn!(error = ?e, "GDPR export sweep failed");
277                        }
278                    }
279                }
280                changed = self.shutdown.changed() => {
281                    if changed.is_ok() && *self.shutdown.borrow() {
282                        tracing::info!("cleanup task received shutdown signal");
283                        return Ok(());
284                    }
285                }
286            }
287        }
288    }

294    /// Sweep users past their scheduled purge date and run the full purge pipeline.
295    ///
296    /// For each user:
297    /// (a) Revoke all sessions (auth artifact cascade via `AuthService`).
298    /// (b) Hard-delete federation links.
299    /// (b2) Hard-delete WebAuthn credentials and password history (SEC-056).
300    /// (b3) Prune `member_of` and `has_role` graph edges (isolated tombstone).
301    /// (c) Compute deterministic GDPR pseudonym via `gdpr_pseudonym`.
302    /// (d)/(e)/(g) Run [`run_erasure_pipeline`]: pseudonymize all audit
303    ///     entries for this user (now FATAL, D-01/D-03/D-04) -> anonymize the
304    ///     user row in-place (D-05) -> insert a PII-free erasure-proof
305    ///     record STRICTLY LAST (D-06, D-03a/D-03b — SECHRD-06).
306    /// (f) Mark the account_deletion row as completed.
307    /// (h) Emit `gdpr.user_pseudonymized` audit event (actor = System).
308    ///
309    /// Returns the count of users purged.
310    async fn sweep_pending_purges(&self) -> Result<u64, AxiamError> {
311        let pepper = match self.gdpr_pepper {
312            Some(p) => p,
313            None => {
314                tracing::warn!("GDPR purge sweep skipped — AXIAM__GDPR_PSEUDONYM_PEPPER not set");
315                return Ok(0);
316            }
317        };
318
319        let now = Utc::now();
320        let due = self.user_repo.find_due_for_purge(now).await?;
321        let mut purged: u64 = 0;
322
323        for user in due {
324            if let Err(e) = self
325                .purge_single_user(user.id, user.tenant_id, pepper)
326                .await
327            {
328                tracing::warn!(
329                    error = ?e,
330                    user_id = %user.id,
331                    tenant_id = %user.tenant_id,
332                    "purge failed for user — skipping"
333                );
334            } else {
335                purged += 1;
336            }
337        }
338
339        Ok(purged)
340    }

342    /// Run the full purge pipeline for a single user.
343    async fn purge_single_user(
344        &self,
345        user_id: Uuid,
346        tenant_id: Uuid,
347        pepper: [u8; 32],
348    ) -> Result<(), AxiamError> {
349        // (a) Revoke all sessions and OAuth2 refresh tokens.
350        self.auth_svc
351            .revoke_all_sessions(tenant_id, user_id)
352            .await?;
353
354        // (b) Hard-delete federation identity links.
355        match self
356            .federation_link_repo
357            .get_by_user_id(tenant_id, user_id)
358            .await
359        {
360            Ok(links) => {
361                for link in links {
362                    if let Err(e) = self.federation_link_repo.delete(tenant_id, link.id).await {
363                        tracing::warn!(
364                            error = %e,
365                            %tenant_id,
366                            link_id = %link.id,
367                            "cleanup: failed to delete expired federation link; will retry next cycle"
368                        );
369                    }
370                }
371            }
372            Err(e) => {
373                tracing::warn!(error = ?e, user_id = %user_id, "failed to list federation links for purge");
374            }
375        }
376
377        // (b2) Hard-delete stored credential material (SEC-056). Done with `?`
378        // (fail-closed): erasure must not be certified while a user's passkeys or
379        // password hashes remain. A transient failure aborts the purge and it is
380        // retried next sweep — every step here is idempotent.
381        let webauthn_creds = self.webauthn_repo.list_by_user(tenant_id, user_id).await?;
382        for cred in webauthn_creds {
383            self.webauthn_repo.delete(tenant_id, cred.id).await?;
384        }
385        // Prune the entire password-history chain (keep_count = 0 deletes all).
386        self.password_history_repo
387            .prune(tenant_id, user_id, 0)
388            .await?;
389
390        // (b3) Prune authorization-graph edges so the anonymized row is left as
391        // an isolated tombstone — no dangling `member_of` / `has_role` relations
392        // skewing group-member queries or leaving live authorization paths.
393        // Remove group memberships first so the subsequent role pass only sees
394        // the user's *direct* has_role edges (inherited ones vanish with the
395        // membership). Fail-closed (`?`): the tombstone must not be created while
396        // graph edges survive; every delete is idempotent and retried on failure.
397        let groups = self.group_repo.get_user_groups(tenant_id, user_id).await?;
398        for group in groups {
399            self.group_repo
400                .remove_member(tenant_id, user_id, group.id)
401                .await?;
402        }
403        let assignments = self
404            .role_repo
405            .get_user_role_assignments(tenant_id, user_id)
406            .await?;
407        for a in assignments {
408            self.role_repo
409                .unassign_from_user(tenant_id, user_id, a.role.id, a.resource_id)
410                .await?;
411        }
412
413        // (c) Compute deterministic pseudonym (keyed HMAC-SHA256, D-02).
414        let pseudonym = gdpr_pseudonym(&pepper, tenant_id, user_id);
415
416        // Derive email hash for anonymize_user from tenant+user IDs (original email
417        // is no longer accessible at purge time — user already marked deletion_pending).
418        use sha2::{Digest, Sha256};
419        let mut h = Sha256::new();
420        h.update(tenant_id.as_bytes());
421        h.update(user_id.as_bytes());
422        let email_hash = hex::encode(h.finalize());
423
424        // (d)/(e)/(g) Run the erasure pipeline: pseudonymize audit actor refs
425        // (now FATAL, was swallowed) -> anonymize the user row -> write the
426        // erasure proof STRICTLY LAST (SECHRD-06 / D-03a, T-25-13). A failure
427        // anywhere in this call propagates with `?` and aborts the purge:
428        // `anonymize_user` (the only step that clears `deletion_pending`)
429        // never ran, so the user stays due for a retry via
430        // `find_due_for_purge`, and NO erasure proof is ever written for an
431        // incomplete erasure.
432        run_erasure_pipeline(
433            self.audit_repo.as_ref(),
434            self.erasure_proof_repo.as_ref(),
435            self.user_repo.as_ref(),
436            tenant_id,
437            user_id,
438            &pseudonym,
439            &email_hash,
440        )
441        .await?;
442
443        // (f) Mark the account_deletion row as completed (lookup by user_id).
444        // Run AFTER the erasure pipeline succeeds — if the pipeline failed
445        // above, this is never reached and the deletion row stays pending
446        // for the next retry too.
447        match self
448            .account_deletion_repo
449            .find_pending_by_user_id(tenant_id, user_id)
450            .await
451        {
452            Ok(Some(deletion)) => {
453                if let Err(e) = self
454                    .account_deletion_repo
455                    .mark_completed(tenant_id, deletion.id)
456                    .await
457                {
458                    tracing::warn!(
459                        error = %e,
460                        %tenant_id,
461                        deletion_id = %deletion.id,
462                        "cleanup: failed to mark account_deletion completed"
463                    );
464                }
465            }
466            Ok(None) => {
467                tracing::debug!(user_id = %user_id, "no pending account_deletion row found at purge time");
468            }
469            Err(e) => {
470                tracing::warn!(error = ?e, user_id = %user_id, "failed to find account_deletion row");
471            }
472        }
473
474        // (h) Emit gdpr.user_pseudonymized audit event. actor_id is the
475        // erased subject's own (now-anonymized) row id — a real, resolvable
476        // identifier rather than a fabricated nil UUID (Task 1 fix); the row
477        // still exists (anonymize-in-place preserves referential integrity),
478        // so this is a legitimate audit-trail reference, not PII leakage.
479        // A DB-write failure here is dead-lettered to BOTH an append-only
480        // file AND a structured audit event (SECHRD-12 / D-02, T-24-61) —
481        // this legally-significant record must never be silently lost.
482        write_erasure_audit_with_dlq(
483            self.audit_repo.as_ref(),
484            CreateAuditLogEntry {
485                tenant_id,
486                actor_id: user_id,
487                actor_type: ActorType::System,
488                action: "gdpr.user_pseudonymized".into(),
489                resource_id: None,
490                outcome: AuditOutcome::Success,
491                ip_address: None,
492                metadata: Some(serde_json::json!({
493                    "pseudonym": pseudonym,
494                })),
495            },
496        )
497        .await;
498
499        tracing::info!(
500            pseudonym = %pseudonym,
501            tenant_id = %tenant_id,
502            "user purged and pseudonymized"
503        );
504
505        Ok(())
506    }

512    /// Sweep queued export jobs and generate the encrypted blobs.
513    ///
514    /// For each job:
515    /// (a) Aggregate all Art. 15 sections from the DB (secrets excluded, D-10).
516    /// (b) Serialize to a single sectioned JSON object (D-11).
517    /// (c) Encrypt with `export_encryption_key` (AES-256-GCM, D-12).
518    /// (d) Call `set_ready` with a SHA-256-hashed 24h single-use download token.
519    /// (e) Enqueue `ExportReady` mail with the raw token (D-12).
520    ///
521    /// Returns the count of jobs processed.
522    async fn sweep_pending_exports(&self) -> Result<u64, AxiamError> {
523        let key = match self.export_encryption_key {
524            Some(k) => k,
525            None => {
526                tracing::warn!("GDPR export sweep skipped — AXIAM__EMAIL_ENCRYPTION_KEY not set");
527                return Ok(0);
528            }
529        };
530
531        let queued = self.export_job_repo.find_queued().await?;
532        let mut processed: u64 = 0;
533
534        for job in queued {
535            if let Err(e) = self
536                .process_export_job(job.id, job.tenant_id, job.user_id, key)
537                .await
538            {
539                tracing::warn!(
540                    error = ?e,
541                    job_id = %job.id,
542                    "export job processing failed — marking Failed (CQ-B38)"
543                );
544                // Mark as Failed so the job does not stay stuck as Queued
545                // (CQ-B38 / REQ-14 AC-5).
546                if let Err(mark_err) = self.export_job_repo.mark_failed(job.id).await {
547                    tracing::warn!(
548                        error = ?mark_err,
549                        job_id = %job.id,
550                        "failed to mark export job as Failed"
551                    );
552                }
553            } else {
554                processed += 1;
555            }
556        }
557
558        Ok(processed)
559    }

561    /// Process a single queued export job.
562    async fn process_export_job(
563        &self,
564        job_id: Uuid,
565        tenant_id: Uuid,
566        user_id: Uuid,
567        key: [u8; 32],
568    ) -> Result<(), AxiamError> {
569        // (a)/(b) Aggregate Art. 15 inventory into one sectioned JSON.
570        let export_json = self.aggregate_export_data(tenant_id, user_id).await?;
571        let export_bytes = export_json.to_string().into_bytes();
572
573        // (c) Encrypt the blob (D-12).
574        let (nonce_b64, ct_b64) = encrypt_separate(&key, &export_bytes)
575            .map_err(|e| AxiamError::Internal(format!("export encrypt failed: {e}")))?;
576
577        // (d) Generate single-use 24h download token (D-13).
578        let raw_download_token = Uuid::new_v4().to_string();
579        let token_hash = {
580            use sha2::{Digest, Sha256};
581            let mut h = Sha256::new();
582            h.update(raw_download_token.as_bytes());
583            hex::encode(h.finalize())
584        };
585        let expires_at = Utc::now() + chrono::Duration::hours(24);
586
587        self.export_job_repo
588            .set_ready(
589                job_id,
590                token_hash,
591                Some(ct_b64),
592                None, // file_path: stored as DB blob
593                Some(nonce_b64),
594                expires_at,
595            )
596            .await?;
597
598        // Resolve the real org_id from the tenant before enqueuing the mail
599        // (SECHRD-08 / D-05d) — a Uuid::nil() org_id leaves the mail
600        // consumer unable to resolve the sending organization, producing
601        // undeliverable ExportReady mail.
602        let org_id = match self.tenant_repo.get_by_id(tenant_id).await {
603            Ok(tenant) => tenant.organization_id,
604            Err(e) => {
605                tracing::warn!(
606                    error = %e,
607                    %tenant_id,
608                    "failed to resolve org_id for ExportReady mail — falling back to nil (mail may be undeliverable)"
609                );
610                Uuid::nil()
611            }
612        };
613
614        // (e) Enqueue ExportReady email.
615        let download_url = format!("/api/v1/account/export/{}", raw_download_token);
616        let msg = OutboundMailMessage {
617            mail_type: MailType::ExportReady,
618            tenant_id,
619            org_id,
620            user_id,
621            to_address: String::new(), // mail consumer resolves from user_id
622            template_context: serde_json::json!({
623                "action_url": download_url,
624                "expiry_time": expires_at.to_rfc3339(),
625            }),
626            attempt_count: 0,
627            enqueued_at: Utc::now(),
628        };
629        if let Err(e) = self.mail_publisher.publish(msg).await {
630            tracing::warn!(error = %e, job_id = %job_id, "failed to enqueue ExportReady mail");
631        }
632
633        tracing::info!(job_id = %job_id, "export job completed");
634        Ok(())
635    }

637    /// Aggregate Art. 15 personal-data inventory for a user.
638    ///
639    /// EXCLUDED (D-10): `password_hash`, `mfa_secret`, any token_hash values.
640    async fn aggregate_export_data(
641        &self,
642        tenant_id: Uuid,
643        user_id: Uuid,
644    ) -> Result<serde_json::Value, AxiamError> {
645        // Profile — no password_hash or mfa_secret.
646        let user = self.user_repo.get_by_id(tenant_id, user_id).await?;
647        let profile = serde_json::json!({
648            "id": user.id,
649            "username": user.username,
650            "email": user.email,
651            "status": user.status,
652            "mfa_enabled": user.mfa_enabled,
653            "metadata": user.metadata,
654            "created_at": user.created_at,
655            "updated_at": user.updated_at,
656        });
657
658        // Consents — propagate errors (CQ-B38 / SEC-056): a section that fails to
659        // query must fail the whole export job rather than emit a legally
660        // incomplete Art. 15 inventory. `process_export_job` marks the job Failed.
661        let consents = self.consent_repo.list_by_user(tenant_id, user_id).await?;
662        let consents_json: Vec<_> = consents
663            .iter()
664            .map(|c| {
665                serde_json::json!({
666                    "consent_type": c.consent_type,
667                    "version": c.version,
668                    "accepted_at": c.accepted_at,
669                    "ip_address": c.ip_address,
670                })
671            })
672            .collect();
673
674        // Audit entries where this user was the actor — paginated to collect ALL
675        // entries regardless of volume (CQ-B38 / REQ-14 AC-5).
676        const AUDIT_PAGE_SIZE: u64 = 1_000;
677        let mut audit_items = Vec::new();
678        let mut offset: u64 = 0;
679        loop {
680            let page = self
681                .audit_repo
682                .list(
683                    tenant_id,
684                    AuditLogFilter {
685                        actor_id: Some(user_id),
686                        action: None,
687                        outcome: None,
688                        resource_id: None,
689                        from: None,
690                        to: None,
691                    },
692                    Pagination {
693                        offset,
694                        limit: AUDIT_PAGE_SIZE,
695                    },
696                )
697                .await?;
698            let fetched = page.items.len() as u64;
699            audit_items.extend(page.items);
700            offset += fetched;
701            if fetched < AUDIT_PAGE_SIZE {
702                break;
703            }
704        }
705        let audit_json: Vec<_> = audit_items
706            .iter()
707            .map(|e| {
708                serde_json::json!({
709                    "action": e.action,
710                    "outcome": e.outcome,
711                    "timestamp": e.timestamp,
712                    "resource_id": e.resource_id,
713                })
714            })
715            .collect();
716
717        // Sessions — real, metadata-only rows (D-03c, SECHRD-06). Token/hash
718        // material is live credential data and must NEVER appear in a GDPR
719        // export, so `token_hash` is deliberately excluded from the
720        // projection below. Propagate errors (CQ-B38 / SEC-056 convention):
721        // a section that fails to query must fail the whole export job.
722        let sessions = self.session_repo.list_by_user(tenant_id, user_id).await?;
723        let sessions_json: Vec<_> = sessions
724            .iter()
725            .map(|s| {
726                serde_json::json!({
727                    "id": s.id,
728                    "created_at": s.created_at,
729                    "expires_at": s.expires_at,
730                    "ip_address": s.ip_address,
731                    "user_agent": s.user_agent,
732                })
733            })
734            .collect();
735
736        // Federation identities — propagate errors (CQ-B38 / SEC-056).
737        let fed_links = self
738            .federation_link_repo
739            .get_by_user_id(tenant_id, user_id)
740            .await?;
741        let fed_json: Vec<_> = fed_links
742            .iter()
743            .map(|l| {
744                serde_json::json!({
745                    "federation_config_id": l.federation_config_id,
746                    "external_subject": l.external_subject,
747                    "created_at": l.created_at,
748                })
749            })
750            .collect();
751
752        // Role assignments (direct + inherited via groups), incl. resource scope.
753        let assignments = self
754            .role_repo
755            .get_user_role_assignments(tenant_id, user_id)
756            .await?;
757        let assignments_json: Vec<_> = assignments
758            .iter()
759            .map(|a| {
760                serde_json::json!({
761                    "role_id": a.role.id,
762                    "role_name": a.role.name,
763                    "is_global": a.role.is_global,
764                    "resource_id": a.resource_id,
765                })
766            })
767            .collect();
768
769        // Group memberships.
770        let groups = self.group_repo.get_user_groups(tenant_id, user_id).await?;
771        let groups_json: Vec<_> = groups
772            .iter()
773            .map(|g| {
774                serde_json::json!({
775                    "group_id": g.id,
776                    "name": g.name,
777                    "description": g.description,
778                })
779            })
780            .collect();
781
782        // WebAuthn credentials — metadata only; the encrypted `passkey_json`
783        // secret material is intentionally EXCLUDED (D-10).
784        let webauthn_creds = self.webauthn_repo.list_by_user(tenant_id, user_id).await?;
785        let webauthn_json: Vec<_> = webauthn_creds
786            .iter()
787            .map(|c| {
788                serde_json::json!({
789                    "id": c.id,
790                    "credential_id": c.credential_id,
791                    "name": c.name,
792                    "credential_type": c.credential_type,
793                    "created_at": c.created_at,
794                    "last_used_at": c.last_used_at,
795                })
796            })
797            .collect();
798
799        let export = serde_json::json!({
800            "export_metadata": {
801                "generated_at": Utc::now(),
802                "tenant_id": tenant_id,
803                "subject_id": user_id,
804                "schema_version": "1.0",
805            },
806            "profile": profile,
807            "consents": consents_json,
808            "sessions": sessions_json, // metadata only; NO token_hash (D-03c)
809            "mfa": { "enabled": user.mfa_enabled }, // NO mfa_secret
810            "federation_identities": fed_json,
811            "assignments": assignments_json,
812            "group_memberships": groups_json,
813            "audit_entries": audit_json,
814            "webauthn_credentials": webauthn_json,
815        });
816
817        Ok(export)
818    }
