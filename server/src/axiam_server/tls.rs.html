1//! Optional direct-TLS support for the REST API listener (ASVS V9.1.2/V9.1.3).
2//!
3//! TLS termination at a proxy/load balancer remains the recommended deployment
4//! pattern (D-06). This module provides an *opt-in* alternative for deployments
5//! that terminate TLS in-process: when `server.tls.enabled` is set, the server
6//! builds a rustls [`ServerConfig`] restricted to **TLS 1.3 only**. Restricting
7//! to TLS 1.3 also satisfies V9.1.3 — every TLS 1.3 cipher suite is
8//! ASVS-approved, so no manual cipher-suite filtering is required.
9//!
10//! The `ring` crypto provider is selected explicitly (rather than relying on a
11//! process-default provider) so `build_rustls_server_config` is self-contained
12//! and deterministic regardless of what other crates in the tree pull in.
13

25/// Number of resumption entries kept in the in-process session cache.
26///
27/// TLS 1.3 resumption is ticket-based (stateless — the ticket travels with the
28/// client), so this cache mainly bounds any non-ticket session state; a small
29/// value is plenty for a benchmark/service workload and caps memory.
30const RESUMPTION_CACHE_SIZE: usize = 512;
31
32/// The ALPN protocol list the rustls listener is *built with*, given the
33/// `http2` knob.
34///
35/// **Important caveat (B2):** for the REST listener this list is only fully
36/// authoritative when the rustls `ServerConfig` is used directly. actix-web's
37/// `HttpServer::bind_rustls_0_23` path funnels through
38/// `actix_http::HttpService::rustls_0_23_with_config`, which unconditionally
39/// **prepends** `["h2", "http/1.1"]` to whatever `alpn_protocols` we set. So
40/// with the current actix bind, `http2 = false` narrows the config's list to
41/// `http/1.1` but h2 is re-added and still wins negotiation. A *true*
42/// http/1.1-only TLS listener therefore requires either fronting with the
43/// `tls13-h1` nginx edge (see the benchmarks) or an H1-only actix service.
44/// This helper is unit-tested and records operator intent; it is authoritative
45/// for any non-actix consumer of the config.
46fn alpn_protocols(http2: bool) -> Vec<Vec<u8>> {
47    if http2 {
48        vec![b"h2".to_vec(), b"http/1.1".to_vec()]
49    } else {
50        vec![b"http/1.1".to_vec()]
51    }
52}

54/// Build a rustls [`ClientCertVerifier`] from the configured client-CA bundle
55/// (D3 native mTLS).
56///
57/// Loads the PEM bundle at `client_ca_path` into a [`RootCertStore`] and builds
58/// a [`WebPkiClientVerifier`] over it, using the same [`CryptoProvider`] as the
59/// server config. For [`ClientAuth::Optional`] the verifier still *offers* and
60/// *verifies* client certs but permits anonymous clients
61/// (`allow_unauthenticated`); for [`ClientAuth::Required`] a verified client
62/// cert is mandatory.
63///
64/// Fails fast (aborting startup) when the CA path is unset, missing/unreadable,
65/// empty, or malformed — matching this file's existing `io::Error` style so a
66/// misconfigured mTLS server never starts.
67fn build_client_cert_verifier(
68    tls: &TlsConfig,
69    provider: &Arc<rustls::crypto::CryptoProvider>,
70) -> io::Result<Arc<dyn ClientCertVerifier>> {
71    let ca_path = tls.client_ca_path.as_ref().ok_or_else(|| {
72        io::Error::new(
73            io::ErrorKind::InvalidInput,
74            "server.tls.client_auth is optional/required but \
75             server.tls.client_ca_path is not set",
76        )
77    })?;
78
79    let ca_file = File::open(ca_path).map_err(|e| {
80        io::Error::new(
81            e.kind(),
82            format!(
83                "failed to open TLS client CA bundle {}: {e}",
84                ca_path.display()
85            ),
86        )
87    })?;
88    let ca_certs: Vec<CertificateDer<'static>> = CertificateDer::pem_reader_iter(ca_file)
89        .collect::<Result<Vec<_>, _>>()
90        .map_err(|e| {
91            io::Error::new(
92                io::ErrorKind::InvalidData,
93                format!(
94                    "failed to parse client CA certificates from {}: {e}",
95                    ca_path.display()
96                ),
97            )
98        })?;
99    if ca_certs.is_empty() {
100        return Err(io::Error::new(
101            io::ErrorKind::InvalidData,
102            format!("no client CA certificates found in {}", ca_path.display()),
103        ));
104    }
105
106    let mut roots = RootCertStore::empty();
107    for cert in ca_certs {
108        roots.add(cert).map_err(|e| {
109            io::Error::new(
110                io::ErrorKind::InvalidData,
111                format!(
112                    "invalid client CA certificate in {}: {e}",
113                    ca_path.display()
114                ),
115            )
116        })?;
117    }
118
119    let builder = WebPkiClientVerifier::builder_with_provider(Arc::new(roots), provider.clone());
120    let verifier = match tls.client_auth {
121        // Verify presented certs but accept anonymous clients.
122        ClientAuth::Optional => builder.allow_unauthenticated().build(),
123        // Require a verified client cert (Off is unreachable here).
124        _ => builder.build(),
125    }
126    .map_err(|e| io::Error::other(format!("failed to build client cert verifier: {e}")))?;
127
128    Ok(verifier)
129}

131/// Build a TLS 1.3-only rustls [`ServerConfig`] from the configured PEM files.
132///
133/// Fails fast (returning an `io::Error` that aborts startup) when TLS is enabled
134/// but misconfigured: a missing `cert_path`/`key_path`, an unreadable or
135/// malformed PEM file, an empty cert chain, or a cert/key mismatch.
136pub fn build_rustls_server_config(tls: &TlsConfig) -> io::Result<ServerConfig> {
137    let cert_path = tls.cert_path.as_ref().ok_or_else(|| {
138        io::Error::new(
139            io::ErrorKind::InvalidInput,
140            "server.tls.enabled is true but server.tls.cert_path is not set",
141        )
142    })?;
143    let key_path = tls.key_path.as_ref().ok_or_else(|| {
144        io::Error::new(
145            io::ErrorKind::InvalidInput,
146            "server.tls.enabled is true but server.tls.key_path is not set",
147        )
148    })?;
149
150    // Open the files explicitly so a missing/unreadable path yields a clean
151    // io::Error (NotFound / PermissionDenied) before any PEM parsing. Cert and
152    // key are parsed via rustls-pki-types' `PemObject` trait directly —
153    // rustls-pemfile is unmaintained (RUSTSEC-2025-0134) and is a thin wrapper
154    // over this same code.
155    let cert_file = File::open(cert_path).map_err(|e| {
156        io::Error::new(
157            e.kind(),
158            format!("failed to open TLS cert file {}: {e}", cert_path.display()),
159        )
160    })?;
161    let cert_chain: Vec<CertificateDer<'static>> = CertificateDer::pem_reader_iter(cert_file)
162        .collect::<Result<Vec<_>, _>>()
163        .map_err(|e| {
164            io::Error::new(
165                io::ErrorKind::InvalidData,
166                format!(
167                    "failed to parse TLS certificates from {}: {e}",
168                    cert_path.display()
169                ),
170            )
171        })?;
172    if cert_chain.is_empty() {
173        return Err(io::Error::new(
174            io::ErrorKind::InvalidData,
175            format!("no certificates found in {}", cert_path.display()),
176        ));
177    }
178
179    let key_file = File::open(key_path).map_err(|e| {
180        io::Error::new(
181            e.kind(),
182            format!("failed to open TLS key file {}: {e}", key_path.display()),
183        )
184    })?;
185    let key = PrivateKeyDer::from_pem_reader(key_file).map_err(|e| {
186        io::Error::new(
187            io::ErrorKind::InvalidData,
188            format!(
189                "failed to read a TLS private key from {}: {e}",
190                key_path.display()
191            ),
192        )
193    })?;
194
195    // TLS 1.3 only (ASVS V9.1.2). ring provider selected explicitly.
196    let provider = Arc::new(rustls::crypto::ring::default_provider());
197    let builder = ServerConfig::builder_with_provider(provider.clone())
198        .with_protocol_versions(&[&rustls::version::TLS13])
199        .map_err(|e| io::Error::other(format!("rustls TLS 1.3 configuration failed: {e}")))?;
200
201    // Client-certificate (mTLS) policy (D3). `off` keeps the server-auth-only
202    // behaviour (`with_no_client_auth`); `optional`/`required` install a
203    // WebPkiClientVerifier over the configured CA bundle so rustls verifies the
204    // client cert during the handshake and the *verified* cert (not a header)
205    // drives certificate-based identity.
206    let builder = match tls.client_auth {
207        ClientAuth::Off => builder.with_no_client_auth(),
208        ClientAuth::Optional | ClientAuth::Required => {
209            let verifier = build_client_cert_verifier(tls, &provider)?;
210            builder.with_client_cert_verifier(verifier)
211        }
212    };
213
214    let mut config = builder.with_single_cert(cert_chain, key).map_err(|e| {
215        io::Error::new(
216            io::ErrorKind::InvalidData,
217            format!("invalid TLS certificate/key pair: {e}"),
218        )
219    })?;
220
221    // ALPN (B2): advertise h2+http/1.1 by default, or http/1.1-only when the
222    // operator disables h2. See `alpn_protocols` for the actix-bind caveat.
223    config.alpn_protocols = alpn_protocols(tls.http2);
224
225    // TLS 1.3 session resumption (B2). Without a ticketer + session store a
226    // rustls server does a *full* handshake on every connection; k6 opens many
227    // short-lived connections per VU, so a full ECDHE handshake per request is a
228    // per-request fixed cost that shows up as the ~2× p50 inflation on the token
229    // endpoints. Enabling stateless TLS 1.3 tickets lets repeat connections
230    // resume (PSK) instead. We deliberately do NOT enable 0-RTT/early-data:
231    // rustls' `max_early_data_size` stays at its default 0 because the token
232    // endpoints are non-idempotent POSTs and early data is replayable
233    // (see docs/security-profiles.md).
234    config.session_storage = ServerSessionMemoryCache::new(RESUMPTION_CACHE_SIZE);
235    match rustls::crypto::ring::Ticketer::new() {
236        Ok(ticketer) => config.ticketer = ticketer,
237        Err(e) => tracing::warn!(
238            error = %e,
239            "failed to construct a TLS ticketer; session resumption disabled \
240             (falling back to full handshakes)"
241        ),
242    }
243
244    Ok(config)
245}

323    /// Write `contents` to a unique temp file and return its path.
324    fn write_tmp(tag: &str, contents: &str) -> PathBuf {
325        let n = TMP_COUNTER.fetch_add(1, Ordering::Relaxed);
326        let path =
327            std::env::temp_dir().join(format!("axiam-d3-{}-{n}-{tag}.pem", std::process::id()));
328        let mut f = File::create(&path).expect("create temp pem");
329        f.write_all(contents.as_bytes()).expect("write temp pem");
330        path
331    }

341    /// Generate a throwaway CA plus a server leaf (SAN `localhost`) and a client
342    /// leaf (SAN `URI:spiffe://axiam/device-01`), all signed by the CA.
343    fn gen_test_pki() -> TestPki {
344        let ca_key = KeyPair::generate_for(&rcgen::PKCS_ED25519).unwrap();
345        let mut ca_params = CertificateParams::new(Vec::<String>::new()).unwrap();
346        ca_params.is_ca = IsCa::Ca(BasicConstraints::Unconstrained);
347        ca_params.key_usages = vec![KeyUsagePurpose::KeyCertSign, KeyUsagePurpose::CrlSign];
348        let ca_cert = ca_params.self_signed(&ca_key).unwrap();
349        let issuer = Issuer::from_params(&ca_params, &ca_key);
350
351        // Server leaf: SAN localhost so the in-process client can verify it.
352        let server_key = KeyPair::generate_for(&rcgen::PKCS_ED25519).unwrap();
353        let mut server_params = CertificateParams::new(vec!["localhost".to_string()]).unwrap();
354        server_params.is_ca = IsCa::NoCa;
355        let server_cert = server_params.signed_by(&server_key, &issuer).unwrap();
356
357        // Client leaf: carries a URI SAN we assert on after extraction.
358        let client_key = KeyPair::generate_for(&rcgen::PKCS_ED25519).unwrap();
359        let mut client_params = CertificateParams::new(Vec::<String>::new()).unwrap();
360        client_params.is_ca = IsCa::NoCa;
361        client_params
362            .subject_alt_names
363            .push(SanType::URI("spiffe://axiam/device-01".try_into().unwrap()));
364        let client_cert = client_params.signed_by(&client_key, &issuer).unwrap();
365
366        TestPki {
367            ca_pem: ca_cert.pem(),
368            server_cert_pem: server_cert.pem(),
369            server_key_pem: server_key.serialize_pem(),
370            client_cert_pem: client_cert.pem(),
371            client_key_pem: client_key.serialize_pem(),
372        }
373    }

457    /// Pump handshake records between two in-memory rustls connections until
458    /// both finish handshaking or one errors. Returns the first processing
459    /// error (e.g. the server rejecting a missing required client cert).
460    fn drive_handshake(
461        client: &mut rustls::Connection,
462        server: &mut rustls::Connection,
463    ) -> Result<(), rustls::Error> {
464        for _ in 0..16 {
465            let mut buf = Vec::new();
466            while client.wants_write() {
467                client.write_tls(&mut buf).unwrap();
468            }
469            let mut rd: &[u8] = &buf;
470            while !rd.is_empty() {
471                server.read_tls(&mut rd).unwrap();
472            }
473            server.process_new_packets()?;
474
475            let mut buf = Vec::new();
476            while server.wants_write() {
477                server.write_tls(&mut buf).unwrap();
478            }
479            let mut rd: &[u8] = &buf;
480            while !rd.is_empty() {
481                client.read_tls(&mut rd).unwrap();
482            }
483            client.process_new_packets()?;
484
485            if !client.is_handshaking() && !server.is_handshaking() {
486                return Ok(());
487            }
488        }
489        Ok(())
490    }

505    /// Client config trusting the CA; `client_cert` toggles whether it presents
506    /// its own certificate.
507    fn make_client(pki: &TestPki, present_cert: bool) -> rustls::ClientConnection {
508        use rustls::pki_types::pem::PemObject;
509        use rustls::pki_types::{CertificateDer, PrivateKeyDer, ServerName};
510
511        let mut roots = RootCertStore::empty();
512        roots
513            .add(CertificateDer::from_pem_slice(pki.ca_pem.as_bytes()).unwrap())
514            .unwrap();
515
516        let provider = Arc::new(rustls::crypto::ring::default_provider());
517        let builder = rustls::ClientConfig::builder_with_provider(provider)
518            .with_protocol_versions(&[&rustls::version::TLS13])
519            .unwrap()
520            .with_root_certificates(roots);
521
522        let config = if present_cert {
523            let chain =
524                vec![CertificateDer::from_pem_slice(pki.client_cert_pem.as_bytes()).unwrap()];
525            let key = PrivateKeyDer::from_pem_slice(pki.client_key_pem.as_bytes()).unwrap();
526            builder.with_client_auth_cert(chain, key).unwrap()
527        } else {
528            builder.with_no_client_auth()
529        };
530
531        let name = ServerName::try_from("localhost").unwrap();
532        rustls::ClientConnection::new(Arc::new(config), name).expect("client connection")
533    }

547    #[test]
548    fn handshake_accepts_bench_client_cert_and_exposes_verified_peer_cert() {
549        let pki = gen_test_pki();
550        let mut server = rustls::Connection::Server(make_server(&pki));
551        let mut client = rustls::Connection::Client(make_client(&pki, true));
552        drive_handshake(&mut client, &mut server)
553            .expect("handshake with a CA-signed client cert must succeed");
554        assert!(
555            !server.is_handshaking(),
556            "server handshake should complete with a valid client cert"
557        );
558
559        // The VERIFIED peer certificate is what handlers consume (never a header).
560        let peer = server
561            .peer_certificates()
562            .expect("verified client cert must be present after mTLS handshake");
563        let leaf = peer.first().expect("at least one peer cert");
564
565        // SAN extraction (the axiam-api-rest side of D3) must find the URI SAN.
566        let verified = axiam_api_rest::VerifiedClientCert::from_der(leaf.as_ref())
567            .expect("verified client cert must parse");
568        assert!(
569            verified
570                .sans
571                .iter()
572                .any(|s| s == "URI:spiffe://axiam/device-01"),
573            "expected the client URI SAN to be extracted, got {:?}",
574            verified.sans
575        );
576        assert_eq!(
